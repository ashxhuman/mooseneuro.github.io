17LookupTable::LookupTable(
18 double min, double max, unsigned int nDivs, unsigned int nSpecies )
19{
20 min_ = min;
21 max_ = max;
22 // Number of points is 1 more than number of divisions.
23 // Then add one more since we may interpolate at the last point in the table.
24 nPts_ = nDivs + 1 + 1;
25 dx_ = ( max - min ) / nDivs;
26 // Every row has 2 entries for each type of gate
27 nColumns_ = 2 * nSpecies;
28
29 //~ interpolate_.resize( nSpecies );
30 table_.resize( nPts_ * nColumns_ );
31}

33void LookupTable::addColumns(
34 int species,
35 const vector< double >& C1,
36 const vector< double >& C2 )
37 //~ const vector< double >& C2,
38 //~ bool interpolate )
39{
40 vector< double >::const_iterator ic1 = C1.begin();
41 vector< double >::const_iterator ic2 = C2.begin();
42 vector< double >::iterator iTable = table_.begin() + 2 * species;
43 // Loop until last but one point
44 for ( unsigned int igrid = 0; igrid < nPts_ - 1 ; ++igrid ) {
45 *( iTable ) = *ic1;
46 *( iTable + 1 ) = *ic2;
47
48 iTable += nColumns_;
49 ++ic1, ++ic2;
50 }
51 // Then duplicate the last point
52 *( iTable ) = C1.back();
53 *( iTable + 1 ) = C2.back();
54
55 //~ interpolate_[ species ] = interpolate;
56}

64void LookupTable::row( double x, LookupRow& row )
65{
66 if(table_.empty()) {
67 cerr << "LookupTable::row : Error: table is empty" << endl;
68 return;
69 }
70 if ( x < min_ )
71 x = min_;
72 else if ( x > max_ )
73 x = max_;
74
75 double div = ( x - min_ ) / dx_;
76 unsigned int integer = ( unsigned int )( div );
77
78 row.fraction = div - integer;
79 row.row = &( table_.front() ) + integer * nColumns_;
80}

82void LookupTable::lookup(
83 const LookupColumn& column,
84 const LookupRow& row,
85 double& C1,
86 double& C2 )
87{
88 double a, b;
89 double *ap, *bp;
90
91 ap = row.row + column.column;
92
93 //~ if ( ! column.interpolate ) {
94 //~ C1 = *ap;
95 //~ C2 = *( ap + 1 );
96 //~
97 //~ return;
98 //~ }
99
100 bp = ap + nColumns_;
101
102 a = *ap;
103 b = *bp;
104 C1 = a + ( b - a ) * row.fraction;
105
106 a = *( ap + 1 );
107 b = *( bp + 1 );
108 C2 = a + ( b - a ) * row.fraction;
109}
