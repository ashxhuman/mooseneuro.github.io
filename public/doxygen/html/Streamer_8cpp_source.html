168/**
169 * @brief Reinit.
170 *
171 * @param e
172 * @param p
173 */
174void Streamer::reinit(const Eref& e, ProcPtr p)
175{
176
177 if( tables_.size() == 0 )
178 {
179 moose::showWarn( "Zero tables in streamer. Disabling Streamer" );
180 e.element()->setTick( -2 ); /* Disable process */
181 return;
182 }
183
184 Clock* clk = reinterpret_cast<Clock*>( Id(1).eref().data() );
185 for (unsigned int i = 0; i < tableIds_.size(); i++)
186 {
187 int tickNum = tableIds_[i].element()->getTick();
188 double tick = clk->getTickDt( tickNum );
189 tableDt_.push_back( tick );
190 // Make sure that all tables have the same tick.
191 if( i > 0 )
192 {
193 if( tick != tableDt_[0] )
194 {
195 moose::showWarn( "Table " + tableIds_[i].path() + " has "
196 " different clock dt. "
197 " Make sure all tables added to Streamer have the same "
198 " dt value."
199 );
200 }
201 }
202 }
203
204
205 // Push each table dt_ into vector of dt
206 for( unsigned int i = 0; i < tables_.size(); i++)
207 {
208 Id tId = tableIds_[i];
209 int tickNum = tId.element()->getTick();
210 tableDt_.push_back( clk->getTickDt(tickNum) );
211 }
212
213 // Make sure all tables have same dt_ else disable the streamer.
214 vector<unsigned int> invalidTables;
215 for (unsigned int i = 1; i < tableTick_.size(); i++)
216 {
217 if( tableTick_[i] != tableTick_[0] )
218 {
219 LOG( moose::warning
220 , "Table " << tableIds_[i].path()
221 << " has tick (dt) which is different than the first table."
222 << endl
223 << " Got " << tableTick_[i] << " expected " << tableTick_[0]
224 << endl << " Disabling this table."
225 );
226 invalidTables.push_back( i );
227 }
228 }
229
230 for (unsigned int i = 0; i < invalidTables.size(); i++)
231 {
232 tables_.erase( tables_.begin() + i );
233 tableDt_.erase( tableDt_.begin() + i );
234 tableIds_.erase( tableIds_.begin() + i );
235 }
236
237 if( ! isOutfilePathSet_ )
238 {
239 string defaultPath = "_tables/" + moose::moosePathToUserPath( e.id().path() );
240 setDatafilePath( defaultPath );
241 }
242
243 // Prepare data. Add columns names and write whatever values are available
244 // write now.
245 currTime_ = 0.0;
246 zipWithTime( );
247 StreamerBase::writeToOutFile(datafilePath_, format_, WRITE, data_, columns_);
248 data_.clear( );
249}

278/**
279 * @brief Add a table to streamer.
280 *
281 * @param table Id of table.
282 */
283void Streamer::addTable( ObjId table )
284{
285 // If this table is not already in the vector, add it.
286 for( unsigned int i = 0; i < tableIds_.size(); i++)
287 if( table.path() == tableIds_[i].path() )
288 return; /* Already added. */
289
290 Table* t = reinterpret_cast<Table*>(table.eref().data());
291 tableIds_.push_back( table );
292 tables_.push_back( t );
293 tableTick_.push_back( table.element()->getTick() );
294
295 // NOTE: Table can also have name. If name is set by User, use the name to
296 // for column name, else use modify the table path to generate a suitable
297 // column name.
298 if(t->getColumnName().size() > 0)
299 columns_.push_back(t->getColumnName());
300 else
301 columns_.push_back(moose::moosePathToColumnName(table.path()));
302}

319 * @brief Remove a table from Streamer.
320 *
321 * @param table. Id of table.
322 */
323void Streamer::removeTable( ObjId table )
324{
325 int matchIndex = -1;
326 for (unsigned int i = 0; i < tableIds_.size(); i++)
327 if( table.path() == tableIds_[i].path() )
328 {
329 matchIndex = i;
330 break;
331 }
332
333 if( matchIndex > -1 )
334 {
335 tableIds_.erase( tableIds_.begin() + matchIndex );
336 tables_.erase( tables_.begin() + matchIndex );
337 columns_.erase( columns_.begin() + matchIndex );
338 }
339}

407/**
408 * @brief This function prepares data to be written to a file.
409 */
410void Streamer::zipWithTime( )
411{
412 unsigned int numEntriesInEachTable = tables_[0]->getVecSize( );
413
414 //LOG( moose::debug, "Entries in each table " << numEntriesInEachTable );
415
416 // Collect data from all table. If some table does not have enough data,
417 // fill it with nan
418 vector< vector< double > > collectedData;
419 for( unsigned int i = 0; i < tables_.size( ); i++ )
420 {
421 vector<double> tVec( tables_[i]->getVector( ) );
422 if( tVec.size( ) <= numEntriesInEachTable )
423 {
424#if 0
425 LOG( moose::debug
426 , "Table " << tables_[i]->getName( ) << " is not functional. Filling with zero "
427 );
428#endif
429 tVec.resize( numEntriesInEachTable, 0 );
430 }
431 collectedData.push_back( tVec );
432 }
433
434 // Turn it into a table format. Its like taking a transpose of vector<
435 // vector >.
436 double allTableDt = tableDt_[ 0 ];
437 for( unsigned int i = 0; i < collectedData[0].size( ); i++ )
438 {
439 data_.push_back( currTime_ );
440 currTime_ += allTableDt;
441 for( unsigned int ii = 0; ii < collectedData.size(); ii++ )
442 data_.push_back( collectedData[ ii ][ i ] );
443 }
444
445 // After collection data from table, clear tables.
446 for(unsigned int i = 0; i < tables_.size(); i++ )
447 tables_[i]->clearVec( );
448
449 return;
450}
