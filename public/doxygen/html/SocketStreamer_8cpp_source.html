244void SocketStreamer::initUDSServer( void )
245{
246 // PF_UNIX means that sockets are local.
247 sockfd_ = socket(PF_UNIX, SOCK_STREAM, 0);
248 if( ! sockfd_)
249 {
250 isValid_ = false;
251 perror( "Socket" );
252 }
253
254 if( sockfd_ > 0 )
255 {
256 bzero(&sockAddrUDS_, sizeof(sockAddrUDS_));
257 sockAddrUDS_.sun_family = AF_UNIX;
258 strncpy(sockAddrUDS_.sun_path, sockInfo_.filepath.c_str(), sizeof(sockAddrUDS_.sun_path)-1);
259 configureSocketServer();
260
261 // Bind. Make sure bind is not std::bind
262 if(0 > ::bind(sockfd_, (struct sockaddr*) &sockAddrUDS_, sizeof(sockAddrUDS_)))
263 {
264 isValid_ = false;
265 LOG(moose::warning, "Warn: Failed to create socket at " << sockInfo_.filepath
266 << ". File descriptor: " << sockfd_
267 << ". Erorr: " << strerror(errno)
268 );
269 }
270
271 if(access(sockInfo_.filepath.c_str(), F_OK) != 0)
272 {
273 LOG( moose::warning, "No file " << sockInfo_.filepath << " exists." );
274 isValid_ = false;
275 }
276 }
277
278 if( (! isValid_) || (sockfd_ < 0) )
279 {
280 LOG( moose::warning, "Failed to create socket : " << sockInfo_ );
281 ::unlink( sockInfo_.filepath.c_str() );
282 }
283}

285void SocketStreamer::initTCPServer( void )
286{
287 // Create a blocking socket.
288 LOG( moose::debug, "Creating TCP socket on port: " << sockInfo_.port );
289 sockfd_ = socket(AF_INET, SOCK_STREAM, 0);
290 if( 0 > sockfd_ )
291 {
292 perror("socket");
293 isValid_ = false;
294 return;
295 }
296
297 configureSocketServer();
298 bzero((char*) &sockAddrTCP_, sizeof(sockAddrTCP_));
299 sockAddrTCP_.sin_family = AF_INET;
300 sockAddrTCP_.sin_addr.s_addr = INADDR_ANY;
301 sockAddrTCP_.sin_port = htons( sockInfo_.port );
302
303 // Bind. Make sure bind is not std::bind
304 if(0 > ::bind(sockfd_, (struct sockaddr*) &sockAddrTCP_, sizeof(sockAddrTCP_)))
305 {
306 isValid_ = false;
307 LOG(moose::warning, "Warn: Failed to create server at "
308 << sockInfo_.host << ":" << sockInfo_.port
309 << ". File descriptor: " << sockfd_
310 << ". Erorr: " << strerror(errno)
311 );
312 return;
313 }

334/* --------------------------------------------------------------------------*/
335/**
336 * @Synopsis Stream data over socket.
337 *
338 * @Returns True of success, false otherwise. It is callee job to clean up data_
339 * on a successful return from this function.
340 */
341/* ----------------------------------------------------------------------------*/
342int SocketStreamer::streamData( )
343{
344 map<string, vector<double>> data;
345 dataToStream(data);
346
347 if(data.empty())
348 {
349 LOG(moose::debug, "No data in tables.");
350 return 0;
351 }
352
353 // Construct a void* array to send over the socket. Serialize the data.
354 // e.g. H 10 / a / t a b l e / a V 4 0.1 0.2 0.3 0.2
355 // H => Header start (chars)
356 // V => Value starts (double)
357 for(auto v: data)
358 {
359 // First header.
360 vecToStream_.push_back((double)'H');
361 vecToStream_.push_back(v.first.size());
362 addStringToDoubleVec(vecToStream_, v.first);
363
364 // Now data.
365 vecToStream_.push_back((double)'V');
366 vecToStream_.push_back(v.second.size());
367 vecToStream_.insert(vecToStream_.end(), v.second.begin(), v.second.end());
368 }
369
370 unsigned int dtypeSize = sizeof(double);
371 int sent = send(clientfd_, (void*) &vecToStream_[0], dtypeSize*vecToStream_.size(), MSG_MORE);
372 LOG(moose::debug, "Sent " << sent << " bytes." );
373 if( sent < 0 )
374 return errno;
375
376 vecToStream_.erase(vecToStream_.begin(), vecToStream_.begin()+(sent/dtypeSize));
377 return 0;
378}

389void SocketStreamer::connect( )
390{
391 // If server was invalid then there is no point.
392 if( ! isValid_ )
393 return;
394
395 struct sockaddr_storage clientAddr;
396 socklen_t addrLen = sizeof(clientAddr);
397 clientfd_ = ::accept(sockfd_,(struct sockaddr*) &clientAddr, &addrLen);
398 assert( clientfd_ );
399
400 char ipstr[INET6_ADDRSTRLEN];
401 int port = -1;
402 // deal with both IPv4 and IPv6:
403 if (clientAddr.ss_family == AF_INET)
404 {
405 struct sockaddr_in *s = (struct sockaddr_in *)&clientAddr;
406 port = ntohs(s->sin_port);
407 inet_ntop(AF_INET, &s->sin_addr, ipstr, sizeof ipstr);
408 }
409 else
410 { // AF_INET6
411 struct sockaddr_in6 *s = (struct sockaddr_in6 *)&clientAddr;
412 port = ntohs(s->sin6_port);
413 inet_ntop(AF_INET6, &s->sin6_addr, ipstr, sizeof ipstr);
414 }
415
416 LOG(moose::info, "Connected to " << ipstr << ':' << port);
417 return;
418}

434/**
435 * @brief Reinit. We make sure it is non-blocking.
436 *
437 * @param e
438 * @param p
439 */
440void SocketStreamer::reinit(const Eref& e, ProcPtr p)
441{
442 if( tables_.size() == 0 )
443 {
444 moose::showWarn( "No table found. Disabling SocketStreamer.\nDid you forget"
445 " to call addTables() on SocketStreamer object. " + e.objId().path()
446 );
447 e.element()->setTick( -2 ); /* Disable process */
448 return;
449 }
450
451 thisDt_ = clk_->getTickDt( e.element()->getTick() );
452
453 // Push each table dt_ into vector of dt
454 for( unsigned int i = 0; i < tables_.size(); i++)
455 {
456 Id tId = tableIds_[i];
457 int tickNum = tId.element()->getTick();
458 tableDt_.push_back( clk_->getTickDt( tickNum ) );
459 }
460
461 // This should only be called once.
462 initServer();
463
464 // Launch a thread in background which monitors the any client trying to
465 // make connection to server.
466 processThread_ = std::thread(&SocketStreamer::connect, this);
467
468 // NOw introduce some delay.
469 timeStamp_ = std::chrono::high_resolution_clock::now();
470}

486/**
487 * @brief Add a table to streamer.
488 *
489 * @param table Id of table.
490 */
491void SocketStreamer::addTable( ObjId table )
492{
493 // If this table is not already in the vector, add it.
494 for( unsigned int i = 0; i < tableIds_.size(); i++)
495 if( table.path() == tableIds_[i].path() )
496 return; /* Already added. */
497
498 Table* t = reinterpret_cast<Table*>(table.eref().data());
499 tableIds_.push_back( table );
500 tables_.push_back( t );
501 tableTick_.push_back( table.element()->getTick() );
502
503 // NOTE: If user can make sure that names are unique in table, using name is
504 // better than using the full path.
505 if( t->getColumnName().size() > 0 )
506 columns_.push_back( t->getColumnName( ) );
507 else
508 columns_.push_back( moose::moosePathToUserPath( table.path() ) );
509}
