450void PoolBase::setN( const Eref& e, double v )
451{

566void PoolBase::setSolvers( const Eref& e, ObjId ks, ObjId ds )
567{
568 if ( ks == ObjId() ) {
569 if ( ksolve_ )
570 ksolve_->notifyRemovePool( e );
571 ksolve_ = 0;
572 } else if ( ! ks.bad() ) {
573 string nm = ks.element()->cinfo()->name();
574 if ( nm == "Ksolve" || nm == "Gsolve" ) {
575 KsolveBase* k = reinterpret_cast< KsolveBase *>(ks.data() );
576 if ( k && k != ksolve_ ) {
577 if ( ksolve_ )
578 ksolve_->notifyRemovePool( e );
579 ksolve_ = k;
580 k->notifyAddPool( e );
581 }
582 }
583 }
584 if ( ds == ObjId() ) {
585 if ( dsolve_ ) {
586 dsolve_->notifyRemovePool( e );
587 }
588 dsolve_ = 0;
589 } else if ( ! ds.bad() ) {
590 string nm = ds.element()->cinfo()->name();
591 if ( nm == "Dsolve" ) {
592 KsolveBase* d = reinterpret_cast< KsolveBase *>(ds.data() );
593 if ( d && d != dsolve_ ) {
594 if ( dsolve_ )
595 dsolve_->notifyRemovePool( e );
596 dsolve_ = d;
597 d->notifyAddPool( e );
598 }
599 }
600 }

603/**
604 * Changing the buffering flag changes the class between Pool and BufPool.
605 */
606void PoolBase::setIsBuffered( const Eref& e, bool v )
607{
608 Element* elm = e.element();
609 bool isBuf = ( elm->cinfo() == bufPoolCinfo );
610 if ( v == isBuf ) return;
611
612 if ( v ) {
613 elm->replaceCinfo( bufPoolCinfo );
614 } else {
615 elm->replaceCinfo( poolCinfo );
616 }
617
618 if ( ksolve_ )
619 ksolve_->setIsBuffered( e, v );
620 if ( dsolve_ )
621 dsolve_->setIsBuffered( e, v );
622}
