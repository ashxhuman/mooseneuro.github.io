305void CubeMesh::fillTwoDimSurface()
306{
307 unsigned int size = nx_ * ny_ * nz_;
308 if ( nx_ == 1 )
309 {
310 for ( unsigned int j = 0; j < ny_; ++j )
311 surface_.push_back( j );
312 for ( unsigned int j = size - ny_; j < size; ++j )
313 surface_.push_back( j );
314 for ( unsigned int i = 1; i < nz_ - 1; ++i )
315 surface_.push_back( i * ny_ );
316 for ( unsigned int i = 1; i < nz_ - 1; ++i )
317 surface_.push_back( ny_ - 1 + i * ny_ );
318 }
319 else if ( ny_ == 1 )
320 {
321 for ( unsigned int k = 0; k < nx_; ++k )
322 surface_.push_back( k );
323 for ( unsigned int k = size - nx_; k < size; ++k )
324 surface_.push_back( k );
325 for ( unsigned int i = 1; i < nz_ - 1; ++i )
326 surface_.push_back( i * nx_ );
327 for ( unsigned int i = 1; i < nz_ - 1; ++i )
328 surface_.push_back( nx_ - 1 + i * nx_ );
329 }
330 else if ( nz_ == 1 )
331 {
332 for ( unsigned int k = 0; k < nx_; ++k )
333 surface_.push_back( k );
334 for ( unsigned int k = size - nx_; k < size; ++k )
335 surface_.push_back( k );
336 for ( unsigned int j = 1; j < ny_ - 1; ++j )
337 surface_.push_back( j * nx_ );
338 for ( unsigned int j = 1; j < ny_ - 1; ++j )
339 surface_.push_back( j * nx_ + nx_ - 1 );
340 }
341 // Ah, C++ STL. Look on my works, ye mighty, and despair.
342 sort( surface_.begin(), surface_.end() );
343 surface_.erase( unique( surface_.begin(), surface_.end() ),
344 surface_.end() );
345}

347void CubeMesh::fillThreeDimSurface() // Need to fix duplicate points.
348{
349 unsigned int size = nx_ * ny_ * nz_;
350 // z == 0 plane
351 for ( unsigned int j = 0; j < ny_; ++j )
352 for ( unsigned int k = 0; k < nx_; ++k )
353 surface_.push_back( j * nx_ + k );
354 // z == nz_-1 plane
355 unsigned int offset = size - nx_ * ny_;
356 for ( unsigned int j = 0; j < ny_; ++j )
357 for ( unsigned int k = 0; k < nx_; ++k )
358 surface_.push_back( offset + j * nx_ + k );
359
360 // y == 0 plane
361 for ( unsigned int i = 0; i < nz_; ++i )
362 for ( unsigned int k = 0; k < nx_; ++k )
363 surface_.push_back( i * nx_ * ny_ + k );
364 // y == ny_-1 plane
365 offset = nx_ * ( ny_ - 1 );
366 for ( unsigned int i = 0; i < nz_; ++i )
367 for ( unsigned int k = 0; k < nx_; ++k )
368 surface_.push_back( offset + i * nx_ * ny_ + k );
369
370 // x == 0 plane
371 for ( unsigned int i = 0; i < nz_; ++i )
372 for ( unsigned int j = 0; j < ny_; ++j )
373 surface_.push_back( ( i * ny_ + j ) * nx_ );
374 // x == nx_-1 plane
375 offset = nx_ - 1;
376 for ( unsigned int i = 0; i < nz_; ++i )
377 for ( unsigned int j = 0; j < ny_; ++j )
378 surface_.push_back( offset + ( i * ny_ + j ) * nx_ );
379
380 sort( surface_.begin(), surface_.end() );
381 surface_.erase( unique( surface_.begin(), surface_.end() ),
382 surface_.end() );
383}

385/**
386 * This assumes that dx, dy, dz are the quantities to preserve, over
387 * numEntries.
388 * So when the compartment changes volume, so does numEntries. dx, dy, dz
389 * do not change, some of the sub-cuboids will partially be outside.
390 */
391void CubeMesh::updateCoords()
392{
393 swapIfBackward( x0_, x1_ );
394 swapIfBackward( y0_, y1_ );
395 swapIfBackward( z0_, z1_ );
396 if ( preserveNumEntries_ )
397 {
398 dx_ = ( x1_ - x0_ ) / nx_;
399 dy_ = ( y1_ - y0_ ) / ny_;
400 dz_ = ( z1_ - z0_ ) / nz_;
401 }
402 else
403 {
404 nx_ = static_cast<unsigned int>(round( (x1_ - x0_) / dx_ ));
405 ny_ = static_cast<unsigned int>(round( (y1_ - y0_) / dy_ ));
406 nz_ = static_cast<unsigned int>(round( (z1_ - z0_) / dz_ ));
407
408 if ( nx_ == 0 ) nx_ = 1;
409 if ( ny_ == 0 ) ny_ = 1;
410 if ( nz_ == 0 ) nz_ = 1;
411 }
412
413 /// Temporarily fill out the whole cube for m2s and s2m. These
414 // will change for none-cube geometries.
415 unsigned int size = nx_ * ny_ * nz_;
416 m2s_.resize( size );
417 s2m_.resize( size );
418 for ( unsigned int i = 0; i < size; ++i )
419 {
420 m2s_[i] = s2m_[i] = i;
421 }
422
423 // Fill out surface vector
424 surface_.resize( 0 );
425 /*
426 if ( numDims() == 0 ) {
427 surface_.push_back( 0 );
428 } else if ( numDims() == 1 ) {
429 surface_.push_back( 0 );
430 if ( size > 1 )
431 surface_.push_back( size - 1 );
432 } else if ( numDims() == 2 ) {
433 fillTwoDimSurface();
434 } else if ( numDims() == 3 ) {
435 fillThreeDimSurface();
436 }
437 */
438 fillThreeDimSurface();
439
440 // volume_ = ( x1_ - x0_ ) * ( y1_ - y0_ ) * ( z1_ - z0_ );
441 assert( size >= 0 );
442
443 buildStencil();
444}

716/**
717 * Builds something as close to a cube as can get. This needs a
718 * smarter boundary handling code than I have here. For now, goes for
719 * the nearest cube
720 */
721void CubeMesh::innerBuildDefaultMesh( const Eref& e,
722 double volume, unsigned int numEntries )
723{
724 double approxN = numEntries;
725 approxN = pow( approxN, 1.0 / 3.0 );
726 unsigned int smaller = static_cast<unsigned int>(floor( approxN ));
727 unsigned int bigger = static_cast<unsigned int>(ceil( approxN ));
728 unsigned int numSide;
729 if ( smaller != bigger )
730 {
731 numSide = smaller;
732 }
733 else
734 {
735 unsigned int smallerVol = smaller * smaller * smaller;
736 unsigned int biggerVol = bigger * bigger * bigger;
737 if ( numEntries - smallerVol < biggerVol - numEntries )
738 numSide = smaller;
739 else
740 numSide = bigger;
741 }
742 double side = pow( volume, 1.0 / 3.0 );
743 vector< double > coords( 9, side );
744 coords[0] = coords[1] = coords[2] = 0;
745 coords[6] = coords[7] = coords[8] = side / numSide;
746 nx_ = ny_ = nz_ = numSide;
747 setCoords( e, coords );
748}

820{

826 unsigned int iz = (spaceIndex / ( nx_ * ny_ )) % nz_;

830 ret[1] = y0_ + iy * dy_;
831 ret[2] = z0_ + iz * dz_;
832

834 ret[4] = y0_ + iy * dy_ + dx_;
835 ret[5] = z0_ + iz * dz_ + dx_;

1224 unsigned int& index ) const
1225{
1226 if ( x > x0_ && x < x1_ && y > y0_ && y < y1_ && z > z0_ && z < z1_ )
1227 {
1228 unsigned int ix = static_cast<unsigned int>(( x - x0_ ) / dx_);
1229 unsigned int iy = static_cast<unsigned int>(( y - y0_ ) / dy_);
1230 unsigned int iz = static_cast<unsigned int>(( z - z0_ ) / dz_);
1231 index = ( iz * ny_ + iy ) * nx_ + ix;
1232 unsigned int innerIndex = s2m_[ index ];
1233 if ( innerIndex != EMPTY ) // Inside filled volume
1234 {
1235 index = innerIndex;
1236 double tx = x0_ + ix * dx_ + dx_ * 0.5;
1237 double ty = y0_ + iy * dy_ + dy_ * 0.5;
1238 double tz = z0_ + iz * dz_ + dz_ * 0.5;
1239 return distance( x - tx, y - ty, z - tz );
1240 }
1241 else // Outside volume. Look over surface for nearest.
1242 {
1243 double rmin = 1e99;
1244 for ( vector< unsigned int >::const_iterator
1245 i = surface_.begin(); i != surface_.end(); ++i )
1246 {
1247 double tx, ty, tz;
1248 indexToSpace( *i, tx, ty, tz );
1249 double r = distance( tx - x, ty - y, tz - z );
1250 if ( rmin > r )
1251 {
1252 rmin = r;
1253 index = *i;
1254 }
1255 }
1256 return -rmin; // Negative distance indicates xyz is outside vol
1257 }
1258 }
1259 // Should really figure out nearest corner anyway.
1260 index = 0;
1261 return -1.0;
1262}

1286void CubeMesh::defineIntersection( const CubeMesh* other,
1287 double& xmin, double &xmax,
1288 double& ymin, double &ymax,
1289 double& zmin, double &zmax )
1290const
1291{
1292 const double meshSlop = 0.2;
1293 xmin = ( x0_ > other->x0_ ) ? x0_ : other->x0_;
1294 xmax = ( x1_ < other->x1_ ) ? x1_ : other->x1_;
1295 ymin = ( y0_ > other->y0_ ) ? y0_ : other->y0_;
1296 ymax = ( y1_ < other->y1_ ) ? y1_ : other->y1_;
1297 zmin = ( z0_ > other->z0_ ) ? z0_ : other->z0_;
1298 zmax = ( z1_ < other->z1_ ) ? z1_ : other->z1_;
1299 // Align to coarser mesh
1300 double temp = ( xmin - x0_) / dx_;
1301 if ( temp - floor( temp ) > meshSlop )
1302 xmin = floor( temp ) * dx_;
1303 temp = ( ymin - y0_) / dy_;
1304 if ( temp - floor( temp ) > meshSlop )
1305 ymin = floor( temp ) * dy_;
1306 temp = ( zmin - z0_) / dz_;
1307 if ( temp - floor( temp ) > meshSlop )
1308 zmin = floor( temp ) * dz_;
1309
1310 // Provide 1 voxel padding all around.
1311 xmin -= dx_;
1312 xmax += dx_;
1313 ymin -= dy_;
1314 ymax += dy_;
1315 zmin -= dz_;
1316 zmax += dz_;
1317 swapIfBackward( xmin, xmax );
1318 swapIfBackward( ymin, ymax );
1319 swapIfBackward( zmin, zmax );
1320}

1379/**
1380 * checkAbut checks the intersect vector for the current position
1381 * ix, iy, iz, to determine how many diffusion terms to extract. It
1382 * then puts each of the extracted terms into the ret vector. There is
1383 * a minor efficiency for one and two diffusion terms as they are
1384 * encoded within the intersect vector. Higher-order surface alignments
1385 * require an in-line scan of neighboring voxels.
1386 * In all casesl the function inserts a flag indicating surface direction
1387 * into the diffScale
1388 * field of the VoxelJunction. 0 = x; 1 = y; 2 = z.
1389 */
1390void checkAbut(
1391 const vector< PII >& intersect,
1392 unsigned int ix, unsigned int iy, unsigned int iz,
1393 unsigned int nx, unsigned int ny, unsigned int nz,
1394 unsigned int meshIndex,
1395 vector< VoxelJunction >& ret
1396)
1397{
1398 unsigned int index = ( iz * ny + iy ) * nx + ix;
1399 unsigned int localFlag = intersect[index].second;
1400
1401 if ( localFlag == CubeMesh::EMPTY || localFlag == CubeMesh::SURFACE )
1402 return; // Nothing to put into the ret vector
1403 if ( localFlag == CubeMesh::ABUTX )
1404 {
1405 ret.push_back(
1406 VoxelJunction( intersect[index].first, meshIndex, 0 ));
1407 }
1408 else if ( localFlag == CubeMesh::ABUTY )
1409 {
1410 ret.push_back(
1411 VoxelJunction( intersect[index].first, meshIndex, 1 ));
1412 }
1413 else if ( localFlag == CubeMesh::ABUTZ )
1414 {
1415 ret.push_back(
1416 VoxelJunction( intersect[index].first, meshIndex, 2 ));
1417 }
1418 else if ( localFlag == CubeMesh::MULTI ) // go through all 6 cases.
1419 {
1420 if ( ix > 0 )
1421 {
1422 index = ( iz * ny + iy ) * nx + ix - 1;
1423 if ( intersect[index].second == CubeMesh::SURFACE )
1424 ret.push_back(
1425 VoxelJunction( intersect[index].first, meshIndex, 0 ));
1426 }
1427 if ( ix + 1 < nx )
1428 {
1429 index = ( iz * ny + iy ) * nx + ix + 1;
1430 if ( intersect[index].second == CubeMesh::SURFACE )
1431 ret.push_back(
1432 VoxelJunction( intersect[index].first, meshIndex, 0 ));
1433 }
1434 if ( iy > 0 )
1435 {
1436 index = ( iz * ny + iy -1 ) * nx + ix;
1437 if ( intersect[index].second == CubeMesh::SURFACE )
1438 ret.push_back(
1439 VoxelJunction( intersect[index].first, meshIndex, 1 ));
1440 }
1441 if ( iy + 1 < ny )
1442 {
1443 index = ( iz * ny + iy + 1 ) * nx + ix;
1444 if ( intersect[index].second == CubeMesh::SURFACE )
1445 ret.push_back(
1446 VoxelJunction( intersect[index].first, meshIndex, 1 ));
1447 }
1448 if ( iz > 0 )
1449 {
1450 index = ( (iz-1) * ny + iy ) * nx + ix;
1451 if ( intersect[index].second == CubeMesh::SURFACE )
1452 ret.push_back(
1453 VoxelJunction( intersect[index].first, meshIndex, 2 ));
1454 }
1455 if ( iz + 1 < nz )
1456 {

1510void CubeMesh::matchCubeMeshEntries( const CubeMesh* other,
1511 vector< VoxelJunction >& ret ) const
1512{
1513 // Flip meshes if the current grid is finer.
1514 // There are still problems here because a finer grid will end
1515 // up with 2 levels deep defined as being abutting.
1516 if ( compareMeshSpacing( other ) == -1 )
1517 {
1518 other->matchMeshEntries( this, ret );
1519 flipRet( ret );
1520 return;
1521 }
1522 ret.resize( 0 );
1523 // Define intersecting cuboid
1524 double xmin, xmax, ymin, ymax, zmin, zmax;
1525 defineIntersection( other, xmin, xmax, ymin, ymax, zmin, zmax );
1526
1527 // Allocate intersecting cuboid
1528 unsigned int nx = static_cast<unsigned int>(0.5 + ( xmax - xmin ) / dx_);
1529 unsigned int ny = static_cast<unsigned int>(0.5 + ( ymax - ymin ) / dy_);
1530 unsigned int nz = static_cast<unsigned int>(0.5 + ( zmax - zmin ) / dz_);
1531 vector< PII > intersect( nx * ny * nz, PII( EMPTY, EMPTY ) );
1532 assignVoxels( intersect, xmin, xmax, ymin, ymax, zmin, zmax );
1533
1534 // Scan through finer mesh surface, check for occupied voxels.
1535 for ( vector< unsigned int >::const_iterator i =
1536 other->surface_.begin();
1537 i != other->surface_.end(); ++i )
1538 {
1539 double x, y, z;
1540 other->indexToSpace( *i, x, y, z );
1541 if ( x >= xmin && x <= xmax && y >= ymin && y <= ymax &&
1542 z >= zmin && z <= zmax )
1543 {
1544 unsigned int ix = static_cast<unsigned int>(( x - xmin ) / dx_);
1545 unsigned int iy = static_cast<unsigned int>(( y - ymin ) / dy_);
1546 unsigned int iz = static_cast<unsigned int>(( z - zmin ) / dz_);
1547 unsigned int meshIndex = other->s2m_[ *i ];
1548 checkAbut( intersect, ix, iy, iz, nx, ny, nz, meshIndex, ret );
1549 }
1550 }
1551
1552 // Scan through the VoxelJunctions and populate their diffScale field
1553 setDiffScale( other, ret );
1554 // Scan through the VoxelJunctions and populate their volume field
1555 setJunctionVol( other, ret );
1556 sort( ret.begin(), ret.end() );
1557}
