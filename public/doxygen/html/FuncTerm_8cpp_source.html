40void FuncTerm::setReactantIndex( const vector< unsigned int >& mol )
41{
42 reactantIndex_ = mol;
43 if ( args_ )
44 {
45 delete[] args_;
46 parser_.ClearAll();
47 }
48
49 args_ = new double[mol.size()+1];
50 for ( unsigned int i = 0; i < mol.size(); ++i )
51 {
52 args_[i] = 0.0;
53 parser_.DefineVar( 'x'+to_string(i), &args_[i] );
54 }
55
56 // Define a 't' variable even if we don't always use it.
57 args_[mol.size()] = 0.0;
58 parser_.DefineVar( "t", &args_[mol.size()] );
59 setExpr(expr_);
60}

82void FuncTerm::setExpr( const string& expr )
83{
84 // Empty expression are not allowed.
85 if(expr.empty())
86 return;
87
88 try
89 {
90 if(! parser_.SetExpr(expr))
91 MOOSE_WARN("Failed to set expression: '" << expr << "'");
92 expr_ = expr;
93 }
94 catch(moose::Parser::exception_type &e)
95 {
96 showError(e);
97 return;
98 }
99}

138/**
139 * This computes the value. The time t is an argument needed by
140 * some functions.
141 */
142double FuncTerm::operator() ( const double* S, double t ) const
143{
144 if ( ! args_ )
145 return 0.0;
146
147 unsigned int i = 0;
148 for ( i = 0; i < reactantIndex_.size(); ++i )
149 args_[i] = S[reactantIndex_[i]];
150 args_[i] = t;
151
152 try
153 {
154 return parser_.Eval() * volScale_;
155 }
156 catch (moose::Parser::exception_type &e )
157 {
158 cerr << "Error: " << e.GetMsg() << endl;
159 throw e;
160 }
161}
