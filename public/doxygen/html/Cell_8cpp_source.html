203/**
204 * This function performs a depth-first search of the tree under the current
205 * cell. First compartment found is returned as the seed.
206 */
207Id Cell::findCompt(Id cell)
208{
209 /* 'curr' is the current element under consideration. 'cstack' is a list
210 * of all elements (and their immediate siblings) found on the path from
211 * the root element (the Cell) to the current element.
212 */
213 vector<vector<Id>> cstack;
214 Id seed;
215
216 const Cinfo* compartmentCinfo = Cinfo::find("Compartment");
217
218 cstack.push_back(children(cell));
219 while(!cstack.empty()) {
220 const vector<Id>& child = cstack.back();
221
222 if(child.empty()) {
223 cstack.pop_back();
224 if(!cstack.empty())
225 cstack.back().pop_back();
226 } else {
227 moose::showWarn("TODO: Commented out code. ");
228 Id curr = child.back();
229
230#if 0 /* ----- #if 0 : If0Label_1 ----- */
231
232 //~ string className = Field< string >::get( curr, "class" );
233 if ( curr()->cinfo() == compartmentCinfo )
234 {
235 seed = curr;
236 break;
237 }
238#endif /* ----- #if 0 : If0Label_1 ----- */
239
240 cstack.push_back(children(curr));
241 }
242 }
243
244 return seed;
245}
