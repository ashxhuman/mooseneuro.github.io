60/* Matrix inversion routine.
61 * Uses lu_factorize and lu_substitute in uBLAS to invert a matrix
62 */
63
64template<class T>
65bool inverse(const ublas::matrix<T>& input, ublas::matrix<T>& inverse)
66{
67 using namespace boost::numeric::ublas;
68 typedef permutation_matrix<std::size_t> pmatrix;
69 // create a working copy of the input
70 matrix<T> A(input);
71 // create a permutation matrix for the LU-factorization
72 pmatrix pm(A.size1());
73
74 // perform LU-factorization
75 int res = lu_factorize(A,pm);
76
77 if( res != 0 ) return false;
78
79 // create identity matrix of "inverse"
80 inverse.assign(ublas::identity_matrix<T>(A.size1()));
81
82 // backsubstitute to get the inverse
83 lu_substitute(A, pm, inverse);
84
85 return true;
86}

94 NonlinearSystem( size_t systemSize ) : size_( systemSize )

108 // Find machine epsilon to decide on dx. Machine eps is typically 2.22045e-16 on 64 bit machines/intel.
109 // Square root of this value is a very good value. The values matches with GSL solver nicely.
110 // Making this value too small or too big gonna cause error in blas methods especially when
111 // computing inverse.
112 // Gnu-GSL uses GSL_SQRT_DBL_EPSILON which is 1.4901161193847656e-08
113 // dx_ = 1.4901161193847656e-08;
114 dx_ = std::sqrt( numeric_limits<double>::epsilon() );
115 }

117 bool compute_at(const vector_type& x, vector_type& result)
118 {
119 return system(x, result);
120 }

122 int apply( )
123 {
124 return system(x_, f_);
125 }

127 int compute_jacobians( const vector_type& x, bool compute_inverse = false )
128 {
129 for( size_t i = 0; i < size_; i++)
130 {
131 // This trick I leart by looking at GSL implmentation.
132 double dx = dx_ * std::fabs(x[i]);
133 if( dx == 0 )
134 dx = dx_;
135
136 for( size_t j = 0; j < size_; j++)
137 {
138 vector_type temp = x;
139 temp[j] += dx;
140 vector_type res1, res2;
141 auto r1 = compute_at(temp, res1);
142 auto r2 = compute_at(x, res2);
143
144 if( 0 == r1 && 0 == r2 )
145 {
146 J_(i, j) = (res1[i] - res2[i]) / dx;
147 if( std::isnan(J_(i,j)) || std::isinf(J_(i,j)) )
148 {
149 /* Try increasing dx */
150 J_.clear();
151 return -1;
152 }
153 }
154 else
155 {
156 J_.clear();
157 return -1;
158 }
159 }
160 }
161
162 if( compute_inverse )
163 {
164 try {
165 inverse( J_, invJ_ );
166 } catch ( exception & e ) {
167 J_.clear();
168 invJ_.clear();
169 return -1;
170 }
171 }
172 return 0;
173 }

209 int system( const vector_type& x, vector_type& f )
210 {
211 size_t num_consv = ri.num_mols - ri.rank;
212 for ( size_t i = 0; i < ri.num_mols; ++i )
213 {
214 double temp = x[i] * x[i] ;
215
216 // if overflow
217 if ( std::isnan( temp ) || std::isinf( temp ) )
218 return -1;
219 else
220 ri.nVec[i] = temp;
221 }
222
223 vector< double > vels;
224 ri.pool->updateReacVelocities( &ri.nVec[0], vels );
225
226 assert( vels.size() == static_cast< unsigned int >( ri.num_reacs ) );
227
228 // y = Nr . v
229 // Note that Nr is row-echelon: diagonal and above.
230 f.resize( ri.rank + num_consv);
231 for ( size_t i = 0; i < ri.rank; ++i )
232 {
233 double temp = 0;
234 for ( size_t j = i; j < ri.num_reacs; ++j )
235 temp += ri.Nr(i, j ) * vels[j];
236 f[i] = temp ;
237 }
238
239 // dT = gamma.S - T
240 for ( size_t i = 0; i < num_consv; ++i )
241 {
242 double dT = - ri.T[i];
243 for ( size_t j = 0; j < ri.num_mols; ++j )
244 dT += ri.gamma(i, j) * x[j] * x[j];
245 f[ i + ri.rank] = dT ;
246 }
247 return 0;
248 }

251 /**
252 * @brief Find roots using Newton-Raphson method.
253 *
254 * @param tolerance 1e-7
255 * @param max_iter Maximum number of iteration allowed , default 100
256 *
257 * @return If successful, return true. Check the variable `x_` at
258 * which the system f_ is close to zero (within the tolerance).
259 */
260 bool find_roots_gnewton( double tolerance, size_t max_iter)
261 {
262 double norm2OfDiff = 1.0;
263 size_t iter = 0;
264 if(0 != apply() )
265 return false;
266
267 while( ublas::norm_2(f_) > tolerance )
268 {
269 iter += 1;
270 ri.nIter = iter;
271
272 if( 0 != compute_jacobians( x_, true ) )
273 {
274 J_.clear();
275 invJ_.clear();
276 return false;
277 }
278
279 vector_type correction = ublas::prod( invJ_, f_ );
280 x_ -= correction;
281
282 // If could not compute the value of system successfully.
283 if( 0 != apply() )
284 {
285 x_.clear();
286 return false;
287 }
288
289 if( iter >= max_iter )
290 break;
291
292 }
293 return true;
294 }
