268void ChemCompt::setEntireVolume( const Eref& e, double volume )
269{
270 // If the reac system is not solved, then explicitly do scaling
271 vector< ObjId > tgtVec =
272 e.element()->getMsgTargets( e.dataIndex(), voxelVolOut() );
273 if ( tgtVec.size() == 0 )
274 {
275 vector< double > childConcs;
276 getChildConcs( e, childConcs );
277 if ( vSetVolumeNotRates( volume ) )
278 {
279 setChildConcs( e, childConcs, 0 );
280 }
281 }
282 else
283 {
284 vSetVolumeNotRates( volume );
285 voxelVolOut()->send( e, this->vGetVoxelVolume() );
286 }
287}

294void ChemCompt::getChildConcs( const Eref& e, vector< double >& childConcs ) const
295{
296 vector< Id > kids;
297 Neutral::children( e, kids );
298 for ( vector < Id >::iterator i = kids.begin(); i != kids.end(); ++i )
299 {
300 if ( i->element()->cinfo()->isA( "PoolBase" ) )
301 {
302 childConcs.push_back( Field< double >::get( *i, "conc" ) );
303 childConcs.push_back( Field< double >::get( *i, "concInit" ) );
304 }
305 else if ( i->element()->cinfo()->isA( "Reac" ) )
306 {
307 childConcs.push_back( Field< double >::get( *i, "Kf" ) );
308 childConcs.push_back( Field< double >::get( *i, "Kb" ) );
309 }
310 else if ( i->element()->cinfo()->isA( "EnzBase" ) )
311 {
312 childConcs.push_back( Field< double >::get( *i, "Km" ) );
313 }
314 else if ( i->element()->cinfo()->isA( "ChemCompt" ) )
315 {
316 // Do NOT traverse into child ChemCompts, they look after their
317 // own volumes.
318 continue;
319 }
320 getChildConcs( i->eref(), childConcs );
321 }
322}

324unsigned int ChemCompt::setChildConcs( const Eref& e,
325 const vector< double >& conc, unsigned int start) const
326{
327 vector< Id > kids;
328 Neutral::children( e, kids );
329 for ( vector < Id >::iterator i = kids.begin(); i != kids.end(); ++i )
330 {
331 if ( i->element()->cinfo()->isA( "PoolBase" ) )
332 {
333 Field< double >::set( *i, "conc", conc[ start++ ] );
334 Field< double >::set( *i, "concInit", conc[start++] );
335 }
336 else if ( i->element()->cinfo()->isA( "Reac" ) )
337 {
338 Field< double >::set( *i, "Kf", conc[ start++ ] );
339 Field< double >::set( *i, "Kb", conc[ start++ ] );
340 }
341 else if ( i->element()->cinfo()->isA( "EnzBase" ) )
342 {
343 Field< double >::set( *i, "Km", conc[ start++ ] );
344 }
345 else if ( i->element()->cinfo()->isA( "ChemCompt" ) )
346 {
347 // Do NOT traverse into child ChemCompts, they look after their
348 // own volumes.
349 continue;
350 }
351 start = setChildConcs( i->eref(), conc, start );
352 }
353 return start;
354}

371vector< double > ChemCompt::getOneVoxelMidpoint( unsigned int vox ) const
372{
373 const vector< double >& v = vGetVoxelMidpoint();
374 vector< double > ret = { 0.0, 0.0, 0.0 };
375 unsigned int numVoxels = v.size() / 3;
376 if ( vox < numVoxels ) {
377 ret[0] = v[vox];
378 ret[1] = v[vox + numVoxels];
379 ret[2] = v[vox + numVoxels * 2];
380 }
381 return ret;
382}

466void ChemCompt::flipRet( vector< VoxelJunction >& ret ) const
467{
468 vector< VoxelJunction >::iterator i;
469 for ( i = ret.begin(); i != ret.end(); ++i )
470 {
471 unsigned int temp = i->first;
472 i->first = i->second;
473 i->second = temp;
474 double vol = i->firstVol;
475 i->firstVol = i->secondVol;
476 i->secondVol = vol;
477 }
478}
