113void MMenz::setSolver( const Eref& e, ObjId solver )
114{
115 static const DestFinfo* enzFinfo = dynamic_cast< const DestFinfo* >(
116 EnzBase::initCinfo()->findFinfo( "enzDest" ) );
117 static const SrcFinfo* subFinfo = dynamic_cast< const SrcFinfo* >(
118 EnzBase::initCinfo()->findFinfo( "subOut" ) );
119 static const SrcFinfo* prdFinfo = dynamic_cast< const SrcFinfo* >(
120 EnzBase::initCinfo()->findFinfo( "prdOut" ) );
121 assert( enzFinfo );
122 assert( subFinfo );
123 assert( prdFinfo );
124
125 if ( solver == ObjId() ) { // Clear solver
126 if ( stoich_ )
127 stoich_->notifyRemoveMMenz( e );
128 stoich_ = 0;
129 return;
130 }
131
132 assert( solver.element()->cinfo()->isA( "Stoich" ) );
133 Stoich* stoichPtr = reinterpret_cast< Stoich* >( solver.data() );
134 if ( stoich_ == stoichPtr )
135 return;
136 if (stoich_)
137 stoich_->notifyRemoveMMenz( e );
138 stoich_ = stoichPtr;
139
140 /// Now set up the RateTerm
141 vector< Id > enzvec;
142 vector< Id > subvec;
143 vector< Id > prdvec;
144 unsigned int num = e.element()->getNeighbors( enzvec, enzFinfo );
145 num = e.element()->getNeighbors( subvec, subFinfo );
146 num = e.element()->getNeighbors( prdvec, prdFinfo );
147 stoich_->installMMenz( e.id(), enzvec, subvec, prdvec );
148 stoich_->setMMenzKm( e, Km_ );
149 stoich_->setMMenzKcat( e, kcat_ );
150}
