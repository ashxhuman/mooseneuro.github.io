15/// Returns the Id of the root of the copied tree upon success.
16Id Shell::doCopy(Id orig, ObjId newParent, string newName, unsigned int n,
17 bool toGlobal, bool copyExtMsg)
18{

24

26 cout << "Error: Shell::doCopy: Cannot copy object to descendant in "

36 cout << "Error: Shell::doCopy: Cannot copy object '" << newName

38 << "' since object with same name already present.\n";
39 return Id();
40 }
41
42 Eref sheller(shelle_, 0);
43 Id newElm = Id::nextId();
44 vector<ObjId> args;
45 args.push_back(orig);
46 args.push_back(newParent);
47 args.push_back(newElm);
48 SetGet5<vector<ObjId>, string, unsigned int, bool, bool>::set(

59/** Runs in parallel on all nodes.
60 * Note that 'n' is the number of complete duplicates. If there were
61 * 10 dataEntries in the original, there will now be 10 x n.
62 */
63Element* innerCopyElements(Id orig, ObjId newParent, Id newId, unsigned int n,
64 bool toGlobal, map<Id, Id>& tree)
65{
66 // Element* e = new Element( newId, orig.element(), n, toGlobal );
67 unsigned int newNumData = orig.element()->numData() * n;
68 Element* e =
69 orig.element()->copyElement(newParent, newId, newNumData, toGlobal);
70 assert(e);
71 Shell::adopt(newParent, newId, 0);
72 e->setTick(Clock::lookupDefaultTick(e->cinfo()->name()));
73 SetGet1< ObjId >::set( newId, "notifyCopy", orig );
74
75 // cout << Shell::myNode() << ": Copy: orig= " << orig << ", newParent = "
76 // << newParent << ", newId = " << newId << endl;
77 tree[orig] = e->id();
78
79 // cout << Shell::myNode() << ": ice, pa = " << newParent << ", pair= (" <<
80 // orig << "," << e->id() << ")\n";
81 vector<Id> kids;
82 Neutral::children(orig.eref(), kids);
83
84 for (vector<Id>::iterator i = kids.begin(); i != kids.end(); ++i) {
85 // Needed in case parent is not on zero dataIndex.
86 ObjId pa = Neutral::parent(*i);
87 ObjId newParent(e->id(), pa.dataIndex);
88 innerCopyElements(*i, newParent, Id::nextId(), n, toGlobal, tree);
89 }
90 return e;
91}

149bool Shell::innerCopy(const vector<ObjId>& args, const string& newName,
150 unsigned int n, bool toGlobal, bool copyExtMsgs)
151{
152 map<Id, Id> tree;
153 // args are orig, newParent, newElm.
154 assert(args.size() == 3);
155 Element* e =
156 innerCopyElements(args[0], args[1], args[2], n, toGlobal, tree);
157 if (!e) {
158 return 0;
159 }
160 if (newName != "") e->setName(newName);
161 // innerCopyData( orig, newParent );
162 innerCopyMsgs(tree, n, copyExtMsgs);
163 return 1;
