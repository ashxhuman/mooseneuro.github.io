49void KsolveBase::setCompartment( Id compt )
50{
51 isBuilt_ = false; // We will have to now rebuild the whole thing.
52 if ( compt.element()->cinfo()->isA( "ChemCompt" ) )
53 {
54 compartment_ = compt;
55 vector<double> vols = Field<vector<double>>::get( compt, "voxelVolume" );
56
57 if (vols.size() > 0)
58 {
59 setNumAllVoxels( vols.size() );
60 for ( unsigned int i = 0; i < vols.size(); ++i )
61 pools(i)->setVolume( vols[i] );
62 }
63 }
64}
