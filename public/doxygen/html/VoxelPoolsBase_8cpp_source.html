109void VoxelPoolsBase::scaleVolsBufsRates(double ratio, const Stoich* stoichPtr)
110{
111 volume_ *= ratio; // Scale vol
112 // Here we also need to set the Ns for the buffered pools.
113 unsigned int start = stoichPtr_->getNumVarPools();
114 unsigned int end = start + stoichPtr_->getNumBufPools();
115 assert( end == Cinit_.size() );
116 for ( unsigned int i = start; i < end; ++i )
117 {
118 // Must not reassign pools that are controlled by functions.
119 if ( !stoichPtr->isFuncTarget(i) )
120 S_[i] = Cinit_[i] * NA * volume_;
121 }
122
123 // Scale rates. Start by clearing out old rates if any
124 for ( unsigned int i = 0; i < rates_.size(); ++i )
125 delete( rates_[i] );
126
127 unsigned int numCoreRates = stoichPtr->getNumCoreRates();
128 const vector< RateTerm* >& rates = stoichPtr->getRateTerms();
129 rates_.resize( rates.size() );
130
131 for ( unsigned int i = 0; i < numCoreRates; ++i )
132 rates_[i] = rates[i]->copyWithVolScaling( getVolume(), 1, 1 );
133
134 for ( unsigned int i = numCoreRates; i < rates.size(); ++i )
135 {
136 rates_[i] = rates[i]->copyWithVolScaling( getVolume(),
137 getXreacScaleSubstrates(i - numCoreRates),
138 getXreacScaleProducts(i - numCoreRates ) );
139 }
140}

194void VoxelPoolsBase::xferIn(
195 const vector< unsigned int >& poolIndex,
196 const vector< double >& values,
197 const vector< double >& lastValues,
198 unsigned int voxelIndex )
199{
200 unsigned int offset = voxelIndex * poolIndex.size();
201 vector< double >::const_iterator i = values.begin() + offset;
202 vector< double >::const_iterator j = lastValues.begin() + offset;
203 for ( vector< unsigned int >::const_iterator
204 k = poolIndex.begin(); k != poolIndex.end(); ++k )
205 {
206 S_[*k] += *i++ - *j++;
207 }
208}

210void VoxelPoolsBase::xferInOnlyProxies(
211 const vector< unsigned int >& poolIndex,
212 const vector< double >& values,
213 unsigned int numProxyPools,
214 unsigned int voxelIndex )
215{
216 unsigned int offset = voxelIndex * poolIndex.size();
217 vector< double >::const_iterator i = values.begin() + offset;
218 unsigned int proxyEndIndex = stoichPtr_->getNumVarPools() +
219 stoichPtr_->getNumProxyPools();
220 for ( vector< unsigned int >::const_iterator
221 k = poolIndex.begin(); k != poolIndex.end(); ++k )
222 {
223 if ( *k >= stoichPtr_->getNumVarPools() && *k < proxyEndIndex )
224 {
225 Cinit_[*k] = *i / ( NA * volume_ );
226 S_[*k] = *i;
227 }
228 i++;
229 }
230}

273bool VoxelPoolsBase::isVoxelJunctionPresent( Id i1, Id i2 ) const
274{
275 if ( i1 == Id () )
276 return false;
277 if ( proxyComptMap_.find( i1 ) == proxyComptMap_.end() )
278 return false;
279 if ( i2 == Id() ) // This is intentionally blank, only one jn.
280 return true;
281 // If there is an i2 but it isn't on the map, then not connected.
282 if ( proxyComptMap_.find( i2 ) == proxyComptMap_.end() )
283 return false;
284 return true;

321/**
322 * Zeroes out rate terms that are involved in cross-reactions that
323 * are not present on current voxel.
324 */
325void VoxelPoolsBase::filterCrossRateTerms(
326 const vector< Id >& offSolverReacs,
327 const vector< pair< Id, Id > >& offSolverReacCompts )
328{
329 assert (offSolverReacs.size() == offSolverReacCompts.size() );
330 // unsigned int numCoreRates = stoichPtr_->getNumCoreRates();
331 for ( unsigned int i = 0; i < offSolverReacCompts.size(); ++i )
332 {
333 const pair< Id, Id >& p = offSolverReacCompts[i];
334 if ( !isVoxelJunctionPresent( p.first, p.second) )
335 {
336 Id reacId = offSolverReacs[i];
337 const Cinfo* reacCinfo = reacId.element()->cinfo();
338 unsigned int k = stoichPtr_->convertIdToReacIndex( offSolverReacs[i] );
339 // Start by replacing the immediate cross reaction term.
340 if ( rates_[k] )
341 delete rates_[k];
342 rates_[k] = new ExternReac;
343 if ( stoichPtr_->getOneWay() )
344 {
345 k++; // Delete the next entry too, it is the reverse reacn.
346 assert( k < rates_.size() );
347 if ( reacCinfo->isA( "Reac" ) )
348 {
349 if ( rates_[k] )
350 delete rates_[k];
351 rates_[k] = new ExternReac;
352 }
353 if ( reacCinfo->isA( "Enz" ) ) // Delete next two.
354 {
355 if ( rates_[k] )
356 delete rates_[k];
357 rates_[k] = new ExternReac;
358 k++;
359 assert( k < rates_.size() );
360 if ( rates_[k] )
361 delete rates_[k];
362 rates_[k] = new ExternReac;
363 }
364 }
365 else
366 {
367 if ( reacCinfo->isA( "Enz" ) ) // Delete next one.
368 {
369 k++;
370 assert( k < rates_.size() );
371 if ( rates_[k] )
372 delete rates_[k];
373 rates_[k] = new ExternReac;
374 }
375 }
376 }
377 }
