280// Checks for numbers bigger than the size. Replaces with
281// values within the range that have not yet been used.
282void SeqSynHandler::fixSynapseOrder()
283{
284 unsigned int sz = synapseOrder_.size();
285 vector< unsigned int > availableEntries( sz );
286 std::iota( availableEntries.begin(), availableEntries.end(), 0 );
287 for( unsigned int i = 0; i < sz; ++i )
288 {
289 if ( synapseOrder_[i] < sz )
290 availableEntries[ synapseOrder_[i] ] = sz;
291 }
292 vector< unsigned int > ae;
293 for( unsigned int i = 0; i < sz; ++i )
294 if ( availableEntries[i] < sz )
295 ae.push_back( availableEntries[i] );
296
297 auto jj = ae.begin();
298 for( unsigned int i = 0; i < sz; ++i )
299 {
300 if ( synapseOrder_[i] >= sz )
301 synapseOrder_[i] = *jj++;
302 }
303}

305// Beautiful snippet from Lukasz Wiklendt on StackOverflow. Returns order
306// of entries in a vector.
307template <typename T> vector<unsigned int> sort_indexes(const vector<T> &v)
308{
309 // initialize original index locations
310 vector<unsigned int> idx(v.size());
311 iota(idx.begin(), idx.end(), 0);
312 // sort indexes based on comparing values in v
313 sort(idx.begin(), idx.end(),
314 [&v](unsigned int i1, unsigned int i2)
315 {
316 return v[i1] < v[i2];
317 });
318 return idx;
319}

321void SeqSynHandler::refillSynapseOrder( unsigned int newSize )
322{
323 if ( synapseOrderOption_ <= -2 ) // User order
324 {
325 synapseOrder_.resize( newSize, newSize );
326 fixSynapseOrder();
327 }
328 else if ( synapseOrderOption_ == -1 ) // Ordered
329 {
330 synapseOrder_.resize( newSize );
331 for ( unsigned int i = 0 ; i < newSize; ++i )
332 synapseOrder_[i] = i;
333 }
334 else
335 {
336 synapseOrder_.resize( newSize );
337 if ( synapseOrderOption_ > 0 ) // Specify seed explicitly
338 {
339 moose::mtseed( synapseOrderOption_ );
340 }
341 vector< double > x;
342 for ( unsigned int i = 0; i < newSize; ++i )
343 x.push_back( moose::mtrand() );
344 auto idx = sort_indexes< double >( x );
345 for ( unsigned int i = 0; i < newSize; ++i )
346 synapseOrder_[i] = idx[i];
347 }

350void SeqSynHandler::updateKernel()
351{
352 if ( kernelEquation_ == "" || kernelEquation_ == "assigned using setKernel" || seqDt_ < 1e-9 || historyTime_ < 1e-9 )
353 return;
354 double x = 0;
355 double t = 0;
356 moose::MooseParser p;
357 p.DefineVar("x", &x);
358 p.DefineVar("t", &t);
359 p.SetExpr( kernelEquation_ );
360 kernel_.clear();
361 int nh = numHistory();
362 kernel_.resize( nh );
363 for ( int i = 0; i < nh; ++i )
364 {
365 kernel_[i].resize( kernelWidth_ );
366 t = i * seqDt_;
367 for ( unsigned int j = 0; j < kernelWidth_; ++j )
368 {
369 x = j;
370 kernel_[i][j] = p.Eval();
371 }
372 }
373}

472void SeqSynHandler::setKernel( vector< double > kern)
473{
474 int nh = numHistory();
475 if ( kern.size() != nh * kernelWidth_ ) {
476 cout << "Warning: Kernel size must be numHistory X width (" << nh << " X " << kernelWidth_ << "). Assignment not done.\n";
477 return;
478 }
479 kernelEquation_ = "assigned using setKernel";
480 kernel_.resize( nh );
481 for ( int i = 0; i < nh; ++i )
482 {
483 kernel_[i].resize( kernelWidth_ );
484 for ( unsigned int j = 0; j < kernelWidth_; ++j )
485 {
486 kernel_[i][j] = kern[ i*kernelWidth_ + j];
487 }
488 }
489}

580void SeqSynHandler::vProcess( const Eref& e, ProcPtr p )
581{
582 // Here we look at the correlations and do something with them.
583 int nh = numHistory();
584 vector< PreSynEvent > newEvents;
585 while( !events_.empty() && events_.top().time <= p->currTime )
586 {
587 newEvents.push_back( events_.top() );
588 latestSpikes_[ synapseOrder_[ events_.top().synIndex ] ] +=
589 events_.top().weight;
590 events_.pop();
591 }
592
593 // Check if we need to do correlations at all.
594 if ( nh > 0 && kernel_.size() > 0 )
595 {
596 // Check if timestep rolls over a seqDt boundary
597 if ( static_cast< int >( p->currTime / seqDt_ ) >
598 static_cast< int >( (p->currTime - p->dt) / seqDt_ ) )
599 {
600 history_.rollToNextRow();
601 history_.sumIntoRow( latestSpikes_, 0 );
602 latestSpikes_.assign( vGetNumSynapses(), 0.0 );
603
604 /** Reinstated 12 Oct 2022.
605 */
606 // Build up the sum of correlations over time
607 vector< double > correlVec( vGetNumSynapses(), 0.0 );
608 for ( int i = 0; i < nh; ++i ) {
609 history_.correl( correlVec, kernel_[i], i );
610 // The correl function does a sweep of dot products
611 // across all columns, ie, all synapses.
612 }
613 if ( sequenceScale_ > 0.0 ) // Sum all responses, send to chan
614 {
615 seqActivation_ = 0.0;
616 for (const auto y : correlVec )
617 seqActivation_ += pow( y, sequencePower_ );
618
619 // We'll use the seqActivation_ to send a special msg.
620 seqActivation_ *= sequenceScale_;
621 }
622
623 /** This version works across history but does not move the
624 * kernel across the range of different synapses
625 * Removed 12 Oct 2022.
626 seqActivation_ = 0.0;
627 if ( sequenceScale_ > 0.0 ) // Sum all responses, send to chan
628 {
629 for ( int i = 0; i < nh; ++i )
630 seqActivation_ += pow( history_.dotProduct( kernel_[i], i, 0 ), sequencePower_ );
631 seqActivation_ *= sequenceScale_;
632 }
633 */
634
635 /*
636 * I think this is altering the correl vec for next cycle.
637 if ( plasticityScale_ > 0.0 ) // Short term changes in individual wts
638 {
639 weightScaleVec_ = correlVec;
640 for ( vector< double >::iterator y=weightScaleVec_.begin();
641 y != weightScaleVec_.end(); ++y )
642 *y *= plasticityScale_;
643 }
644 */
645 }
646 }
647
648 // Here we go through the regular synapse activation calculations.
649 // We can't leave it to the base class vProcess, because we need
650 // to scale the weights individually in some cases.
651 double activation = seqActivation_; // Start with seq activation
652 if ( plasticityScale_ > 0.0 )
653 {
654 for (const auto ee : newEvents ) {
655 activation += ee.weight * baseScale_ *
656 weightScaleVec_[ ee.synIndex ] / p->dt;
657 }
658 /*
659 while( !events_.empty() && events_.top().time <= p->currTime )
660 {
661 activation += events_.top().weight * baseScale_ *
662 weightScaleVec_[ events_.top().synIndex ] / p->dt;
663 events_.pop();
664 }
665 */
666 }
667 else
668 {
669 for (const auto ee : newEvents ) {
670 activation += baseScale_ * ee.weight / p->dt;
671 }
672 /*
673 while( !events_.empty() && events_.top().time <= p->currTime )
674 {
675 activation += baseScale_ * events_.top().weight / p->dt;
676 events_.pop();
677 }
678 */
679 }
680 if ( activation != 0.0 )
681 SynHandlerBase::activationOut()->send( e, activation );
682}
