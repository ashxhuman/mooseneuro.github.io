91/**
92 * If the gate exists and has only this element for input, then change
93 * the gate power.
94 * If the gate exists and has multiple parents, then make a new gate,
95 * set its power.
96 * If the gate does not exist, make a new gate, set its power.
97 *
98 * The function is designed with the idea that if copies of this
99 * channel are made, then they all point back to the original HHGate.
100 * (Unless they are cross-node copies).
101 * It is only if we subsequently alter the HHGate of this channel that
102 * we need to make our own variant of the HHGate, or disconnect from
103 * an existing one.
104 * \todo: May need to convert to handling arrays and Erefs.
105 */
106// Assuming that the elements are simple elements. Use Eref for
107// general case
108bool HHChannel::checkOriginal(Id chanId) const {
109 bool isOriginal = true;
110 // cerr << "# HHChannel::checkOriginal(Id chanId) chanId: " << chanId << ", xGate: " << xGate_ << endl;
111 if (xGate_ != nullptr) {
112 isOriginal = xGate_->isOriginalChannel(chanId);
113 } else if (yGate_ != nullptr) {
114 isOriginal = yGate_->isOriginalChannel(chanId);
115 } else if (zGate_ != nullptr) {
116 isOriginal = zGate_->isOriginalChannel(chanId);
117 }
118 return isOriginal;
119}

132void HHChannel::vCreateGate(const Eref& e, string gateType) {
133 if (!checkOriginal(e.id())) {
134 cout << "Warning: HHChannel::createGate: Not allowed from copied "
135 "channel:\n"
136 << e.id().path() << "\n";
137 return;
138 }
139
140 if (gateType == "X")
141 innerCreateGate("xGate", &xGate_, e.id(), Id(e.id().value() + 1));
142 else if (gateType == "Y")
143 innerCreateGate("yGate", &yGate_, e.id(), Id(e.id().value() + 2));
144 else if (gateType == "Z")
145 innerCreateGate("zGate", &zGate_, e.id(), Id(e.id().value() + 3));
146 else
147 cout << "Warning: HHChannel::createGate: Unknown gate type '"
148 << gateType << "'. Ignored\n";
149}

208void HHChannel::vProcess(const Eref& e, ProcPtr info) {
209 g_ += ChanCommon::vGetGbar(e);
210 double A = 0;
211 double B = 0;
212 if (Xpower_ > 0) {
213 xGate_->lookupBoth(Vm_, &A, &B);
214 if (instant_ & INSTANT_X)
215 X_ = A / B;
216 else
217 X_ = integrate(X_, info->dt, A, B);
218 g_ *= takeXpower_(X_, Xpower_);
219 }
220
221 if (Ypower_ > 0) {
222 yGate_->lookupBoth(Vm_, &A, &B);
223 if (instant_ & INSTANT_Y)
224 Y_ = A / B;
225 else
226 Y_ = integrate(Y_, info->dt, A, B);
227
228 g_ *= takeYpower_(Y_, Ypower_);
229 }
230
231 if (Zpower_ > 0) {
232 if (useConcentration_)
233 zGate_->lookupBoth(conc_, &A, &B);
234 else
235 zGate_->lookupBoth(Vm_, &A, &B);
236 if (instant_ & INSTANT_Z)
237 Z_ = A / B;
238 else
239 Z_ = integrate(Z_, info->dt, A, B);
240
241 g_ *= takeZpower_(Z_, Zpower_);
242 }
243
244 ChanCommon::vSetGk(e, g_ * ChanCommon::vGetModulation(e));
245 ChanCommon::updateIk();
246 // Gk_ = g_;
247 // Ik_ = ( Ek_ - Vm_ ) * g_;
248
249 // Send out the relevant channel messages.
250 ChanCommon::sendProcessMsgs(e, info);
251
252 g_ = 0.0;
253}

255/**
256 * Here we get the steady-state values for the gate (the 'instant'
257 * calculation) as A_/B_.
258 */
259void HHChannel::vReinit(const Eref& er, ProcPtr info) {
260 g_ = ChanCommon::vGetGbar(er);
261 Element* e = er.element();
262
263 double A = 0.0;
264 double B = 0.0;
265 if (Xpower_ > 0) {
266 assert(xGate_);
267 if((xGate_->getMax(er) == 1) && (xGate_->getMin(er) == 0)) {
268 cout << "Warning: " << er.objId().path()
269 << ".gateX: `min` and `max` have default values. Did you forget to "
270 << "set them?"
271 << endl;
272 }
273 if(xGate_->getForm() > 0) // explicit string expression set, fill the tables by evaluating it
274 {
275 xGate_->fillFromExpr(er);
276 }
277 xGate_->lookupBoth(Vm_, &A, &B);
278 if (B < EPSILON) {
279 cout << "Warning: B_ value for " << e->getName()
280 << " is ~0. Check X table\n";
281 return;
282 }
283 if (!xInited_) X_ = A / B;
284 g_ *= takeXpower_(X_, Xpower_);
285 }
286
287 if (Ypower_ > 0) {
288 assert(yGate_);
289 if((yGate_->getMax(er) == 1) && (yGate_->getMin(er) == 0)) {
290 cout << "Warning: " << er.objId().path()
291 << ".gateY: `min` and `max` have default values. Did you forget to "
292 << "set them?"
293 << endl;
294 }
295 if(yGate_->getForm() > 0) // explicit string expression set, fill the tables by evaluating it
296 {
297 yGate_->fillFromExpr(er);
298 }
299 yGate_->lookupBoth(Vm_, &A, &B);
300 if (B < EPSILON) {
301 cout << "Warning: B value for " << e->getName()
302 << " is ~0. Check Y table\n";
303 return;
304 }
305 if (!yInited_) Y_ = A / B;
306 g_ *= takeYpower_(Y_, Ypower_);
307 }
308
309 if (Zpower_ > 0) {
310 assert(zGate_);
311 if((zGate_->getMax(er) == 1) && (zGate_->getMin(er) == 0)) {
312 cout << "Warning: " << er.objId().path()
313 << ".gateZ: `min` and `max` have default values. Did you forget to "
314 << "set them?"
315 << endl;
316 }
317 if(zGate_->getForm() > 0) // explicit string expression set, fill the tables by evaluating it
318 {
319 zGate_->fillFromExpr(er);
320 }
321 if (useConcentration_){
322 zGate_->lookupBoth(conc_, &A, &B);
323 }
324 else {
325 cout << "Warning: " << er.objId().path()
326 << ".gateZ: `useConcentration=False`, though this gate is"
327 << " commonly used for concentration-dependency."
328 << " Did you forget to set it?"
329 << endl;
330 zGate_->lookupBoth(Vm_, &A, &B);
331 }
332 if (B < EPSILON) {
333 cout << "Warning: B value for " << e->getName()
334 << " is ~0. Check Z table\n";
335 return;
336 }
337 if (!zInited_) Z_ = A / B;
338 g_ *= takeZpower_(Z_, Zpower_);
339 }
340
341 ChanCommon::vSetGk(er, g_ * ChanCommon::vGetModulation(er));
342 ChanCommon::updateIk();
343 // Gk_ = g_;
344 // Ik_ = ( Ek_ - Vm_ ) * g_;
345
346 // Send out the relevant channel messages.
347 // Same for reinit as for process.
348 ChanCommon::sendReinitMsgs(er, info);
349
350 g_ = 0.0;
351}
