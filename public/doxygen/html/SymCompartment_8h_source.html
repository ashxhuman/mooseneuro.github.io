14/**
15 * The SymCompartment class sets up a symmetric compartment for
16 * branched nerve calculations. Handles electronic structure and
17 * also channels. This version splits the Ra between either end of the
18 * compartment and is hence slightly cleaner than the asymmetric
19 * compartment.
20 * The default EE method is not a particularly efficient way of doing
21 * the calculations, so we should use a solver for any substantial
22 * calculations.
23 */

42 void vInitReinit( const Eref& e, ProcPtr p );
43 private:

48};
