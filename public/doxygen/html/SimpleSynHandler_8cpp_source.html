53SimpleSynHandler& SimpleSynHandler::operator=(const SimpleSynHandler& ssh)
54{
55 synapses_ = ssh.synapses_;
56 for (auto i = synapses_.begin(); i != synapses_.end(); ++i)
57 i->setHandler(this);
58
59 // For no apparent reason, priority queues don't have a clear operation.
60 while (!events_.empty()) events_.pop();
61
62 return *this;
63}

98void SimpleSynHandler::vProcess(const Eref& e, ProcPtr p)
99{
100 double activation = 0.0;
101 while (!events_.empty() && events_.top().time <= p->currTime) {
102 // Send out weight / dt for every spike
103 // Since it is an impulse active only for one dt,
104 // need to send it divided by dt.
105 // Can connect activation to SynChan (double exp)
106 // or to LIF as an impulse to voltage.
107 // See:
108 // http://www.genesis-sim.org/GENESIS/Hyperdoc/Manual-26.html#synchan
109 activation += events_.top().weight / p->dt;
110 events_.pop();
111 }
112 if (activation != 0.0) SynHandlerBase::activationOut()->send(e, activation);
113}
