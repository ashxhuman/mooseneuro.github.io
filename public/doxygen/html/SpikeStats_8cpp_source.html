107void SpikeStats::Vm( double v )
108{
109 if ( fired_ ) { // Wait for it to go below threshold
110 if ( v < threshold_ )
111 fired_ = false;
112 } else {
113 if ( v > threshold_ ) { // wait for it to go above threshold.
114 fired_ = true;
115 numSpikes_++;
116 }
117 }
118}
