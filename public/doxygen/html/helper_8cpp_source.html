39Id initShell()
40{
41 Cinfo::rebuildOpIndex();
42
43 Id shellId;
44
45 Element *shelle =
46 new GlobalDataElement(shellId, Shell::initCinfo(), "/", 1);
47
48 Id clockId = Id::nextId();
49 assert(clockId.value() == 1);
50 Id classMasterId = Id::nextId();
51 Id postMasterId = Id::nextId();
52
53 Shell *s = reinterpret_cast<Shell *>(shellId.eref().data());
54 s->setHardware(1, 1, 0);
55 s->setShellElement(shelle);
56
57 /// Sets up the Elements that represent each class of Msg.
58 auto numMsg = Msg::initMsgManagers();
59
60 new GlobalDataElement(clockId, Clock::initCinfo(), "clock", 1);
61 new GlobalDataElement(classMasterId, Neutral::initCinfo(), "classes", 1);
62 new GlobalDataElement(postMasterId, PostMaster::initCinfo(), "postmaster",
63 1);
64
65 assert(shellId == Id());
66 assert(clockId == Id(1));
67 assert(classMasterId == Id(2));
68 assert(postMasterId == Id(3));
69
70 Shell::adopt(shellId, clockId, numMsg++);
71 Shell::adopt(shellId, classMasterId, numMsg++);
72 Shell::adopt(shellId, postMasterId, numMsg++);
73 assert(numMsg == 10); // Must be the same on all nodes.
74
75 Cinfo::makeCinfoElements(classMasterId);
76 return shellId;
77}

79map<string, Finfo *> getFinfoDict(const Cinfo *cinfo, const string &fieldType)
80{
81 // All field types - available via Cinfo.finfoMap()
82 if(fieldType == "*") {
83 return cinfo->finfoMap();
84 }
85 map<string, Finfo *> ret;
86 // Other cases - for each field type XYZ, Cinfo has
87 // getNumXYZFinfo() to get the number of fields of that type,
88 // and getXYZFinfo(n) to get the n-th field of that type.
89 Finfo *(Cinfo::*finfoGetter)(unsigned int) const = nullptr;
90 unsigned int numFinfo{0};
91 if(fieldType == "valueFinfo" || fieldType == "value") {
92 numFinfo = cinfo->getNumValueFinfo();
93 finfoGetter = &Cinfo::getValueFinfo;
94 }
95 else if(fieldType == "srcFinfo" || fieldType == "src") {
96 numFinfo = cinfo->getNumSrcFinfo();
97 finfoGetter = &Cinfo::getSrcFinfo;
98 }
99 else if(fieldType == "destFinfo" || fieldType == "dest") {
100 numFinfo = cinfo->getNumDestFinfo();
101 finfoGetter = &Cinfo::getDestFinfo;
102 }
103 else if(fieldType == "lookupFinfo" || fieldType == "lookup") {
104 numFinfo = cinfo->getNumLookupFinfo();
105 finfoGetter = &Cinfo::getLookupFinfo;
106 }
107 else if(fieldType == "sharedFinfo" || fieldType == "shared") {
108 numFinfo = cinfo->getNumSharedFinfo();
109 finfoGetter = &Cinfo::getSharedFinfo;
110 }
111 else if(fieldType == "element" || fieldType == "elementFinfo" ||
112 fieldType == "field" || fieldType == "fieldElement" ||
113 fieldType == "fieldElementFinfo") {
114 numFinfo = cinfo->getNumFieldElementFinfo();
115 finfoGetter = &Cinfo::getFieldElementFinfo;
116 }
117 if(!finfoGetter) {
118 throw nb::value_error("Invalid field type");
119 }
120 for(unsigned int ii = 0; ii < numFinfo; ++ii) {
121 Finfo *finfo = (cinfo->*finfoGetter)(ii);
122 ret[finfo->name()] = finfo;
123 }
124 return ret;
125}

127vector<string> getFieldNames(const string &className, const string &fieldType)
128{
129 auto cinfo = Cinfo::find(className);
130 if(!cinfo) {
131 throw nb::key_error((className + ": no such class found").c_str());
132 }
133 auto finfoDict = getFinfoDict(cinfo, fieldType);
134 vector<string> ret;
135 ret.reserve(finfoDict.size());
136 for(const auto &it : finfoDict) {
137 ret.push_back(it.first);
138 }

166string getClassFieldsDoc(const Cinfo *cinfo, const string &ftype,
167 const string &prefix)
168{
169 stringstream ss;
170 map<string, Finfo *> fmap = getFinfoDict(cinfo, ftype);
171 if(fmap.size() == 0)
172 return "\n";
173
174 ss << moose::underlined<'-'>(moose::capitalize(ftype) + " Attributes:");
175
176 for(auto v : fmap) {
177 ss << fieldDocFormatted(v.first, cinfo, v.second, prefix);
178 }
179
180 // There are from base classes.
181 const Cinfo *baseClassCinfo = cinfo->baseCinfo();
182 while(baseClassCinfo) {
183 ss << prefix << "Attributes inherited from " << baseClassCinfo->name()
184 << ":\n";
185 auto baseFmap = getFinfoDict(baseClassCinfo, ftype);
186 for(const auto &vv : baseFmap) {
187 if(fmap.find(vv.first) == fmap.end()) {
188 fmap[vv.first] = vv.second;
189 ss << fieldDocFormatted(vv.first, baseClassCinfo, vv.second,
190 prefix);
191 }
192 }
193 baseClassCinfo = baseClassCinfo->baseCinfo();
194 }
195 return ss.str();
196}

215 ss << getClassFieldsDoc(cinfo, f, "");
216
217 return ss.str();

220string getDoc(const string &query)
221{
222 auto getClassAttributeDoc = [](const Cinfo *cinfo,
223 const string &fname) -> string {
224 const Finfo *finfo = cinfo->findFinfo(fname);
225 if(!finfo)
226 return "Error: '" + fname + "' not found.";
227 return fmt::format("{0}: {1} - {2}\n{3}", fname, finfo->rttiType(),
228 cinfo->getFinfoType(finfo), finfo->docs());
229 };
230
231 vector<string> tokens;
232 moose::tokenize(query, ".", tokens);
233
234 auto cinfo = Cinfo::find(tokens[0]);
235 string msg = "Class '" + tokens[0] + "' is not a valid MOOSE class.";
236 if(!cinfo)
237 throw nb::key_error(msg.c_str());
238
239 if(tokens.size() == 1)
240 return getClassDoc(tokens[0]);
241
242 if(tokens.size() == 2) {
243 cout << "Query: " << tokens[1] << endl;
244 return getClassAttributeDoc(cinfo, tokens[1]);
245 }
246
247 throw runtime_error(__func__ + string(":: Not supported '" + query + "'"));
248}

250bool setFieldGeneric(const ObjId &oid, const string &fieldName,
251 const nb::object &val)
252{
253 // if(fieldName == "dt") {
254 // throw nb::attribute_error(
255 // "Read-only property. Use `moose.setcClock(...)`");

265 "'.")

267 return false;

269
270 auto fieldType = finfo->rttiType();
271
272 // Remove any space in fieldType
273 fieldType.erase(
274 std::remove_if(fieldType.begin(), fieldType.end(), ::isspace),
275 fieldType.end());
276
277 if(fieldType == "double") {
278 return Field<double>::set(oid, fieldName, nb::cast<double>(val));
279 }
280 if(fieldType == "vector<double>")
281 return Field<vector<double>>::set(oid, fieldName,
282 nb::cast<vector<double>>(val));
283 if(fieldType == "float")
284 return Field<float>::set(oid, fieldName, nb::cast<float>(val));
285 if(fieldType == "unsignedint")
286 return Field<unsigned int>::set(oid, fieldName,
287 nb::cast<unsigned int>(val));
288 if(fieldType == "unsignedlong")
289 return Field<unsigned long>::set(oid, fieldName,
290 nb::cast<unsigned long>(val));
291 if(fieldType == "int")
292 return Field<int>::set(oid, fieldName, nb::cast<int>(val));
293 if(fieldType == "bool")
294 return Field<bool>::set(oid, fieldName, nb::cast<bool>(val));
295 if(fieldType == "string")
296 return Field<string>::set(oid, fieldName, nb::cast<string>(val));
297 if(fieldType == "vector<string>")
298 return Field<vector<string>>::set(oid, fieldName,
299 nb::cast<vector<string>>(val));
300 if(fieldType == "char")
301 return Field<char>::set(oid, fieldName, nb::cast<char>(val));
302 if(fieldType == "vector<ObjId>")
303 return Field<vector<ObjId>>::set(oid, fieldName,
304 nb::cast<vector<ObjId>>(val));
305 if(fieldType == "ObjId")
306 return Field<ObjId>::set(oid, fieldName, nb::cast<ObjId>(val));
307 if(fieldType == "Id") {
308 // NB: Handle MooseVec as well. Note that we send ObjId to the set
309 // function. The C++ implicit conversion takes care of the rest.
310 // Id tgt;
311 if(nb::isinstance<MooseVec>(val)) {
312 return Field<Id>::set(oid.id, fieldName,

317 }
318 else if(nb::isinstance<Id>(val)) {
319 return Field<Id>::set(oid.id, fieldName, nb::cast<Id>(val));
320 }
321 }
322 if(fieldType == "vector<double>") {
323 // NB: Note that we cast to ObjId here and not to Id.
324 return Field<vector<double>>::set(oid.id, fieldName,
325 nb::cast<vector<double>>(val));
326 }
327 if(fieldType == "vector<vector<double>>") {
328 // NB: Note that we cast to ObjId here and not to Id.
329 return Field<vector<vector<double>>>::set(
330 oid.id, fieldName, nb::cast<vector<vector<double>>>(val));
331 }
332 if(fieldType == "Variable")
333 if(fieldType == "Variable")
334 return Field<Variable>::set(oid, fieldName,
335 nb::cast<Variable>(val));
336
337 throw runtime_error("NotImplemented::setField: '" + fieldName +
338 "' with value type '" + fieldType + "'.");
339 return false;

342nb::object getFieldValue(const ObjId &oid, const Finfo *f)
343{
344 auto rttType = f->rttiType();
345 auto fname = f->name();
346 nb::object r = nb::none();
347 if(rttType == "double" || rttType == "float") {
348 r = nb::float_(Field<double>::get(oid, fname));
349 }
350 else if(rttType == "vector<double>") {
351 vector<double> val = Field<vector<double>>::get(oid, fname);
352 // Must copy data - val goes out of scope!
353 size_t size = val.size();
354
355 // Allocate numpy array and copy data into it
356 auto *data = new double[size];
357 std::copy(val.begin(), val.end(), data);
358
359 nb::capsule owner(
360 data, [](void *p) noexcept { delete[] static_cast<double *>(p); });
361 r = nb::cast(nb::ndarray<nb::numpy, double>(data, {size}, owner));
362 }
363 else if(rttType == "vector<unsigned int>") {
364 vector<unsigned int> val = Field<vector<unsigned int>>::get(oid, fname);
365 size_t size = val.size();
366
367 auto *data = new unsigned int[size];
368 std::copy(val.begin(), val.end(), data);
369
370 nb::capsule owner(data, [](void *p) noexcept {
371 delete[] static_cast<unsigned int *>(p);
372 });
373
374 r = nb::cast(nb::ndarray<nb::numpy, unsigned int>(data, {size}, owner));
375 }
376 else if(rttType == "vector<int>") {
377 vector<int> val = Field<vector<int>>::get(oid, fname);
378 size_t size = val.size();
379
380 auto *data = new int[size];
381 std::copy(val.begin(), val.end(), data);
382
383 nb::capsule owner(
384 data, [](void *p) noexcept { delete[] static_cast<int *>(p); });
385
386 r = nb::cast(nb::ndarray<nb::numpy, int>(data, {size}, owner));
387 }
388 else if(rttType == "string")
389 r = nb::cast(Field<string>::get(oid, fname));
390 else if(rttType == "char")
391 r = nb::int_(Field<char>::get(oid, fname));
392 else if(rttType == "int")
393 r = nb::int_(Field<int>::get(oid, fname));
394 else if(rttType == "unsigned int")
395 r = nb::int_(Field<unsigned int>::get(oid, fname));
396 else if(rttType == "unsigned long")
397 r = nb::int_(Field<unsigned long>::get(oid, fname));
398 else if(rttType == "bool")
399 r = nb::bool_(Field<bool>::get(oid, fname));
400 else if(rttType == "Id")
401 r = nb::cast(Field<Id>::get(oid, fname));
402 else if(rttType == "ObjId")
403 r = nb::cast(Field<ObjId>::get(oid, fname));
404 else if(rttType == "Variable")
405 r = nb::cast(Field<Variable>::get(oid, fname));
406 else if(rttType == "vector<Id>")
407 r = nb::cast(Field<vector<Id>>::get(oid, fname));
408 else if(rttType == "vector<ObjId>")
409 r = nb::cast(Field<vector<ObjId>>::get(oid, fname));
410 else if(rttType == "vector<string>")
411 r = nb::cast(Field<vector<string>>::get(oid, fname));
412 else {
413 cerr << "Warning: getValueFinfo:: Unsupported type '" + rttType + "'"
414 << endl;
415 r = nb::none();
416 }
417 return r;
418}

420nb::object createDestFunction(const ObjId &oid, const Finfo *finfo)
421{
422 auto rttiType = finfo->rttiType();
423 auto fname = finfo->name();
424
425 // Zero parameters
426 if(rttiType == "void") {
427 return nb::cpp_function(
428 [oid, fname]() { return SetGet0::set(oid, fname); });
429 }
430
431 // One parameter
432// Shorthand for defining single arg dest function
433#define DEST_FUNC_1(TYPE)
434 if(rttiType == #TYPE) {
435 return nb::cpp_function(
436 [oid, fname](TYPE val) {
437 return SetGet1<TYPE>::set(oid, fname, val);
438 },
439 nb::arg("value"));
440 }
441
442 DEST_FUNC_1(double)
443 DEST_FUNC_1(unsigned int)
444 DEST_FUNC_1(int)
445 DEST_FUNC_1(long)
446 DEST_FUNC_1(unsigned long)
447 DEST_FUNC_1(bool)
448 DEST_FUNC_1(string)
449 DEST_FUNC_1(Id)
450 DEST_FUNC_1(ObjId)
451 DEST_FUNC_1(vector<double>)
452 DEST_FUNC_1(vector<int>)
453 DEST_FUNC_1(vector<unsigned int>)
454 DEST_FUNC_1(vector<Id>)
455 DEST_FUNC_1(vector<ObjId>)
456 DEST_FUNC_1(vector<string>)
457#undef DEST_FUNC_1
458
459 vector<string> types;
460 moose::tokenize(rttiType, ",", types);
461
462 // Two parameters
463 // Short hand for 2-arg functions
464#define DEST_FUNC_2(T1, T2, TOKENS)
465 if(TOKENS[0] == #T1 && TOKENS[1] == #T2) {
466 return nb::cpp_function(
467 [oid, fname](const T1 &a, const T2 &b) {
468 return SetGet2<T1, T2>::set(oid, fname, a, b);
469 },
470 nb::arg("a"), nb::arg("b"));
471 }
472
473 if(types.size() == 2) {
474 DEST_FUNC_2(double, double, types)
475 DEST_FUNC_2(unsigned int, unsigned int, types)
476 DEST_FUNC_2(double, unsigned int, types)
477 DEST_FUNC_2(unsigned int, unsigned int, types)
478 DEST_FUNC_2(unsigned int, double, types)
479 DEST_FUNC_2(double, long, types)
480 DEST_FUNC_2(string, string, types)
481 DEST_FUNC_2(ObjId, ObjId, types)
482 DEST_FUNC_2(Id, double, types)
483 DEST_FUNC_2(vector<double>, string, types)
484 }
485#undef DEST_FUNC_2
486
487 // Three param destFinfo - rarely used (just the 6 specific cases)
488 //
489 // | CompartmentBase | displace | double, double, double |
490 // | SparseMsg | setEntry | unsigned int, unsigned int, unsigned int |
491 // | SparseMsg | tripletFill | vector<unsigne int>, vector<unsigned int>, vector<unsigned int> |
492 // | TableBase | compareXplot | string, string, string |
493 // | MarkovRateTable | set2d | unsigned int, unsigned int, Id |
494 // | MarkovRateTable | setconst | unsigned int, unsigned int, double |
495
496#define DEST_FUNC_3(T1, T2, T3, TYPES)
497 if(TYPES[0] == #T1 && TYPES[1] == #T2 && TYPES[2] == #T3) {
498 return nb::cpp_function(
499 [oid, fname](const T1 &a, const T2 &b, const T3 &c) {
500 return SetGet3<T1, T2, T3>::set(oid, fname, a, b, c);
501 },
502 nb::arg("a"), nb::arg("b"), nb::arg("c"));
503 }
504 if(types.size() == 3) {
505 DEST_FUNC_3(double, double, double, types)
506 DEST_FUNC_3(unsigned int, unsigned int, unsigned int, types)
507 DEST_FUNC_3(unsigned int, unsigned int, double, types)
508 DEST_FUNC_3(unsigned int, unsigned int, Id, types)
509 DEST_FUNC_3(string, string, string, types)
510 DEST_FUNC_3(vector<unsigned int>, vector<unsigned int>,
511 vector<unsigned int>, types)
512 }
513#undef DEST_FUNC_3
514
515 // 4-arg (just the 4 specific cases)
516 //
517 // | Class | Function | Types |
518 // |-----------------|------------|----------------------------------------------|
519 // | CubeMesh | buildMesh | Id, double, double, double |
520 // | TableBase | loadCSV | string, int, int, char |
521 // | TableBase | compareVec | string, string, unsigned int, unsigned int |
522 // | MarkovRateTable | set1d | unsigned int, unsigned int, Id, unsigned int |
523 //
524#define DEST_FUNC_4(T1, T2, T3, T4, TYPES)
525 if(TYPES[0] == #T1 && TYPES[1] == #T2 && TYPES[2] == #T3 &&
526 TYPES[3] == #T4) {
527 return nb::cpp_function(
528 [oid, fname](const T1 &a, const T2 &b, const T3 &c, const T4 &d) {
529 return SetGet4<T1, T2, T3, T4>::set(oid, fname, a, b, c, d);
530 },
531 nb::arg("a"), nb::arg("b"), nb::arg("c"), nb::arg("d"));
532 }
533
534 DEST_FUNC_4(Id, double, double, double, types)
535 DEST_FUNC_4(string, int, int, char, types)
536 DEST_FUNC_4(string, string, unsigned int, unsigned int, types)
537 DEST_FUNC_4(unsigned int, unsigned int, Id, unsigned int, types)
538#undef DEST_FUNC_4
539
540 throw nb::type_error(("Unsupported DestFinfo type: " + rttiType).c_str());
541}

543nb::object getFieldGeneric(const ObjId &oid, const string &fieldName)
544{
545
546 // Special fields that do not depend on a valid element being there: empty
547 // FieldElements
548 if(fieldName == "numData") {
549 return nb::cast(Field<unsigned int>::get(oid, "numData"));
550 }
551 else if(fieldName == "numFields") {
552 return nb::cast(Field<unsigned int>::get(oid, "numField"));
553 }
554
555 auto cinfo = oid.element()->cinfo();
556 auto finfo = cinfo->findFinfo(fieldName);
557
558 if(!finfo) {
559 throw nb::attribute_error(
560 (fieldName + " is not found on '" + oid.path() + "'.").c_str());
561 }
562
563 string finfoType = cinfo->getFinfoType(finfo);
564
565 if(finfoType == "ValueFinfo") {
566 return getFieldValue(oid, finfo);
567 }
568 else if(finfoType == "FieldElementFinfo") {
569 return nb::cast(ElementField(oid, finfo));
570 }
571 else if(finfoType == "LookupValueFinfo") {
572 return nb::cast(LookupField(oid, finfo));
573 }
574 else if(finfoType == "DestFinfo") {
575 return createDestFunction(oid, finfo);
576 }
577
578 throw runtime_error("getFieldGeneric::NotImplemented : " + fieldName +
579 " with rttType " + finfo->rttiType() + " and type: '" +
580 finfoType + "'");
581 return nb::none();
582}

584ObjId createElementFromPath(const string &type, const string &p,
585 unsigned int numdata)
586{
587
588 // NOTE: This function is bit costly because of regex use. One can replace
589 // it with bit more efficient one if required.
590 auto path = moose::normalizePath(p);
591
592 if(path.at(0) != '/') {
593 string cwe = getShellPtr()->getCwe().path();
594 if(cwe.back() != '/')
595 cwe += '/';
596 path = cwe + path;
597 }
598
599 // Split into dirname and basename component.
600 auto pp = moose::splitPath(path);
601 string name(pp.second);
602 if(name.empty()) {
603 throw nb::value_error(
604 ("path= " + path + ": path must not end with '/' except for root.")
605 .c_str());
606 }
607 if(name.back() == ']')
608 name = name.substr(0, name.find_last_of('['));
609
610 // Check if parent exists.
611 auto parent = ObjId(pp.first);
612 if(parent.bad()) {
613 throw std::runtime_error("Parent '" + pp.first +
614 "' is not found. Not creating...");
615 return Id();
616 }
617
618 // If path exists and user is asking for the same type then return the
619 // underlying object else raise an exception.
620 auto oid = ObjId(path);
621 if(!oid.bad()) {
622 if(oid.element()->cinfo()->name() == type)
623 return oid;
624 else
625 throw runtime_error("An object with path'" + path +
626 "' already "
627 "exists. Use moose.element to access it.");
628 }
629
630 Id newId = getShellPtr()->doCreate2(type, ObjId(pp.first), name, numdata);
631 return ObjId(newId);
632}

639ObjId convertToObjId(const nb::object &arg)
640{
641 ObjId ret;
642 if(nb::isinstance<nb::str>(arg)) {

745void showMsg(const nb::object &arg, MsgDirection direction)
746{
747 stringstream ss;
748 ObjId obj = convertToObjId(arg);
749
750 auto formatMessages = [&](const vector<ObjId> &msgs, bool isIncoming) {
751 const char *arrow = isIncoming ? "<--" : "-->";
752
753 for(const auto &msgObj : msgs) {
754 const Msg *msg = Msg::getMsg(msgObj);
755 if(!msg) {
756 ss << " (invalid message)\n";
757 continue;
758 }
759
760 Id e1 = msg->getE1();
761 Id e2 = msg->getE2();
762 bool objIsE1 = (obj.id == e1);
763
764 ObjId self = obj;
765 ObjId other = objIsE1 ? ObjId(e2) : ObjId(e1);
766
767 vector<string> selfFields, otherFields;
768 if(isIncoming) {
769 selfFields = objIsE1 ? msg->getDestFieldsOnE1()
770 : msg->getDestFieldsOnE2();
771 otherFields =
772 objIsE1 ? msg->getSrcFieldsOnE2() : msg->getSrcFieldsOnE1();
773 }
774 else {
775 selfFields =
776 objIsE1 ? msg->getSrcFieldsOnE1() : msg->getSrcFieldsOnE2();
777 otherFields = objIsE1 ? msg->getDestFieldsOnE2()
778 : msg->getDestFieldsOnE1();
779 }
780
781 ss << fmt::format(" {0} [{1}] {2} {3} [{4}]\n", self.path(),
782 moose::vectorToCSV<string>(selfFields), arrow,
783 other.path(),
784 moose::vectorToCSV<string>(otherFields));
785 }
786 };
787 if(direction != MsgDirection::Out) {
788 ss << "INCOMING:\n";
789 auto inmsgs = Field<vector<ObjId>>::get(obj, "msgIn");
790 formatMessages(inmsgs, true);
791 ss << "\n";
792 }
793
794 if(direction != MsgDirection::In) {
795 ss << "OUTGOING:\n";
796 auto outmsgs = Field<vector<ObjId>>::get(obj, "msgOut");
797 formatMessages(outmsgs, false);
798 }
799
800 nb::print(ss.str().c_str());
801}

803vector<ObjId> getNeighbors(const nb::object &arg, const string &fieldName,
804 const string &msgType,
805 MsgDirection direction = MsgDirection::Both)
806{
807 vector<ObjId> result;
808 ObjId obj = convertToObjId(arg);
809 // Normalize msgType once
810 string lowerMsgType = msgType;
811 std::transform(lowerMsgType.begin(), lowerMsgType.end(),
812 lowerMsgType.begin(), ::tolower);
813
814 // Collect messages based on direction
815 vector<ObjId> msgList;
816 if(direction != MsgDirection::Out) {
817 auto in = Field<vector<ObjId>>::get(obj, "msgIn");
818 msgList.insert(msgList.end(), in.begin(), in.end());
819 }
820 if(direction != MsgDirection::In) {
821 auto out = Field<vector<ObjId>>::get(obj, "msgOut");
822 msgList.insert(msgList.end(), out.begin(), out.end());
823 }
824
825 for(const auto &mobj : msgList) {
826 const Msg *msg = Msg::getMsg(mobj);
827 if(!msg)
828 continue;
829
830 // Filter by message type
831 if(!lowerMsgType.empty()) {
832 bool matches = (lowerMsgType == "single" &&
833 dynamic_cast<const SingleMsg *>(msg)) ||
834 (lowerMsgType == "onetoone" &&
835 dynamic_cast<const OneToOneMsg *>(msg)) ||
836 (lowerMsgType == "onetoall" &&
837 dynamic_cast<const OneToAllMsg *>(msg)) ||
838 (lowerMsgType == "diagonal" &&
839 dynamic_cast<const DiagonalMsg *>(msg)) ||
840 (lowerMsgType == "sparse" &&
841 dynamic_cast<const SparseMsg *>(msg));
842 if(!matches)
843 continue;
844 }
845
846 // Determine which end we're on
847 Id e1 = msg->getE1();
848 Id e2 = msg->getE2();
849 bool isE1 = (obj.id == e1);
850
851 // Collect relevant fields based on direction
852 set<string> fields;
853 if(direction != MsgDirection::Out) {
854 auto dest =
855 isE1 ? msg->getDestFieldsOnE1() : msg->getDestFieldsOnE2();
856 fields.insert(dest.begin(), dest.end());
857 }
858 if(direction != MsgDirection::In) {
859 auto src = isE1 ? msg->getSrcFieldsOnE1() : msg->getSrcFieldsOnE2();
860 fields.insert(src.begin(), src.end());
861 }
862
863 // Check field match and add neighbor
864 if(fieldName == "*" || fields.count(fieldName)) {
865 result.push_back(isE1 ? e2 : e1);
866 }
867 }
868
869 return result;
870}
