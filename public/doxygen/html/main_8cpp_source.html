86unsigned int getNumCores()
87{
88 unsigned int numCPU = 0;
89
90#ifdef LINUX
91 numCPU = sysconf( _SC_NPROCESSORS_ONLN );
92#endif
93
94#ifdef MACOSX
95 int mib[4];
96 size_t len = sizeof(numCPU);
97
98 /* set the mib for hw.ncpu */
99 mib[0] = CTL_HW;
100 mib[1] = HW_AVAILCPU; // alternatively, try HW_NCPU;
101
102 /* get the number of CPUs from the system */
103 sysctl(mib, 2, &numCPU, &len, NULL, 0);
104
105 if( numCPU < 1 )
106 {
107 mib[1] = HW_NCPU;
108 sysctl( mib, 2, &numCPU, &len, NULL, 0 );
109 }
110#endif
111 if ( numCPU < 1 )
112 {
113 cout << "No CPU information available. Assuming single core." << endl;
114 numCPU = 1;
115 }
116 return numCPU;
117}

137Id init(int argc, char** argv, bool& doUnitTests)
138{
139 unsigned int numCores = getNumCores();
140 int numNodes = 1;
141 int myNode = 0;
142 bool isInfinite = 0;
143 int opt;
144 Cinfo::rebuildOpIndex();
145
146#ifdef USE_MPI
147 MPI_Init( &argc, &argv );
148 MPI_Comm_size( MPI_COMM_WORLD, &numNodes );
149 MPI_Comm_rank( MPI_COMM_WORLD, &myNode );
150#endif
151
152 /**
153 * Here we allow the user to override the automatic identification
154 * of processor configuration
155 */
156 while ( ( opt = getopt( argc, argv, "hiqun:b:B:" ) ) != -1 )
157 {
158 switch ( opt )
159 {
160 case 'i' : // infinite loop, used for multinode debugging, to give gdb something to attach to.
161 isInfinite = 1;
162 break;
163 case 'n': // Multiple nodes
164 numNodes = (unsigned int)atoi( optarg );
165 break;
166 case 'B': // Benchmark plus dump data: handle later.
167 break;
168 case 'u': // Do unit tests, pass back.
169 doUnitTests = true;
170 break;
171 case 'q': // quit immediately after completion.
172 quitFlag = 1;
173 break;
174 case 'h': // help
175 default:
176 cout << "Usage: moose -help -infiniteLoop -unit_tests -quit -n numNodes\n";
177
178 exit( 1 );
179 }
180 }
181 if ( myNode == 0 )
182 {
183#ifndef QUIET_MODE
184 cout << "on node " << myNode << ", numNodes = "
185 << numNodes << ", numCores = " << numCores << endl;
186#endif
187 }
188
189 Id shellId;
190 Element* shelle =
191 new GlobalDataElement( shellId, Shell::initCinfo(), "root", 1 );
192
193 Id clockId = Id::nextId();
194 assert( clockId.value() == 1 );
195 Id classMasterId = Id::nextId();
196 Id postMasterId = Id::nextId();
197
198 Shell* s = reinterpret_cast< Shell* >( shellId.eref().data() );
199 s->setShellElement( shelle );
200 s->setHardware( numCores, numNodes, myNode );
201 s->loadBalance();
202
203 /// Sets up the Elements that represent each class of Msg.
204 unsigned int numMsg = Msg::initMsgManagers();
205
206 new GlobalDataElement( clockId, Clock::initCinfo(), "clock", 1 );
207 new GlobalDataElement( classMasterId, Neutral::initCinfo(), "classes", 1);
208 new GlobalDataElement( postMasterId, PostMaster::initCinfo(), "postmaster", 1 );
209
210 assert ( shellId == Id() );
211 assert( clockId == Id( 1 ) );
212 assert( classMasterId == Id( 2 ) );
213 assert( postMasterId == Id( 3 ) );
214
215 Shell::adopt( shellId, clockId, numMsg++ );
216 Shell::adopt( shellId, classMasterId, numMsg++ );
217 Shell::adopt( shellId, postMasterId, numMsg++ );
218
219 assert( numMsg == 10 ); // Must be the same on all nodes.
220
221 Cinfo::makeCinfoElements( classMasterId );
222
223 while ( isInfinite ) // busy loop for debugging under gdb and MPI.
224 ;
225
226 return shellId;
227}
