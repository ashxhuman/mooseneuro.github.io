108void VoxelPools::advance( const ProcInfo* p )
109{
110 double t = p->currTime - p->dt;
111 Ksolve* k = reinterpret_cast<Ksolve*>( stoichPtr_->getKsolve().eref().data() );
112
113 if( getMethod() == "lsoda" )
114 {
115 // True if first step or restart, or if diffusion. Tells LSODA to
116 // recalculate using new pool n values, which slows it down a bit.
117 if ( p->isStart() || (numVoxels_ > 1) )
118 lsodaState_ = 1;
119 size_t totVar = stoichPtr_->getNumVarPools() + stoichPtr_->getNumProxyPools();
120 vector<double> yout(size()+1);
121 pLSODA->lsoda_update( &VoxelPools::lsodaSys, size()
122 , Svec(), yout , &t
123 , p->currTime, &lsodaState_, this
124 );
125
126 // Now update the y from yout. This is different thant normal GSL or
127 // BOOST based approach.
128 // totVar += stoichPtr_->getNumFuncPools();
129 for (size_t i = 0; i < totVar; i++)
130 varS()[i] = yout[i+1];
131
132 if( lsodaState_ == 0 )
133 {
134 cerr << "Error: VoxelPools::advance: LSODA integration error at time "
135 << t << "\n";
136 assert(0);
137 }
138 }
139 else
140 {
141
142#ifdef USE_GSL
143 int status = gsl_odeiv2_driver_apply( driver_, &t, p->currTime, varS());
144 if ( status != GSL_SUCCESS )
145 {
146 cerr << "Error: VoxelPools::advance: GSL integration error at time "
147 << t << "\n";
148 cerr << "Error info: " << status << ", " <<
149 gsl_strerror( status ) << endl;
150 if ( status == GSL_EMAXITER )
151 cerr << "Max number of steps exceeded\n";
152 else if ( status == GSL_ENOPROG )
153 cerr << "Timestep has gotten too small\n";
154 else if ( status == GSL_EBADFUNC )
155 cerr << "Internal error\n";
156 assert( 0 );
157 }
158
159#elif USE_BOOST_ODE
160 // NOTE: Make sure to assing vp to BoostSys vp. In next call, it will be used by
161 // updateRates func. Unlike gsl call, we can't pass extra void* to gslFunc.
162 VoxelPools* vp = reinterpret_cast< VoxelPools* >( this );
163
164 /*-----------------------------------------------------------------------------
165NOTE: 04/21/2016 11:31:42 AM
166
167We need to call updateFuncs here (unlike in GSL solver) because there
168is no way we can update const vector_type_& y in evalRatesUsingBoost
169function. In gsl implmentation one could do it, because const_cast can
170take away the constantness of double*. This probably makes the call bit
171cleaner.
172 *-----------------------------------------------------------------------------*/
173 stoichPtr_->updateFuncs( &Svec()[0], p->currTime );
174
175 /*-----------------------------------------------------------------------------
176 * Using integrate function works with with default stepper type.
177 *
178 * NOTICE to developer:
179 * If you are planning your own custom typdedef of stepper_type_ (see
180 * file BoostSystem.h), the you may run into troble. Have a look at this
181 * http://boostw.boost.org/doc/libs/1_56_0/boost/numeric/odeint/integrate/integrate.hpp
182 *
183 * To make numerical results comparable with "gsl" solvers,
184 * by default, we pick adaptive step-size control. Use a suffix 'c' to
185 * force a constant step size. The numerical accuracy of constant step
186 * size solver is low.
187 * More details can be found here:
188 * https://www.boost.org/doc/libs/1_72_0/libs/numeric/odeint/doc/html/boost_numeric_odeint/odeint_in_detail/steppers.html
189 */
190
191 const double fixedDt = 0.1;
192
193 if( method_ == "rk2" )
194 odeint::integrate_const( rk_midpoint_stepper_type_()
195 , [this](const vector_type_& dy, vector_type_& dydt, const double t) {

200 );

341/**
342 * @Synopsis Function to pass LSODA::lsoda_update function. Since it is
343 a * static function, we have to make sure void* param holds the value of
344 pointer * to VoxelPools. * * @Param t * @Param y * @Param dydt * @Param
345 params Address of VoxelPools as void*. */
346/*
347----------------------------------------------------------------------------*/
348void VoxelPools::lsodaSys( double t, double* y, double* dydt, void* param)
349{
350 VoxelPools* vp = reinterpret_cast< VoxelPools* >( param );
351 // Fill in the values.
352 // Ensure that the buffered values are assigned to y.
353 size_t totVar = vp->stoichPtr_->getNumVarPools() + vp->stoichPtr_->getNumProxyPools();
354 for( size_t ii = totVar + vp->stoichPtr_->getNumFuncPools(); ii < vp->size(); ii++ ) {
355 y[ii] = vp->Svec()[ii];
356 }
357
358 vp->stoichPtr_->updateFuncs( y, t );
359 for( size_t ii = totVar; ii < totVar + vp->stoichPtr_->getNumFuncPools(); ii++ ) {
360 vp->Svec()[ii] = y[ii];
361 }
362 vp->updateRates( y, dydt );
363}

369void VoxelPools::updateAllRateTerms( const vector< RateTerm* >& rates,
370 unsigned int numCoreRates )
371{
372 // Clear out old rates if any
373 for ( unsigned int i = 0; i < rates_.size(); ++i )
374 delete( rates_[i] );
375
376 rates_.resize(rates.size());
377
378 for ( unsigned int i = 0; i < numCoreRates; ++i )
379 {
380 rates_[i] = rates[i]->copyWithVolScaling( getVolume(), 1, 1 );
381 }
382
383 for ( unsigned int i = numCoreRates; i < rates.size(); ++i )
384 {
385 rates_[i] = rates[i]->copyWithVolScaling( getVolume(),
386 getXreacScaleSubstrates(i-numCoreRates),
387 getXreacScaleProducts(i-numCoreRates)
388 );
389 }
390}

392void VoxelPools::updateRateTerms( const vector< RateTerm* >& rates,
393 unsigned int numCoreRates, unsigned int index )
394{
395 // During setup or expansion of the reac system, it is possible to
396 // call this function before the rates_ term is assigned. Disable.
397 if ( index >= rates_.size() )
398 return;
399 delete( rates_[index] );
400 if ( index >= numCoreRates )
401 {
402 rates_[index] = rates[index]->copyWithVolScaling(
403 getVolume(),
404 getXreacScaleSubstrates(index - numCoreRates),
405 getXreacScaleProducts(index - numCoreRates ) );
406 }
407 else
408 rates_[index] = rates[index]->copyWithVolScaling(getVolume(), 1.0, 1.0);
409}

411void VoxelPools::updateRates( const double* s, double* yprime ) const
412{
413 const KinSparseMatrix& N = stoichPtr_->getStoichiometryMatrix();
414 vector< double > v( N.nColumns(), 0.0 );
415 vector< double >::iterator j = v.begin();
416 // totVar should include proxyPools only if this voxel uses them
417 unsigned int totVar = stoichPtr_->getNumVarPools() + stoichPtr_->getNumProxyPools();
418 // totVar should include proxyPools if this voxel does not use them
419 unsigned int totInvar = stoichPtr_->getNumBufPools();
420 assert( N.nColumns() == 0 || N.nRows() == stoichPtr_->getNumAllPools() );
421 assert( N.nColumns() == rates_.size() );
422
423 for ( auto i = rates_.cbegin(); i != rates_.end(); i++)
424 *j++ = (**i)( s );
425 for (unsigned int i = 0; i < totVar; ++i)
426 {
427 auto rate = N.computeRowRate( i, v );
428 assert(! std::isnan(rate));
429 *yprime++ = rate;
430 }
431 for (unsigned int i = 0; i < totInvar ; ++i)
432 *yprime++ = 0.0;
433}

440void VoxelPools::updateReacVelocities(const double* s, vector< double >& v) const

443 assert( N.nColumns() == rates_.size() );
444
445 v.clear();
446 v.resize( rates_.size(), 0.0 );
447
448 vector< RateTerm* >::const_iterator i;
449 vector< double >::iterator j = v.begin();
450
451 for ( i = rates_.begin(); i != rates_.end(); i++)

454 assert(! std::isnan(*(j-1)));
455 }
456}
