74 /**
75 * processFunc handles the update and calculations every
76 * clock tick. It first sends the request for evaluation of
77 * the gate variables to the respective gate objects and
78 * recieves their response immediately through a return
79 * message. This is done so that many channel instances can
80 * share the same gate lookup tables, but do so cleanly.
81 * Such messages should never go to a remote node.
82 * Then the function does its own little calculations to
83 * send back to the parent compartment through regular
84 * messages.
85 */
86 void vProcess(const Eref& e, ProcPtr p) override;
87
88 /**
89 * Reinitializes the values for the channel. This involves
90 * computing the steady-state value for the channel gates
91 * using the provided Vm from the parent compartment. It
92 * involves a similar cycle through the gates and then
93 * updates to the parent compartment as for the processFunc.
94 */
95 void vReinit(const Eref& e, ProcPtr p) override;
96 /**
97 * Assign the local conc_ to the incoming conc from the
98 * concentration calculations for the compartment. Typically
99 * the message source will be a CaConc object, but there
100 * are other options for computing the conc.
101 */
102 void conc1(double conc);
103 void conc2(double conc);
104
105
106 /**
107 * Function for safely creating each gate, identified by strings
108 * as X, Y and Z. Will only work on a new channel, not on a
109 * copy. The idea is that the gates are always referred to the
110 * original 'library' channel, and their contents cannot be touched
111 * except by the original.
112 */
113 void vCreateGate(const Eref& e, string gateType) override;
114
115 /// Inner utility function for creating the gate.
