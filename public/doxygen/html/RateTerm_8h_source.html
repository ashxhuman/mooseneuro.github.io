18class RateTerm
19{
20public:
21 RateTerm() {;}
22 virtual ~RateTerm() {;}
23 /// Computes the rate. The argument is the molecule array.
24 virtual double operator() ( const double* S ) const = 0;
25
26 /**
27 * Assign the rates.
28 */
29 virtual void setRates( double k1, double k2 ) = 0;
30
31 /// Used by Zombie to assign rate terms
32 virtual void setR1( double k1 ) = 0;
33
34 /// Used by Zombie to assign rate terms
35 virtual void setR2( double k2 ) = 0;
36
37 /// Used by Zombie to return rate terms
38 virtual double getR1() const = 0;
39
40 /// Used by Zombie to return rate terms
41 virtual double getR2() const = 0;
42
43 /**
44 * This function finds the reactant indices in the vector
45 * S. It returns the number of substrates found, which are the
46 * first entries in molIndex. The products are the remaining ones.
47 * Note that it does NOT find products for unidirectional
48 * reactions, which is a bit of a problem.
49 */
50 virtual unsigned int getReactants(
51 vector< unsigned int >& molIndex ) const = 0;
52 static const double EPSILON;
53
54 /**
55 * This is used to rescale the RateTerm kinetics when the
56 * compartment volume changes. This is needed because the kinetics
57 * are in extensive units, that is, mol numbers, rather than in
58 * intensive units like concentration. So when the volume changes
59 * the rate terms change. Each Rate term checks if any of its
60 * reactant molecules are affected, and if so, rescales.
61 * Ratio is newVol / oldVol
62 */
63 virtual void rescaleVolume( short comptIndex,
64 const vector< short >& compartmentLookup, double ratio ) = 0;
65
66 /**
67 * Duplicates rate term and then applies volume scaling.
68 * Arguments are volume of reference voxel,
69 * product of vol/refVol for all substrates: applied to R1
70 * product of vol/refVol for all products: applied to R2
71 *
72 * Note that unless the reaction is cross-compartment, the
73 * vol/refVol will be one.
74 */
75 virtual RateTerm* copyWithVolScaling(
76 double vol, double sub, double prd ) const = 0;
77};

145// Single substrate MMEnzyme: by far the most common.
146class MMEnzyme1: public MMEnzymeBase
147{
148public:
149 MMEnzyme1( double Km, double kcat,
150 unsigned int enz, unsigned int sub )
151 : MMEnzymeBase( Km, kcat, enz ), sub_( sub )
152 {
153 ;
154 }
155
156 double operator() ( const double* S ) const
157 {
158 // assert( S[ sub_ ] >= -EPSILON
159 auto val = ( kcat_ * S[ sub_ ] * S[ enz_ ] ) / ( Km_ + S[ sub_ ] );
160 assert(! std::isnan(val));
161 return val;
162 }
163
164 unsigned int getReactants( vector< unsigned int >& molIndex ) const
165 {
166 molIndex.resize( 2 );
167 molIndex[0] = enz_;
168 molIndex[1] = sub_;
169 return 2;
170 }
171
172 RateTerm* copyWithVolScaling(
173 double vol, double sub, double prd ) const
174 {
175 double ratio = vol * sub * NA;
176 return new MMEnzyme1( ratio * Km_, kcat_, enz_, sub_);
177 }
178
179private:
180 unsigned int sub_;
181};

184{
185public:
186 MMEnzyme( double Km, double kcat,
187 unsigned int enz, RateTerm* sub )
188 : MMEnzymeBase( Km, kcat, enz ), substrates_( sub )
189 {
190 ;
191 }
192
193 double operator() ( const double* S ) const
194 {
195 double sub = (*substrates_)( S );
196 // the subtrates_() operator returns the conc product.
197 assert( sub >= -EPSILON );
198 auto val = ( sub * kcat_ * S[ enz_ ] ) / ( Km_ + sub );
199 assert(! std::isnan(val));
200 return val;
201 }
202
203 unsigned int getReactants( vector< unsigned int >& molIndex ) const
204 {
205 substrates_->getReactants( molIndex );
206 molIndex.insert( molIndex.begin(), enz_ );
207 return molIndex.size();
208 }
209
210 RateTerm* copyWithVolScaling(
211 double vol, double sub, double prd ) const
212 {
213 double ratio = sub * vol * NA;
214 return new MMEnzyme( ratio * Km_, kcat_, enz_, substrates_ );
215 }
216private:
217 RateTerm* substrates_;
218};

463/**
464 * The reason for this class is that if we have a single substrate
465 * acting in 2nd order, we must account for the removal of one molecule
466 * halfway through the reaction. This has a major effect when there
467 * is only one molecule to start with: the eventual # would become
468 * negative if we didn't do this correction!
469 */
470class StochSecondOrderSingleSubstrate: public ZeroOrder
471{
472public:
473 StochSecondOrderSingleSubstrate( double k, unsigned int y )
474 : ZeroOrder( k ), y_( y )
475 {;}
476

484

489 molIndex[1] = y_;

492

511class NOrder: public ZeroOrder
512{
513public:
514 NOrder( double k, vector< unsigned int > v )
515 : ZeroOrder( k ), v_( v )
516 {;}
517
518 double operator() ( const double* S ) const
519 {
520 double ret = k_;
521 vector< unsigned int >::const_iterator i;
522 for ( i = v_.begin(); i != v_.end(); i++)
523 {
524 assert(! std::isnan( S[ *i ] ) );
525 ret *= S[ *i ];
526 }
527 return ret;
528 }
529
530 unsigned int getReactants( vector< unsigned int >& molIndex ) const
531 {
532 molIndex = v_;
533 return v_.size();
534 }
535
536 void rescaleVolume( short comptIndex,
537 const vector< short >& compartmentLookup, double ratio )
538 {
539 for ( unsigned int i = 1; i < v_.size(); ++i )
540 {
541 if ( comptIndex == compartmentLookup[ v_[i] ] )
542 k_ /= ratio;
543 }
544 }
545
546 RateTerm* copyWithVolScaling(
547 double vol, double sub, double prd ) const
548 {
549 assert( v_.size() > 0 );
550 double ratio = sub * pow( NA * vol, (int)( v_.size() ) - 1 );
551 return new NOrder( k_ / ratio, v_ );
552 }
553
554protected:
555 vector< unsigned int > v_;
556};
