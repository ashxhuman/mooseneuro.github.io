228void CaConcBase::updateDimensions(const Eref& e)
229{
230 double vol{0};
231 if(length_ > 0) {
232 vol = PI * diameter_ * diameter_ * length_ * 0.25;
233 }
234 else {
235 vol = PI * diameter_ * diameter_ * diameter_ * 0.5 / 3;
236 }
237 if(thickness_ > 0 && thickness_ < diameter_ / 2.0) {
238 double coreRadius = diameter_ / 2.0 - thickness_;
239 if(length_ > 0) {
240 vol -= PI * coreRadius * coreRadius * length_;
241 }
242 else {
243 vol -= 4 * PI * coreRadius * coreRadius * coreRadius / 3;
244 }
245 }
246 double B = 1.0 / (VALENCE * FaradayConst * vol);
247 vSetB(e, B);
248}

326// static func
327void CaConcBase::zombify(Element* orig, const Cinfo* zClass, Id hsolve)
328{
329 if(orig->cinfo() == zClass)
330 return;
331 unsigned int start = orig->localDataStart();
332 unsigned int num = orig->numLocalData();
333 if(num == 0)
334 return;
335 vector<double> data(num * 9);
336
337 unsigned int j = 0;
338 for(unsigned int i = 0; i < num; ++i) {
339 Eref er(orig, i + start);
340 const CaConcBase* cb = reinterpret_cast<const CaConcBase*>(er.data());
341 data[j + 0] = cb->getCa(er);
342 data[j + 1] = cb->getCaBasal(er);
343 data[j + 2] = cb->getTau(er);
344 data[j + 3] = cb->getB(er);
345 data[j + 4] = cb->getCeiling(er);
346 data[j + 5] = cb->getFloor(er);
347 data[j + 6] = cb->getThickness(er);
348 data[j + 7] = cb->getLength(er);
349 data[j + 8] = cb->getDiameter(er);
350 j += 9;
351 }
352 orig->zombieSwap(zClass);
353 j = 0;
354 for(unsigned int i = 0; i < num; ++i) {
355 Eref er(orig, i + start);
356 CaConcBase* cb = reinterpret_cast<CaConcBase*>(er.data());
357 cb->vSetSolver(er, hsolve);
358 cb->setCa(er, data[j + 0]);
359 cb->setCaBasal(er, data[j + 1]);
360 cb->setTau(er, data[j + 2]);
361 cb->setB(er, data[j + 3]);
362 cb->setCeiling(er, data[j + 4]);
363 cb->setFloor(er, data[j + 5]);
364 cb->setThickness(er, data[j + 6]);
365 cb->setLength(er, data[j + 7]);
366 cb->setDiameter(er, data[j + 8]);
367 j += 9; // was 7?
368 }
369}
