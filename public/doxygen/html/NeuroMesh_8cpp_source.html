336/**
337 * This assumes that lambda is the quantity to preserve, over numEntries.
338 * So when the compartment changes size, so does numEntries.
339 * Assumes that the soma node is at index 0.
340 */
341void NeuroMesh::updateCoords()
342{
343 unsigned int startFid = 0;
344 if ( nodes_.size() <= 1 ) // One for soma and one for dummy pa of soma
345 {
346 buildStencil();
347 return;
348 }
349 for ( vector< NeuroNode >::iterator i = nodes_.begin();
350 i != nodes_.end(); ++i )
351 {
352 if ( !i->isDummyNode() )
353 {
354 double len = i->getLength();
355 unsigned int numDivs = static_cast<unsigned int>(floor( 0.5 + len / diffLength_ ));
356 if ( numDivs < 1 )
357 numDivs = 1;
358 i->setNumDivs( numDivs );
359 i->setStartFid( startFid );
360 startFid += numDivs;
361 }
362 }
363 nodeIndex_.resize( startFid );
364 for ( unsigned int i = 0; i < nodes_.size(); ++i )
365 {
366 if ( !nodes_[i].isDummyNode() )
367 {
368 unsigned int end = nodes_[i].startFid() +nodes_[i].getNumDivs();
369 assert( end <= startFid );
370 assert( nodes_[i].getNumDivs() > 0 );
371 for ( unsigned int j = nodes_[i].startFid(); j < end; ++j )
372 nodeIndex_[j] = i;
373 }
374 }
375 // Assign volumes and areas
376 vs_.resize( startFid );
377 area_.resize( startFid );
378 length_.resize( startFid );
379 for ( unsigned int i = 0; i < nodes_.size(); ++i )
380 {
381 const NeuroNode& nn = nodes_[i];
382 if ( !nn.isDummyNode() )
383 {
384 assert( nn.parent() < nodes_.size() );
385 const NeuroNode& parent = nodes_[ nn.parent() ];
386 for ( unsigned int j = 0; j < nn.getNumDivs(); ++j )
387 {
388 vs_[j + nn.startFid()] = nn.voxelVolume( parent, j );
389 area_[j + nn.startFid()] = nn.getMiddleArea( parent, j);
390 length_[j + nn.startFid()] = nn.getVoxelLength();
391 }
392 }
393 }
394 buildStencil();
395}

411 // someone has overloaded tolower so it can have one or two args, so

429 for ( vector< NeuroNode >::iterator
430 i = nodes_.begin(); i != nodes_.end(); ++i )
431 i->setIsCylinder( isCylinder );

467{
468 Id soma = origSoma;
469 if ( nodes_[maxDiaIndex].elecCompt() == soma ) // Happy, happy

471 ;

473 else if ( soma == Id() )
474 {
475 soma = nodes_[maxDiaIndex].elecCompt();
476 }
477 else // Disagreement. Ugh.
478 {
479 string name = nodes_[ maxDiaIndex ].elecCompt().element()->getName();

482 {
483 soma = nodes_[maxDiaIndex].elecCompt();
484 }
485 else
486 {
487 cout << "Warning: NeuroMesh::putSomaAtStart: named 'soma' compartment isn't biggest\n";
488 soma = nodes_[maxDiaIndex].elecCompt();
489 }
490 }
491 // Move soma to start of nodes_ vector.
492 if ( maxDiaIndex != 0 )
493 {
494 NeuroNode temp = nodes_[0];
495 nodes_[0] = nodes_[maxDiaIndex];
496 nodes_[maxDiaIndex] = temp;
497 }
498 return soma;

501void NeuroMesh::insertSingleDummy(
502 unsigned int parent, unsigned int self,
503 double x, double y, double z )
504{
505 static const double EPSILON = 1e-8;
506 NeuroNode dummy( nodes_[ self ] );
507 dummy.clearChildren();
508 dummy.setNumDivs( 0 );
509 bool isCylinder = (geometryPolicy_ == "cylinder" );
510 dummy.setIsCylinder( isCylinder );
511 dummy.setX( x );
512 dummy.setY( y );
513 dummy.setZ( z );
514 // Now insert the dummy as a surrogate parent.
515 dummy.setParent( parent );
516 dummy.addChild( self );
517 nodes_[ self ].setParent( nodes_.size() );
518 // Idiot check for a bad dimensioned compartment.
519 if ( nodes_[self].calculateLength( dummy ) < EPSILON )
520 {
521 double length = nodes_[self].getLength();
522 dummy.setX( x - length );
523 assert( doubleEq( nodes_[self].calculateLength( dummy ), length) );
524 }
525 nodes_.push_back( dummy );
526}

528void NeuroMesh::insertDummyNodes()
529{
530 // First deal with the soma, always positioned at node 0.
531 unsigned int num = nodes_.size();
532 for ( unsigned int i = 0; i < num; ++i )
533 {
534 if ( nodes_[i].parent() == ~0U )
535 {
536 Id elec = nodes_[i].elecCompt();
537 double x = Field< double >::get( elec, "x0" );
538 double y = Field< double >::get( elec, "y0" );
539 double z = Field< double >::get( elec, "z0" );
540 insertSingleDummy( ~0U, i, x, y, z );
541 }
542 }
543
544 // Second pass: insert dummy nodes for children.
545 // Need to know if parent has multiple children, because each of
546 // them will need a dummyNode to connect to.
547 // In all the policies so far, the dummy nodes take the same diameter
548 // as the children that they host.
549 for ( unsigned int i = 0; i < nodes_.size(); ++i )
550 {
551 vector< unsigned int > kids = nodes_[i].children();
552 if ( (!nodes_[i].isDummyNode()) && kids.size() > 1 )
553 {
554 for( unsigned int j = 0; j < kids.size(); ++j )
555 {
556 double x = nodes_[i].getX(); // use coords of parent.
557 double y = nodes_[i].getY();
558 double z = nodes_[i].getZ();
559 insertSingleDummy( i, kids[j], x, y, z );
560 // Replace the old kid entry with the dummy
561 kids[j] = nodes_.size() - 1;
562 }
563 // Connect up the parent to the dummy nodes.
564 nodes_[i].clearChildren();
565 for( unsigned int j = 0; j < kids.size(); ++j )
566 nodes_[i].addChild( kids[j] );
567 }
568 }

614/**
615 * converts the parents_ vector from identifying the parent NeuroNode
616 * to identifying the parent voxel, for each shaft entry.
617 */
618void NeuroMesh::updateShaftParents()
619{
620 assert( shaft_.size() == parent_.size() );
621 vector< unsigned int > pa = parent_;
622 for ( unsigned int i = 0; i < shaft_.size(); ++i )
623 {
624 const NeuroNode& nn = nodes_[ pa[i] ];
625 double x0 = Field< double >::get( shaft_[i], "x0" );
626 double y0 = Field< double >::get( shaft_[i], "y0" );
627 double z0 = Field< double >::get( shaft_[i], "z0" );
628 const NeuroNode& pn = nodes_[ nn.parent() ];
629 unsigned int index = 0;
630 double r = nn.nearest( x0, y0, z0, pn, index );
631 if ( r >= 0.0 )
632 {
633 parent_[i] = index + nn.startFid();
634 }
635 else
636 {
637 cout << "Warning: NeuroMesh::updateShaftParents: may be"
638 " misaligned on " << i << ", r=" << r <<
639 "\n pt=(" << x0 << "," << y0 << "," << z0 << ")" <<
640 "pa=(" << nn.getX() << "," << nn.getY() << "," << nn.getZ() << ")" <<
641 "\n";
642 parent_[i] = index + nn.startFid();
643 }
644 }

705unsigned int NeuroMesh::getNumSegments() const
706{
707 unsigned int ret = 0;
708 for ( vector< NeuroNode >::const_iterator i = nodes_.begin();
