30double StochNOrder::operator() ( const double* S ) const
31{
32 double ret = k_;
33 vector< unsigned int >::const_iterator i;
34 unsigned int lasty = ~0U;
35 double y = 0.0;
36 for ( i = v_.begin(); i != v_.end(); i++)
37 {
38 assert( !std::isnan( S[ *i ] ) );
39 if ( lasty == *i ) {
40 y -= 1.0;
41 } else {
42 y = S[ *i ];
43 }
44 ret *= y;
45 if ( ret < 0.0 ) {
46 return 0.0;
47 }
48 lasty = *i;
49 }
50 return ret;
51}
