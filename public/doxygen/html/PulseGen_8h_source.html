34/**
35 * PulseGen acts as a pulse generator. It generates square pulses of
36 * specified duration and amplitude. Two consecutive pulses are
37 * separated by specified delay.
38 */
39
40class PulseGen {
41 /**
42 With trigMode = FREE_RUN the pulse generator will create a
43 train of pulses determined by the firstDealy, firstWidth,
44 firstLevel, secondDelay, secondWidth and secondLevel.
45 */
46 static const int FREE_RUN = 0;
47 /**
48 With trigMode = EXT_TRIG the pulse generator may be triggered
49 or may generate a single pulse and then stop. The logic
50 depends on trigTime field. If trigTime is negative ( which it
51 is by default and remains so unless there is a non zero input
52 trigger to the PulseGen object ), the output remains at
53 baselevel. For trigTime >= 0, if there is no input trigger, the
54 PulseGen object will produce pair of pulses, the first one
55 starting at firstDelay and staying at firstLevel for firstWidth
56 time. And then the second one starting at secondDelay time
57 after the start of the first pulse, staying at secondLevel for
58 secondWidth time. If secondDelay is less than firstWidth, there
59 will be no second pulse.
60
61 */
62 static const int EXT_TRIG = 1;
63 /**
64 With trigMode = EXT_GATE, the pulse occurs firstDelay later
65 from the leading edge of the input.
66 */
67 static const int EXT_GATE = 2;
