227void PyRun::trigger(const Eref& e, double input)
228{
229 if(!runcompiled_.is_valid() || mode_ == RUNPROC) {
230 return;
231 }
232
233 try {
234 locals_[inputvar_.c_str()] = nb::cast(input);
235
236 // Execute the compiled code
237 PyObject* result =
238 PyEval_EvalCode(runcompiled_.ptr(), globals_.ptr(), locals_.ptr());
239 if(!result) {
240 throw nb::python_error();
241 }
242 Py_DECREF(result);
243
244 // Get output variable
245 if(locals_.contains(outputvar_.c_str())) {
246 double output = nb::cast<double>(locals_[outputvar_.c_str()]);
247 outputOut()->send(e, output);
248 }
249 }
250 catch(nb::python_error& err) {
251 std::cerr << "ERROR: PyRun::trigger(): " << err.what() << std::endl;
252 throw;
253 }
254}

300void PyRun::reinit(const Eref& e, ProcPtr p)
301{
302
303 nb::gil_scoped_acquire gil;
304
305 try {
306 // Initialize globals from __main__ if not set
307 if(!globals_.is_valid() || globals_.empty()) {
308 nb::object main_module = nb::module_::import_("__main__");
309 globals_ =
310 nb::borrow<nb::dict>(PyModule_GetDict(main_module.ptr()));
311 }
312 if(!globals_.contains("__builtins__")) {
313 globals_["__builtins__"] = nb::borrow(PyEval_GetBuiltins());
314 }
315 // Initialize locals if not set
316 if(!locals_.is_valid()) {
317 locals_ = nb::dict();
318 }
319 // Compile and run init string
320 if(!initstr_.empty()) {
321 PyObject* compiled = Py_CompileString(
322 initstr_.c_str(), "moose.PyRun::reinit", Py_file_input);
323 if(!compiled) {
324 std::cerr << "Error compiling initString" << std::endl;
325 throw nb::python_error();
326 }
327 initcompiled_ = nb::steal(compiled);
328
329 PyObject* result = PyEval_EvalCode(initcompiled_.ptr(),
330 globals_.ptr(), locals_.ptr());
331 if(!result) {
332 throw nb::python_error();
333 }
334 Py_DECREF(result);
335 }
336
337 // Compile and run runString
338 PyObject* compiled = Py_CompileString(
339 runstr_.c_str(), "moose.PyRun::reinit", Py_file_input);
340 if(!compiled) {
341 std::cerr << "Error compiling runString" << std::endl;
342 throw nb::python_error();
343 }
344 runcompiled_ = nb::steal(compiled);
345 /// init should only run `initStr`
346 if(evalOnReinit_) {
347 PyObject* result = PyEval_EvalCode(initcompiled_.ptr(),
348 globals_.ptr(), locals_.ptr());
349 if(!result) {
350 throw nb::python_error();
351 }
352 Py_DECREF(result);
353 }
354 }
355 catch(nb::python_error& err) {
356 std::cerr << "ERROR: PyRun::reinit(): " << err.what() << std::endl;
357 throw;
358 }
359}
