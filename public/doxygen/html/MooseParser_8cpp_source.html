166/* --------------------------------------------------------------------------*/
167/**
168 * @Synopsis EXPRTK does not have && and || but have 'and' and 'or' symbol.
169 * Replace && with 'and' and '||' with 'or'.
170 *
171 * @Param user_expr
172 *
173 * @Returns
174 */
175/* ----------------------------------------------------------------------------*/
176string MooseParser::Reformat( const string user_expr )
177{
178 string expr( user_expr );
179
180 // Replate || with 'or'
181 moose::str_replace_all( expr, "||", " or " );
182 // Replace && with 'and'
183 moose::str_replace_all( expr, "&&", " and " );
184
185 // Trickt business: Replace ! with not but do not change !=
186 moose::str_replace_all( expr, "!=", "@@@" ); // placeholder
187 moose::str_replace_all( expr, "!", " not " );
188 moose::str_replace_all( expr, "@@@", "!=" ); // change back @@@ to !=
189
190 return expr;
191}

209bool MooseParser::ParseVariables(const string& expr, vector<string>& vars)
210{
211
212 ASSERT_FALSE(expr.empty(),
213 __func__ << ": Empty expression not allowed here");
214
215 Parser::symbol_table_t symtab;
216 Parser::expression_t expression;
217 Parser::parser_t parser;
218 parser.enable_unknown_symbol_resolver();
219 symtab.add_constants();
220 init_symtab(symtab);
221 expression.register_symbol_table(symtab);
222 bool res = parser.compile(expr, expression);
223 if (!res) {
224 Parser::varmap_type varmap;
225 stringstream ss;
226 ss << "Failed to parse '" << expr << "' :" << endl;
227 for (unsigned int i = 0; i < parser.error_count(); ++i) {
228 Parser::error_t error = parser.get_error(i);
229 ss << "Error[" << i << "] Position: " << error.token.position
230 << " Type: [" << exprtk::parser_error::to_str(error.mode)
231 << "] Msg: " << error.diagnostic << endl;
232
233 // map is
234 auto n = symtab.get_variable_list(varmap);
235 ss << "More Information:\nTotal variables " << n << ".";
236 for (auto i : varmap)
237 ss << "\t" << i.first << "=" << i.second << " "
238 << symtab.get_variable(i.first)->ref();
239 ss << endl;
240 }
241 // Throw the error, this is handled in callee.
242 throw moose::Parser::exception_type(ss.str());
243 }
244 vector<string> varlist;
245 symtab.get_variable_list(varlist);
246 for (auto name : varlist) {
247 if (!symtab.is_constant_node(name)) {
248 vars.push_back(name);
249 }
250 }
251 return res;
252}

255/**
256 * @Synopsis Compile a given expression.
257 *
258 * @Returns Return true if successful, throws exception if compilation fails.
259 * Exception includes a detailed diagnostic.
260 */
261/* ----------------------------------------------------------------------------*/
262bool MooseParser::CompileExpr()
263{
264 // User should make sure that symbol table has been setup. Do not raise
265 // exception here. User can set expression again.
266 // GCC specific
267 ASSERT_FALSE(expr_.empty(),
268 __func__ << ": Empty expression not allowed here");
269
270 // expression_.release();
271 // symbolTable_.clear_variables();
272 Parser::parser_t parser;
273 parser.enable_unknown_symbol_resolver();
274 valid_ = parser.compile(expr_, expression_);
275 // This should never occur, as we are running this as a second pass
276 if (!valid_) {
277 expr_ = "";
278 stringstream ss;
279 ss << "Failed to parse '" << expr_ << "' :" << endl;
280 for (unsigned int i = 0; i < parser.error_count(); ++i)
281 {
282 Parser::error_t error = parser.get_error(i);
283 ss << "Error[" << i << "] Position: " << error.token.position
284 << " Type: [" << exprtk::parser_error::to_str(error.mode)
285 << "] Msg: " << error.diagnostic << endl;
286
287 // map is
288 Parser::varmap_type vars;
289 auto n = symbolTable_.get_variable_list(vars);
290 ss << "More Information:\nTotal variables " << n << ".";
291 for (auto i : vars)
292 ss << "\t" << i.first << "=" << i.second << " "
293 << symbolTable_.get_variable(i.first)->ref();
294 ss << endl;
295 }
296 valid_ = false;
297 throw moose::Parser::exception_type(ss.str());
298 }
299 return valid_;
300}

303double MooseParser::Derivative(const string& name, unsigned int nth) const
304{
