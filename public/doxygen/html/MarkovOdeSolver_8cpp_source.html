167#ifdef USE_GSL
168int MarkovOdeSolver::evalSystem( double t, const double* state, double* f, void *params)
169{
170 vector< vector< double > >* Q = static_cast< vector< vector< double > >* >( params );
171 unsigned int nVars = Q->size();
172
173 //Matrix being accessed along columns, which is a very bad thing in terms of
174 //cache optimality. Transposing the matrix during reinit() would be a good idea.
175 for ( unsigned int i = 0; i < nVars; ++i)
176 {
177 f[i] = 0;
178 for ( unsigned int j = 0; j < nVars; ++j)
179 f[i] += state[j] * ((*Q)[j][i]);
180 }
181 return 0;
182}
183#endif /* ----- not USE_GSL ----- */

193/* ----------------------------------------------------------------------------*/
194void MarkovOdeSolver::OdeSystem( const vector<double>& y, vector<double>& dydt )
195{
196 unsigned int nVars = Q_.size();
197 //Matrix being accessed along columns, which is a very bad thing in terms of
198 //cache optimality. Transposing the matrix during reinit() would be a good idea.
199 for ( unsigned int i = 0; i < nVars; ++i)
200 {
201 dydt[i] = 0;
202 for ( unsigned int j = 0; j < nVars; ++j)
203 dydt[i] += y[j] * Q_[j][i];
204 }
205}

311//Handles data from MarkovChannel class.
312void MarkovOdeSolver::init( vector< double > initialState )
313{
314
315 nVars_ = initialState.size();
316 state_ = initialState;
317 initialState_ = initialState;
318 Q_.resize( nVars_ );
319 for ( unsigned int i = 0; i < nVars_; ++i )
320 Q_[i].resize( nVars_, 0.0 );
321
322 stateOde_.resize( nVars_ );
323 isInitialized_ = 1;
324
325#ifdef USE_GSL
326 assert( gslStepType_ != 0 );
327 if ( gslStep_ )
328 gsl_odeiv_step_free(gslStep_);
329 gslStep_ = gsl_odeiv_step_alloc( gslStepType_, nVars_ );
330 assert( gslStep_ != 0 );
331 if ( !gslEvolve_ )
332 gslEvolve_ = gsl_odeiv_evolve_alloc(nVars_);
333 else
334 gsl_odeiv_evolve_reset(gslEvolve_);
335 assert(gslEvolve_ != 0);
336 if ( !gslControl_ )
337 gslControl_ = gsl_odeiv_control_y_new( absAccuracy_, relAccuracy_ );
338 else
339 gsl_odeiv_control_init(gslControl_,absAccuracy_, relAccuracy_, 1, 0);
340 assert(gslControl_!= 0);
341 gslSys_.function = &MarkovOdeSolver::evalSystem;
342 gslSys_.jacobian = 0;
343 gslSys_.dimension = nVars_;
344 gslSys_.params = static_cast< void * >( &Q_ );
345#endif /* ----- not USE_GSL ----- */
346
347}

352void MarkovOdeSolver::process( const Eref& e, ProcPtr p )
353{
354 double nextt = p->currTime + p->dt;
355 double t = p->currTime;
356 double sum = 0;
357
358 for ( unsigned int i = 0; i < nVars_; ++i )
359 stateOde_[i] = state_[i];
360
361
362#ifdef USE_GSL
363 while( t < nextt )
364 {
365 int status = gsl_odeiv_evolve_apply ( gslEvolve_, gslControl_, gslStep_, &gslSys_,
366 &t, nextt, &internalStepSize_, &stateOde_[0]
367 );
368 //Simple idea borrowed from Dieter Jaeger's implementation of a Markov
369 //channel to deal with potential round-off error.
370 sum = 0;
371 for ( unsigned int i = 0; i < nVars_; i++ )
372 sum += stateOde_[i];
373
374 for ( unsigned int i = 0; i < nVars_; i++ )
375 stateOde_[i] /= sum;
376
377 if( status != GSL_SUCCESS )
378 break;
379 }
380#endif
381
382#if USE_BOOST_ODE
383 auto sys = [this](const vector<double>& dy, vector<double>& dydt, const double t)
384 {
385 this->OdeSystem(dy, dydt);
386 };
387
388 // It works well in practice for this setup.
389 if( method_ == "rk5" || method_ == "gsl" || method_ == "boost" )
390 odeint::integrate( sys, stateOde_, t, nextt, p->dt );
391 else if( method_ == "rk5a" || method_ == "adaptive" )
392 odeint::integrate_adaptive(
393 odeint::make_controlled<rk_dopri_stepper_type_>( absAccuracy_, relAccuracy_ )
394 , sys, stateOde_, t, nextt, p->dt
395 );
396 else if( method_ == "rk4" )
397 odeint::integrate_const( rk4_stepper_type_()
398 , sys, stateOde_, t, nextt, p->dt
399 );
400 else if ("rk54" == method_ )
401 odeint::integrate_const( rk_karp_stepper_type_()
402 , sys, stateOde_, t, nextt, p->dt
403 );
404 else if ("rkck" == method_ || "rkcka" == method_)
405 odeint::integrate_adaptive(
406 odeint::make_controlled<rk_karp_stepper_type_>( absAccuracy_, relAccuracy_ )
407 , sys, stateOde_, t, nextt, p->dt
408 );
409 else if( method_ == "rk8a" || "rk8" == method_ )
410 odeint::integrate_adaptive( rk_felhberg_stepper_type_()
411 , sys, stateOde_, t, nextt, p->dt
412 );
413 else
414 odeint::integrate( sys, stateOde_, t, nextt, p->dt );
415#endif /* ----- not USE_GSL ----- */
416
417
418 for ( unsigned int i = 0; i < nVars_; ++i )
419 state_[i] = stateOde_[i];
420
421 stateOut()->send( e, state_ );
422}
