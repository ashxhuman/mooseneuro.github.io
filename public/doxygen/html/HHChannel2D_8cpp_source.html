246int HHChannel2D::dependency(string index, unsigned int dim)
247{
248 static vector<map<string, int>> dep;
249 if(dep.empty()) {
250 dep.resize(2);
251
252 dep[0]["VOLT_INDEX"] = 0;
253 dep[0]["C1_INDEX"] = 1;
254 dep[0]["C2_INDEX"] = 2;
255
256 dep[0]["VOLT_C1_INDEX"] = 0;
257 dep[0]["VOLT_C2_INDEX"] = 0;
258 dep[0]["C1_C2_INDEX"] = 1;
259
260 dep[1]["VOLT_INDEX"] = -1;
261 dep[1]["C1_INDEX"] = -1;
262 dep[1]["C2_INDEX"] = -1;
263
264 dep[1]["VOLT_C1_INDEX"] = 1;
265 dep[1]["VOLT_C2_INDEX"] = 2;
266 dep[1]["C1_C2_INDEX"] = 2;
267 }
268
269 if(dep[dim].find(index) == dep[dim].end())
270 return -1;
271
272 if(dep[dim][index] == 0)
273 return 0;
274 if(dep[dim][index] == 1)
275 return 1;
276 if(dep[dim][index] == 2)
277 return 2;
278
279 return -1;
280}

300void HHChannel2D::vProcess(const Eref &e, ProcPtr info)
301{
302 g_ += ChanBase::getGbar(e);
303 double A = 0;
304 double B = 0;
305 if(Xpower_ > 0) {
306 xGate_->lookupBoth(depValue(Xdep0_), depValue(Xdep1_), &A, &B);
307 if(instant_ & INSTANT_X)
308 X_ = A / B;
309 else
310 X_ = integrate(X_, info->dt, A, B);
311 g_ *= takeXpower_(X_, Xpower_);
312 }
313
314 if(Ypower_ > 0) {
315 yGate_->lookupBoth(depValue(Ydep0_), depValue(Ydep1_), &A, &B);
316 if(instant_ & INSTANT_Y)
317 Y_ = A / B;
318 else
319 Y_ = integrate(Y_, info->dt, A, B);
320
321 g_ *= takeYpower_(Y_, Ypower_);
322 }
323
324 if(Zpower_ > 0) {
325 zGate_->lookupBoth(depValue(Zdep0_), depValue(Zdep1_), &A, &B);
326 if(instant_ & INSTANT_Z)
327 Z_ = A / B;
328 else
329 Z_ = integrate(Z_, info->dt, A, B);
330
331 g_ *= takeZpower_(Z_, Zpower_);
332 }
333
334 ChanBase::setGk(e, g_ * vGetModulation(e));
335 updateIk();
336 // Gk_ = g_;
337 // Ik_ = ( Ek_ - Vm_ ) * g_;
338
339 // Send out the relevant channel messages.
340 sendProcessMsgs(e, info);
341 g_ = 0.0;
342}

344/**
345 * Here we get the steady-state values for the gate (the 'instant'
346 * calculation) as A_/B_.
347 */
348void HHChannel2D::vReinit(const Eref &er, ProcPtr info)
349{
350 g_ = ChanBase::getGbar(er);
351 Element *e = er.element();
352
353 double A = 0.0;
354 double B = 0.0;
355 if(Xpower_ > 0) {
356 xGate_->lookupBoth(depValue(Xdep0_), depValue(Xdep1_), &A, &B);
357 if(B < EPSILON) {
358 cout << "Warning: B_ value for " << e->getName()
359 << " is ~0. Check X table\n";
360 return;
361 }
362 if(!xInited_)
363 X_ = A / B;
364 g_ *= takeXpower_(X_, Xpower_);
365 }
366
367 if(Ypower_ > 0) {
368 yGate_->lookupBoth(depValue(Ydep0_), depValue(Ydep1_), &A, &B);
369 if(B < EPSILON) {
370 cout << "Warning: B value for " << e->getName()
371 << " is ~0. Check Y table\n";
372 return;
373 }
374 if(!yInited_)
375 Y_ = A / B;
376 g_ *= takeYpower_(Y_, Ypower_);
377 }
378
379 if(Zpower_ > 0) {
380 zGate_->lookupBoth(depValue(Zdep0_), depValue(Zdep1_), &A, &B);
381 if(B < EPSILON) {
382 cout << "Warning: B value for " << e->getName()
383 << " is ~0. Check Z table\n";
384 return;
385 }
386 if(!zInited_)
387 Z_ = A / B;
388 g_ *= takeZpower_(Z_, Zpower_);
389 }
390
391 ChanBase::setGk(er, g_ * vGetModulation(er));
392 updateIk();
393 // Gk_ = g_;
394 // Ik_ = ( Ek_ - Vm_ ) * g_;
395
396 // Send out the relevant channel messages.
397 // Same for reinit as for process.
398 sendReinitMsgs(er, info);
399 g_ = 0.0;
400}

406/**
407 * If the gate exists and has only this element for input, then change
408 * the gate power.
409 * If the gate exists and has multiple parents, then make a new gate,
410 * set its power.
411 * If the gate does not exist, make a new gate, set its power.
412 *
413 * The function is designed with the idea that if copies of this
414 * channel are made, then they all point back to the original HHGate.
415 * (Unless they are cross-node copies).
416 * It is only if we subsequently alter the HHGate of this channel that
417 * we need to make our own variant of the HHGate, or disconnect from
418 * an existing one.
419 * \todo: May need to convert to handling arrays and Erefs.
420 */
421// Assuming that the elements are simple elements. Use Eref for
422// general case
423
424bool HHChannel2D::checkOriginal(Id chanId) const
425{
426 bool isOriginal = 1;
427 if(xGate_) {
428 isOriginal = xGate_->isOriginalChannel(chanId);
429 }
430 else if(yGate_) {
431 isOriginal = yGate_->isOriginalChannel(chanId);
432 }
433 else if(zGate_) {
434 isOriginal = zGate_->isOriginalChannel(chanId);
435 }
436 return isOriginal;
437}

451void HHChannel2D::vCreateGate(const Eref &e, string gateType)
452{
453 if(!checkOriginal(e.id())) {
454 cout << "Warning: HHChannel2D::createGate: Not allowed from copied "
455 "channel:\n"
456 << e.id().path() << "\n";
457 return;
458 }
459
460 if(gateType == "X")
461 innerCreateGate("xGate", &xGate_, e.id(), Id(e.id().value() + 1));
462 else if(gateType == "Y")
463 innerCreateGate("yGate", &yGate_, e.id(), Id(e.id().value() + 2));
464 else if(gateType == "Z")
465 innerCreateGate("zGate", &zGate_, e.id(), Id(e.id().value() + 3));
466 else
467 cout << "Warning: HHChannel2D::createGate: Unknown gate type '"
468 << gateType << "'. Ignored\n";
469}
