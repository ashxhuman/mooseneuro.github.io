357void testStats()
358{
359 Shell* shell = reinterpret_cast< Shell* >( Id().eref().data() );
360 unsigned int size = 1000;
361 vector< double > sinewave( size, 0 );
362 for ( unsigned int i = 0; i < size; ++i ) {
363 sinewave[i] = sin( 2 * i * PI / static_cast< double >( size ) );
364 }
365 ObjId tabid = shell->doCreate( "StimulusTable", ObjId(), "tab", 1 );
366 assert( tabid != ObjId() );
367 Field< vector< double > >::set( tabid, "vector", sinewave );
368 Field< double >::set( tabid, "startTime", 0 );
369 Field< double >::set( tabid, "stopTime", size );
370 Field< double >::set( tabid, "loopTime", size );
371 Field< double >::set( tabid, "stepSize", 0 );
372 Field< double >::set( tabid, "stepPosition", 0 );
373 Field< bool >::set( tabid, "doLoop", true );
374
375 ObjId stat1 = shell->doCreate( "Stats", ObjId(), "stat1", 1 );
376 Field< unsigned int >::set( stat1, "windowLength", size );
377
378 ObjId mid = shell->doAddMsg( "Single", tabid, "output",
379 stat1, "input" );
380
381 shell->doUseClock( "/tab", "process", 0 );
382 shell->doUseClock( "/stat1", "process", 1 );
383 shell->doSetClock( 0, 1 );
384 shell->doSetClock( 1, 1 );
385
386 shell->doReinit();
387 shell->doStart( size/2 );
388 double mean = Field< double >::get( stat1, "mean" );
389 double sdev = Field< double >::get( stat1, "sdev" );
390 double sum = Field< double >::get( stat1, "sum" );
391 unsigned int num = Field< unsigned int >::get( stat1, "num" );
392 double wmean = Field< double >::get( stat1, "wmean" );
393 double wsdev = Field< double >::get( stat1, "wsdev" );
394 double wsum = Field< double >::get( stat1, "wsum" );
395 unsigned int wnum = Field< unsigned int >::get( stat1, "wnum" );
396 assert( doubleApprox( mean, 2.0 / PI ) );
397 assert( doubleApprox( sdev, 0.3077627 ) );
398 assert( doubleApprox( sum, size / PI ) );
399 assert( doubleApprox( num, size/2 ) );
400 assert( doubleApprox( wmean, 2.0 / PI ) );
401 // assert( doubleApprox( wsdev, 1.0/sqrt( 2.0 ) ) );
402 assert( doubleApprox( wsum, size / PI ) );
403 assert( wnum == size/2 );
404
405 shell->doStart( size );
406 mean = Field< double >::get( stat1, "mean" );
407 sdev = Field< double >::get( stat1, "sdev" );
408 sum = Field< double >::get( stat1, "sum" );
409 num = Field< unsigned int >::get( stat1, "num" );
410 wmean = Field< double >::get( stat1, "wmean" );
411 wsdev = Field< double >::get( stat1, "wsdev" );
412 wsum = Field< double >::get( stat1, "wsum" );
413 wnum = Field< unsigned int >::get( stat1, "wnum" );
414 assert( doubleApprox( mean, 2.0 / (PI * 3.0) ) );
415 assert( doubleApprox( sdev, 0.6745136 ) );
416 assert( doubleApprox( sum, size / PI ) );
417 assert( doubleApprox( num, 3 * size/2 ) );
418 assert( doubleApprox( wmean, 0 ) );
419 assert( doubleApprox( wsdev, 1.0/sqrt( 2.0 ) ) );
420 assert( doubleApprox( wsum, 0 ) );
421 assert( wnum == size );
422
423 shell->doDelete( tabid );
424 shell->doDelete( stat1 );
425 cout << "." << flush;
426}
