202void SpikeGen::process( const Eref& e, ProcPtr p )
203{
204 double t = p->currTime;
205 if ( V_ > threshold_ ) {
206 if ((t + p->dt/2.0) >= (lastEvent_ + refractT_)) {
207 if ( !( edgeTriggered_ && fired_ ) ) {
208 spikeOut()->send( e, t );
209 lastEvent_ = t;
210 fired_ = true;
211 }
212 }
213 } else {
214 fired_ = false;
215 }
216}

219void SpikeGen::reinit( const Eref& e, ProcPtr p )
220{
221 lastEvent_ = -refractT_ ;
222}

234void testSpikeGen()
235{
236 Shell* shell = reinterpret_cast< Shell* >( Id().eref().data() );
237 Id sid = shell->doCreate( "SpikeGen", Id(), "spike", 1, MooseGlobal );
238 SpikeGen& sg = *( reinterpret_cast< SpikeGen* >( sid.eref().data() ) );
239
240 Eref er( sid.eref() );
241 ProcInfo p;
242 p.dt = 0.001;
243 p.currTime = 0.0;
244 sg.setThreshold( 1.0 );
245 sg.setRefractT( 0.005 );
246
247 sg.reinit( er, &p );
248 sg.handleVm( 0.5 );
249 sg.process( er, &p );
250 assert( !sg.getFired() );
251 p.currTime += p.dt;
252
253 sg.handleVm( 0.999 );
254 sg.process( er, &p );
255 assert( !sg.getFired() );
256 p.currTime += p.dt;
257
258 sg.handleVm( 1.001 );
259 sg.process( er, &p );
260 assert( sg.getFired() );
261 p.currTime += p.dt;
262
263 sg.handleVm( 0.999 );
264 sg.process( er, &p );
265 assert( !sg.getFired() );
266 p.currTime += p.dt;
267
268 sg.handleVm( 2.0 ); // Too soon, refractory
269 sg.process( er, &p );
270 assert( !sg.getFired() );
271
272 p.currTime += 0.005; // Now post-refractory
273 sg.handleVm( 2.0 ); // Now not refractory
274 sg.process( er, &p );
275 assert( sg.getFired() );
276
277 sid.destroy();
278 cout << "." << flush;
279}
