117double HHGate2D::lookupA(vector<double> v) const
118{
119 if(v.size() < 2) {
120 cerr << "Error: HHGate2D::getAValue: 2 real numbers needed to lookup "
121 "2D table.\n";
122 return 0.0;
123 }
124
125 if(v.size() > 2) {
126 cerr << "Error: HHGate2D::getAValue: Only 2 real numbers needed to "
127 "lookup 2D table. "
128 "Using only first 2.\n";
129 }
130
131 return A_.innerLookup(v[0], v[1]);
132}
133
134double HHGate2D::lookupB(vector<double> v) const
135{
136 if(v.size() < 2) {
137 cerr << "Error: HHGate2D::getAValue: 2 real numbers needed to lookup "
138 "2D table.\n";
139 return 0.0;
140 }
141
142 if(v.size() > 2) {
143 cerr << "Error: HHGate2D::getAValue: Only 2 real numbers needed to "
144 "lookup 2D table. "
145 "Using only first 2.\n";
146 }
147
148 return B_.innerLookup(v[0], v[1]);
149}

151void HHGate2D::lookupBoth(double v, double c, double* A, double* B) const
152{
153 *A = A_.innerLookup(v, c);
154 *B = B_.innerLookup(v, c);
155}
