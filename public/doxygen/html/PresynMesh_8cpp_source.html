319/// Virtual function to return coords of mesh Entry.
320/// Assume that it is a hemisphere with the flat face toward postsyn.
321/// midpoint, direction, diameter.
322vector< double > PresynMesh::getCoordinates( unsigned int fid ) const
323{
324 assert( fid < boutons_.size() );
325 const Bouton& b = boutons_[fid];
326 vector< double > ret(7);
327 ret[0] = b.x_; // Midpoint
328 ret[1] = b.y_;
329 ret[2] = b.z_;
330 ret[3] = b.vx_; // direction
331 ret[4] = b.vy_;
332 ret[5] = b.vz_;
333 ret[6] = 2.0 * pow( b.volume_ * 1.5 / PI, 1.0/3.0 );
334
335 return ret;
336}

381/// Requires that all objects in v are the head compartments of spines.
382void PresynMesh::buildOnSpineHeads( vector< ObjId > v )
383{
384 isOnSpines_ = true;
385 spacing_ = 0; // Indicate that the spacing term isn't being used.
386 boutons_.resize( v.size() );
387 for (unsigned int i = 0; i < v.size(); ++i ) {
388 if ( !v[i].element()->cinfo()->isA( "CompartmentBase" ) ) {
389 cout << "Error: Attempt to assign PresynMesh to a non_compartment: " << v[i].id.path() << endl;
390 assert( 0 );
391 }
392
393 Bouton& b = boutons_[i];
394 b.postsynCompt_ = v[i];
395 double length = Field< double >::get( v[i], "length" );
396 double diameter = Field< double >::get( v[i], "diameter" );
397 // Set up coords on end of compt
398 b.x_ = Field< double >::get( v[i], "x" );
399 b.y_ = Field< double >::get( v[i], "y" );
400 b.z_ = Field< double >::get( v[i], "z" );
401 // Define unit vector of direction of presyn from x to x0. (toward)
402 b.vx_ = (Field< double >::get( v[i], "x0" ) - b.x_) / length;
403 b.vy_ = (Field< double >::get( v[i], "y0" ) - b.y_) / length;
404 b.vz_ = (Field< double >::get( v[i], "z0" ) - b.z_) / length;
405 // Shift bouton coords 20 nanometres away from spine end.
406 b.x_ -= b.vx_ * 20e-9;
407 b.y_ -= b.vy_ * 20e-9;
408 b.z_ -= b.vz_ * 20e-9;
409 b.volume_ = length * diameter * diameter * 0.25 * PI;
410 }
411}

413void PresynMesh::buildOnDendrites( vector< ObjId > compts, double spacing )
414{
415 isOnSpines_ = false;
416 spacing_ = spacing;
417 boutons_.clear();
418 for ( ObjId& v : compts ) {
419 if ( !v.element()->cinfo()->isA( "CompartmentBase" ) ) {
420 cout << "Error: Attempt to assign PresynMesh to a non_compartment: " << v.id.path() << endl;
421 assert( 0 );
422 }
423 double length = Field< double >::get( v, "length" );
424 double radius = Field< double >::get( v, "diameter" ) / 2.0;
425 double x = Field< double >::get( v, "x" );
426 double y = Field< double >::get( v, "y" );
427 double z = Field< double >::get( v, "z" );
428 double x0 = Field< double >::get( v, "x0" );
429 double y0 = Field< double >::get( v, "y0" );
430 double z0 = Field< double >::get( v, "z0" );
431
432 Vec v0( x0, y0, z0 );
433 Vec v1( x, y, z );
434 Vec vAxial( v1 - v0 );
435 Vec vRadial1( 0, 0, 0);
436 Vec vRadial2( 0, 0, 0);
437 vAxial.orthogonalAxes( vRadial1, vRadial2 );
438
439 for (double r = 0.5 * spacing; r < length; r += spacing) {
440 Bouton b;
441 b.postsynCompt_ = v;
442 b.vx_ = -vRadial1.a0(); // Components of the radial unit vector
443 b.vy_ = -vRadial1.a1();
444 b.vz_ = -vRadial1.a2();
445 b.x_ = x0 + (r/length) * (x - x0) - (radius + 20e-9) * b.vx_;
446 b.y_ = y0 + (r/length) * (y - y0) - (radius + 20e-9) * b.vy_;
447 b.z_ = z0 + (r/length) * (z - z0) - (radius + 20e-9) * b.vz_;
448 b.volume_ = DEFAULT_BOUTON_VOLUME;
449 boutons_.push_back( b );
450 }
451 }
452}

454/**
455 * setRadiusStats is meant to be called before the presynMesh is populated
456 * with the chemical system, but after the boutons are created.
457 */
458void PresynMesh::setRadiusStats( double r, double sdev )
459{
460 // As a placeholder, just go through assigning vols without the sdev.
461 if ( isOnSpines_ ) {
462 for ( Bouton& b : boutons_ )
463 b.volume_ *= r*r*r; // Scale relative to postsyn.
464 } else {
465 double v = r*r*r*PI*2.0/3.0; // Hemisphere
466 for ( Bouton& b : boutons_ )
467 b.volume_ = v; // Absolute assignment.
468 }

572// This function returns the distance to the nearest mesh entry, and
573// passes back its index.
574double PresynMesh::nearest( double x, double y, double z,
575 unsigned int& index ) const
576{
577 double distance = 1e19;
578 index = 0;
579 for ( unsigned int i = 0; i < boutons_.size(); i++ ) {
580 const Bouton& b = boutons_[i];
581 double r = (b.x_ - x) * (b.x_ - x) + (b.y_ - y) * (b.y_ - y ) + (b.z_ - z) * (b.z_ - z);
582 if ( distance > r ) {
583 index = i;
584 distance = r;
585 }
586 }
587 if (distance == 1e19)
588 return -1;
589 return distance;
590}
