63typedef struct {
64 string containerPath;
65 // For elec models this is the path of the Neuron object.
66 // For chem models this is the path of the Mesh object.
67 // This is used to provide a single set of coords.
68 string relPath; // Relative to container
69 // for elec models it is wildcard path of all objects wrt container
70 // for chem models it is rel path to pool[]. Braces for indices.
71 string nsdfContainerPath; // NSDF makes a group of the container.
72 // Sits on /data/uniform.
73 // Converts / to %
74 string nsdfRelPath; // nsdf makes subgroups identifed by nsdfRelPath,
75 // with respect to thei container.
76 // Sits on /data/uniform/container
77 // It is the relpath with the / replaced by %
78 vector< string > objPathList; // relative paths of all objects here.
79 // For elec compts, it is the compt names
80 // For chem obj it is just the indices.
81 string field; // Regular MOOSE value field.
82 string getField; // name of call to get the field.
83 string className; // All obj in a block should be of same class.
84 vector< vector< double > > data; // data_[objIdx][timeStep]
85 vector< ObjId > objVec;
86 hid_t container; // reference to container
87 hid_t relPathContainer; // reference to objects on nsdfRelPath
88 // hid_t filespace;
89 hid_t dataset;
90 bool hasMsg;
91 bool hasContainer;
92} Block;

94/**
95 NSDFWriter dumps data in NSDF file format.
96
97 - As of June 2015, MOOSE uses fixed time steps for updating field
98 values. So we support only Uniform data.
99
100 - This class has one SrcFinfo to request data. Multiple get{Field}
101 DestFinfos can be connected to this and the NSDFWriter class will put
102 together the ones with same name in one 2D dataset.
103
104 - One DestFinfo where SrcFinfos sending out event times can be
105 connected. These will go under Event data.
106
107 */

178 Storing uniform data:
179
180 The data coming in return for requestOut gets stored in
181 dataBuffer in the same sequence as the connections,

183
