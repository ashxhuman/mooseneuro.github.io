81vector< string > Msg::getSrcFieldsOnE1() const
82{
83 vector< pair< BindIndex, FuncId > > ids;
84 vector< string > ret;
85
86 e1_->getFieldsOfOutgoingMsg( mid_, ids );
87
88 for ( unsigned int i = 0; i < ids.size(); ++i )
89 {
90 string name = e1_->cinfo()->srcFinfoName( ids[i].first );
91 if ( name == "" )
92 {
93 cout << "Error: Msg::getSrcFieldsOnE1: Failed to find field on msg " <<
94 e1_->getName() << "-->" << e2_->getName() << endl;
95 }
96 else
97 {
98 ret.push_back( name );
99 }
100 }
101 return ret;
102}

253// Static func. Returns the index to use for msgIndex.
254unsigned int Msg::initMsgManagers()
255{
256 Dinfo< short > dummyDinfo;
257
258 // This is to be the parent of all the msg managers.
259 msgManagerId_ = Id::nextId();
260 new GlobalDataElement(
261 msgManagerId_, Neutral::initCinfo(), "Msgs", 1 );
262
263 SingleMsg::managerId_ = Id::nextId();
264 new MsgElement( SingleMsg::managerId_, SingleMsg::initCinfo(),
265 "singleMsg", &SingleMsg::numMsg, &SingleMsg::lookupMsg );
266
267 OneToOneMsg::managerId_ = Id::nextId();
268 new MsgElement( OneToOneMsg::managerId_, OneToOneMsg::initCinfo(),
269 "oneToOneMsg", &OneToOneMsg::numMsg, &OneToOneMsg::lookupMsg );
270
271 OneToAllMsg::managerId_ = Id::nextId();
272 new MsgElement( OneToAllMsg::managerId_, OneToAllMsg::initCinfo(),
273 "oneToAllMsg", &OneToAllMsg::numMsg, &OneToAllMsg::lookupMsg );
274
275 DiagonalMsg::managerId_ = Id::nextId();
276 new MsgElement( DiagonalMsg::managerId_, DiagonalMsg::initCinfo(),
277 "diagonalMsg", &DiagonalMsg::numMsg, &DiagonalMsg::lookupMsg );
278
279 SparseMsg::managerId_ = Id::nextId();
280 new MsgElement( SparseMsg::managerId_, SparseMsg::initCinfo(),
281 "sparseMsg", &SparseMsg::numMsg, &SparseMsg::lookupMsg );
282
283 OneToOneDataIndexMsg::managerId_ = Id::nextId();
284 new MsgElement( OneToOneDataIndexMsg::managerId_,
285 OneToOneDataIndexMsg::initCinfo(),
286 "oneToOneDataIndexMsg",
287 &OneToOneDataIndexMsg::numMsg,
288 &OneToOneDataIndexMsg::lookupMsg );
289
290 // Do the 'adopt' only after all the message managers exist - we need
291 // the OneToAll manager for the adoption messages themselves.
292 assert( OneToAllMsg::numMsg() == 0 );
293 unsigned int n = 1;
294 Shell::adopt( Id(), msgManagerId_, n++ );
295 Shell::adopt( msgManagerId_, SingleMsg::managerId_, n++ );
296 Shell::adopt( msgManagerId_, OneToOneMsg::managerId_, n++ );
297 Shell::adopt( msgManagerId_, OneToAllMsg::managerId_, n++ );
298 Shell::adopt( msgManagerId_, DiagonalMsg::managerId_, n++ );
299 Shell::adopt( msgManagerId_, SparseMsg::managerId_, n++ );
300
301 return n;
302}

304void Msg::clearAllMsgs()
305{
306 lastTrump_ = true;
307 for ( unsigned int i = 0; i < SingleMsg::numMsg(); ++i )
308 {
309 Msg* m = reinterpret_cast< Msg* >( SingleMsg::lookupMsg( i ) );
310 if ( m ) delete m;
311 }
312 for ( unsigned int i = 0; i < OneToOneMsg::numMsg(); ++i )
313 {
314 Msg* m = reinterpret_cast< Msg* >( OneToOneMsg::lookupMsg( i ) );
315 if ( m ) delete m;
316 }
317 for ( unsigned int i = 0; i < OneToAllMsg::numMsg(); ++i )
318 {
319 Msg* m = reinterpret_cast< Msg* >( OneToAllMsg::lookupMsg( i ) );
320 if ( m ) delete m;
321 }
322 for ( unsigned int i = 0; i < DiagonalMsg::numMsg(); ++i )
323 {
324 Msg* m = reinterpret_cast< Msg* >( DiagonalMsg::lookupMsg( i ) );
325 if ( m ) delete m;
326 }
327 for ( unsigned int i = 0; i < SparseMsg::numMsg(); ++i )
328 {
329 Msg* m = reinterpret_cast< Msg* >( SparseMsg::lookupMsg( i ) );
330 if ( m ) delete m;
331 }
332}
