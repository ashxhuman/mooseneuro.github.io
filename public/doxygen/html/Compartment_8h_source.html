14/**
15 * The Compartment class sets up an asymmetric compartment for
16 * branched nerve calculations. Handles electronic structure and
17 * also channels. This is not a particularly efficient way of doing
18 * this, so we should use a solver for any substantial calculations.
19 */
20namespace moose
21{
22class Compartment: public CompartmentBase
23{
24public:
25 Compartment();
26 virtual ~Compartment();
27
28 // Value Field access function definitions.
29 virtual void vSetVm( const Eref& e, double Vm );
30 virtual double vGetVm( const Eref& e ) const;
31 virtual void vSetEm( const Eref& e, double Em );
32 virtual double vGetEm( const Eref& e ) const;
33 virtual void vSetCm( const Eref& e, double Cm );
34 virtual double vGetCm( const Eref& e ) const;
35 virtual void vSetRm( const Eref& e, double Rm );
36 virtual double vGetRm( const Eref& e ) const;
37 virtual void vSetRa( const Eref& e, double Ra );
38 virtual double vGetRa( const Eref& e ) const;
39 virtual double vGetIm( const Eref& e ) const;
40 virtual void vSetInject( const Eref& e, double Inject );
41 virtual double vGetInject( const Eref& e ) const;
42 virtual void vSetInitVm( const Eref& e, double initVm );
43 virtual double vGetInitVm( const Eref& e ) const;
44
45 // Dest function definitions.
46 /**
47 * The process function does the object updating and sends out
48 * messages to channels, nernsts, and so on.
49 */
50 void vProcess( const Eref& e, ProcPtr p );
51
52 /**
53 * The reinit function reinitializes all fields.
54 */
55 void vReinit( const Eref& e, ProcPtr p );
56
57 /**
58 * The initProc function is for a second phase of 'process'
59 * operations. It sends the axial and raxial messages
60 * to other compartments. It has to be executed out of phase
61 * with the main process so that all compartments are
62 * equivalent and there is no calling order dependence in
63 * the results.
64 */
65 void vInitProc( const Eref& e, ProcPtr p );
66
67 /**
68 * Empty function to do another reinit step out of phase
69 * with the main one. Nothing needs doing there.
70 */
71 void vInitReinit( const Eref& e, ProcPtr p );
72
73 /**
74 * handleChannel handles information coming from the channel
75 * to the compartment
76 */
77 void vHandleChannel( const Eref& e, double Gk, double Ek);
78
79 /**
80 * handleRaxial handles incoming raxial message data.
81 */
82 void vHandleRaxial( double Ra, double Vm);
83
84 /**
85 * handleAxial handles incoming axial message data.
86 */
87 void vHandleAxial( double Vm);
88
89 /**
90 * Injects a constantly updated current into the compartment.
91 * Unlike the 'inject' field, this injected current is
92 * applicable only for a single timestep. So this is meant to
93 * be used as the destination of a message rather than as a
94 * one-time assignment.
95 */
96 void vInjectMsg( const Eref& e, double current);
97
98 /**
99 * Injects a constantly updated current into the
100 * compartment, with a probability prob. Note that it isn't
101 * the current amplitude that is random, it is the presence
102 * or absence of the current that is probabilistic.
103 */
104 void vRandInject( const Eref& e, double prob, double current);
105
106 /**
107 * Dummy function to act as recipient of 'cable' message,
108 * which is just for grouping compartments.
109 */
110 void cable();
111
112
113 /**
114 * Initializes the class info.
115 */
116 static const Cinfo* initCinfo();
117
118protected:
119 double Vm_;
120 double initVm_;
121 double Em_;
122 double Cm_;
123 double Rm_;
124 double Ra_;
125 double Im_;
126 double lastIm_;
127 double inject_;
128 double A_;
129 double B_;
130 double sumInject_;
131
132private:
133 double invRm_;
134 double dt_;
135 static const double EPSILON;
136
137};
138}
