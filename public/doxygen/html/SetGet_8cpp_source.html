15const OpFunc* SetGet::checkSet( const string& field, ObjId& tgt, FuncId& fid )
16{
17 // string field = "set_" + destField;
18 const Finfo* f = tgt.element()->cinfo()->findFinfo( field );
19 if ( !f ) // Could be a child element? Note that field name will
20 {
21 // change from set_<name> to just <name>
22 string f2 = field.substr( 3 );
23 Id child = Neutral::child( tgt.eref(), f2 );
24 if ( child == Id() )
25 {
26 cout << "Error: SetGet:checkSet:: No field or child named '" <<
27 field << "' was found on\n" << tgt.id.path() << endl;
28 }
29 else
30 {
31 if ( field.substr( 0, 3 ) == "set" )
32 f = child.element()->cinfo()->findFinfo( "setThis" );
33 else if ( field.substr( 0, 3 ) == "get" )
34 f = child.element()->cinfo()->findFinfo( "getThis" );
35 assert( f ); // should always work as Neutral has the field.
36 if ( child.element()->numData() == tgt.element()->numData() )
37 {
38 tgt = ObjId( child, tgt.dataIndex, tgt.fieldIndex );
39 if ( !tgt.isDataHere() )
40 return 0;
41 }
42 else if ( child.element()->numData() <= 1 )
43 {
44 tgt = ObjId( child, 0 );
45 if ( !tgt.isDataHere() )
46 return 0;
47 }
48 else
49 {
50 cout << "SetGet::checkSet: child index mismatch\n";
51 return 0;
52 }
53 }
54 }
55
56 const DestFinfo* df = dynamic_cast< const DestFinfo* >( f );
57 if ( !df )
58 return 0;
59
60 fid = df->getFid();
61 const OpFunc* func = df->getOpFunc();
62 assert( func );
63 return func;
64}
