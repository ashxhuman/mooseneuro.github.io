10/**
11 * This program works out a steady-state value for a reaction system.
12 * It uses GSL heavily, and isn't even compiled if the flag isn't set.
13 * It finds the ss value closest to the initial conditions.
14 *
15 * If you want to find multiple stable states, it is best to do this
16 * in Python as it gives a lot of flexibility in working out how to
17 * find steady states.
18 * Likewise, if you want to carry out a dose-response calculation.
19 */

562void SteadyState::setupSSmatrix()
563{
564#ifdef USE_GSL
565 if ( numVarPools_ == 0 || nReacs_ == 0 )
566 return;
567
568 int nTot = numVarPools_ + nReacs_;
569 gsl_matrix* N = gsl_matrix_calloc (numVarPools_, nReacs_);
570 if ( LU_ ) // Clear out old one.
571 {
572 gsl_matrix_free( LU_ );
573 }
574 LU_ = gsl_matrix_calloc (numVarPools_, nTot);
575 vector< int > entry = Field< vector< int > >::get(
576 stoich_, "matrixEntry" );
577 vector< unsigned int > colIndex = Field< vector< unsigned int > >::get(
578 stoich_, "columnIndex" );
579 vector< unsigned int > rowStart = Field< vector< unsigned int > >::get(
580 stoich_, "rowStart" );
581
582 // cout << endl << endl;
583 for ( unsigned int i = 0; i < numVarPools_; ++i )
584 {
585 gsl_matrix_set (LU_, i, i + nReacs_, 1 );
586 unsigned int k = rowStart[i];
587 // cout << endl << i << ": ";
588 for ( unsigned int j = 0; j < nReacs_; ++j )
589 {
590 double x = 0;
591 if ( j == colIndex[k] && k < rowStart[i+1] )
592 {
593 x = entry[k++];
594 }
595 // cout << " " << x;
596 gsl_matrix_set (N, i, j, x);
597 gsl_matrix_set (LU_, i, j, x );
598 }
599 }
600 cout << endl << endl;
601
602 rank_ = myGaussianDecomp( LU_ );
603
604 unsigned int nConsv = numVarPools_ - rank_;
605 if ( nConsv == 0 )
606 {
607 cout << "SteadyState::setupSSmatrix(): Number of conserved species == 0. Aborting\n";
608 return;
609 }
610
611 if ( Nr_ ) // Clear out old one.
612 {
613 gsl_matrix_free( Nr_ );
614 }
615 Nr_ = gsl_matrix_calloc ( rank_, nReacs_ );
616 // Fill up Nr.
617 for ( unsigned int i = 0; i < rank_; i++)
618 for ( unsigned int j = i; j < nReacs_; j++)
619 gsl_matrix_set (Nr_, i, j, gsl_matrix_get( LU_, i, j ) );
620
621 if ( gamma_ ) // Clear out old one.
622 {
623 gsl_matrix_free( gamma_ );
624 }
625 gamma_ = gsl_matrix_calloc (nConsv, numVarPools_ );
626
627 // Fill up gamma
628 for ( unsigned int i = rank_; i < numVarPools_; ++i )
629 for ( unsigned int j = 0; j < numVarPools_; ++j )
630 gsl_matrix_set( gamma_, i - rank_, j,
631 gsl_matrix_get( LU_, i, j + nReacs_ ) );
632
633 // Fill up boundary condition values
634 total_.resize( nConsv );
635 total_.assign( nConsv, 0.0 );
636
637 Id ksolve = Field< Id >::get( stoich_, "ksolve" );
638 vector< double > nVec =
639 LookupField< unsigned int, vector< double > >::get(
640 ksolve,"nVec", 0 );
641
642 if ( nVec.size() >= numVarPools_ )
643 {
644 for ( unsigned int i = 0; i < nConsv; ++i )
645 for ( unsigned int j = 0; j < numVarPools_; ++j )
646 total_[i] += gsl_matrix_get( gamma_, i, j ) * nVec[ j ];
647 isSetup_ = 1;
648 }
649 else
650 {
651 cout << "Error: SteadyState::setupSSmatrix(): unable to get"
652 "pool numbers from ksolve.\n";
653 isSetup_ = 0;
654 }
655
656 gsl_matrix_free( N );
657#endif
658}

660static double op( double x )
661{
662 return x * x;

665static double invop( double x )
666{
667 if ( x > 0.0 )
668 return sqrt( x );
669 return 0.0;

673/**
674 * This does the iteration, using the specified method.
675 * First try gsl_multiroot_fsolver_hybrids
676 * If that doesn't work try gsl_multiroot_fsolver_dnewton
677 * Returns the gsl status.
678 */
679#ifdef USE_GSL
680int iterate( const gsl_multiroot_fsolver_type* st, struct reac_info *ri,
681 int maxIter )
682{
683 int status = 0;
684 gsl_vector* x = gsl_vector_calloc( ri->num_mols );
685 gsl_multiroot_fsolver *solver =
686 gsl_multiroot_fsolver_alloc( st, ri->num_mols );
687 gsl_multiroot_function func = {&ss_func, ri->num_mols, ri};
688
689 // This gives the starting point for finding the solution
690 for ( unsigned int i = 0; i < ri->num_mols; ++i )
691 gsl_vector_set( x, i, invop( ri->nVec[i] ) );
692
693 gsl_multiroot_fsolver_set( solver, &func, x );
694
695 ri->nIter = 0;
696 do
697 {
698 ri->nIter++;
699 status = gsl_multiroot_fsolver_iterate( solver );
700 if (status ) break;
701 status = gsl_multiroot_test_residual(
702 solver->f, ri->convergenceCriterion);
703 }
704 while (status == GSL_CONTINUE && ri->nIter < maxIter );
705
706 gsl_multiroot_fsolver_free( solver );
707 gsl_vector_free( x );
708 return status;
709}
710#endif

712void SteadyState::classifyState( const double* T )
713{
714#ifdef USE_GSL
715 // unsigned int nConsv = numVarPools_ - rank_;
716 gsl_matrix* J = gsl_matrix_calloc ( numVarPools_, numVarPools_ );
717 // double* yprime = new double[ numVarPools_ ];
718 // vector< double > yprime( numVarPools_, 0.0 );
719 // Generate an approximation to the Jacobean by generating small
720 // increments to each of the molecules in the steady state, one
721 // at a time, and putting the resultant rate vector into a column
722 // of the J matrix.
723 // This needs a bit of heuristic to decide what is a 'small' increment.
724 // Use the CoInits for this. Stoichiometry shouldn't matter too much.
725 // I used the totals from consv rules earlier, but that can have
726 // negative values.
727 double tot = 0.0;
728 Stoich* s = reinterpret_cast< Stoich* >( stoich_.eref().data() );
729 vector< double > nVec = LookupField< unsigned int, vector< double > >::get(
730 s->getKsolve(), "nVec", 0 );
731 for ( unsigned int i = 0; i < numVarPools_; ++i )
732 {
733 tot += nVec[i];
734 }
735 tot *= DELTA;
736
737 vector< double > yprime( nVec.size(), 0.0 );
738 // Fill up Jacobian
739 for ( unsigned int i = 0; i < numVarPools_; ++i )
740 {
741 double orig = nVec[i];
742 if ( isNaN( orig ) )
743 {
744 cout << "Warning: SteadyState::classifyState: orig=nan\n";
745 solutionStatus_ = 2; // Steady state OK, eig failed
746 gsl_matrix_free ( J );
747 return;
748 }
749 if ( isNaN( tot ) )
750 {
751 cout << "Warning: SteadyState::classifyState: tot=nan\n";
752 solutionStatus_ = 2; // Steady state OK, eig failed
753 gsl_matrix_free ( J );
754 return;
755 }
756 nVec[i] = orig + tot;
757
758 pool_.updateRates( &nVec[0], &yprime[0] );
759 nVec[i] = orig;
760
761 // Assign the rates for each mol.
762 for ( unsigned int j = 0; j < numVarPools_; ++j )
763 {
764 gsl_matrix_set( J, i, j, yprime[j] );
765 }
766 }
767
768 // Jacobian is now ready. Find eigenvalues.
769 gsl_vector_complex* vec = gsl_vector_complex_alloc( numVarPools_ );
770 gsl_eigen_nonsymm_workspace* workspace =
771 gsl_eigen_nonsymm_alloc( numVarPools_ );
772 int status = gsl_eigen_nonsymm( J, vec, workspace );
773 eigenvalues_.clear();
774 eigenvalues_.resize( numVarPools_, 0.0 );
775 if ( status != GSL_SUCCESS )
776 {
777 cout << "Warning: SteadyState::classifyState failed to find eigenvalues. Status = " <<
778 status << endl;
779 solutionStatus_ = 2; // Steady state OK, eig classification failed
780 }
781 else // Eigenvalues are ready. Classify state.
782 {
783 nNegEigenvalues_ = 0;
784 nPosEigenvalues_ = 0;
785 for ( unsigned int i = 0; i < numVarPools_; ++i )
786 {
787 gsl_complex z = gsl_vector_complex_get( vec, i );
788 double r = GSL_REAL( z );
789 nNegEigenvalues_ += ( r < -EPSILON );
790 nPosEigenvalues_ += ( r > EPSILON );
791 eigenvalues_[i] = r;
792 // We have a problem here because numVarPools_ usually > rank
793 // This means we have several zero eigenvalues.
794 }
795
796 if ( nNegEigenvalues_ == rank_ )
797 stateType_ = 0; // Stable
798 else if ( nPosEigenvalues_ == rank_ ) // Never see it.
799 stateType_ = 1; // Unstable
800 else if (nPosEigenvalues_ == 1)
801 stateType_ = 2; // Saddle
802 else if ( nPosEigenvalues_ >= 2 )
803 stateType_ = 3; // putative oscillatory
804 else if ( nNegEigenvalues_ == ( rank_ - 1) && nPosEigenvalues_ == 0 )
805 stateType_ = 4; // one zero or unclassified eigenvalue. Messy.
806 else
807 stateType_ = 5; // Other
808 }
809
810 gsl_vector_complex_free( vec );
811 gsl_matrix_free ( J );
812 gsl_eigen_nonsymm_free( workspace );
813#endif
814}

831 * The settle function computes the steady state nearest the initial
832 * conditions.
833 */
834void SteadyState::settle( bool forceSetup )
835{
836#ifdef USE_GSL
837 gsl_set_error_handler_off();
838
839 if ( !isInitialized_ )
840 {
841 cout << "Error: SteadyState object has not been initialized. No calculations done\n";
842 return;
843 }
844 if ( forceSetup || isSetup_ == 0 )
845 {
846 setupSSmatrix();
847 }
848
849 // Setting up matrices and vectors for the calculation.
850 unsigned int nConsv = numVarPools_ - rank_;
851 double * T = (double *) calloc( nConsv, sizeof( double ) );
852
853 unsigned int i, j;
854
855
856 Id ksolve = Field< Id >::get( stoich_, "ksolve" );
857 struct reac_info ri;
858 ri.rank = rank_;
859 ri.num_reacs = nReacs_;
860 ri.num_mols = numVarPools_;
861 ri.T = T;
862 ri.Nr = Nr_;
863 ri.gamma = gamma_;
864 ri.pool = &pool_;
865 ri.nVec =
866 LookupField< unsigned int, vector< double > >::get(
867 ksolve,"nVec", 0 );
868 ri.convergenceCriterion = convergenceCriterion_;
869
870 // Fill up boundary condition values
871 if ( reassignTotal_ ) // The user has defined new conservation values.
872 {
873 for ( i = 0; i < nConsv; ++i )
874 T[i] = total_[i];
875 reassignTotal_ = 0;
876 }
877 else
878 {
879 for ( i = 0; i < nConsv; ++i )
880 for ( j = 0; j < numVarPools_; ++j )
881 T[i] += gsl_matrix_get( gamma_, i, j ) * ri.nVec[ j ];
882 total_.assign( T, T + nConsv );
883 }
884
885 vector< double > repair( numVarPools_, 0.0 );
886 for ( unsigned int j = 0; j < numVarPools_; ++j )
887 repair[j] = ri.nVec[j];
888
889 int status = iterate( gsl_multiroot_fsolver_hybrids, &ri, maxIter_ );
890 if ( status ) // It failed. Fall back with the Newton method
891 status = iterate( gsl_multiroot_fsolver_dnewton, &ri, maxIter_ );
892 status_ = string( gsl_strerror( status ) );
893 nIter_ = ri.nIter;
894 if ( status == GSL_SUCCESS && isSolutionPositive( ri.nVec ) )
895 {
896 solutionStatus_ = 0; // Good solution
897 LookupField< unsigned int, vector< double > >::set(
898 ksolve,"nVec", 0, ri.nVec );
899 classifyState( T );
900 }
901 else
902 {
903 cout << "Warning: SteadyState iteration failed, status = " <<
904 status_ << ", nIter = " << nIter_ << endl;
905 // Repair the mess
906 for ( unsigned int j = 0; j < numVarPools_; ++j )
907 ri.nVec[j] = repair[j];
908 solutionStatus_ = 1; // Steady state failed.
909 LookupField< unsigned int, vector< double > >::set(
910 ksolve,"nVec", 0, ri.nVec );
911 }
912
913 // Clean up.
914 free( T );
915#endif
916}

920int ss_func( const gsl_vector* x, void* params, gsl_vector* f )
921{
922 struct reac_info* ri = (struct reac_info *)params;
923 // Stoich* s = reinterpret_cast< Stoich* >( ri->stoich.eref().data() );
924 int num_consv = ri->num_mols - ri->rank;
925
926 for ( unsigned int i = 0; i < ri->num_mols; ++i )
927 {
928 double temp = op( gsl_vector_get( x, i ) );
929 if ( isNaN( temp ) || isInfinity( temp ) )
930 return GSL_ERANGE;
931 else
932 ri->nVec[i] = temp;
933 }
934 vector< double > vels;
935 ri->pool->updateReacVelocities( &ri->nVec[0], vels );
936 assert( vels.size() == static_cast< unsigned int >( ri->num_reacs ) );
937
938 // y = Nr . v
939 // Note that Nr is row-echelon: diagonal and above.
940 for ( int i = 0; i < ri->rank; ++i )
941 {
942 double temp = 0;
943 for ( int j = i; j < ri->num_reacs; ++j )
944 temp += gsl_matrix_get( ri->Nr, i, j ) * vels[j];
945 gsl_vector_set( f, i, temp );
946 }
947
948 // dT = gamma.S - T
949 for ( int i = 0; i < num_consv; ++i )
950 {
951 double dT = - ri->T[i];
952 for ( unsigned int j = 0; j < ri->num_mols; ++j )
953 dT += gsl_matrix_get( ri->gamma, i, j) *
954 op( gsl_vector_get( x, j ) );
955
956 gsl_vector_set( f, i + ri->rank, dT );
957 }
958
959 return GSL_SUCCESS;

962/**
963 * eliminateRowsBelow:
964 * Eliminates leftmost entry of all rows below 'start'.
965 * Returns the row index of the row which has the leftmost nonzero
966 * entry. If this sticks out beyond numReacs, then the elimination is
967 * complete, and it returns a zero to indicate this.
968 * In leftCol it returns the column # of this leftmost nonzero entry.
969 * Zeroes out anything below EPSILON.
970 */
971void eliminateRowsBelow( gsl_matrix* U, int start, int leftCol )
972{
973 int numMols = U->size1;
974 double pivot = gsl_matrix_get( U, start, leftCol );
975 assert( fabs( pivot ) > SteadyState::EPSILON );
976 for ( int i = start + 1; i < numMols; ++i )
977 {
978 double factor = gsl_matrix_get( U, i, leftCol );
979 if ( fabs ( factor ) > SteadyState::EPSILON )
980 {
981 factor = factor / pivot;
982 for ( size_t j = leftCol + 1; j < U->size2; ++j )
983 {
984 double x = gsl_matrix_get( U, i, j );
985 double y = gsl_matrix_get( U, start, j );
986 x -= y * factor;
987 if ( fabs( x ) < SteadyState::EPSILON )
988 x = 0.0;
989 gsl_matrix_set( U, i, j, x );
990 }
991 }
992 gsl_matrix_set( U, i, leftCol, 0.0 ); // Cleaning up.
993 }
994}
995
996/**
997 * reorderRows:
998 * Finds the leftmost row beginning from start, ignoring everything to the
999 * left of leftCol. Puts this row at 'start', swapping with the original.
1000 * Assumes that the matrix is set up as [N I].
1001 * Returns the new value of leftCol
1002 * If there are no appropriate rows, returns numReacs.
1003 */
1004int reorderRows( gsl_matrix* U, int start, int leftCol )
1005{
1006 int leftMostRow = start;
1007 int numReacs = U->size2 - U->size1;
1008 int newLeftCol = numReacs;
1009 for ( size_t i = start; i < U->size1; ++i )
1010 {
1011 for ( int j = leftCol; j < numReacs; ++j )
1012 {
1013 if ( fabs( gsl_matrix_get( U, i, j ) ) > SteadyState::EPSILON )
1014 {
1015 if ( j < newLeftCol )
1016 {
1017 newLeftCol = j;
1018 leftMostRow = i;
1019 }
1020 break;
1021 }
1022 }
1023 }
1024 if ( leftMostRow != start ) // swap them.
1025 {
1026 gsl_matrix_swap_rows( U, start, leftMostRow );
1027 }
1028 return newLeftCol;
1029}
1030
1031/**
1032 * Does a simple gaussian decomposition. Assumes U has nice clean numbers
1033 * so I can apply a generous EPSILON to zero things out.
1034 * Assumes that the U matrix is the N matrix padded out by an identity
1035 * matrix on the right.
1036 * Returns rank.
1037 */
1038int myGaussianDecomp( gsl_matrix* U )
1039{
1040 int numMols = U->size1;
1041 int numReacs = U->size2 - numMols;
1042 int i;
1043 // Start out with a nonzero entry at 0,0
1044 int leftCol = reorderRows( U, 0, 0 );
1045
1046 for ( i = 0; i < numMols - 1; ++i )
1047 {
1048 eliminateRowsBelow( U, i, leftCol );
1049 leftCol = reorderRows( U, i + 1, leftCol );
1050 if ( leftCol == numReacs )
1051 break;
1052 }
1053 return i + 1;
1054}
