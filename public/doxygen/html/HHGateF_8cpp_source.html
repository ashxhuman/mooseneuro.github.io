146HHGateF& HHGateF::operator=(const HHGateF& rhs)
147{
148 // protect from self-assignment.
149 if(this == &rhs)
150 return *this;
151
152 v_ = rhs.v_;
153 symTab_.add_variable("v", v_);
154 symTab_.add_variable("alpha", alphav_);
155 symTab_.add_variable("beta", betav_);
156 symTab_.add_variable("tau", tauv_);
157 symTab_.add_variable("inf", infv_);
158 symTab_.add_constants();
159 alpha_.register_symbol_table(symTab_);
160 beta_.register_symbol_table(symTab_);
161 alphaExpr_ = rhs.alphaExpr_;
162 betaExpr_ = rhs.betaExpr_;
163 parser_.compile(alphaExpr_, alpha_);
164 parser_.compile(betaExpr_, beta_);
165 tauInf_ = rhs.tauInf_;
166 return *this;
167}

173double HHGateF::lookupA(double v) const
174{
175 // TODO: check for divide by zero?
176 v_ = v;
177 return tauInf_ ? beta_.value() / alpha_.value() : alpha_.value();
178}
179
180double HHGateF::lookupB(double v) const
181{
182 // TODO: check for divide by zero?
183 v_ = v;
184 return tauInf_ ? 1.0 / alpha_.value() : alpha_.value() + beta_.value();
185}
