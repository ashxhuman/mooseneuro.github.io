12/**
13 * This defines a single segment in an SWC file used in NeuroMorpho
14 * Note that this is not going to work well for dendritic spines which
15 * come off in the middle of a parent compartment, and which have to be
16 * started from the surface rather than the axis of the dendrite.
17 */

121 // This is the original SWC set.
122 static const short UNDEF;
123 static const short SOMA;
124 static const short AXON;

133 protected:
134 unsigned int myIndex_; /// Index of self
135 /**
136 * The type of the segment is supposedly as below for SWC files.
137 * Seems to be honored in the breach in actual files.
138 * 0 = undefined
139 * 1 = soma
140 * 2 = axon
141 * 3 = dendrite
142 * 4 = apical dendrite
143 * 5 = fork point
144 * 6 = end point
145 * 7 = custom
146 */
147 short type_;
148 Vec v_; /// coordinates of end of segment
149 double radius_; /// Radius of segment
150 double length_; /// Length of segment
151 double L_; /// Number of length constants in segment.
152 unsigned int parent_; /// Index of parent. Is ~0 for soma.
153
154 /// dist from soma: not direct, but threaded along dend
155 double pathDistanceFromSoma_;
156
157 /// geometrical distance from soma.
158 double geometricalDistanceFromSoma_;
159
160 /// electrotonic dist from soma, summed along dend.
161 double electrotonicDistanceFromSoma_;
162
163 vector< int > kids_; // Indices of all children of segment.
164};
