34/**
35 * Check 2 floating-point numbers for "equality".
36 * Algorithm (from Knuth) 'a' and 'b' are close if:
37 * | ( a - b ) / a | < e AND | ( a - b ) / b | < e
38 * where 'e' is a small number.
39 *
40 * In this function, 'e' is computed as:
41 * e = tolerance * machine-epsilon
42 */
43 template< class T >
44bool isClose( T a, T b, T tolerance )
45{
46 T epsilon = std::numeric_limits< T >::epsilon();
47
48 if ( a == b )
49 return true;
50
51 if ( a == 0 || b == 0 )
52 return ( fabs( a - b ) < tolerance * epsilon );
53
54 return (
55 fabs( ( a - b ) / a ) < tolerance * epsilon
56 &&
57 fabs( ( a - b ) / b ) < tolerance * epsilon
58 );
59}
