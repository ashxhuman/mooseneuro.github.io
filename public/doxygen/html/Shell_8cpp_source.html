159/**
160 * This is the version used by the parser. Acts as a blocking,
161 * serial-like interface to a potentially multithread, multinode call.
162 * Returns the new Id index upon success, otherwise returns Id().
163 * The data of the new Element is not necessarily allocated at this point,
164 * that can be deferred till the global Instantiate or Reset calls.
165 * Idea is that the model should be fully defined before load balancing.
166 *
167 */
168Id Shell::doCreate(string type, ObjId parent, string name, unsigned int numData,
169 NodePolicy nodePolicy, unsigned int preferredNode)
170{
171
172 const Cinfo* c = Cinfo::find(type);
173 if (!isNameValid(name)) {
174 stringstream ss;
175 ss << "Shell::doCreate: bad character in the name '" << name
176 << "'. No Element created.";
177 warning(ss.str());
178 return Id();
179 }
180
181 if (c) {
182 if (c->banCreation()) {
183 stringstream ss;
184 ss << "Shell::doCreate: Cannot create an object of class '" << type
185 << "' because it is an abstract base class or a FieldElement.\n";
186 warning(ss.str());
187 return Id();
188 }
189
190 Element* pa = parent.element();
191 if (!pa) {
192 cerr << "Shell::doCreate: Parent Element'" << parent
193 << "' not found. No Element created." << endl;
194 return Id();
195 }
196
197 // TODO: This should be an error in future.
198 // This logic of handling already existing path is now handled in
199 // melements.cpp . Calling this section should become an error in
200 // future.
201 if (Neutral::child(parent.eref(), name) != Id()) {
202 string msg = "Object with path '" + parent.path() + "/" +
203 name + "' already exists. "
204 " Use `moose.element` to access the existing element.";
205 throw runtime_error(msg);
206 return Id();
207 }
208 // Get the new Id ahead of time and pass to all nodes.
209 Id ret = Id::nextId();
210 NodeBalance nb(numData, nodePolicy, preferredNode);
211 // Get the parent MsgIndex ahead of time and pass to all nodes.
212 unsigned int parentMsgIndex = OneToAllMsg::numMsg();
213
214 SetGet6<string, ObjId, Id, string, NodeBalance, unsigned int>::set(
215 ObjId(), // Apply command to Shell
216 "create", // Function to call.
217 type, // class of new object
218 parent, // Parent
219 ret, // id of new object
220 name, // name of new object
221 nb, // Node balance configuration
222 parentMsgIndex // Message index of child-parent msg.
223 );
224
225 // innerCreate( type, parent, ret, name, numData, isGlobal );
226
227 return ret;
228 } else {
229 stringstream ss;
230 ss << "Shell::doCreate: Class '" << type
231 << "' not known. No Element created";
232 warning(ss.str());
233 }
234
235 return Id();
236}

391bool extractIndex(const string& s, unsigned int& index)
392{
393 vector<unsigned int> open;
394 vector<unsigned int> close;
395
396 index = 0;
397 if (s.length() == 0) // a plain slash is OK
398 return true;
399
400 if (s[0] == '[') // Cannot open with a brace
401 return false;
402
403 for (unsigned int i = 0; i < s.length(); ++i) {
404 if (s[i] == '[')
405 open.push_back(i + 1);
406 else if (s[i] == ']')
407 close.push_back(i);
408 }
409
410 if (open.size() != close.size()) return false;
411 if (open.size() == 0) return true; // the index was set already to zero.
412 int j = atoi(s.c_str() + open[0]);
413 if (j >= 0) {
414 index = j;
415 return true;
416 }
417 return false;
418}

420/**
421 * Static func to subdivide a string at the specified separator.
422 */
423bool Shell::chopString(const string& path, vector<string>& ret, char separator)
424{
425 // /foo/bar/zod
426 // foo/bar/zod
427 // ./foo/bar/zod
428 // ../foo/bar/zod
429 // .
430 // /
431 // ..
432 ret.resize(0);
433 if (path.length() == 0) return 1; // Treat it as an absolute path
434
435 bool isAbsolute = 0;
436 string temp = path;
437 if (path[0] == separator) {
438 isAbsolute = 1;
439 if (path.length() == 1) return 1;
440 temp = temp.substr(1);
441 }
442
443 auto pos = temp.find_first_of(separator);
444 ret.push_back(temp.substr(0, pos));
445 while (pos != string::npos) {
446 temp = temp.substr(pos + 1);
447 if (temp.length() == 0) break;
448 pos = temp.find_first_of(separator);
449 ret.push_back(temp.substr(0, pos));
450 }
451 return isAbsolute;
452}

463/**
464 * static func.
465 *
466 * Example: /foo/bar[10]/zod[3] would return:
467 * ret: {"foo", "bar", "zod" }
468 * index: { 0, 10, 3 }
469 */
470bool Shell::chopPath(const string& path, vector<string>& ret,
471 vector<unsigned int>& index)
472{
473 bool isAbsolute = chopString(path, ret, '/');
474 if (isAbsolute) {
475 index.clear();
476 } else {
477 index.clear();
478 }
479 for (unsigned int i = 0; i < ret.size(); ++i) {
480 index.push_back(0);
481 if (ret[i] == ".") continue;
482 if (ret[i] == "..") {
483 continue;
484 }
485 if (!extractIndex(ret[i], index[i])) {
486 cout << "Error: Shell::chopPath: Failed to parse indices in path '"
487 << path << "'\n";
488 ret.resize(0);
489 index.resize(0);
490 return isAbsolute;
491 }
492 unsigned int pos = ret[i].find_first_of('[');
493 if (ret[i].find_first_of('[') != string::npos)
494 ret[i] = ret[i].substr(0, pos);
495 }
496
497 return isAbsolute;
498}

500ObjId Shell::doFind(const string& path) const
501{
502 if (path == "/" || path == "/root") return ObjId();
503
504 ObjId curr;
505 vector<string> names;
506 vector<unsigned int> indices;
507 bool isAbsolute = chopPath(path, names, indices);
508 assert(names.size() == indices.size());
509
510 if (!isAbsolute) curr = cwe_;
511
512 for (unsigned int i = 0; i < names.size(); ++i) {
513 if (names[i] == ".") {
514 } else if (names[i] == "..") {
515 curr = Neutral::parent(curr.eref());
516 } else {
517 ObjId pa = curr;
518 curr = Neutral::child(curr.eref(), names[i]);
519 if (curr == ObjId()) // Neutral::child returned Id(), ie, bad.
520 return ObjId(0, BADINDEX);
521 if (curr.element()->hasFields()) {
522 curr.dataIndex = pa.dataIndex;
523 curr.fieldIndex = indices[i];
524 } else {
525 curr.dataIndex = indices[i];
526 if (curr.element()->numData() <= curr.dataIndex)
527 return ObjId(0, BADINDEX);
528 }
529 }
530 }
531
532 assert(curr.element());
533 if (curr.element()->numData() <= curr.dataIndex) return ObjId(0, BADINDEX);
534 if (curr.fieldIndex > 0 && !curr.element()->hasFields())
535 return ObjId(0, BADINDEX);
536
537 return curr;
538}

688/**
689 * The actual function that adds messages. Does NOT send an ack.
690 * The msgIndex specifies the index on which to place this message. If the
691 * value is zero it does an automatic placement.
692 * Returns zero on failure.
693 */
694const Msg* Shell::innerAddMsg(string msgType, ObjId src, string srcField,
695 ObjId dest, string destField,
696 unsigned int msgIndex)
697{
698 /*
699 cout << myNode_ << ", Shell::handleAddMsg: " <<
700 msgType << ", " << mid <<
701 ", src =" << src << "." << srcField <<
702 ", dest =" << dest << "." << destField << "\n";
703 */
704 const Finfo* f1 = src.id.element()->cinfo()->findFinfo(srcField);
705 if (f1 == 0) return 0;
706 const Finfo* f2 = dest.id.element()->cinfo()->findFinfo(destField);
707 if (f2 == 0) return 0;
708
709 // Should have been done before msgs request went out.
710 assert(f1->checkTarget(f2));
711
712 Msg* m = 0;
713 if (msgType == "diagonal" || msgType == "Diagonal") {
714 m = new DiagonalMsg(src.id.element(), dest.id.element(), msgIndex);
715 } else if (msgType == "sparse" || msgType == "Sparse") {
716 m = new SparseMsg(src.id.element(), dest.id.element(), msgIndex);
717 } else if (msgType == "Single" || msgType == "single") {
718 m = new SingleMsg(src.eref(), dest.eref(), msgIndex);
719 } else if (msgType == "OneToAll" || msgType == "oneToAll") {
720 m = new OneToAllMsg(src.eref(), dest.id.element(), msgIndex);
721 } else if (msgType == "AllToOne" || msgType == "allToOne") {
722 m = new OneToAllMsg(dest.eref(), src.id.element(),

803// Static function
804void Shell::dropClockMsgs(const vector<ObjId>& list, const string& field)
805{
806 vector<ObjId> msgs; // These are the messages to delete.
807 for (vector<ObjId>::const_iterator i = list.begin(); i != list.end(); ++i) {
808 // Sanity check: shouldn't try to operate on already deleted objects
809 if (i->element()) {
810 const Finfo* f = i->element()->cinfo()->findFinfo(field);
811 const DestFinfo* df = dynamic_cast<const DestFinfo*>(f);
812 if (df) {
813 FuncId fid = df->getFid();
814
815 // These are the messages to be zapped
816 vector<ObjId> caller;
817 if (i->element()->getInputMsgs(caller, fid) > 0) {
818 msgs.insert(msgs.end(), caller.begin(), caller.end());
819 }
820 } else {
821 insertSharedMsgs(f, i->element(), msgs);
822 }
823 }
824 }
825 // Do the unique/erase bit. My favourite example of C++ hideousity.
826 sort(msgs.begin(), msgs.end());
827 msgs.erase(unique(msgs.begin(), msgs.end()), msgs.end());
828 // Delete them.
829 for (vector<ObjId>::iterator i = msgs.begin(); i != msgs.end(); ++i)
830 Msg::deleteMsg(*i);
831}
