276void Neutral::setName(const Eref& e, string name)
277{
278 if(e.id().value() <= 3) {
279 cout << "Warning: Neutral::setName on '" << e.id().path()
280 << "'. Cannot rename core objects\n";
281 return;
282 }
283 if(!Shell::isNameValid(name)) {
284 cout << "Warning: Neutral::setName on '" << e.id().path()
285 << "'. Illegal character in name.\n";
286 return;
287 }
288 ObjId pa = parent(e);
289 Id sibling = Neutral::child(pa.eref(), name);
290 if(sibling == Id()) // OK, no existing object with same name.
291 {
292 e.element()->setName(name);
293 } else {
294 cout << "Warning: Neutral::setName: an object with the name '" << name
295 << "'\n already exists on the same parent. Not changed\n";
296 }
297}

557unsigned int Neutral::buildTree(const Eref& e, vector<Id>& tree) const
558{
559 unsigned int ret = 1;
560 Eref er(e.element(), ALLDATA);
561 vector<Id> kids = getChildren(er);
562 sort(kids.begin(), kids.end());
563 kids.erase(unique(kids.begin(), kids.end()), kids.end());
564 for(vector<Id>::iterator i = kids.begin(); i != kids.end(); ++i)
565 ret += buildTree(i->eref(), tree);
566 tree.push_back(e.element()->id());
567 return ret;
568}

588//
589// Stage 0: Check if it is a Msg. This is deleted by Msg::deleteMsg( ObjId )
590// Stage 1: mark for deletion. This is done by setting cinfo = 0
591// Stage 2: Clear out outside-going msgs
592// Stage 3: delete self and attached msgs,
593void Neutral::destroy(const Eref& e, int stage)
594{
595 if(e.element()->cinfo()->isA("Msg")) {
596 Msg::deleteMsg(e.objId());
597 return;
598 }
599 vector<Id> tree;
600 Eref er(e.element(), ALLDATA);
601 unsigned int numDescendants = buildTree(er, tree);
602 /*
603 cout << "Neutral::destroy: id = " << e.id() <<
604 ", name = " << e.element()->getName() <<
605 ", numDescendants = " << numDescendants << endl;
606 */
607 assert(numDescendants == tree.size());
608 Element::destroyElementTree(tree);
609}

633// static function
634bool Neutral::isDescendant(Id me, Id ancestor)
635{
636 static const Finfo* pf = neutralCinfo->findFinfo("parentMsg");
637 static const DestFinfo* pf2 = dynamic_cast<const DestFinfo*>(pf);
638 static const FuncId pafid = pf2->getFid();
639
640 Eref e = me.eref();
641
642 while(e.element()->id() != Id() && e.element()->id() != ancestor) {
643 ObjId mid = e.element()->findCaller(pafid);
644 assert(mid != ObjId());
645 ObjId fid = Msg::getMsg(mid)->findOtherEnd(e.objId());
646 e = fid.eref();
647 }
648 return (e.element()->id() == ancestor);
649}

651// static function
652Id Neutral::child(const Eref& e, const string& name)
653{
654 static const Finfo* pf = neutralCinfo->findFinfo("parentMsg");
655 static const DestFinfo* pf2 = dynamic_cast<const DestFinfo*>(pf);
656 static const FuncId pafid = pf2->getFid();
657 static const Finfo* cf = neutralCinfo->findFinfo("childOut");
658 static const SrcFinfo* cf2 = dynamic_cast<const SrcFinfo*>(cf);
659 static const BindIndex bi = cf2->getBindIndex();
660
661 const vector<MsgFuncBinding>* bvec = e.element()->getMsgAndFunc(bi);
662
663 vector<Id> ret;
664
665 for(vector<MsgFuncBinding>::const_iterator i = bvec->begin();
666 i != bvec->end(); ++i) {
667 if(i->fid == pafid) {
668 const Msg* m = Msg::getMsg(i->mid);
669 assert(m);
670 Element* e2 = m->e2();
671 if(e2->getName() == name) {
672 if(e.dataIndex() == ALLDATA) // Child of any index is OK
673 {
674 return e2->id();
675 } else {
676 ObjId parent = m->findOtherEnd(m->getE2());
677 // If child is a fieldElement, then all parent indices
678 // are permitted. Otherwise insist parent dataIndex OK.
679 if(e2->hasFields() || parent == e.objId())
680 return e2->id();
681 }
682 }
683 }
684 }
685 return Id();
686}
