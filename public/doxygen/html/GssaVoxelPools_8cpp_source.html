24/**
25 * The SAFETY_FACTOR Protects against the total propensity exceeding
26 * the cumulative
27 * sum of propensities, atot. We do a lot of adding and subtracting of
28 * dependency terms from atot. Roundoff error will eventually cause
29 * this to drift from the true sum. To guarantee that we never lose
30 * the propensity of the last reaction, this safety factor scales the
31 * first calculation of atot to be slightly larger. Periodically this
32 * will cause the reaction picking step to exceed the last reaction
33 * index. This is safe, we just pick another random number.
34 * This will happen rather infrequently.
35 * That is also a good time to update the cumulative sum.
36 * A double should have >15 digits, so cumulative error will be much
37 * smaller than this.
38 */
39const double SAFETY_FACTOR = 1.0 + 1.0e-9;

56void GssaVoxelPools::updateDependentMathExpn(
57 const GssaSystem* g, unsigned int rindex, double time )
58{
59 // The issue is that if the expression depends on t, we really need
60 // to update it every timestep. But then a cascading set of reacs
61 // should also be updated.
62 // The lower commented block has all funcs updated every time step,
63 // but this too
64 // doesn't update the cascading reacs. So this is now handled by the
65 // useClockedUpdate flag, and we use the lower block here instead.
66 /*
67 const vector< unsigned int >& deps = g->dependentMathExpn[ rindex ];
68 for( vector< unsigned int >::const_iterator
69 i = deps.begin(); i != deps.end(); ++i ) {
70 g->stoich->funcs( *i )->evalPool( varS(), time );
71 }
72 */
73 /*
74 unsigned int numFuncs = g->stoich->getNumFuncs();
75 for( unsigned int i = 0; i < numFuncs; ++i )
76 {
77 g->stoich->funcs( i )->evalPool( varS(), time );
78 }
79 */
80 // This function is equivalent to the loop above.
81 g->stoich->updateFuncs( varS(), time );
82}

84void GssaVoxelPools::updateDependentRates(
85 const vector< unsigned int >& deps, const Stoich* stoich )
86{
87 for ( auto i = deps.cbegin(); i != deps.end(); ++i )
88 {
89 atot_ -= fabs( v_[ *i ] );
90 atot_ += fabs( v_[ *i ] = getReacVelocity( *i, S() ) );
91 }

95unsigned int GssaVoxelPools::pickReac()
96{
97 double r = rng_.uniform( ) * atot_;
98 double sum = 0.0;
99
100 // This is an inefficient way to do it. Can easily get to
101 // log time or thereabouts by doing one or two levels of
102 // subsidiary tables. Too many levels causes slow-down because
103 // of overhead in managing the tree.
104 // Slepoy, Thompson and Plimpton 2008
105 // report a linear time version.
106 for ( auto i = v_.cbegin(); i != v_.end(); ++i )
107 {
108 if ( r < ( sum += fabs( *i ) ) )
109 {
110 // double vel = fabs( getReacVelocity( i - v_.begin(), S() ) );
111 // assert( vel >= 0 );
112 return static_cast< unsigned int >( i - v_.begin() );
113 }
114 }
115 return v_.size();
116}

125/**
126 * Cleans out all reac rates and recalculates atot. Needed whenever a
127 * mol conc changes, or if there is a roundoff error.
128 * Returns true if OK, returns false if it is in a stuck state and atot<=0
129 */
130bool GssaVoxelPools::refreshAtot( const GssaSystem* g )
131{
132 g->stoich->updateFuncs( varS(), t_ );
133 updateReacVelocities( g, S(), v_ );
134 atot_ = 0;
135 for ( auto i = v_.cbegin(); i != v_.cend(); ++i )
136 atot_ += fabs(*i);
137
138 atot_ *= SAFETY_FACTOR;
139
140 // Check if the system is in a stuck state. If so, terminate.
141 if ( atot_ <= 0.0 )
142 return false;
143 return true;
144}

146/**
147 * Recalculates the time for the next event. Used when we have a clocked
148 * update of rates due to timed functions. In such cases the propensities
149 * may change invisibly, so we need to update time estimates
150 */
151void GssaVoxelPools::recalcTime( const GssaSystem* g, double currTime )
152{
153 refreshAtot( g );
154 assert( t_ > currTime );
155 t_ = currTime;
156 double r = rng_.uniform( );
157 while( r == 0.0 )
158 r = rng_.uniform( );
159 t_ -= ( 1.0 / atot_ ) * log( r );
160}

162void GssaVoxelPools::advance( const ProcInfo* p, const GssaSystem* g )
163{
164 double nextt = p->currTime;
165 while ( t_ < nextt )
166 {
167 if ( atot_ <= 0.0 ) // reac system is stuck, will not advance.
168 {
169 t_ = nextt;
170 g->stoich->updateFuncs( varS(), t_ );
171 return;
172 }
173 unsigned int rindex = pickReac();
174 assert( g->stoich->getNumRates() == v_.size() );
175 if ( rindex >= g->stoich->getNumRates() )
176 {
177 // probably cumulative roundoff error here.
178 // Recalculate atot to avoid, and redo.
179 if ( !refreshAtot( g ) ) // Stuck state.
180 {
181 t_ = nextt;
182 g->stoich->updateFuncs( varS(), t_ );
183 return;
184 }
185 // We had a roundoff error, fixed it, but now need to be sure
186 // we only fire a reaction where this is permissible.
187 for ( unsigned int i = v_.size(); i > 0; --i )
188 {
189 if ( fabs( v_[i-1] ) > 0.0 )
190 {
191 rindex = i - 1;
192 break;
193 }
194 }
195 assert( rindex < v_.size() );
196 }
197
198 double sign = std::copysign( 1, v_[rindex] );
199
200 g->transposeN.fireReac( rindex, Svec(), sign );
201 numFire_[rindex]++;
202
203 double r = rng_.uniform();
204 while ( r <= 0.0 )
205 r = rng_.uniform();
206
207 t_ -= ( 1.0 / atot_ ) * log( r );
208 g->stoich->updateFuncs( varS(), t_ );
209 updateDependentRates( g->dependency[ rindex ], g->stoich );
210 }
211}

213void GssaVoxelPools::reinit( const GssaSystem* g, int rngSeedOffset )
214{
215 rng_.setSeed( moose::getGlobalSeed() + rngSeedOffset );
216 VoxelPoolsBase::reinit(); // Assigns S = NA * vol * Cinit;
217 unsigned int numVarPools = g->stoich->getNumVarPools();
218 g->stoich->updateFuncs( varS(), 0 );
219
220 double* n = varS();
221
222 double error = 0.0;
223 double _prev = 0.0;
224 if( g->useRandInit )
225 {
226 map<double, vector<Eref>> groupByVal;
227 for ( unsigned int i = 0; i < numVarPools; ++i )
228 {
229 _prev = n[i];
230 n[i] = approximateWithInteger(_prev, rng_);
231 error += fabs(_prev - n[i]);
232 }
233
234 // Show warning to user if extra molecules in the system after
235 // converting floats to integer is larger than 1.0.
236 // Jan 2024: Disable until I come up with a good way to handle.
237 // Note also that this is likely when we have more pools.
238 if( 0 && std::abs(error) >= 1.0 )
239 {
240 MOOSE_WARN( "Extra " << error
241 << " molecules in system after converting fractional to integer."
242 );
243 }
244 }
245 else // Just round to the nearest int.
246 {
247 for ( unsigned int i = 0; i < numVarPools; ++i )
248 n[i] = std::round(n[i]);
249 }
250
251 t_ = 0.0;
252 refreshAtot( g );
253 numFire_.assign( v_.size(), 0 );
254}

265void GssaVoxelPools::updateAllRateTerms( const vector< RateTerm* >& rates,
266 unsigned int numCoreRates )
267{
268 for ( unsigned int i = 0; i < rates_.size(); ++i )
269 delete( rates_[i] );
270
271 rates_.resize( rates.size() );
272
273 for ( unsigned int i = 0; i < numCoreRates; ++i )
274 rates_[i] = rates[i]->copyWithVolScaling( getVolume(), 1, 1 );
275
276 for ( unsigned int i = numCoreRates; i < rates.size(); ++i )
277 rates_[i] = rates[i]->copyWithVolScaling( getVolume(),
278 getXreacScaleSubstrates(i - numCoreRates),
279 getXreacScaleProducts(i - numCoreRates ) );
280}

282void GssaVoxelPools::updateRateTerms( const vector< RateTerm* >& rates,
283 unsigned int numCoreRates, unsigned int index )
284{
285 // During setup or expansion of the reac system, this might be called
286 // before the local rates_ term is assigned. If so, ignore.
287 if ( index >= rates_.size() )
288 return;
289 delete( rates_[index] );
290 if ( index >= numCoreRates )
291 rates_[index] = rates[index]->copyWithVolScaling(
292 getVolume(),
293 getXreacScaleSubstrates(index - numCoreRates),
294 getXreacScaleProducts(index - numCoreRates)
295 );
296 else
297 rates_[index] = rates[index]->copyWithVolScaling(getVolume(), 1.0, 1.0);
298}
299/**
300 * updateReacVelocities computes the velocity *v* of each reaction.
301 * This is a utility function for programs like SteadyState that need
302 * to analyze velocity.
303 */
304void GssaVoxelPools::updateReacVelocities( const GssaSystem* g,
305 const double* s, vector< double >& v ) const
306{
307 const KinSparseMatrix& N = g->stoich->getStoichiometryMatrix();
308 assert( N.nColumns() == rates_.size() );
309
310 vector< RateTerm* >::const_iterator i;
311 v.clear();
312 v.resize( rates_.size(), 0.0 );
313 vector< double >::iterator j = v.begin();
314
315 for ( i = rates_.begin(); i != rates_.end(); i++)
316 {
317 *j++ = (**i)( s );
318 assert( !std::isnan( *( j-1 ) ) );
319 }
320}

342void GssaVoxelPools::xferIn( XferInfo& xf, unsigned int voxelIndex, const GssaSystem* g )
343{
344 unsigned int offset = voxelIndex * xf.xferPoolIdx.size();
345 auto i = xf.values.cbegin() + offset;
346 auto j = xf.lastValues.cbegin() + offset;
347 auto m = xf.subzero.begin() + offset;
348 double* s = varS();
349
350 for ( auto k = xf.xferPoolIdx.cbegin(); k != xf.xferPoolIdx.end(); ++k )
351 {
352 double& x = s[*k];
353 double dx = *i++ - *j++;
354 // x += approximateWithInteger_debug(__FUNCTION__, dx, rng_);
355 x += approximateWithInteger(dx, rng_);
356
357 if ( x < *m )
358 {
359 *m -= x;
360 x = 0;
361 }
362 else
363 {
364 x -= *m;
365 *m = 0;
366 }
367 m++;
368 }
369 // If S has changed, then I should update rates of all affected reacs.
370 // Go through stoich matrix to find affected reacs for each mol
371 // Store in list, since some may be hit more than once in this func.
372 // When done, go through and update all affected ones.
373
374 // Does this fix the problem of negative concs?
375 refreshAtot( g );
376}
