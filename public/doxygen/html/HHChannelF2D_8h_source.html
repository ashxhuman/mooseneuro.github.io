69 /**
70 * processFunc handles the update and calculations every
71 * clock tick. It first sends the request for evaluation of
72 * the gate variables to the respective gate objects and
73 * recieves their response immediately through a return
74 * message. This is done so that many channel instances can
75 * share the same gate lookup tables, but do so cleanly.
76 * Such messages should never go to a remote node.
77 * Then the function does its own little calculations to
78 * send back to the parent compartment through regular
79 * messages.
80 */
81 void vProcess(const Eref& e, ProcPtr p) override;
82
83 /**
84 * Reinitializes the values for the channel. This involves
85 * computing the steady-state value for the channel gates
86 * using the provided Vm from the parent compartment. It
87 * involves a similar cycle through the gates and then
88 * updates to the parent compartment as for the processFunc.
89 */
90 void vReinit(const Eref& e, ProcPtr p) override;
91 /**
92 * Assign the local conc_ to the incoming conc from the
93 * concentration calculations for the compartment. Typically
94 * the message source will be a CaConc object, but there
95 * are other options for computing the conc.
96 */
97 void conc1(double conc);
98 void conc2(double conc);
99
100 /**
101 * Function for safely creating each gate, identified by strings
102 * as X, Y and Z. Will only work on a new channel, not on a
103 * copy. The idea is that the gates are always referred to the
104 * original 'library' channel, and their contents cannot be touched
105 * except by the original.
106 */
107 void vCreateGate(const Eref& e, string gateType) override;
