22/**
23 * This is the base class for voxels used in reac-diffusion systems.
24 * Each voxel manages all the molecular pools that live in it. This
25 * is the S_ and Cinit_ vectors.
26 * Additionally, the last set of entries in S_ and Cinit_ refer to the
27 * proxy pools that participate in cross-node reactions, but are really
28 * located on other compartments.
29 */
30
31class VoxelPoolsBase
32{
33public:
34 VoxelPoolsBase();
35 virtual ~VoxelPoolsBase();
36
37 //////////////////////////////////////////////////////////////////
38 // Compute access operations.
39 //////////////////////////////////////////////////////////////////
40 /// Allocate # of pools
41 void resizeArrays( unsigned int totNumPools );

43 /**
44 * Returns # of pools

46 unsigned int size() const;
47
48 /**
49 * Returns the array of doubles of current mol #s at the specified
50 * mesh index
51 */
52 const double* S() const;
53
54 /// Returns a handle to the mol # vector.
55 vector< double >& Svec();
56
57 /**
58 * Returns the array of doubles of current mol #s at the specified
59 * mesh index. Dangerous, allows one to modify the values.
60 */
61 double* varS();

63 /**
64 * Returns the array of doubles of initial mol #s at the specified
65 * mesh index
66 */
67 const double* Cinit() const;
68
69 /**
70 * Returns the array of doubles of initial mol #s at the specified
71 * mesh index, as a writable array.
72 */
73 double* varCinit();
74
75 /**
76 * Assigns S = NA * vol * Cinit and sets up initial dt.
77 */
78 void reinit();
79
80 /// Just assigns the volume without any cascading to other values.
81 void setVolume( double vol );
82 /// Return the volume of the voxel.
83 double getVolume() const;

89 virtual void setVolumeAndDependencies( double vol );

94
95 void setN( unsigned int i, double v );
96 double getN( unsigned int ) const;

103 // Functions to set up rates vector
104 //////////////////////////////////////////////////////////////////
105 /**
106 * Reassign entire rate vector.
107 */
108 virtual void updateAllRateTerms( const vector< RateTerm* >& rates,
109 unsigned int numCoreRates ) = 0;

111 /**
112 * Update specified index on rate terms. The entire rates vector is
113 * passed in for the source values, the index specifies which

170 bool isVoxelJunctionPresent( Id i1, Id i2 ) const;

172 // Functions for cross-compartment rate scaling.

178 void resetXreacScale( unsigned int size );

207
208 void setNumVoxels( unsigned int );
209
210 /// Debugging utility
211 void print() const;
212
213protected:
214 const Stoich* stoichPtr_;
215 vector< RateTerm* > rates_;
216 /**
217 * Number of voxels. If > 1, set flag for LSODA to handle molecule
218 * flux on each timestep during diffusion, which slows it down.
219 */
220 unsigned int numVoxels_;
221
222private:
223 /**
224 * S_ is the array of molecules. Stored as n, number of molecules
225 * per mesh entry.
226 * The poolIndex specifies which molecular species pool to use.
227 *
228 * The first numVarPools_ in the poolIndex are state variables and
229 * are integrated using the ODE solver or the GSSA solver.
230 * The next numBufPools_ are fixed but can be changed by the script.
231 * The next numFuncPools_ are computed using arbitrary functions of
232 * any of the molecule levels, and the time.
233 * The functions evaluate _before_ the solver.
234 * The functions should not cascade as there is no guarantee of
235 * execution order.
236 */
237 vector< double > S_;
238
239 /**
240 * Cinit_ specifies initial conc at t = 0. Whenever the reac
241 * system is rebuilt or reinited, all S_ values become set to Cinit.
242 * scaled for volume and NA.
243 * Also used for buffered molecules as the fixed values of these
244 * molecules.
245 */
246 vector< double > Cinit_;
247
248 /**
249 * proxyPoolVoxels_[comptIndex][#]
250 * Used to build the transfer vector.
251 * Stores the voxels of proxy pools in specified compartment,
252 */
253 vector< vector< unsigned int > > proxyPoolVoxels_;
254 /**
255 * proxyTransferIndex_[comptIndex][#]
256 * Stores the index in the transfer vector for proxy pools that
257 * live on this voxel. The poolIndex and # of pools is known from
258 * the xferPoolIdx_ vector stored on the Ksolve.
259 */
260 vector< vector< unsigned int > > proxyTransferIndex_;
261
262 /**
263 * Looks up comptIndex from compt. Actually just a set would do.
264 */
265 map< Id, unsigned int > proxyComptMap_;
266
267 /** This is what I would need to fully identify proxy compts and
268 * voxels for each proxy reac.
269 vector< Id, Id, unsigned int, unsigned int> proxyComptVoxel
270 */
271
272 /**
273 * Volume of voxel.
274 */
275 double volume_;
276
277 /**
278 * xReacScaleSubstrates_[crossRateTermIndex]
279 * Product of substrateVol/voxelVol for each of the substrates,
280 * for forward reacs.
281 * Applied to R1 of the RateTerm. Used only for cross reactions.
282 */
283 vector< double > xReacScaleSubstrates_;
284
285 /**
286 * xReacScaleProducts_[crossRateTermIndex]
287 * Product of productVol/voxelVol for each of the products,
288 * for reverse reacs.
289 * Applied to R2 of the RateTerm. Used only for cross reactions.
290 */
291 vector< double > xReacScaleProducts_;
292};
