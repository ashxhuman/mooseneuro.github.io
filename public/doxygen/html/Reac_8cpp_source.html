293void Reac::setSolver( const Eref& e, ObjId newStoich )
294{
295 if ( newStoich.bad() ) {
296 cout << "Warning: Reac::setSolver: Bad Stoich " <<
297 e.id().path() << endl;
298 return;
299 }
300 if ( newStoich == ObjId() ) { // Unsetting stoich.
301 if ( stoich_ != 0 )
302 stoich_->notifyRemoveReac( e );
303 stoich_ = 0;
304 return;
305 }
306 if ( !newStoich.element()->cinfo()->isA( "Stoich" ) ) {
307 cout << "Warning: Reac::setSolver: object " << newStoich.path() << "is not a Stoich for " <<
308 e.id().path() << endl;
309 return;
310 }
311
312 assert( newStoich.element()->cinfo()->isA( "Stoich" ) );
313 Stoich* stoichPtr = reinterpret_cast< Stoich* >( newStoich.eref().data( ) );
314 if ( stoich_ == stoichPtr )
315 return;
316
317 if ( stoich_ != 0 )
318 stoich_->notifyRemoveReac( e );
319
320 stoich_ = stoichPtr;
321 /// Some of this should go into notifyAddReac
322 vector< Id > sub;
323 vector< Id > prd;
324 e.element()->getNeighbors( sub, subOut() );
325 e.element()->getNeighbors( prd, prdOut() );
326
327 stoich_->installReaction( e.id(), sub, prd );
328 stoich_->setReacKf( e, concKf_ );
329 stoich_->setReacKb( e, concKb_ );
330}
