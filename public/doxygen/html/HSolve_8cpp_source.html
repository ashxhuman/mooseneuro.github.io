236{
237 vector< Id >::const_iterator i;

240 for ( i = compartmentId_.begin(); i != compartmentId_.end(); ++i )

243 CompartmentBase::zombify( i->eref().element(),
244 ZombieCompartment::initCinfo(), hsolve.id() );
245 }
246

248 for ( i = caConcId_.begin(); i != caConcId_.end(); ++i )
249 temp.push_back( ObjId( *i, 0 ) );
250 // Shell::dropClockMsgs( temp, "process" );

252 CaConcBase::zombify( i->eref().element(), ZombieCaConc::initCinfo(), hsolve.id() );
253 }
254
255 temp.clear();
256 for ( i = channelId_.begin(); i != channelId_.end(); ++i )
257 temp.push_back( ObjId( *i, 0 ) );

341/**
342 * This function performs a depth-first search (for a compartment) in the tree
343 * with its root at 'base'. Returns (Id of) a compartment if found, else a
344 * blank Id.
345 */
346Id HSolve::deepSearchForCompartment( Id base )
347{
348 /*
349 * 'cstack' is a stack-of-stacks used to perform the depth-first search.
350 * The 0th entry in 'cstack' is a stack containing simply the base.
351 * The i-th entry in 'cstack' contains children of the node at the top
352 * of the stack at position ( i - 1 ).
353 * Hence, at any time, the top of the i-th stack is the i-th node on
354 * the ancestral path from the 'base' node to the 'current' node
355 * (more below) which is being examined. Also, the remaining nodes in
356 * the i-th stack are the siblings of this ancestor.
357 *
358 * 'current' is the node at the top of the top of 'cstack'. If this node is
359 * a Compartment, then the search is completed, returning 'current'.
360 * Otherwise, the children of 'current' are pushed onto 'cstack' for a
361 * deeper search. If the deeper search yields nothing, then this
362 * 'current' node is discarded. When an entire stack of siblings is
363 * exhausted in this way, then this empty stack is discarded, and
364 * the search moves 1 level up.
365 *
366 * 'result' is a blank Id (moose root element) if the search failed.
367 * Otherwise, it is a compartment that was found under 'base'.
368 */
369 vector< vector< Id > > cstack( 1, vector< Id >( 1, base ) );
370 Id current;
371 Id result;
372
373 while ( !cstack.empty() )
374 if ( cstack.back().empty() )
375 {
376 cstack.pop_back();
377
378 if ( !cstack.empty() )
379 cstack.back().pop_back();
380 }
381 else
382 {
383 current = cstack.back().back();
384
385 // if ( current()->cinfo() == moose::Compartment::initCinfo() )
386 // Compartment is base class for SymCompartment.
387 if ( current.element()->cinfo()->isA( "Compartment" ) )
388 {
389 result = current;
390 break;
391 }
392 cstack.push_back( children( current ) );
393 }
394
395 return result;
396}

517/**
518 Utility function to delete incoming messages on orig.
519 To be used in zombifying elements.
520*/
521void HSolve::deleteIncomingMessages( Element * orig, const string finfo)
522{
523 const DestFinfo * concenDest = dynamic_cast<const DestFinfo*>(orig->cinfo()->findFinfo(finfo));
524 assert(concenDest);
525 ObjId mid = orig->findCaller(concenDest->getFid());
526 while (! mid.bad())
527 {
528 const Msg * msg = Msg::getMsg(mid);
529 assert(msg);
530 ObjId other = msg->findOtherEnd(orig->id());
531 Element * otherEl = other.id.element();
532 if (otherEl && HSolve::handledClasses().find(otherEl->cinfo()->name()) != HSolve::handledClasses().end())
533 {
534 Msg::deleteMsg(mid);
535 }
536 else
537 {
538 break; // Have to do this otherwise it is an infinite loop
539 }
540 mid = orig->findCaller(concenDest->getFid());
541 }
542}
