206vector< double > SpineEntry::psdCoords() const
207{
208 vector< double > ret( 8, 0.0 );
209 double m0, m1, m2;
210 mid( m0, m1, m2 );

212 ret[1] = m1;
213 ret[2] = m2;
214 ret[3] = head_.getX() - m0;
215 ret[4] = head_.getY() - m1;
216 ret[5] = head_.getZ() - m2;
217 /*
218 ret[0] = head_.getX();
219 ret[1] = head_.getY();
220 ret[2] = head_.getZ();
221 double m0, m1, m2;
222 mid( m0, m1, m2 );

224 ret[4] = ret[1] - m1;
225 ret[5] = ret[2] - m2;

227 ret[6] = head_.getDia();
228 ret[7] = sqrt( ret[3] * ret[3] + ret[4] * ret[4] + ret[5] * ret[5] );

234void SpineEntry::setVolume( double volume )
235{
236 double volscale = volume / head_.volume( shaft_ );
237 double linscale = pow( volscale, 1.0 / 3.0 );
238 head_.setLength( head_.getLength() * linscale );
239 head_.setDia( head_.getDia() * linscale );
240 double dx = head_.getX() - shaft_.getX();
241 double dy = head_.getY() - shaft_.getY();
242 double dz = head_.getZ() - shaft_.getZ();
243 head_.setX( dx * linscale + shaft_.getX() );
244 head_.setY( dy * linscale + shaft_.getY() );
245 head_.setZ( dz * linscale + shaft_.getZ() );
246}

249{

258 shaft_.setZ( dz + shaft_.getZ() );
259 head_.setX( dx + head_.getX() );
260 head_.setY( dy + head_.getY() );
261 head_.setZ( dz + head_.getZ() );
