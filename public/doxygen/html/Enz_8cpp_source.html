213// v is in number units.
214void Enz::setK1( const Eref& e, double v )
215{
216 double volScale =
217 convertConcToNumRateUsingMesh( e, subOut, true );
218
219 double concK1 = v * volScale;
220 if ( concK1 < EPSILON )
221 concK1 = EPSILON;
222 Km_ = (k2_ + kcat_)/concK1;
223 if (stoich_)
224 stoich_->setEnzK1( e, concK1 );
225}
226
227// k1 is In number units.
228double Enz::getK1( const Eref& e ) const
229{
230 double volScale =
231 convertConcToNumRateUsingMesh( e, subOut, true );
232
233 return (k2_ + kcat_) / (volScale * Km_);
234}

250void Enz::vSetKcat( const Eref& e, double v )
251{
252 // Assumes kcat_ has been set by EnzBase::setKcat
253 // Tries to preserve 'ratio'.
254 double ratio = 4.0;
255 if ( kcat_ > EPSILON ) {
256 ratio = k2_/kcat_;
257 }
258 kcat_ = v;
259 k2_ = v * ratio;
260 double concK1 = v * (1.0 + ratio) / Km_;
261
262 if (stoich_ ) {
263 stoich_->setEnzK1( e, concK1 );
264 stoich_->setEnzK3( e, kcat_ );
265 stoich_->setEnzK2( e, k2_ );
266 }
267}

269void Enz::vSetKm( const Eref& e, double v )
270{
271 Km_ = v;
272 double concK1 = ( k2_ + kcat_ ) / v;
273 if ( stoich_ )
274 stoich_->setEnzK1( e, concK1 );
275}

277void Enz::setRatio( const Eref& e, double v )
278{
279 k2_ = v * kcat_;
280
281 if ( stoich_ ) {
282 stoich_->setEnzK2( e, k2_ );
283 double k1 = ( k2_ + kcat_ ) / Km_;
284 setConcK1( e, k1 );
285 }
286}

317// static func
318void Enz::setSolver( const Eref& e, ObjId stoich )
319{
320 static const Finfo* subFinfo = Cinfo::find("Enz")->findFinfo( "subOut");
321 static const Finfo* prdFinfo = Cinfo::find("Enz")->findFinfo( "prdOut");
322 static const Finfo* enzFinfo = Cinfo::find("Enz")->findFinfo( "enzOut");
323 static const Finfo* cplxFinfo= Cinfo::find("Enz")->findFinfo("cplxOut");
324
325 assert( subFinfo );
326 assert( prdFinfo );
327 assert( enzFinfo );
328 assert( cplxFinfo );
329 vector< Id > enzMols;
330 vector< Id > cplxMols;
331
332 if ( stoich == ObjId() ) { // Clear solver
333 if ( stoich_ )
334 stoich_->notifyRemoveEnz( e );
335 stoich_ = 0;
336 return;
337 }
338
339 bool isOK = true;
340 unsigned int numReactants;
341 numReactants = e.element()->getNeighbors( enzMols, enzFinfo );
342 bool hasEnz = ( numReactants == 1 );
343 vector< Id > subs;
344 numReactants = e.element()->getNeighbors( subs, subFinfo );
345 bool hasSubs = ( numReactants > 0 );
346 numReactants = e.element()->getNeighbors( cplxMols, cplxFinfo );
347 bool hasCplx = ( numReactants == 1 );
348 vector< Id > prds;
349 numReactants = e.element()->getNeighbors( prds, prdFinfo );
350 bool hasPrds = ( numReactants > 0 );
351 assert( stoich.element()->cinfo()->isA( "Stoich" ) );
352 stoich_ = reinterpret_cast< Stoich* >( stoich.data() );
353
354 if ( hasEnz && hasSubs && hasCplx && hasPrds ) {
355 stoich_->installEnzyme( e.id(), enzMols[0], cplxMols[0], subs, prds );
356 double concK1 = ( k2_ + kcat_ ) / Km_;
357 stoich_->setEnzK1( e, concK1 );
358 stoich_->setEnzK2( e, k2_ );
359 stoich_->setEnzK3( e, kcat_ );
360 } else {
361 stoich_->installDummyEnzyme( e.id(), Id() );
362 string msg = "";
363 if ( !hasEnz ) msg = msg + " enzyme";
364 if ( !hasCplx ) msg = msg + " enzyme-substrate complex";
365 if ( !hasSubs ) msg = msg + " substrates";
366 if ( !hasPrds ) msg = msg + " products";
367 cout << "Warning: Enz:setSolver: Dangling Enz '" <<
368 e.objId().path() << "':\nMissing " << msg << endl;
369 }
370}
