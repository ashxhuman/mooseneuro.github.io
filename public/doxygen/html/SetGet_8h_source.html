22/**
23 * Similar to Field< A >::fastGet(), except that an existing Msg is not needed.
24 *
25 * Instant-return call for a single value. Bypasses all the queueing stuff.
26 * It is hardcoded so type safety will have to be coded in too:
27 * the dynamic_cast will catch it only at runtime.
28 *
29 * Perhaps analogous localSet(), localLookupGet(), localGetVec(), etc. should
30 * also be added.
31 *
32 * Also, will be nice to change this to Field< A >::localGet() to make things
33 * more uniform.
34 */
35template< class T, class A >
36A localGet( const Eref& er, string field )
37{
38 string fullFieldName = "get" + field;
39 fullFieldName[3] = std::toupper( fullFieldName[3] );
40 const Finfo* finfo = er.element()->cinfo()->findFinfo( fullFieldName );
41 assert( finfo );
42
43 const DestFinfo* dest = dynamic_cast< const DestFinfo* >( finfo );
44 assert( dest );
45
46 const OpFunc* op = dest->getOpFunc();
47 assert( op );
48
49 const GetOpFunc< T, A >* gop =
50 dynamic_cast< const GetOpFunc< T, A >* >( op );
51 assert( gop );
52
53 return gop->reduceOp( er );
54}
