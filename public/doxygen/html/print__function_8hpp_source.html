90 /*
91 * === FUNCTION ==============================================================
92 * Name: mapToString
93 * Description: GIven a map, return a string representation of it.
94 *
95 * If the second argument is true then print the value with key. But default it
96 * is true.
97 * ==============================================================================
98 */
99
100 template<typename A, typename B>
101 string mapToString(const map<A, B>& m, bool value=true)
102 {
103 unsigned int width = 81;
104 unsigned int size = 0;
105
106 vector<string> row;
107
108 /* Get the maximum size of any entry in map */
109 stringstream ss;
110 typename map<A, B>::const_iterator it;
111 for(it = m.begin(); it != m.end(); it++)
112 {
113 ss.str("");
114 ss << it->first;
115 if(value)
116 ss << ": " << it->second;
117 row.push_back(ss.str());
118 if(ss.str().size() > size)
119 size = ss.str().size()+1;
120 }
121
122 unsigned int colums = width / size;
123 ss.str("");
124
125 size_t i = 0;
126 for(unsigned int ii = 0; ii < row.size(); ii++)
127 {
128 if(i < colums)
129 {
130 ss << setw(size+1) << row[ii];
131 i++;
132 }
133 else
134 {
135 ss << endl;
136 i = 0;
137 }
138 }
139 return ss.str();
140 }

170 /*-----------------------------------------------------------------------------
171 * This function __dump__ a message onto console. Fills appropriate colors as
172 * needed.
173 *-----------------------------------------------------------------------------*/
174
175 inline void __dump__(string msg, serverity_level_ type = debug, bool autoFormat = true)
176 {
177 stringstream ss;
178 ss << "[" << levels_[type] << "] ";
179 bool set = false;
180 bool reset = true;
181 string color = T_GREEN;
182 if(type == warning || type == fixme )
183 color = T_YELLOW;
184 else if(type == debug )
185 color = T_CYAN;
186 else if(type == error || type == failed )
187 color = T_RED;
188 else if(type == info )
189 color = T_MAGENTA;
190
191 for(unsigned int i = 0; i < msg.size(); ++i)
192 {
193 if('`' == msg[i])
194 {
195 if(!set && reset)
196 {
197 set = true;
198 reset = false;
199 ss << color;
200 }
201 else if(set && !reset)
202 {
203 reset = true;
204 set = false;
205 ss << T_RESET;
206 }
207 }
208 else if('\n' == msg[i])
209 ss << "\n | ";
210 else
211 ss << msg[i];
212 }
213
214 /* Be safe than sorry */
215 if(!reset)
216 ss << T_RESET;
217
218 cout << ss.str() << endl;
219 }

272 /**
273 * @brief Log to console (and to a log-file)
274 *
275 * @param msg String, message to be written.
276 * @param type Type of the message.
277 * @param redirectToConsole
278 * @param removeTicks
279 */
280 inline void log(string msg, serverity_level_ type = debug
281 , bool redirectToConsole = true
282 , bool removeTicks = true
283 )
284 {
285
286 if(redirectToConsole)
287 __dump__(msg, type, true);
288
289 /* remove any backtick from the string. */
290 formattedMsg( msg );
291
292 fstream logF;
293 logF.open( "__moose__.log", ios::app);
294 time_t rawtime; time(&rawtime);
295 struct tm* timeinfo;
296 timeinfo = localtime(&rawtime);
297
298 logF << asctime(timeinfo) << ": " << msg;
299
300 logF.close();
301 }
