331void SymCompartment::sumRaxial( double Ra )

338void SymCompartment::raxialSym( double Ra, double Vm)
339{
340 // cout << "SymCompartment " << ":: raxialSym: Ra = " << Ra << ", Vm = " << Vm << endl;
341 /*
342 Ra *= coeff_;
343 A_ += Vm / Ra;
344 B_ += 1.0 / Ra;
345 Im_ += ( Vm - Vm_ ) / Ra;
346 */
347
348 double R = Ra * coeff_;
349 // cout << "raxialSym:R=" << R << endl;
350 A_ += Vm / R;
351 B_ += 1.0 / R;
352 Im_ += (Vm - Vm_) / R;
353 // double invR = 2.0 / ( Ra + Ra_ );
354 // A_ += Vm * invR;
355 // B_ += invR;
356 // Im_ += ( Vm - Vm_ ) * invR;
357}
358
359void SymCompartment::raxialSphere( double Ra, double Vm)
360{
361 double invR = 2.0 / ( Ra + Ra_ );
362 A_ += Vm * invR;
363 B_ += invR;
364 Im_ += ( Vm - Vm_ ) * invR;
365}
366
367void SymCompartment::raxialCylinder( double Ra, double Vm)
368{
369 double invR = 2.0 / Ra;
370 A_ += Vm * invR;
371 B_ += invR;
372 Im_ += ( Vm - Vm_ ) * invR;
373}
