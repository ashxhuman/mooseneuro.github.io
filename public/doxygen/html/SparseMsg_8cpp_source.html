238void SparseMsg::setEntryPairs( vector< unsigned int > v )
239{
240 vector< unsigned int > src( v.begin(), v.begin() + v.size()/2 );
241 vector< unsigned int > dest( v.begin() + v.size()/2, v.end() );
242 pairFill( src, dest );

245vector< unsigned int > SparseMsg::getEntryPairs() const
246{
247 vector< unsigned int > cols = matrix_.colIndex();
248 vector< unsigned int > y;
249 for ( unsigned int row = 0; row < matrix_.nRows(); ++row )
250 {
251 unsigned int begin = matrix_.rowStart()[row];
252 unsigned int end = matrix_.rowStart()[row+1];
253 for ( unsigned int j = begin; j < end; ++j )
254 y.push_back( row );
255 }
256 assert( cols.size() == y.size() );
257 y.insert( y.end(), cols.begin(), cols.end() );
258 return y;
259}

295void SparseMsg::updateAfterFill()
296{
297 unsigned int startData = e2_->localDataStart();
298 unsigned int endData = startData + e2_->numLocalData();
299 SparseMatrix< unsigned int > temp( matrix_);
300 temp.transpose();
301 for ( unsigned int i = 0; i < temp.nRows(); ++ i )
302 {
303 const unsigned int* colIndex;
304 const unsigned int* entry;
305 unsigned int num = temp.getRow( i, &entry, &colIndex );
306 if ( i >= startData && i < endData )
307 {
308 // Inefficient. Better to do it in one pass after getting
309 // the max num
310 e2_->resizeField( i - startData, num + 1 );
311 }
312 }
313 e1()->markRewired();
314 e2()->markRewired();
315}

317void SparseMsg::pairFill( vector< unsigned int > src,
318 vector< unsigned int> dest )
319{
320 // Sanity check
321 vector< unsigned int >::const_iterator i;
322 for ( i = src.begin(); i != src.end(); ++i )
323 {
324 if (*i >= e1()->numData() )
325 {
326 cout << "Warning: SparseMsg::pairFill: Src index " << *i <<
327 " exceeds Src array size " << e1()->numData() <<
328 ". Aborting\n";
329 return;
330 }
331 }
332 for ( i = dest.begin(); i != dest.end(); ++i )
333 {
334 if (*i >= e2()->numData() )
335 {
336 cout << "Warning: SparseMsg::pairFill: Dest index " << *i <<
337 " exceeds Dest array size " << e2()->numData() <<
338 ". Aborting\n";
339 return;
340 }
341 }
342
343 vector< unsigned int > numAtDest( dest.size(), 0 );
344 vector< unsigned int > fieldIndex( dest.size(), 0 );
345 for ( unsigned int i = 0; i < dest.size(); ++i )
346 {
347 fieldIndex[i] = numAtDest[ dest[i] ];
348 // Could do on previous line, but clarity
349 ++numAtDest[ dest[i] ];
350 }
351
352 /**
353 * We set retainSize flag to true since the # of src/dest objects
354 * doesn't change. We can explicitly assign it elsewhere if needed.
355 */
356 matrix_.tripletFill( src, dest, fieldIndex, true );
357 updateAfterFill();
358}

428Eref SparseMsg::firstTgt( const Eref& src ) const
429{
430 if ( matrix_.nEntries() == 0 )
431 return Eref( 0, 0 );
432
433 if ( src.element() == e1_ )
434 {
435 const unsigned int* fieldIndex;
436 const unsigned int* colIndex;
437 unsigned int n = matrix_.getRow( src.dataIndex(),
438 &fieldIndex, &colIndex );
439 if ( n != 0 )
440 {
441 return Eref( e2_, colIndex[0], fieldIndex[0] );

443 }
444 else if ( src.element() == e2_ )
445 {
446 return Eref( e1_, 0 );
447 }
448 return Eref( 0, 0 );

451/**
452 * Returns number of synapses formed.
453 * Fills it in transpose form, because we need to count and index the
454 * number of synapses on the target, so we need to iterate over the sources
455 * in the inner loop. Once full, does the transpose.
456 * Should really have a seed argument for the random number.
457 * Later need a way to fast-forward mtrand to just the entries we
458 * need to fill.
459 */
460unsigned int SparseMsg::randomConnect( double probability )
461{
462 unsigned int nRows = matrix_.nRows(); // Sources
463 unsigned int nCols = matrix_.nColumns(); // Destinations
464 matrix_.clear();
465 unsigned int totalSynapses = 0;
466 vector< unsigned int > sizes( nCols, 0 );
467 unsigned int totSynNum = 0;
468 Element* syn = e2_;
469 unsigned int startData = syn->localDataStart();
470 unsigned int endData = startData + syn->numLocalData();
471
472 assert( nCols == syn->numData() );
473
474 matrix_.transpose();
475 for ( unsigned int i = 0; i < nCols; ++i )
476 {
477 vector< unsigned int > synIndex;
478 // This needs to be obtained from current size of syn array.
479 // unsigned int synNum = sizes[ i ];
480 unsigned int synNum = 0;
481 for ( unsigned int j = 0; j < nRows; ++j )
482 {
483 double r = rng_.uniform(); // Want to ensure it is called each time round the loop.
484 if ( r < probability )
485 {
486 synIndex.push_back( synNum );
487 ++synNum;
488 ++totSynNum;
489 }
490 else
491 {
492 synIndex.push_back( ~0 );
493 }
494 }
495
496 if ( i >= startData && i < endData )
497 {
498 e2_->resizeField( i - startData, synNum );
499 }
500 totalSynapses += synNum;
501 matrix_.addRow( i, synIndex );
502 /*
503 * This is the correct form, but I need to implement something
504 * to check up for target nodes in order to use this.
505 if ( i >= startData && i < endData ) {
506 e2_->resizeField( i - startData, synNum );
507 totalSynapses += synNum;
508 matrix_.addRow( i, synIndex );
509 } else {
510 synIndex.resize( 0 );
511 synIndex.assign( nRows, ~0 );
512 matrix_.addRow( i, synIndex );
513 }
514 */
515 }
516
517 matrix_.transpose();
518 // cout << Shell::myNode() << ": sizes.size() = " << sizes.size() << ", ncols = " << nCols << ", startSynapse = " << startSynapse << endl;
519 e1()->markRewired();
520 e2()->markRewired();
521 return totalSynapses;
522}

539ObjId SparseMsg::findOtherEnd( ObjId f ) const
540{
541 if ( f.element() == e1() )
542 {
543 const unsigned int* entry;
544 const unsigned int* colIndex;
545 unsigned int num = matrix_.getRow( f.dataIndex, &entry, &colIndex );
546 if ( num > 0 ) // Return the first matching entry.
547 {
548 return ObjId( e2()->id(), colIndex[0] );
549 }
550 return ObjId( 0, BADINDEX );
551 }
552 else if ( f.element() == e2() ) // Bad! Slow! Avoid!
553 {
554 vector< unsigned int > entry;
555 vector< unsigned int > rowIndex;
556 unsigned int num = matrix_.getColumn( f.dataIndex, entry, rowIndex );
557 if ( num > 0 ) // Return the first matching entry.
558 {
559 return ObjId( e1()->id(), DataId( rowIndex[0] ) );
560 }
561 }
562 return ObjId( 0, BADINDEX );

598void fillErefsFromMatrix(
599 const SparseMatrix< unsigned int >& matrix,
600 vector< vector < Eref > >& v, Element* e1, Element* e2 )
601{
602 v.clear();
603 v.resize( e1->numData() );
604 assert( e1->numData() == matrix.nRows() );
605 assert( e2->numData() == matrix.nColumns() );
606 for ( unsigned int i = 0; i < e1->numData(); ++i )
607 {
608 const unsigned int* entry;
609 const unsigned int* colIndex;
610 unsigned int num = matrix.getRow( i, &entry, &colIndex );
611 v[i].resize( num );
612 for ( unsigned int j = 0; j < num; ++j )
613 {
614 v[i][j] = Eref( e2, colIndex[j], entry[j] );
615 }
616 }
617}
