78void HSolveActive::reinitChannels()
79{
80 vector< double >::iterator iv;
81 vector< double >::iterator istate = state_.begin();
82 vector< int >::iterator ichannelcount = channelCount_.begin();
83 vector< ChannelStruct >::iterator ichan = channel_.begin();
84 vector< ChannelStruct >::iterator chanBoundary;
85 vector< unsigned int >::iterator icacount = caCount_.begin();
86 vector< double >::iterator ica = ca_.begin();
87 vector< double >::iterator caBoundary;
88 vector< LookupColumn >::iterator icolumn = column_.begin();
89 vector< LookupRow >::iterator icarowcompt;
90 vector< LookupRow* >::iterator icarow = caRow_.begin();
91 vector< double>::iterator iextca = externalCalcium_.begin();
92
93 LookupRow vRow;
94 LookupRow dRow;
95 double C1, C2;
96 for ( iv = V_.begin(); iv != V_.end(); ++iv )
97 {
98
99 if (!vTable_.empty())
100 vTable_.row( *iv, vRow );
101 icarowcompt = caRowCompt_.begin();
102
103 caBoundary = ica + *icacount;
104 for ( ; ica < caBoundary; ++ica )
105 {
106 caTable_.row( *ica, *icarowcompt );
107
108 ++icarowcompt;
109
110 }
111
112 chanBoundary = ichan + *ichannelcount;
113 for ( ; ichan < chanBoundary; ++ichan )
114 {
115
116 if(!caTable_.empty())
117 caTable_.row( *iextca, dRow );
118
119 if ( ichan->Xpower_ > 0.0 )
120 {
121 vTable_.lookup( *icolumn, vRow, C1, C2 );
122
123 *istate = C1 / C2;
124
125 ++icolumn, ++istate;
126 }
127
128 if ( ichan->Ypower_ > 0.0 )
129 {
130 vTable_.lookup( *icolumn, vRow, C1, C2 );
131
132 *istate = C1 / C2;
133
134 ++icolumn, ++istate;
135 }
136
137 if ( ichan->Zpower_ > 0.0 )
138 {
139 LookupRow* caRow = *icarow;
140
141 if ( caRow )
142 {
143 caTable_.lookup( *icolumn, *caRow, C1, C2 );
144 }
145 else if (*iextca>0)
146 caTable_.lookup( *icolumn, dRow, C1, C2 );
147 else
148 {
149 vTable_.lookup( *icolumn, vRow, C1, C2 );
150 }
151
152 *istate = C1 / C2;
153
154 ++icolumn, ++istate, ++icarow;
155 }
156
157 ++iextca;
158 }
159
160 ++ichannelcount, ++icacount;
161 }

164void HSolveActive::readHHChannels()
165{
166 vector< Id >::iterator icompt;
167 vector< Id >::iterator ichan;
168 int nChannel = 0;
169 double Gbar = 0.0, Ek = 0.0;
170 double X = 0.0, Y = 0.0, Z = 0.0;
171 double Xpower = 0.0, Ypower = 0.0, Zpower = 0.0;
172 int instant = 0;
173
174 for ( icompt = compartmentId_.begin(); icompt != compartmentId_.end(); ++icompt )
175 {
176 nChannel = HSolveUtils::hhchannels( *icompt, channelId_ );
177
178 // todo: discard channels with Gbar = 0.0
179 channelCount_.push_back( nChannel );
180
181 ichan = channelId_.end() - nChannel;
182 for ( ; ichan != channelId_.end(); ++ichan )
183 {
184 channel_.resize( channel_.size() + 1 );
185 ChannelStruct& channel = channel_.back();
186
187 current_.resize( current_.size() + 1 );
188 CurrentStruct& current = current_.back();
189
190 Gbar = Field< double >::get( *ichan, "Gbar" );
191 Ek = Field< double >::get( *ichan, "Ek" );
192 X = Field< double >::get( *ichan, "X" );
193 Y = Field< double >::get( *ichan, "Y" );
194 Z = Field< double >::get( *ichan, "Z" );
195 Xpower = Field< double >::get( *ichan, "Xpower" );
196 Ypower = Field< double >::get( *ichan, "Ypower" );
197 Zpower = Field< double >::get( *ichan, "Zpower" );
198 instant = Field< int >::get( *ichan, "instant" );
199 double modulation = Field< double >::get( *ichan, "modulation");
200
201 current.Ek = Ek;
202
203 channel.Gbar_ = Gbar;
204 channel.setPowers( Xpower, Ypower, Zpower );
205 channel.instant_ = instant;
206 channel.modulation_ = modulation;
207
208 /*
209 * Map channel index to state index. This is useful in the
210 * interface to find gate values.
211 */
212 chan2state_.push_back( state_.size() );
213
214 if ( Xpower > 0.0 )
215 state_.push_back( X );
216 if ( Ypower > 0.0 )
217 state_.push_back( Y );
218 if ( Zpower > 0.0 )
219 state_.push_back( Z );
220
221 /*
222 * Map channel index to compartment index. This is useful in the
223 * interface to generate channel Ik values (since we then need the
224 * compartment Vm).
225 */
226 chan2compt_.push_back( icompt - compartmentId_.begin() );
227 }
228 }
229
230 int nCumulative = 0;
231 currentBoundary_.resize( nCompt_ );
232 for ( unsigned int ic = 0; ic < nCompt_; ++ic )
233 {
234 nCumulative += channelCount_[ ic ];
235 currentBoundary_[ ic ] = current_.begin() + nCumulative;
236 }

341void HSolveActive::createLookupTables()
342{
343 vector< Id > caGate;
344 vector< Id > vGate;
345 map< Id, unsigned int > gateSpecies;
346
347 for ( unsigned int ig = 0; ig < gateId_.size(); ++ig )
348 if ( gCaDepend_[ ig ] )
349 caGate.push_back( gateId_[ ig ] );
350 else
351 vGate.push_back( gateId_[ ig ] );
352
353 for ( unsigned int ig = 0; ig < caGate.size(); ++ig )
354 gateSpecies[ caGate[ ig ] ] = ig;
355 for ( unsigned int ig = 0; ig < vGate.size(); ++ig )
356 gateSpecies[ vGate[ ig ] ] = ig;
357
358 /*
359 * Finding the smallest xmin and largest xmax across all gates' lookup
360 * tables.
361 *
362 * # of divs is determined by finding the smallest dx (highest density).
363 */
364 vMin_ = numeric_limits< double >::max();
365 vMax_ = numeric_limits< double >::min();
366 double vDx = numeric_limits< double >::max();
367 caMin_ = numeric_limits< double >::max();
368 caMax_ = numeric_limits< double >::min();
369 double caDx = numeric_limits< double >::max();
370
371 double min;
372 double max;
373 unsigned int divs;
374 double dx;
375
376 for ( unsigned int ig = 0; ig < caGate.size(); ++ig )
377 {
378 min = Field< double >::get( caGate[ ig ], "min" );
379 max = Field< double >::get( caGate[ ig ], "max" );
380 divs = Field< unsigned int >::get( caGate[ ig ], "divs" );
381 dx = ( max - min ) / divs;
382
383 if ( min < caMin_ )
384 caMin_ = min;
385 if ( max > caMax_ )
386 caMax_ = max;
387 if ( dx < caDx )
388 caDx = dx;
389 }
390 double caDiv = ( caMax_ - caMin_ ) / caDx;
391 caDiv_ = static_cast< int >( caDiv + 0.5 ); // Round-off to nearest int.
392
393 for ( unsigned int ig = 0; ig < vGate.size(); ++ig )
394 {
395 min = Field< double >::get( vGate[ ig ], "min" );
396 max = Field< double >::get( vGate[ ig ], "max" );
397 divs = Field< unsigned int >::get( vGate[ ig ], "divs" );
398 dx = ( max - min ) / divs;
399
400 if ( min < vMin_ )
401 vMin_ = min;
402 if ( max > vMax_ )
403 vMax_ = max;
404 if ( dx < vDx )
405 vDx = dx;
406 }
407 double vDiv = ( vMax_ - vMin_ ) / vDx;
408 vDiv_ = static_cast< int >( vDiv + 0.5 ); // Round-off to nearest int.
409
410 caTable_ = LookupTable( caMin_, caMax_, caDiv_, caGate.size() );
411 vTable_ = LookupTable( vMin_, vMax_, vDiv_, vGate.size() );
412
413 vector< double > A, B;
414 vector< double >::iterator ia, ib;
415 // double a, b;
416 //~ int AMode, BMode;
417 //~ bool interpolate;
418
419 // Calcium-dependent lookup tables
420 HSolveUtils::Grid caGrid( caMin_, caMax_, caDiv_ );
421 //~ if ( !caGate.empty() ) {
422 //~ grid.resize( 1 + caDiv_ );
423 //~ double dca = ( caMax_ - caMin_ ) / caDiv_;
424 //~ for ( int igrid = 0; igrid <= caDiv_; ++igrid )
425 //~ grid[ igrid ] = caMin_ + igrid * dca;
426 //~ }
427
428 for ( unsigned int ig = 0; ig < caGate.size(); ++ig )
429 {
430 HSolveUtils::rates( caGate[ ig ], caGrid, A, B );
431 //~ HSolveUtils::modes( caGate[ ig ], AMode, BMode );
432 //~ interpolate = ( AMode == 1 ) || ( BMode == 1 );
433
434 ia = A.begin();
435 ib = B.begin();
436 for ( unsigned int igrid = 0; igrid < caGrid.size(); ++igrid )
437 {
438 // Use one of the optimized forms below, instead of A and B
439 // directly. Also updated reinit() accordingly (for gate state).
440// a = *ia;
441// b = *ib;
442
443 // *ia = ( 2.0 - dt_ * b ) / ( 2.0 + dt_ * b );
444 // *ib = dt_ * a / ( 1.0 + dt_ * b / 2.0 );
445 // *ia = dt_ * a;
446 // *ib = 1.0 + dt_ * b / 2.0;
447 ++ia, ++ib;
448 }
449
450 //~ caTable_.addColumns( ig, A, B, interpolate );
451 caTable_.addColumns( ig, A, B );
452 }
453
454 // Voltage-dependent lookup tables
455 HSolveUtils::Grid vGrid( vMin_, vMax_, vDiv_ );
456 //~ if ( !vGate.empty() ) {
457 //~ grid.resize( 1 + vDiv_ );
458 //~ double dv = ( vMax_ - vMin_ ) / vDiv_;
459 //~ for ( int igrid = 0; igrid <= vDiv_; ++igrid )
460 //~ grid[ igrid ] = vMin_ + igrid * dv;
461 //~ }
462
463
464 for ( unsigned int ig = 0; ig < vGate.size(); ++ig )
465 {
466 //~ interpolate = HSolveUtils::get< HHGate, bool >( vGate[ ig ], "useInterpolation" );
467 HSolveUtils::rates( vGate[ ig ], vGrid, A, B );
468 //~ HSolveUtils::modes( vGate[ ig ], AMode, BMode );
469 //~ interpolate = ( AMode == 1 ) || ( BMode == 1 );
470
471 ia = A.begin();
472 ib = B.begin();
473 for ( unsigned int igrid = 0; igrid < vGrid.size(); ++igrid )
474 {
475 // Use one of the optimized forms below, instead of A and B
476 // directly. Also updated reinit() accordingly (for gate state).
477// a = *ia;
478// b = *ib;
479
480 // *ia = ( 2.0 - dt_ * b ) / ( 2.0 + dt_ * b );
481 // *ib = dt_ * a / ( 1.0 + dt_ * b / 2.0 );
482 // *ia = dt_ * a;
483 // *ib = 1.0 + dt_ * b / 2.0;
484 ++ia, ++ib;
485 }
486
487 //~ vTable_.addColumns( ig, A, B, interpolate );
488 vTable_.addColumns( ig, A, B );
489 }
490
491 column_.reserve( gateId_.size() );
492 for ( unsigned int ig = 0; ig < gateId_.size(); ++ig )
493 {
494 unsigned int species = gateSpecies[ gateId_[ ig ] ];
495
496 LookupColumn column;
497 if ( gCaDepend_[ ig ] )
498 caTable_.column( species, column );
499 else
500 vTable_.column( species, column );
501
502 column_.push_back( column );
503 }
504
505 ///////////////////!!!!!!!!!!
506 unsigned int maxN = *( max_element( caCount_.begin(), caCount_.end() ) );
507 caRowCompt_.resize( maxN );
508
509 for ( unsigned int ichan = 0; ichan < channel_.size(); ++ichan )
510 {
511 if ( channel_[ ichan ].Zpower_ > 0.0 )
512 {
513 int index = caDependIndex_[ ichan ];
514
515 if ( index == -1 )
516 caRow_.push_back( 0 );
517 else
518 caRow_.push_back( &caRowCompt_[ index ] );
519 }
520 }
521
522
523}

525/**
526 * Reads in SynChans and SpikeGens.
527 *
528* Unlike Compartments, HHChannels, etc., neither of these are zombified.
529 * In other words, their fields are not managed by HSolve, and their "process"
530 * functions are invoked to do their calculations. For SynChans, the process
531 * calls are made by their respective clocks, and hence the process message is
532 * not dropped. On the other hand, we drop the SpikeGen process messages here,
533 * and explicitly call the SpikeGen process() from the HSolve via a pointer.
534 */
535void HSolveActive::readSynapses()
536{
537 vector< Id > spikeId;
538 vector< Id > synId;
539 vector< Id >::iterator syn;
540 vector< Id >::iterator spike;
541 SynChanStruct synchan;
542
543 for ( unsigned int ic = 0; ic < nCompt_; ++ic )
544 {
545 synId.clear();
546 HSolveUtils::synchans( compartmentId_[ ic ], synId );
547 for ( syn = synId.begin(); syn != synId.end(); ++syn )
548 {
549 synchan.compt_ = ic;
550 synchan.elm_ = *syn;
551 synchan_.push_back( synchan );
552 }
553
554 static const Finfo* procDest = SpikeGen::initCinfo()->findFinfo( "process");
555 assert( procDest );
556 const DestFinfo* df = dynamic_cast< const DestFinfo* >( procDest );
557 assert( df );
558
559 spikeId.clear();
560 HSolveUtils::spikegens( compartmentId_[ ic ], spikeId );
561 // Very unlikely that there will be >1 spikegens in a compartment,
562 // but lets take care of it anyway.
563 for ( spike = spikeId.begin(); spike != spikeId.end(); ++spike )
564 {
565 spikegen_.push_back(
566 SpikeGenStruct( &V_[ ic ], spike->eref() )
567 );
568
569 ObjId mid = spike->element()->findCaller( df->getFid() );
570 if ( ! mid.bad() )
571 Msg::deleteMsg( mid );
572 }
573 }

596void HSolveActive::manageOutgoingMessages()
597{
598 vector< Id > targets;
599 vector< string > filter;
600
601 /*
602 * Going through all comparments, and finding out which ones have external
603 * targets through the VmOut msg. External refers to objects that do not
604 * belong the cell being managed by this HSolve. We find these by excluding
605 * any HHChannels and SpikeGens from the VmOut targets. These will then
606 * be used in HSolveActive::sendValues() to send out the messages behalf of
607 * the original objects.
608 */
609 filter.push_back( "HHChannel" );
610 filter.push_back( "SpikeGen" );
611 for ( unsigned int ic = 0; ic < compartmentId_.size(); ++ic )
612 {
613 targets.clear();
614
615 int nTargets = HSolveUtils::targets(
616 compartmentId_[ ic ],
617 "VmOut",
618 targets,
619 filter,
620 false // include = false. That is, use filter to exclude.
621 );
622
623 if ( nTargets )
624 outVm_.push_back( ic );
625 }
626
627 /*
628 * As before, going through all CaConcs, and finding any which have external
629 * targets.
630 */
631 filter.clear();
632 filter.push_back( "HHChannel" );
633 for ( unsigned int ica = 0; ica < caConcId_.size(); ++ica )
