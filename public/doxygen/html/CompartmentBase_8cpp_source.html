488void CompartmentBase::setLength( double value )
489{
490 // If length is assigned correctly, also redo the end coords to match.
491 if ( value > 0 && length_ > 0 &&
492 doubleEq( length_ * length_,
493 (x_-x0_)*(x_-x0_) + (y_-y0_)*(y_-y0_) + (z_-z0_)*(z_-z0_) ) ) {
494 double ratio = value / length_;
495 x_ = x0_ + ratio * ( x_ - x0_ );
496 y_ = y0_ + ratio * ( y_ - y0_ );
497 z_ = z0_ + ratio * ( z_ - z0_ );
498 }
499
500 length_ = value;
501}

505 return length_;
506}

666static bool hasScaleFormula( const Eref& e ) {
667 vector< Id > kids;
668 Neutral::children( e, kids );
669 for ( vector< Id >::iterator j = kids.begin(); j != kids.end(); j++ )
670 if ( j->element()->getName() == "scaleFormula" )
671 return true;
672 return false;
673}

675void CompartmentBase::setGeomAndElec( const Eref& e,
676 double len, double dia )
677{
678 if ( length_ > 0 && diameter_ > 0 && len > 0 && dia > 0 &&
679 doubleEq( length_ * length_,
680 (x_-x0_)*(x_-x0_) + (y_-y0_)*(y_-y0_) + (z_-z0_)*(z_-z0_) ) ) {
681 vSetRm( e, vGetRm( e ) * diameter_ * length_ / ( dia * len ) );
682 vSetCm( e, vGetCm( e ) * dia * len / ( diameter_ * length_ ) );
683 vSetRa( e, vGetRa( e ) * len * (diameter_ * diameter_) /
684 ( length_ * dia * dia ) );
685 // Rescale channel Gbars here
686 vector< ObjId > chans;
687 allChildren( e.objId(), ALLDATA, "ISA=ChanBase", chans );
688 for ( unsigned int i = 0; i < chans.size(); ++i ) {
689 if ( hasScaleFormula( chans[i].eref() ) )
690 continue; // Later we will eval the formula with len and dia
691 double gbar = Field< double >::get( chans[i], "Gbar" );
692 gbar *= len * dia / ( length_ * diameter_ );
693 Field< double >::set( chans[i], "Gbar", gbar );
694 }
695 // Rescale CaConc sizes here
696 vector< ObjId > concs;
697 allChildren( e.objId(), ALLDATA, "ISA=CaConcBase", concs );
698 for ( unsigned int i = 0; i < concs.size(); ++i ) {
699 Field< double >::set( concs[i], "length", len );
700 Field< double >::set( concs[i], "diameter", dia );
701 }
702
703 setLength( len );
704 setDiameter( dia );
705 }
706}

716// static func
717void CompartmentBase::zombify( Element* orig, const Cinfo* zClass,
718 Id hsolve )
719{
720 if ( orig->cinfo() == zClass )
721 return;
722 unsigned int start = orig->localDataStart();
723 unsigned int num = orig->numLocalData();
724 if ( num == 0 )
725 return;
726 vector< CompartmentDataHolder > cdh( num );
727
728 for ( unsigned int i = 0; i < num; ++i ) {
729 Eref er( orig, i + start );
730 const CompartmentBase* cb =
731 reinterpret_cast< const CompartmentBase* >( er.data() );
732 cdh[i].readData( cb, er );
733 }
734 orig->zombieSwap( zClass );
735 for ( unsigned int i = 0; i < num; ++i ) {
736 Eref er( orig, i + start );
737 CompartmentBase* cb = reinterpret_cast< CompartmentBase* >( er.data() );
738 cb->vSetSolver( er, hsolve );
739 cdh[i].writeData( cb, er );
740 }
741}
