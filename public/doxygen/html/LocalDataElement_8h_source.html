15/**
16 * This is the class for handling the local portion of a data element
17 * that is distributed over many nodes.
18 * Does block-wise partitioning between nodes.
19 */
20class LocalDataElement: public DataElement
21{
22 public:
23 /**
24 * This is the main constructor, used by Shell::innerCreate
25 * which makes most Elements. Also used to create base
26 * Elements to init the simulator in main.cpp.
27 * Id is the Id of the new Element
28 * Cinfo is the class
29 * name is its name
30 * numData is the number of data entries, defaults to a singleton.
31 */
32 LocalDataElement( Id id, const Cinfo* c, const string& name,
33 unsigned int numData = 1 );
34
35 /**
36 * This constructor copies over the original n times. It is
37 * used for doing all copies, in Shell::innerCopyElements.
38 */
39 LocalDataElement( Id id, const Element* orig, unsigned int n );
40
41 /**
42 * Virtual Destructor
43 */
44 ~LocalDataElement();
45
46 /**
47 * Virtual copier. Makes a copy of self.
48 */
49 Element* copyElement( Id newParent, Id newId, unsigned int n,
50 bool toGlobal ) const;
51
52 /////////////////////////////////////////////////////////////////
53 // Information access fields
54 /////////////////////////////////////////////////////////////////
55
56 /// Inherited virtual. Returns number of data entries over all nodes
57 unsigned int numData() const;
58
59 /// Inherited virtual. Returns index of first entry on this node
60 unsigned int localDataStart() const;
61
62 /// Inherited virtual. Returns node location of specified object
63 unsigned int getNode( unsigned int dataId ) const;
64
65 /// Inherited virtual. Returns start DataIndex on specified node
66 unsigned int startDataIndex( unsigned int node ) const;
67
68 /// Converts dataId to index on current node.
69 unsigned int rawIndex( unsigned int dataId ) const;
70
71
72 /// Inherited virtual. Reports if this is Global, which it isn't
73 bool isGlobal() const {
74 return false;
75 }
76
77 /// Inherited virtual.
78 unsigned int getNumOnNode( unsigned int node ) const;
79
80 /////////////////////////////////////////////////////////////////
81 // data access stuff
82 /////////////////////////////////////////////////////////////////
83
84 /**
85 * the data() function is inherited from DataElement and used as is.
86 */
87
88 /**
89 * Inherited virtual.
90 * Changes the total number of data entries on Element in entire
91 * simulation. Not permitted for
92 * FieldElements since they are just fields on the data.
93 */
94 void resize( unsigned int newNumData );
95
96 /////////////////////////////////////////////////////////////////
97 // Utility function for figuring out node data assignments when
98 // the data array is resized.
99 /////////////////////////////////////////////////////////////////
100 unsigned int setDataSize( unsigned int numData );
101
102 private:
103 /**
104 * This is the total number of data entries on this Element, in
105 * the entire simulation. Note that these
106 * entries do not have to be on this node, some may be farmed out
107 * to other nodes.
108 */
109 unsigned int numData_;
110
111 /**
112 * This is the number of data entries per node, except for possibly
113 * the last node if they don't divide evenly. Useful for
114 * intermediate calculations.
115 */
116 unsigned int numPerNode_;
117
118 /**
119 * Precomputed value for start index of data on this node.
120 */
121 unsigned int localDataStart_;
122};
