41void testPoolVolumeScaling()
42{
43 Shell* shell = reinterpret_cast< Shell* >( Id().eref().data() );
44 Id comptId = shell->doCreate( "CylMesh", Id(), "cyl", 1 );
45 Id meshId( comptId.value() + 1 );
46 Id poolId = shell->doCreate( "Pool", comptId, "pool", 1 );
47
48 ObjId mid = shell->doAddMsg( "OneToOne",
49 ObjId( poolId, 0 ), "requestVolume",
50 ObjId( meshId, 0 ), "get_volume" );
51
52 assert( mid != ObjId() );
53
54 vector< double > coords( 9, 0.0 );
55 double x1 = 100e-6;
56 double r0 = 10e-6;
57 double r1 = 5e-6;
58 double lambda = x1;
59 coords[3] = x1;
60 coords[6] = r0;
61 coords[7] = r1;
62 coords[8] = lambda;
63
64 Field< vector< double > >::set( comptId, "coords", coords );
65
66 double volume = Field< double >::get( poolId, "volume" );
67 assert( doubleEq( volume, PI * x1 * (r0+r1) * (r0+r1) / 4.0 ) );
68
69 Field< double >::set( poolId, "n", 400 );
70 double volscale = 1 / ( NA * volume );
71 double conc = Field< double >::get( poolId, "conc" );
72 assert( doubleEq( conc, 400 * volscale ) );
73 Field< double >::set( poolId, "conc", 500 * volscale );
74 double n = Field< double >::get( poolId, "n" );
75 assert( doubleEq( n, 500 ) );
76
77 Field< double >::set( poolId, "nInit", 650 );
78 double concInit = Field< double >::get( poolId, "concInit" );
79 assert( doubleEq( concInit, 650 * volscale ) );
80 Field< double >::set( poolId, "concInit", 10 * volscale );
81 n = Field< double >::get( poolId, "nInit" );
82 assert( doubleEq( n, 10 ) );
83
84 shell->doDelete( comptId );
85 cout << "." << flush;
86}

219///////////////////////////////////////////////////////////////////////
220// The equation for conc of substrate of an MMenz reaction is a nasty
221// transcendental, so here we just work backwards and estimate t from
222// the substrate concentration
223///////////////////////////////////////////////////////////////////////
224double estT( double s )
225{
226 double E = 1.0;
227 double Km = 1.0;
228 double kcat = 1.0;
229 double s0 = 1.0;
230 double c = -Km * log( s0 ) - s0;
231 double t = (-1.0 /( E * kcat ) ) * ( ( Km * log( s ) + s ) + c );
232 return t;
233}

235void testMMenzProcess()
236{
237 Shell* shell = reinterpret_cast< Shell* >( Id().eref().data() );
238 //////////////////////////////////////////////////////////////////////
239 // This set is the test kinetic calculation using MathFunc
240 //////////////////////////////////////////////////////////////////////
241 Id nid = shell->doCreate( "Neutral", Id(), "n", 1 );
242 //////////////////////////////////////////////////////////////////////
243 // This set is the reference kinetic calculation using MMEnz
244 //////////////////////////////////////////////////////////////////////
245 Id pid = shell->doCreate( "Pool", nid, "p", 1 ); // substrate
246 Id qid = shell->doCreate( "Pool", nid, "q", 1 ); // enz mol
247 Id rid = shell->doCreate( "Pool", nid, "r", 1 ); // product
248 Id mmid = shell->doCreate( "MMenz", nid, "mm", 1 ); // mmenz
249
250 Id tabid2 = shell->doCreate( "Table", nid, "tab2", 1 ); //output plot
251
252 Field< double >::set( mmid, "Km", 1.0 );
253 Field< double >::set( mmid, "kcat", 1.0 );
254 Field< double >::set( pid, "nInit", 1.0 );
255 Field< double >::set( qid, "nInit", 1.0 );
256 Field< double >::set( rid, "nInit", 0.0 );
257
258 shell->doAddMsg( "Single", ObjId( mmid ), "sub", ObjId( pid ), "reac" );
259 shell->doAddMsg( "Single", ObjId( mmid ), "prd", ObjId( rid ), "reac" );
260 shell->doAddMsg( "Single", ObjId( qid ), "nOut", ObjId( mmid ), "enzDest" );
261 shell->doAddMsg( "Single", ObjId( pid ), "nOut", ObjId( tabid2 ), "input" );
262 shell->doSetClock( 0, 0.01 );
263 shell->doSetClock( 1, 0.01 );
264 shell->doUseClock( "/n/mm,/n/tab2", "process", 0 );
265 shell->doUseClock( "/n/#[ISA=Pool]", "process", 1 );
266
267 //////////////////////////////////////////////////////////////////////
268 // Now run models and compare outputs
269 //////////////////////////////////////////////////////////////////////
270
271 shell->doReinit();
272 shell->doStart( 10 );
273
274 vector< double > vec = Field< vector< double > >::get( tabid2, "vec" );
275 assert( vec.size() == 1001 );
276 for ( unsigned int i = 0; i < vec.size(); ++i ) {
277 double t = 0.01 * i;
278 double et = estT( vec[i] );
279 assert( doubleApprox( t, et ) );
280 }
281
282 shell->doDelete( nid );
283 cout << "." << flush;
284}
