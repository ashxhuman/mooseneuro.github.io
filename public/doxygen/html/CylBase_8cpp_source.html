123/**
124 * The entire volume for a truncated cone is given by:
125 * V = 1/3 pi.length.(r0^2 + r0.r1 + r1^2)
126 * where the length is the length of the cone
127 * r0 is radius at base
128 * r1 is radius at top.
129 * Note that this converges to volume of a cone if r0 or r1 is zero, and
130 * to the volume of a cylinder if r0 == r1.
131 */
132double CylBase::volume( const CylBase& parent ) const
133{
134 if ( isCylinder_ )
135 return length_ * dia_ * dia_ * M_PI / 4.0;
136 double r0 = parent.dia_/2.0;
137 double r1 = dia_/2.0;
138 return length_ * ( r0*r0 + r0 *r1 + r1 * r1 ) * M_PI / 3.0;
139}

141/**
142 * Returns volume of MeshEntry.
143 * This isn't the best subdivision of the cylinder from the viewpoint of
144 * keeping the length constants all the same for different volumes.
145 * Ideally the thinner segments should have a smaller length.
146 * But this is simple and so is the diffusion calculation, so leave it.
147 * Easy to fine-tune later by modifying how one computes frac0 and frac1.
148 */
149double CylBase::voxelVolume( const CylBase& parent, unsigned int fid ) const
150{
151 assert( numDivs_ > fid );
152 if ( isCylinder_ )
153 return length_ * dia_ * dia_ * M_PI / ( 4.0 * numDivs_ );
154
155 double frac0 = ( static_cast< double >( fid ) ) /
156 static_cast< double >( numDivs_ );
157 double frac1 = ( static_cast< double >( fid + 1 ) ) /
158 static_cast< double >( numDivs_ );
159 double r0 = 0.5 * ( parent.dia_ * ( 1.0 - frac0 ) + dia_ * frac0 );
160 double r1 = 0.5 * ( parent.dia_ * ( 1.0 - frac1 ) + dia_ * frac1 );
161 double s0 = length_ * frac0;
162 double s1 = length_ * frac1;
163
164 return (s1 - s0) * ( r0*r0 + r0 *r1 + r1 * r1 ) * M_PI / 3.0;
165}

167/// Virtual function to return coords of mesh Entry.
168/// For Cylindrical mesh, coords are x1y1z1 x2y2z2 r0 r1 phi0 phi1
169vector< double > CylBase::getCoordinates(
170 const CylBase& parent, unsigned int fid ) const
171{
172 assert( numDivs_ > fid );
173 double frac0 = ( static_cast< double >( fid ) ) /
174 static_cast< double >( numDivs_ );
175 double frac1 = ( static_cast< double >( fid + 1 ) ) /
176 static_cast< double >( numDivs_ );
177
178 double r0 = 0.5 * ( parent.dia_ * ( 1.0 - frac0 ) + dia_ * frac0 );
179 // equivalent: double r0 = parent.dia_ + frac0 * ( dia_ - parent.dia_ );
180 double r1 = 0.5 * ( parent.dia_ * ( 1.0 - frac1 ) + dia_ * frac1 );
181
182 vector< double > ret( 10, 0.0 );
183 ret[0] = parent.x_ + frac0 * ( x_ - parent.x_ );
184 ret[1] = parent.y_ + frac0 * ( y_ - parent.y_ );
185 ret[2] = parent.z_ + frac0 * ( z_ - parent.z_ );
186 ret[3] = parent.x_ + frac1 * ( x_ - parent.x_ );
187 ret[4] = parent.y_ + frac1 * ( y_ - parent.y_ );
188 ret[5] = parent.z_ + frac1 * ( z_ - parent.z_ );
189 ret[6] = r0;
190 ret[7] = r1;
191 ret[8] = 0;
192 ret[9] = 0;
193
194 return ret;
195}

197/**
198 * Returns diffusion cross-section from specified index to previous.
199 * For index 0, this is cross-section area of parent.
200 * For index numDivs-1, it is the cross-section from the second-last to
201 * the last voxel in this CylBase.
202 * For index numDivs it is the area of this CylBase.
203 * Thus there is no valid value for (index > numDivs), it has
204 * to be computed external to the CylBase, typically by calling the
205 * getDiffusionArea for the child CylBase.
206 */
207double CylBase::getDiffusionArea(
208 const CylBase& parent, unsigned int fid ) const
209{
210 assert( fid < numDivs_ + 1 );
211 if ( isCylinder_ )
212 return M_PI * dia_ * dia_ / 4.0;
213 double frac0 = ( static_cast< double >( fid ) ) /
214 static_cast< double >( numDivs_ );
215 double r0 = 0.5 * ( parent.dia_ * ( 1.0 - frac0 ) + dia_ * frac0 );
216 return M_PI * r0 * r0;
217}

219/// Return the cross section area of the middle of the specified voxel.
220double CylBase::getMiddleArea(
221 const CylBase& parent, unsigned int fid ) const
222{
223 assert( fid < numDivs_ );
224 if ( isCylinder_ )
225 return M_PI * dia_ * dia_ / 4.0;
226 double frac0 = ( 0.5 + static_cast< double >( fid ) ) /
227 static_cast< double >( numDivs_ );
228 double r0 = 0.5 * ( parent.dia_ * ( 1.0 - frac0 ) + dia_ * frac0 );
229 return M_PI * r0 * r0;

238// Select grid size. Ideally the meshes should be comparable.
239double CylBase::selectGridSize( double h, double dia1,
240 double granularity ) const
241{
242 if ( length_ < 1e-7 && numDivs_ == 1 ) { // It is a disc, not a cylinder
243 return granularity * dia_ / 2.0;
244 }
245
246 double lambda = length_ / numDivs_;
247 if ( h > lambda )
248 h = lambda;
249 if ( h > dia_ / 2.0 )
250 h = dia_ / 2.0;
251 if ( h > dia1/2.0 )
252 h = dia1/2.0;
253 h *= granularity;
254 unsigned int num = static_cast<unsigned int>(ceil( lambda / h ));
255 h = lambda / num;
256
257 return h;
258}

260static void fillPointsOnCircle(
261 const Vec& u, const Vec& v, const Vec& q,
262 double h, double r, vector< double >& area,
263 const CubeMesh* other
264 )
265{
266 // fine-tune the h spacing so it is integral around circle.
267 // This will cause small errors in area estimate but they will
268 // be anisotropic. The alternative will have large errors toward
269 // 360 degrees, but not elsewhere.
270 unsigned int numAngle = static_cast<unsigned int>(floor( 2.0 * M_PI * r / h + 0.5 ));
271 assert( numAngle > 0 );
272 double dtheta = 2.0 * M_PI / numAngle;
273 double dArea = h * dtheta * r;
274 // March along points on surface of circle centred at q.
275 for ( unsigned int j = 0; j < numAngle; ++j ) {
276 double theta = j * dtheta;
277 double c = cos( theta );
278 double s = sin( theta );
279 double p0 = q.a0() + r * ( u.a0() * c + v.a0() * s );
280 double p1 = q.a1() + r * ( u.a1() * c + v.a1() * s );
281 double p2 = q.a2() + r * ( u.a2() * c + v.a2() * s );
282 unsigned int index = other->spaceToIndex( p0, p1, p2 );
283 if ( index != CubeMesh::EMPTY )
284 area[index] += dArea;
285 }
286}

288static void fillPointsOnDisc(
289 const Vec& u, const Vec& v, const Vec& q,
290 double h, double r, vector< double >& area,
291 const CubeMesh* other
292 )
293{
294 unsigned int numRadial = static_cast<unsigned int>(floor( r / h + 0.5 ));
295 double dRadial = r / numRadial;
296 for ( unsigned int i = 0; i < numRadial; ++i ) {
297 double a = ( i + 0.5 ) * dRadial;
298 unsigned int numAngle = static_cast<unsigned int>(floor( 2.0 * M_PI * a / h + 0.5 ));
299 if ( i == 0 )
300 numAngle = 1;
301 double dtheta = 2.0 * M_PI / numAngle;
302 double dArea = dRadial * dtheta * a;
303 for ( unsigned int j = 0; j < numAngle; ++j ) {
304 double theta = j * dtheta;
305 double c = cos( theta );
306 double s = sin( theta );
307 double p0 = q.a0() + a * ( u.a0() * c + v.a0() * s );
308 double p1 = q.a1() + a * ( u.a1() * c + v.a1() * s );
309 double p2 = q.a2() + a * ( u.a2() * c + v.a2() * s );
310 unsigned int index = other->spaceToIndex( p0, p1, p2 );
311 if ( index != CubeMesh::EMPTY )
312 area[index] += dArea;
313 }
314 }

317void CylBase::matchCubeMeshEntries( const ChemCompt* compt,
318 const CylBase& parent,
319 unsigned int startIndex,
320 double granularity,
321 vector< VoxelJunction >& ret,
322 bool useCylinderCurve, bool useCylinderCap ) const
323{
324 const CubeMesh* other = dynamic_cast< const CubeMesh* >( compt );
325 assert( other );
326 const double EPSILON = 1e-18;
327 Vec a( parent.x_ - x_, parent.y_ - y_, parent.z_ - z_ );
328 Vec u;
329 Vec v;
330 a.orthogonalAxes( u, v );
331
332 double h = selectGridSize( other->getDx(), parent.dia_/2, granularity );
333 double lambda = length_ / numDivs_;
334
335 unsigned int num = static_cast<unsigned int>(floor( 0.1 + lambda / h ));
336 // March along axis of cylinder.
337 // q is the location of the point along axis.
338 double rSlope = ( dia_ - parent.dia_ ) * 0.5 / length_;
339 for ( unsigned int i = 0; i < numDivs_; ++i ) {
340 vector< double >area( other->getNumEntries(), 0.0 );
341 if ( useCylinderCurve ) {
342 for ( unsigned int j = 0; j < num; ++j ) {
343 unsigned int m = i * num + j;
344 double frac = ( m * h + h/2.0 ) / length_;
345 double q0 = x_ + a.a0() * frac;
346 double q1 = y_ + a.a1() * frac;
347 double q2 = z_ + a.a2() * frac;
348 // get radius of cylinder at this point.
349 double r = dia_/2.0;
350 if ( !isCylinder_ ) // Use the more complicated conic value
351 r = parent.dia_/2.0 + frac * rSlope;
352 fillPointsOnCircle( u, v, Vec( q0, q1, q2 ),
353 h, r, area, other );
354 }
355 }
356 if ( useCylinderCap && i == numDivs_ - 1 ) {
357 fillPointsOnDisc( u, v, Vec( x_, y_, z_ ),
358 h, dia_/2.0, area, other );
359 }
360 // Go through all cubeMesh entries and compute diffusion
361 // cross-section. Assume this is through a membrane, so the
362 // only factor relevant is area. Not the distance.
363 for ( unsigned int k = 0; k < area.size(); ++k ) {
364 if ( area[k] > EPSILON ) {
365 ret.push_back( VoxelJunction( i + startIndex, k, area[k] ));
366 }
367 }
368 }
369}

371// this is the function that does the actual calculation.
372// Returns radial distance from self to axis formed from parent to self.
373// also sends back linePos, the fraction along the line, and r, the
374// radius of the parent tapering cylinder at the point of linePos.
375double CylBase::nearest( double x, double y, double z,
376 const CylBase& parent,
377 double& linePos, double& r ) const
378{
379 const double EPSILON = 1e-8; // 10 nm is too small for any compt.
380 // Consider a = parent and b = self, and c = x,y,z.
381 // Fraction along cylinder axis ab = k
382 // k = (c - a).(b - a)/((b-a).(b-a))
383 //
384 // Then, point p along line from parent to self is
385 // p = k( self-parent) + parent.
386 //
387 // So distance from c to p is what we want.
388 //
389 // If k is
390 //
391 Vec a( parent.x_, parent.y_, parent.z_ );
392 Vec b( x_, y_, z_ );
393 Vec c( x, y, z );
394
395 double dist = b.distance( a );
396 assert( dist > EPSILON );
397 double k = ( b - a ).dotProduct( c - a );
398 k /= dist * dist;
399 Vec pt = a.pointOnLine( b, k );
400
401 double ret = c.distance(pt);
402 linePos = k;
403 // For now I disable the positioning with a tapering cylinder.
404 // double rSlope = 0.5 * ( dia_ - parent.dia_ ) / numDivs_;
405 double rSlope = 0;
406 r = dia_/2.0 + k * numDivs_ * rSlope;
407 return ret;
408}

410// This function returns the index.
411double CylBase::nearest( double x, double y, double z,
412 const CylBase& parent,
413 unsigned int& index ) const
414{
415 double k = 0.0;
416 double r;
417 double ret = nearest( x, y, z, parent, k, r );
418 if ( k < 0.0 ) {
419 ret = -ret;
420 index = 0;
421 } else if ( k > 1.0 ) {
422 ret = -ret;
423 index = numDivs_ - 1;
424 } else { // Inside length of cylinder, now is it inside radius?
425 index = static_cast<unsigned int>(k * numDivs_);
426 if ( index >= numDivs_ )
427 index = numDivs_ - 1;
428 // double ri = r0_ + (index + 0.5) * rSlope_;
429 if ( ret > r * 1.01 ) // Handle roundoff
430 ret = -ret;
431 }
432 return ret;
433}
