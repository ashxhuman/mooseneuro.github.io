39/**
40 * Figures out all the volumes of the substrates or products on the
41 * specified reaction 'reac'. The SrcFinfo is for the sub or prd msg.
42 * Returns the index of the smallest vol. Passes back a vector of volumes.
43 * The meshIndex is zero. Reasoning is as follows: both in the case of
44 * well-stirred (single mesh entry) models, and in the case of spatial
45 * models with consistent mesh sizes and alignments, the mesh entry
46 * volumes are in the same ratio.
47 * Cases with more complex arrangements may also use the current vols as
48 * a starting point, but will need to add index-specific scaling factors
49 * to their reaction system.
50 */
51
52unsigned int getReactantVols( const Eref& reac, const SrcFinfo* pools,
53 vector< double >& vols )
54{
55 static const unsigned int meshIndex = 0;
56
57 const vector< MsgFuncBinding >* mfb =
58 reac.element()->getMsgAndFunc( pools->getBindIndex() );
59 unsigned int smallIndex = 0;
60
61 vols.resize( 0 );
62 if ( mfb )
63 {
64 for ( unsigned int i = 0; i < mfb->size(); ++i )
65 {
66 double v = 1;
67 Element* pool = Msg::getMsg( (*mfb)[i].mid )->e2();
68 if ( pool == reac.element() )
69 pool = Msg::getMsg( (*mfb)[i].mid )->e1();
70 assert( pool != reac.element() );
71 Eref pooler( pool, meshIndex );
72 if ( pool->cinfo()->isA( "PoolBase" ) )
73 {
74 v = lookupVolumeFromMesh( pooler );
75 }
76 else
77 {
78 cout << "Error: getReactantVols: pool is of unknown type\n";
79 assert( 0 );
80 }
81 vols.push_back( v );
82 if ( v < vols[0] )
83 smallIndex = i;
84 }
85 }
86 return smallIndex;
87}

89/**
90 * Returns conversion factor to convert rates from concentration to
91 * mol# units.
92 * Handles arbitrary combinations of volumes.
93 * Assumes that the reference volume for computing rates is the
94 * smallest volume.
95 * 26 Feb 2013: This is now changed to use the volume of the first entry.
96 * Should only be used for substrates. For products need to find the
97 * first substrate, separately, and use that to scale down the conv factor.
98 * Assumes all calculations are in SI: cubic metres and millimolar.
99 * 27 Feb 2013: This is changed to use the volume of a voxel of the
100 * the home compartment of the reac.
101 * Be warned: this can cause unexpected problems if the home compartment
102 * isn't according to convention. For example, if there is a single
103 * substrate and the home compartment is elsewhere, you will get very odd
104 * Kf:kf values.
105 * 9 Oct 2013: This is now changed to use the volume of the first
106 * substrate. Note that if the conversion is for products, then the
107 * routine has to look up the substrate list to get the first substrate.
108 * Reason is that the home compartment was often wrong in ReadKkit.
109 * Unfortunately this may yet cause problems with SBML. I don't know what
110 * conventions they use for cross-compartment reactions.
111 */
112
113double convertConcToNumRateUsingMesh( const Eref& e, const SrcFinfo* pools,
114 bool doPartialConversion )
115{
116 vector< double > vols;
117 // unsigned int smallest = getReactantVols( e, pools, vols );
118 double conv = 1.0;
119 getReactantVols( e, pools, vols );
120 if ( vols.size() == 0 ) // Should this be an assertion?
121 {
122 // cout << "Warning: convertConcToNumRateUsingMesh: zero reactants on " << e.id().path() << endl;
123 return 1.0;
124 }
125 for ( unsigned int i = 0; i < vols.size(); ++i )
126 {
127 conv *= vols[i] * NA;
128 }
129 if ( !doPartialConversion )
130 {
131 if ( pools->name() == "subOut" )
132 {
133 conv /= vols[0] * NA;
134 }
135 else
136 {
137 const Finfo* f = e.element()->cinfo()->findFinfo( "subOut" );
138 assert( f );
139 const SrcFinfo* toSub = dynamic_cast< const SrcFinfo* >( f );
140 assert( toSub );
141 vector< double > subVols;
142 getReactantVols( e, toSub, subVols );
143 if ( subVols.size() == 0 ) // no substrates!
144 return 1.0;
145 conv /= subVols[0] * NA;
146 }
147 /*
148 Id compt = getCompt( e.id() );
149 if ( compt != Id() ) {
150 Id mesh( compt.value() + 1 );
151 double meshVol = Field< double >::get( mesh, "volume" );
152 conv /= meshVol * NA;
153 }
154 */
155 }
156 return conv;
157}

160/**
161 * Generates conversion factor for rates from concentration to mol# units.
162 * This variant already knows the volume, but has to figure out # of
163 * reactants.
164 */
165double convertConcToNumRateUsingVol( const Eref& e, const SrcFinfo* pools,
166 double volume, double scale, bool doPartialConversion )
167{
168 const vector< MsgFuncBinding >* mfb =
169 e.element()->getMsgAndFunc( pools->getBindIndex() );
170 double conversion = 1.0;
171 if ( mfb && mfb->size() > 0 )
172 {
173 if ( doPartialConversion || mfb->size() > 1 )
174 {
175 conversion = scale * NA * volume;
176 double power = static_cast<double>(doPartialConversion + mfb->size() - 1);
177 if ( power > 1.0 )
178 {
179 conversion = pow( conversion, power );
180 }
181 }
182 if ( conversion <= 0 )
183 conversion = 1.0;
184 }
185
186 return conversion;
187}

189/**
190 * Generates conversion factor for rates from concentration to mol# units.
191 * This variant is used when the reactants are in different compartments
192 * or mesh entries, and may therefore have different volumes.
193 * We already know the reactants and their affiliations.
194 */
195double convertConcToNumRateInTwoCompts( double v1, unsigned int n1,
196 double v2, unsigned int n2, double scale )
197{
198 double conversion = 1.0;
199
200 for ( unsigned int i = 1; i < n1; ++i )
201 conversion *= scale * NA * v1;
202 for ( unsigned int i = 0; i < n2; ++i )
203 conversion *= scale * NA * v2;
204
205 if ( conversion <= 0 )
206 conversion = 1.0;
207
208 return conversion;
209}
