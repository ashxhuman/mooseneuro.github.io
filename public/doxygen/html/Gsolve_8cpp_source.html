408// Process operations.
409//////////////////////////////////////////////////////////////
410void Gsolve::process( const Eref& e, ProcPtr p )
411{
412 // cout << stoichPtr_ << " dsolve = " << dsolvePtr_ << endl;
413 if ( !stoichPtr_ )
414 return;
415
416 // First, handle incoming diffusion values. Note potential for
417 // issues with roundoff if diffusion is not integral.
418 if ( dsolvePtr_ )
419 {
420 vector< double > dvalues( 4 );
421 dvalues[0] = 0;
422 dvalues[1] = getNumLocalVoxels();
423 dvalues[2] = 0;
424 dvalues[3] = stoichPtr_->getNumVarPools();
425 dsolvePtr_->getBlock( dvalues );
426 dsolvePtr_->setPrev();
427
428 // Here we need to convert to integers, just in case. Normally
429 // one would use a stochastic (integral) diffusion method with
430 // the GSSA, but in mixed models it may be more complicated.
431 vector< double >::iterator i = dvalues.begin() + 4;
432 for ( ; i != dvalues.end(); ++i )
433 {
434#if 0
435 *i = std::round( *i );
436#else
437 // *i = approximateWithInteger_debug(__FUNCTION__, *i, rng_);
438 *i = approximateWithInteger(*i, rng_);
439#endif
440 }
441 setBlock( dvalues );
442 }
443
444 if ( dsolvePtr_ )
445 {
446 for ( auto i = pools_.begin(); i != pools_.end(); ++i )
447 i->refreshAtot( &sys_ );
448 }
449
450 if( 1 == numThreads_ || 1 == pools_.size())
451 {
452 if( numThreads_ > 1 )
453 {
454 cerr << "Warn: Not enough voxel. Reverting back to serial mode. " << endl;
455 numThreads_ = 1;
456 }
457
458 for ( size_t i = 0; i < pools_.size(); i++ )
459 pools_[i].advance( p, &sys_ );
460 }
461 else
462 {
463 /*-----------------------------------------------------------------------------
464 * Somewhat complicated computation to compute the number of threads. 1
465 * thread per (at least) voxel pool is ideal situation.
466 *-----------------------------------------------------------------------------*/
467#if USING_ASYNC
468 vector<std::future<size_t>> vecFutures;
469 for (size_t i = 0; i < numThreads_; i++)
470 vecFutures.push_back(
471 std::async( THREAD_LAUNCH_POLICY
472 , [this, i, p](){
473 return this->advance_chunk(i*this->grainSize_, (i+1)*this->grainSize_, p);
474 })
475 );
476 // Block in same order
477 size_t tot = 0;
478 for (auto& fut : vecFutures) tot += fut.get();
479 // We have processed all the pools.
480 assert( tot >= pools_.size() );
481#else
482 vector<std::thread> vecThreads;
483 for (size_t i = 0; i < numThreads_; i++)
484 {
485 // Use lambda. It is roughly 10% faster than std::bind and does not
486 // involve copying data.
487 vecThreads.push_back(
488 std::thread(
489 [this, i, p](){ this->advance_chunk(i*this->grainSize_, (i+1)*this->grainSize_, p); }
490 )
491 );
492 }
493
494 for( auto &v : vecThreads )
495 v.join();
496#endif
497
498 }
499
500 if ( useClockedUpdate_ ) // Check if a clocked stim is to be updated
501 {
502 if(numThreads_ == 1)
503 {
504 for ( auto &v : pools_ )
505 v.recalcTime( &sys_, p->currTime );
506 }
507 else
508 {
509#if USING_ASYNC
510 vector<std::future<size_t>> vecFutures;
511 for (size_t i = 0; i < numThreads_; i++)
512 vecFutures.push_back(
513 std::async( THREAD_LAUNCH_POLICY
514 , [this, i, p](){
515 return this->recalcTimeChunk(i*this->grainSize_, (i+1)*this->grainSize_, p);
516 })
517 );
518 // Block in same order
519 size_t tot = 0;
520 for (auto& fut : vecFutures) tot += fut.get();
521 assert( tot >= pools_.size() ); // We have processed all the pools.
522#else
523 vector<std::thread> vecThreads;
524
525 for (size_t i = 0; i < numThreads_; i++)
526 {
527 // Use lambda. It is roughly 10% faster than std::bind and does not
528 // involve copying data.
529 vecThreads.push_back(
530 std::thread(
531 [this, i, p](){
532 this->recalcTimeChunk(i*this->grainSize_, (i+1)*this->grainSize_, p);
533 }
534 )
535 );
536 }
537 for( auto &v : vecThreads )
538 v.join();
539#endif
540
541 }
542 }
543
544 // Finally, assemble and send the integrated values off for the Dsolve.
545 if ( dsolvePtr_ )
546 {
547 vector< double > kvalues( 4 );
548 kvalues[0] = 0;
549 kvalues[1] = getNumLocalVoxels();
550 kvalues[2] = 0;
551 kvalues[3] = stoichPtr_->getNumVarPools();
552 getBlock( kvalues );
553 dsolvePtr_->setBlock( kvalues );
554
555 // Now use the values in the Dsolve to update junction fluxes
556 // for diffusion, channels, and xreacs
557 dsolvePtr_->updateJunctions( p->dt );
558 // Here the Gsolve may need to do something to convert to integers
559 }
560}

562size_t Gsolve::recalcTimeChunk( const size_t begin, const size_t end, ProcPtr p)
563{
564 assert( begin >= std::min(pools_.size(), end));
565
566 size_t tot = 0;
567 for (size_t i = begin; i < std::min(pools_.size(), end); i++) {
568 tot += 1;
569 pools_[i].recalcTime( &sys_, p->currTime );
570 }
571 return tot;
572}
573
574size_t Gsolve::advance_chunk( const size_t begin, const size_t end, ProcPtr p )
575{
576 assert( begin <= std::min(end, pools_.size()) );
577 size_t tot = 0;
578 for (size_t i = begin; i < std::min(end, pools_.size() ); i++)
579 {
580 pools_[i].advance( p, &sys_ );
581 tot += 1;
582 }
583 return tot;
584}

586void Gsolve::reinit( const Eref& e, ProcPtr p )
587{
588 if ( !stoichPtr_ )
589 return;
590
591 if ( !sys_.isReady )
592 rebuildGssaSystem();
593
594 // First reinit concs. We also assign distinct rng seeds per voxel
595 for( unsigned int i = 0; i < pools_.size(); ++i ) {
596 pools_[i].reinit( &sys_, (i+1) * rngSeedOffset_ );
597 }
598
599 // Second, update the atots.
600 for ( auto i = pools_.begin(); i != pools_.end(); ++i )
601 i->refreshAtot( &sys_ );
602
603
604 // LoadBalancing. Recompute the optimal number of threads.
605 size_t nvPools = pools_.size( );
606 grainSize_ = (size_t) std::ceil((double)nvPools / (double)numThreads_);
607 assert( grainSize_ * numThreads_ >= nvPools);
608 numThreads_ = (size_t) std::ceil((double)nvPools / (double)grainSize_);
609 // MOOSE_DEBUG( "Grain size is " << grainSize_ << ". Num threads " << numThreads_);
610
611 if(1 < numThreads_)
612 cout << "Info: Setting up threaded gsolve with " << getNumThreads( )
613 << " threads. " << endl;
614
615}

635void Gsolve::rebuildGssaSystem()
636{
637 stoichPtr_->convertRatesToStochasticForm();
638 sys_.transposeN = stoichPtr_->getStoichiometryMatrix();
639 sys_.transposeN.transpose();
640 sys_.transposeN.truncateRow( stoichPtr_->getNumVarPools() + stoichPtr_->getNumProxyPools() );
641 vector< vector< unsigned int > > & dep = sys_.dependency;
642 dep.resize( stoichPtr_->getNumRates() );
643 for ( unsigned int i = 0; i < stoichPtr_->getNumRates(); ++i )
644 {
645 sys_.transposeN.getGillespieDependence( i, dep[i] );
646 }
647 fillMmEnzDep();
648 fillPoolFuncDep();
649 fillIncrementFuncDep();
650 makeReacDepsUnique();
651 for ( vector< GssaVoxelPools >::iterator
652 i = pools_.begin(); i != pools_.end(); ++i )
653 {
654 i->setNumReac( stoichPtr_->getNumRates() );
655 i->updateAllRateTerms( stoichPtr_->getRateTerms(),
656 stoichPtr_->getNumCoreRates() );
657 }
658 sys_.isReady = true;
659}
660
661/**
662 * Fill in dependency list for all MMEnzs on reactions.
663 * The dependencies of MMenz products are already in the system,
664 * so here we just need to add cases where any reaction product
665 * is the Enz of an MMEnz.
666 */
667void Gsolve::fillMmEnzDep()
668{
669 unsigned int numRates = stoichPtr_->getNumRates();
670 vector< unsigned int > indices;
671
672 // Make a map to look up enzyme RateTerm using
673 // the key of the enzyme molecule.
674 map< unsigned int, unsigned int > enzMolMap;
675 for ( unsigned int i = 0; i < numRates; ++i )
676 {
677 const MMEnzymeBase* mme = dynamic_cast< const MMEnzymeBase* >(
678 stoichPtr_->rates( i ) );
679 if ( mme )
680 {
681 vector< unsigned int > reactants;
682 mme->getReactants( reactants );
683 if ( reactants.size() > 1 )
684 enzMolMap[ reactants.front() ] = i; // front is enzyme.
685 }
686 }
687
688 // Use the map to fill in deps.
689 for ( unsigned int i = 0; i < numRates; ++i )
690 {
691 // Extract the row of all molecules that depend on the reac.
692 const int* entry;
693 const unsigned int* colIndex;
694
695 unsigned int numInRow =
696 sys_.transposeN.getRow( i, &entry, &colIndex );
697 for( unsigned int j = 0; j < numInRow; ++j )
698 {
699 map< unsigned int, unsigned int >::iterator pos =
700 enzMolMap.find( colIndex[j] );
701 if ( pos != enzMolMap.end() )
702 sys_.dependency[i].push_back( pos->second );
703 }
704 }
705}
706
707/**
708 * Here we fill in the dependencies involving poolFuncs. These are
709 * the functions that evaluate an expression and assign directly to the
710 * # of a target molecule.
711 * There are two dependencies:
712 * 1. When a reaction fires, all the Functions that depend on the reactants
713 * must update their target molecule. This is in sys_.dependentMathExpn[].
714 * 2. All the reactions that depend on the updated target molecules must
715 * now recompute their propensity. This, along with other reac dependencies,
716 * is in sys_.dependency[].
717 */
718void Gsolve::fillPoolFuncDep()
719{
720 // create map of funcs that depend on specified molecule.
721 vector< vector< unsigned int > > funcMap(
722 stoichPtr_->getNumAllPools() );
723 unsigned int numFuncs = stoichPtr_->getNumFuncs();
724 for ( unsigned int i = 0; i < numFuncs; ++i )
725 {
726 const FuncTerm *f = stoichPtr_->funcs( i );
727 vector< unsigned int > molIndex = f->getReactantIndex();
728 for ( unsigned int j = 0; j < molIndex.size(); ++j )
729 funcMap[ molIndex[j] ].push_back( i );
730 }
731 // The output of each func is a mol indexed as
732 // numVarMols + numBufMols + i
733 unsigned int numRates = stoichPtr_->getNumRates();
734 sys_.dependentMathExpn.resize( numRates );
735 vector< unsigned int > indices;
736 for ( unsigned int i = 0; i < numRates; ++i )
737 {
738 vector< unsigned int >& dep = sys_.dependentMathExpn[ i ];
739 dep.resize( 0 );
740 // Extract the row of all molecules that depend on the reac.
741 const int* entry;
742 const unsigned int* colIndex;
743 unsigned int numInRow =
744 sys_.transposeN.getRow( i, &entry, &colIndex );
745 for ( unsigned int j = 0; j < numInRow; ++j )
746 {
747 unsigned int molIndex = colIndex[j];
748 vector< unsigned int >& funcs = funcMap[ molIndex ];
749 dep.insert( dep.end(), funcs.begin(), funcs.end() );
750 for ( unsigned int k = 0; k < funcs.size(); ++k )
751 {
752 // unsigned int outputMol = funcs[k] + funcOffset;
753 unsigned int outputMol = stoichPtr_->funcs( funcs[k] )->getTarget();
754 // Insert reac deps here. Columns are reactions.

767/**
768 * Here we fill in the dependencies involving incrementFuncs. These are
769 * the functions that evaluate an expression that specifies rate of change
770 * of # of a target molecule.
771 * There are two dependencies:
772 * 1. When a reaction fires, all the incrementFuncs that depend on the
773 * reactants must update their rates. This is added to sys_.dependency[]
774 * which is the usual handler for reac dependencies. Note that the inputs
775 * to the incrementFuncs are NOT present in the stoichiometry matrix, so
776 * this has to be done as a separate step.
777 * 2. When the incrementFunc fires, then downstream reacs must update
778 * their rates. This is handled by default with the regular dependency
779 * calculations, since incrementFuncs are treated like regular reactions.
780 */
781void Gsolve::fillIncrementFuncDep()
782{
783 // create map of funcs that depend on specified molecule.
784 vector< vector< unsigned int > > funcMap( stoichPtr_->getNumAllPools() );
785 const vector< RateTerm* >& rates = stoichPtr_->getRateTerms();
786 vector< FuncRate* > incrementRates;
787 vector< unsigned int > incrementRateIndex;
788 const vector< RateTerm* >::const_iterator q;
789 for ( unsigned int i = 0; i < rates.size(); ++i )
790 {
791 FuncRate *term =
792 dynamic_cast< FuncRate* >( rates[i] );
793 if (term)
794 {
795 incrementRates.push_back( term );
796 incrementRateIndex.push_back( i );
797 }
798 }
799
800 for ( unsigned int k = 0; k < incrementRates.size(); ++k )
801 {
802 const vector< unsigned int >& molIndex =
803 incrementRates[k]->getFuncArgIndex();
804 for ( unsigned int j = 0; j < molIndex.size(); ++j )
805 funcMap[ molIndex[j] ].push_back( incrementRateIndex[k] );
806 }
807
808 unsigned int numRates = stoichPtr_->getNumRates();
809 sys_.dependentMathExpn.resize( numRates );
810 vector< unsigned int > indices;
811 for ( unsigned int i = 0; i < numRates; ++i )
812 {
813 // Algorithm:
814 // 1.Go through stoich matrix finding all the poolIndices affected
815 // by each Rate Term.
816 // 2.Use funcMap to look up FuncRateTerms affected by these indices
817 // 3. Add the rateTerm->FuncRateTerm mapping to the dependencies.
818
819 const int* entry;
820 const unsigned int* colIndex;
821 unsigned int numInRow =
822 sys_.transposeN.getRow( i, &entry, &colIndex );
823 // 1.Go through stoich matrix finding all the poolIndices affected
824 // by each Rate Term.
825 for ( unsigned int j = 0; j < numInRow; ++j )
826 {
827 unsigned int molIndex = colIndex[j]; // Affected poolIndex
828
829 // 2.Use funcMap to look up FuncRateTerms affected by these indices
830 vector< unsigned int >& funcs = funcMap[ molIndex ];
831 // 3. Add the rateTerm->FuncRateTerm mapping to the dependencies.
832 vector< unsigned int >& rdep = sys_.dependency[i];
833 rdep.insert( rdep.end(), funcs.begin(), funcs.end() );
834 }
835 }
836}
