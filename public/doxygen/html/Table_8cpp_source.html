255void Table::process( const Eref& e, ProcPtr p )
256{
257 lastTime_ = p->currTime;
258 tvec_.push_back(lastTime_);
259
260 // Copy incoming data to ret and insert into vector.
261 vector< double > ret;
262 requestOut()->send( e, &ret );
263
264 if (useSpikeMode_)
265 {
266 for ( auto i = ret.begin(); i != ret.end(); ++i )
267 spike( *i );
268 }
269 else
270 vec().insert( vec().end(), ret.begin(), ret.end() );
271
272 /* If we are streaming to a file, let's write to a file. And clean the
273 * vector.
274 * Write at every 5 seconds or whenever size of vector is more than 10k.
275 */
276 if( useFileStreamer_ )
277 {
278 if( fmod(lastTime_, 5.0) == 0.0 || getVecSize() >= 10000 )
279 {
280 mergeWithTime( data_ );
281 StreamerBase::writeToOutFile( datafile_, format_, APPEND, data_, columns_ );
282 clearAllVecs();
283 }
284 }
285 }

294/**
295 * @brief Reinitialize
296 *
297 * @param e
298 * @param p
299 */
300void Table::reinit( const Eref& e, ProcPtr p )
301{
302 tablePath_ = e.id().path();
303 unsigned int numTick = e.element()->getTick();
304 Clock* clk = reinterpret_cast<Clock*>(Id(1).eref().data());
305
306
307 dt_ = clk->getTickDt( numTick );
308 fired_ = false;
309
310 // Set column name for this table. It is used in Streamer to generate
311 // column names because path can be pretty verbose and name may not be
312 // unique. This is the default column name.
313 if( tableColumnName_.empty() )
314 tableColumnName_ = moose::moosePathToColumnName(tablePath_);
315
316 columns_ = {"time", tableColumnName_};
317
318 /** Create the default filepath for this table. */
319 if( useFileStreamer_ )
320 {
321 // If user has not set the filepath, then use the table path prefixed
322 // with rootdit as path.
323 if( datafile_.empty() )
324 setDatafile(moose::moosePathToUserPath(tablePath_) + '.' + format_);
325 }
326
327 input_ = 0.0;
328 vec().resize( 0 );
329 lastTime_ = 0;
330 vector< double > ret;
331 requestOut()->send( e, &ret );
332
333 if (useSpikeMode_)
334 {
335 for ( auto i = ret.begin(); i != ret.end(); ++i )
336 spike( *i );
337 }
338 else
339 vec().insert( vec().end(), ret.begin(), ret.end() );
340
341 tvec_.push_back(lastTime_);
342
343 if( useFileStreamer_ )
344 {
345 mergeWithTime( data_ );
346 StreamerBase::writeToOutFile(datafile_, format_, WRITE, data_, columns_);
347 clearAllVecs();
348 }
349}

360void Table::spike( double v )
361{
362 if ( fired_ )
363 { // Wait for it to go below threshold
364 if ( v < threshold_ )
365 fired_ = false;
366 }
367 else
368 {
369 if ( v > threshold_ )
370 {
371 // wait for it to go above threshold.
372 fired_ = true;
373 vec().push_back( lastTime_ );
374 }
375 }

481/* --------------------------------------------------------------------------*/
482/**
483 * @Synopsis. Convert table data to JOSN.
484 *
485 * @Returns string.
486 */
487/* ----------------------------------------------------------------------------*/
488string Table::toJSON(bool withTime, bool clear)
489{
490 stringstream ss;
491 auto v = vec();
492 if( clear )
493 lastN_ = 0;
494
495 for (unsigned int i = lastN_; i < v.size(); i++)
496 {
497 if(withTime)
498 ss << '[' << tvec_[i] << ',' << v[i] << "],";
499 else
500 ss << v[i] << ',';
501 }
502
503 string res = ss.str();
504 if(',' == res.back())
505 res.pop_back();
506
507 if( clear )
508 clearAllVecs();
509 else
510 lastN_ += v.size();
511
512 return res;
513}
