14class Synapse;
15/**
16 * This is a pure virtual base class for accessing and handling synapses.
17 * It provides a uniform interface so that all classes that use synapses
18 * can do so without duplication.
19 * The derived classes have the responsibility of handling the
20 * synaptic events as well as possible return events from the parent.
21 */
22class SynHandlerBase
23{
24public:
25 SynHandlerBase();
26 virtual ~SynHandlerBase();
27
28 ////////////////////////////////////////////////////////////////
29 // Field assignment stuff.
30 ////////////////////////////////////////////////////////////////
31
32 /**
33 * Resizes the synapse storage
34 */
35 void setNumSynapses( unsigned int num );
36
37 /**
38 * Returns number of synapses defined.
39 */
40 unsigned int getNumSynapses() const;
41
42 /**
43 * Gets specified synapse
44 */
45 Synapse* getSynapse( unsigned int i );
46 ////////////////////////////////////////////////////////////////
47
48 void process( const Eref& e, ProcPtr p );
49 void reinit( const Eref& e, ProcPtr p );
50
51 /**
52 * A utility function to check for assignment to fields that
53 * must be > 0
54 */
55 bool rangeWarning( const string& field, double value );
56
57 ////////////////////////////////////////////////////////////////
58 /**
59 * Adds a new synapse, returns its index. This is
60 * triggered by addMsg of inputs to the child synapse. The
61 * SynHandler has to ensure that we have enough synapses allocated
62 * to handle the new message, and the return value is used so that
63 * the new message knows the fieldIndex to send the message to.
64 */
65 virtual unsigned int addSynapse() = 0;
66
67 /**
68 * Drops an existing synapse, triggered by deleteMsg of an input
69 * to the child synapse. This is a little messy because we can't
70 * change the indices of the other synapses. The Handler has to
71 * figure out how to handle the 'holes' in its array of synapses.
72 */
73 virtual void dropSynapse( unsigned int droppedSynNumber ) = 0;
74
75 /**
76 * Record arrival of a new spike event. The 'time' is time for
77 * eventual arrival of the spike, and is typically well in the
78 * future. The index specifies which synapse the spike came to.
79 */
80 virtual void addSpike(
81 unsigned int index, double time, double weight ) = 0;
82 virtual double getTopSpike( unsigned int index ) const = 0;
83 ////////////////////////////////////////////////////////////////
84 // Virtual func definitions for fields.
85 ////////////////////////////////////////////////////////////////
86 virtual void vSetNumSynapses( unsigned int num ) = 0;
87 virtual unsigned int vGetNumSynapses() const = 0;
88 virtual Synapse* vGetSynapse( unsigned int i ) = 0;
89 virtual void vProcess( const Eref& e, ProcPtr p ) = 0;
90 virtual void vReinit( const Eref& e, ProcPtr p ) = 0;
91 ////////////////////////////////////////////////////////////////
92 static SrcFinfo1< double >* activationOut();
93 static const Cinfo* initCinfo();
94
95private:
96
97};
