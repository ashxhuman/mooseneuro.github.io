22void tokenize(const string& str, const string& delimiters,
23 vector<string>& tokens)
24{
25 // Token boundaries
26 string::size_type begin = str.find_first_not_of(delimiters, 0);
27 string::size_type end = str.find_first_of(delimiters, begin);
28
29 while(string::npos != begin || string::npos != end) {
30 // Found a token, add it to the vector.
31 tokens.push_back(str.substr(begin, end - begin));
32
33 // Update boundaries
34 begin = str.find_first_not_of(delimiters, end);
35 end = str.find_first_of(delimiters, begin);
36 }
37}

53std::string trim(const std::string myString, const string& delimiters)
54{
55 if(myString.length() == 0) {
56 return myString;
57 }
58
59 string::size_type end = myString.find_last_not_of(delimiters);
60 string::size_type begin = myString.find_first_not_of(delimiters);
61
62 if(begin != string::npos) {
63 return std::string(myString, begin, end - begin + 1);
64 }
65
66 return "";
67}

69std::string fix(const std::string userPath, const string& delimiters)
70{
71 string trimmedPath = trim(userPath, delimiters);
72
73 string fixedPath;
74 char prev = 0;
75
76 // In this loop, we check if there are more than one '/' together. If yes,
77 // then accept only first one and reject other.
78 for(unsigned int i = 0; i < trimmedPath.size(); ++i) {
79 const char c = trimmedPath[i];
80 if(c != '/' || c != prev)
81 fixedPath.push_back(c);
82 prev = c;
83 }
84 return fixedPath;
85}

139int strncasecmp(const string& a, const string& b, size_t n)
140{
141 for(size_t i = 0; i < std::min(n, b.size()); ++i)
142 if(tolower(a[i]) != tolower(b[i]))
143 return tolower(a[i]) - tolower(b[i]);
144
145 if(b.size() < n)
146 return a.size() - b.size();
147
148 return 0;
149}

168void str_replace_all(string& str, const string& a, const string& b)
169{
170 if(a.size() == 0)
171 return;
172
173 size_t index = 0;
174 while(true) {
175 /* Locate the substring to replace. */
176 index = str.find(a, index);
177
178 if(index == std::string::npos)
179 break;
180
181 /* Make the replacement. */
182 str.erase(index, a.size());
183 str.insert(index, b);
184 }
185}

197 * function is akin to unix'; dirname and basepath.
198 *
199 * @Param path

201 * @Returns A pair of strings <parent, name>

204std::pair<std::string, std::string> splitPath(const std::string& path)
205{
206 assert(path[0] == '/');
207 auto i = path.find_last_of('/');
208 auto parentPath = i > 0 ? path.substr(0, i) : "/";
209 return std::make_pair(parentPath, path.substr(i + 1));
210}

213/**
214 * @Synopsis Normalize a given path by removing multiple repeating // to / and
215 * /./ to /
216 *
217 * @Param path
218 *
219 * @Returns
220 */
221/* ----------------------------------------------------------------------------*/
222string normalizePath(const string& path)
223{
224 string s(path);
225 static std::regex e0("/+"); // Remove multiple / by single /
226 s = std::regex_replace(s, e0, "/");
227 static std::regex e1("/(\\./)+"); // Remove multiple / by single /
228 s = std::regex_replace(s, e1, "/");
229 return s;
230}

232void split(const string& text, const string& delimiter, vector<string>& res)
233{
234 // From https://stackoverflow.com/a/46931770/1805129
235 string token;
236 size_t pos_start = 0, pos_end, delim_len = delimiter.size();
237 while ((pos_end = text.find (delimiter, pos_start)) != string::npos) {
238 token = text.substr (pos_start, pos_end - pos_start);
239 pos_start = pos_end + delim_len;
240 res.push_back (token);
241 }
242 res.push_back (text.substr (pos_start));
243}

245string textwrap(const string& text, const string& prefix, const size_t width)
246{
247 vector<string> words;
248 tokenize(text, " \n", words);
249 string res;
250 size_t size = 0;
251 for(const auto& w : words) {
252 if(size == 0) {
253 res += prefix;
254 size = prefix.size();
255 }
256 res += w + ' ';
257 size += w.size() + 1;
258 if(size > width) {
259 res += '\n';
260 size = 0;
261 }
262 }
263 return res;
264}
