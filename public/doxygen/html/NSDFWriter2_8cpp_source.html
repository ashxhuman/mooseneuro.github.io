237void NSDFWriter2::sortMsgs(const Eref& eref)
238{
239 const Finfo* tmp = eref.element()->cinfo()->findFinfo("requestOut");
240 const SrcFinfo1< vector < double > *>* requestOut = static_cast<const SrcFinfo1< vector < double > * > * >(tmp);
241 vector< ObjId > tgts = eref.element()->getMsgTargets( eref.dataIndex(), requestOut );
242 // Make a map from ObjId to index of obj in objVec.
243 map< ObjId, unsigned int > mapMsgs;
244 for (unsigned int tgtIdx = 0; tgtIdx < tgts.size(); ++tgtIdx)
245 mapMsgs[ tgts[ tgtIdx ] ] = tgtIdx;
246
247 mapMsgIdx_.resize( tgts.size() );
248 unsigned int consolidatedBlockMsgIdx = 0;
249 for (unsigned int blockIdx = 0; blockIdx < blocks_.size(); ++blockIdx) {
250 vector< ObjId >& objVec = blocks_[blockIdx].objVec;
251 for ( auto obj = objVec.begin(); obj != objVec.end(); ++obj ) {
252 mapMsgIdx_[consolidatedBlockMsgIdx] = mapMsgs[*obj];
253 consolidatedBlockMsgIdx++;
254 }
255 }
256 assert( tgts.size() == consolidatedBlockMsgIdx );
257 // make a vector where tgtMsgIdx = mapMsgIdx_[consolidatedBlockMsgIdx]
258}

260void NSDFWriter2::buildUniformSources(const Eref& eref)
261{
262 Shell* shell = reinterpret_cast<Shell*>(Id().eref().data());
263 for ( auto bb = blocks_.begin(); bb != blocks_.end(); ++bb ) {
264 if ( bb->hasMsg )
265 continue;
266 const vector< ObjId >& objVec = bb->objVec;
267 for( vector< ObjId >::const_iterator obj = objVec.begin(); obj != objVec.end(); ++obj ) {
268
269 ObjId ret = shell->doAddMsg( "single", eref.objId(), "requestOut", *obj, bb->getField );
270 if (ret == ObjId() ) {
271 cout << "Error: NSDFWriter2::buildUniformSources: Failed to build msg from '" << eref.id().path() << "' to '" << bb->containerPath << "/" << bb->relPath << "." << bb->field << endl;
272 return;
273 }
274 }
275
276 bb->hasMsg = true;
277 }
278 sortMsgs( eref );
279}

281/**
282 Handle the datasets for the requested fields (connected to
283 requestOut). This is is similar to what HDF5DataWriter does.
284 */
285void NSDFWriter2::openUniformData(const Eref &eref)
286{
287 buildUniformSources(eref);
288 htri_t exists;
289 herr_t status;
290 if (uniformGroup_ < 0){
291 uniformGroup_ = require_group(filehandle_, UNIFORMPATH);
292 }
293 for ( auto bb = blocks_.begin(); bb != blocks_.end(); ++bb ) {
294 if ( bb->hasContainer )
295 continue;
296 // From the documentation:
297 // https://support.hdfgroup.org/HDF5/doc1.6/UG/09_Groups.html
298 // "Component link names may be any string of ASCII characters not containing a slash or a dot (/ and ., which are reserved as noted above)."
299 // So I need to replace path with a string with the slashes
300 bb->container = require_group(uniformGroup_, bb->nsdfContainerPath);
301 bb->relPathContainer = require_group(bb->container,bb->nsdfRelPath);
302 hid_t dataset = createDataset2D(bb->relPathContainer, bb->field.c_str(), bb->data.size());
303 bb->dataset = dataset;
304 writeScalarAttr<string>(dataset, "field", bb->field);
305 H5Gclose(bb->container);
306 H5Gclose(bb->relPathContainer);
307 bb->hasContainer = true;
308 }
309}

425void NSDFWriter2::createEventMap()
426{
427 herr_t status;
428 hid_t eventMapContainer = require_group(filehandle_, MAPEVENTSRC);
429 // Open the container for the event maps
430 // Create the Datasets themselves (one for each field - each row
431 // for one object).
432 for (map< string, vector < string > >::iterator ii = classFieldToEventSrc_.begin();
433 ii != classFieldToEventSrc_.end();
434 ++ii){
435 vector < string > pathTokens;
436 moose::tokenize(ii->first, "/", pathTokens);
437 string className = pathTokens[0];
438 string fieldName = pathTokens[1];
439 hid_t classGroup = require_group(eventMapContainer, className);
440 hid_t strtype = H5Tcopy(H5T_C_S1);
441 status = H5Tset_size(strtype, H5T_VARIABLE);
442 // create file space
443 hid_t ftype = H5Tcreate(H5T_COMPOUND, sizeof(hvl_t) +sizeof(hobj_ref_t));
444 status = H5Tinsert(ftype, "source", 0, strtype);
445 status = H5Tinsert(ftype, "data", sizeof(hvl_t), H5T_STD_REF_OBJ);
446 hsize_t dims[1] = {ii->second.size()};
447 hid_t space = H5Screate_simple(1, dims, NULL);
448 // The dataset for mapping is named after the field
449 hid_t ds = H5Dcreate2(classGroup, fieldName.c_str(), ftype, space,
450 H5P_DEFAULT, H5P_DEFAULT, H5P_DEFAULT);
451 status = H5Sclose(space);
452 map_type * buf = (map_type*)calloc(ii->second.size(), sizeof(map_type));
453 // Populate the buffer entries with source uid and data
454 // reference
455 for (unsigned int jj = 0; jj < ii->second.size(); ++jj){
456 buf->source = ii->second[jj].c_str();
457 char * dsname = (char*)calloc(256, sizeof(char));
458 hsize_t size = H5Iget_name(classFieldToEvent_[ii->first][jj], dsname, 255);
459 if (size > 255){
460 free(dsname);
461 dsname = (char*)calloc(size, sizeof(char));
462 size = H5Iget_name(classFieldToEvent_[ii->first][jj], dsname, 255);
463 }
464 status = H5Rcreate(&(buf->data), filehandle_, dsname, H5R_OBJECT, -1);
465 free(dsname);
466 assert(status >= 0);
467 }
468 // create memory space
469 hid_t memtype = H5Tcreate(H5T_COMPOUND, sizeof(map_type));
470 status = H5Tinsert(memtype, "source",
471 HOFFSET(map_type, source), strtype);
472 status = H5Tinsert(memtype, "data",
473 HOFFSET(map_type, data), H5T_STD_REF_OBJ);
474 status = H5Dwrite(ds, memtype, H5S_ALL, H5S_ALL, H5P_DEFAULT, buf);
475 free(buf);
476 status = H5Tclose(strtype);
477 status = H5Tclose(ftype);
478 status = H5Tclose(memtype);
479 status = H5Dclose(ds);
480 }

483/**
484 Create or retrieve a dataset for an event input. The dataset path
485 will be /data/event/{class}/{srcFinfo}/{id}_{dataIndex}_{fieldIndex}.
486
487 path : {source_object_id}.{source_field_name}
488
489 TODO: check the returned hid_t and show appropriate error messages.
490*/
491hid_t NSDFWriter2::getEventDataset(string srcPath, string srcField)
492{
493 string eventSrcPath = srcPath + string("/") + srcField;
494 map< string, hid_t >::iterator it = eventSrcDataset_.find(eventSrcPath);
495 if (it != eventSrcDataset_.end()){
496 return it->second;
497 }
498 ObjId source(srcPath);
499 herr_t status;
500 htri_t exists = -1;
501 string className = Field<string>::get(source, "className");
502 string path = EVENTPATH + string("/") + className + string("/") + srcField;
503 hid_t container = require_group(filehandle_, path);
504 stringstream dsetname;
505 dsetname << source.id.value() <<"_" << source.dataIndex << "_" << source.fieldIndex;
506 hid_t dataset = createDoubleDataset(container, dsetname.str().c_str());
507 classFieldToEvent_[className + "/" + srcField].push_back(dataset);
508 classFieldToEventSrc_[className + "/" + srcField].push_back(srcPath);
509 status = writeScalarAttr<string>(dataset, "source", srcPath);
510 assert(status >= 0);
511 status = writeScalarAttr<string>(dataset, "field", srcField);
512 assert(status >= 0);
513 eventSrcDataset_[eventSrcPath] = dataset;
514 return dataset;
515}

517void NSDFWriter2::flush()
518{
519 // We need to update the tend on each write since we do not know
520 // when the simulation is getting over and when it is just paused.
521 writeScalarAttr<string>(filehandle_, "tend", iso_time(NULL));
522
523 // append all uniform data
524 for ( vector< Block >::iterator bit = blocks_.begin(); (steps_ > 0) && (bit != blocks_.end()); bit++ ) {
525 assert( steps_ == bit->data[0].size() );
526 double* buffer = (double*)calloc(bit->data.size() * steps_, sizeof(double));
527 for (unsigned int ii = 0; ii < bit->data.size(); ++ii){
528 for (unsigned int jj = 0; jj < steps_; ++jj){
529 buffer[ii * steps_ + jj] = bit->data[ii][jj];
530 }
531 bit->data[ii].clear();
532 }
533 hid_t filespace = H5Dget_space(bit->dataset);
534 if (filespace < 0){
535 cout << "Error: NSDFWriter2::flush(): Failed to open filespace\n";
536 break;
537 }
538 hsize_t dims[2];
539 hsize_t maxdims[2];
540 // retrieve current datset dimensions
541 herr_t status = H5Sget_simple_extent_dims(filespace, dims, maxdims);
542 hsize_t newdims[] = {dims[0], dims[1] + steps_}; // new column count
543 status = H5Dset_extent(bit->dataset, newdims); // extend dataset to new column count
544 if ( status < 0 ) {
545 cout << "Error: NSDFWriter2::flush(): Fail to extend dataset\n";
546 break;
547 }
548 H5Sclose(filespace);
549 filespace = H5Dget_space(bit->dataset); // get the updated filespace
550 hsize_t start[2] = {0, dims[1]};
551 dims[1] = steps_; // change dims for memspace & hyperslab
552 hid_t memspace = H5Screate_simple(2, dims, NULL);
553 H5Sselect_hyperslab(filespace, H5S_SELECT_SET, start, NULL, dims, NULL);
554 status = H5Dwrite(bit->dataset, H5T_NATIVE_DOUBLE, memspace, filespace, H5P_DEFAULT, buffer);
555 if ( status < 0 ) {
556 cout << "Error: NSDFWriter2::flush(): Failed to write data\n";
557 break;
558 }
559 H5Sclose(memspace);
560 H5Sclose(filespace);
561 free(buffer);
562 }
563 steps_ = 0;
564
565 // append all event data
566 for (unsigned int ii = 0; ii < eventSrc_.size(); ++ii){
567 appendToDataset(getEventDataset(eventSrc_[ii], eventSrcFields_[ii]),
568 events_[ii]);
569 events_[ii].clear();
570 }
571 // flush HDF5 nodes.
572 HDF5DataWriter::flush();
573}

609void NSDFWriter2::process(const Eref& eref, ProcPtr proc)
610{
611 if (filehandle_ < 0){
612 return;
613 }
614 vector < double > uniformData;
615 const Finfo* tmp = eref.element()->cinfo()->findFinfo("requestOut");
616 const SrcFinfo1< vector < double > *>* requestOut = static_cast<const SrcFinfo1< vector < double > * > * >(tmp);
617 requestOut->send(eref, &uniformData);
618 assert( uniformData.size() == mapMsgIdx_.size() );
619 // Note that uniformData is ordered by msg tgt order. We want to store
620 // data in block_->objVec order.
621 unsigned int ii = 0;
622 for (unsigned int blockIdx = 0; blockIdx < blocks_.size(); ++blockIdx) {
623 vector< vector< double > >& bjd = blocks_[blockIdx].data;
624 for ( auto jj = bjd.begin(); jj != bjd.end(); ++jj ) {
625 jj->push_back( uniformData[ mapMsgIdx_[ii] ] );
626 ii++;
627 }
628 }
629 ++steps_;
630 if (steps_ < flushLimit_){
631 return;
632 }
633 NSDFWriter2::flush();
634 }

711bool parseBlockString( const string& val, Block& block )
712{
713 string s = val;
714 auto ff = s.find_last_of(".");
715 if (ff == string::npos )
716 return false;
717 block.hasMsg = false;
718 block.hasContainer = false;
719 block.dataset = -1;
720 block.field = s.substr( ff + 1 );
721 string temp = block.field;
722 temp[0] = toupper( temp[0] );
723 block.getField = "get" + temp;
724 s = s.substr( 0, ff );
725 vector< string > svec;
726 moose::tokenize(s, "/", svec);
727 string path = "";
728 unsigned int containerIdx = 0;
729 block.containerPath = "";
730 block.nsdfContainerPath = "";
731 string pct = "";
732 for ( unsigned int ii = 0; ii < svec.size(); ii++ ) {
733 path += "/" + svec[ii];
734 Id id( path );
735 if ( id != Id() ) {
736 if ( id.element()->cinfo()->isA( "Neuron" ) ||
737 id.element()->cinfo()->isA( "ChemCompt" ) )
738 {
739 containerIdx = ii;
740 block.containerPath = path;
741 block.nsdfContainerPath += pct + svec[ii];
742 }
743 pct = "%";
744 } else {
745 cout << "Error: NSDFWriter2: parseBlockString: No object found on '" << path << "'. Ignoring block.\n";
746 }
747 }
748 if( block.containerPath == "" )
749 return false;
750 block.relPath = "";
751 block.nsdfRelPath = "";
752 string slash = "";
753 pct = "";
754 for ( auto jj = containerIdx+1; jj < svec.size(); ++jj) {
755 block.relPath += slash + svec[jj];
756 slash = "/";
757 block.nsdfRelPath += pct + svec[jj];
758 pct = "%";
759 }
760 if( block.relPath == "" )
761 return false;
762 string objWildcardPath = block.containerPath + '/' + block.relPath;
763 block.objVec.clear();
764 simpleWildcardFind( objWildcardPath, block.objVec );
765 if ( block.objVec.size() == 0 ) {
766 cout << "Error: NSDFWriter2:parseBlockString: No objects found on path '" << objWildcardPath << "'\n";
767 return false;
768 }
769 block.className = block.objVec[0].element()->cinfo()->name();
770 for ( auto obj : block.objVec ) {
771 // Nasty workaround for different ways of handling CaConcs in Hsolve
772 if (obj.element()->cinfo()->name().find("CaConc") != string::npos &&
773 block.className.find( "CaConc" ) != string::npos )
774 continue;
775 if (obj.element()->cinfo()->name() != block.className ) {
776 cout << "Error: NSDFWriter2:parseBlockString: different classes found on path '" << objWildcardPath << "': '" << block.className << "' vs. '" << obj.element()->cinfo()->name() << "'\n";
777 return false;
778 }
779 }
780
781 block.data.resize( block.objVec.size() );
782 for( auto i = block.data.begin(); i != block.data.end(); i++ )
783 i->clear();
784 return true;
785}

828void NSDFWriter2::writeStaticCoords()
829{
830 hid_t staticObjContainer = require_group(filehandle_, STATICPATH );
831 for( auto bit = blocks_.begin(); bit != blocks_.end(); bit++ ) {
832 string meshType = "";
833 string coordContainer = bit->nsdfContainerPath + "/" + bit->nsdfRelPath;
834 string fieldName = "coords"; // pathTokens[1] is not relevant.
835 hid_t container = require_group(staticObjContainer, coordContainer);
836 double * buffer =
837 (double*)calloc(bit->data.size() * 7, sizeof(double));
838 if ( bit->className.find( "Pool" ) != string::npos ) {
839 meshType = getMeshType(*bit);
840 for (unsigned int jj = 0; jj < bit->data.size(); ++jj) {
841 ObjId obj = bit->objVec[jj];
842 vector< double > coords = Field< vector< double > >::get( obj, fieldName );
843 // cout << "numCoo = " << coords.size() << ", meshType = " << meshType << endl;
844 if ( meshType == "SpineMesh" ) {
845 for ( unsigned int kk = 0; kk < 6; ++kk) {
846 buffer[jj * 7 + kk] = coords[kk];
847 }
848 buffer[jj * 7 + 6] = coords[9]; // head Dia
849 } else if ( meshType == "EndoMesh" ) { // for EndoMesh
850 for ( unsigned int kk = 0; kk < 3; ++kk) {
851 buffer[jj * 7 + kk] = coords[kk];
852 buffer[jj * 7 + kk+3] = coords[kk];
853 }
854 buffer[jj * 7 + 6] = coords[3];
855 } else if ( meshType == "NeuroMesh") { // For NeuroMesh
856 for ( unsigned int kk = 0; kk < 7; ++kk) {
857 buffer[jj * 7 + kk] = coords[kk];
858 }
859 } else if ( meshType == "PresynMesh" ) { // For PresynMesh
860 for ( unsigned int kk = 0; kk < 3; ++kk) {
861 buffer[jj * 7 + kk] = coords[kk];
862 buffer[jj * 7 + kk + 3] = coords[kk] + coords[6]*coords[kk+3];
863 }
864 buffer[jj * 7 + 6] = coords[6];
865 }
866 }
867 } else if ( bit->className.find( "Compartment" ) != string::npos ) {
868 for (unsigned int jj = 0; jj < bit->data.size(); ++jj) {
869 ObjId obj = bit->objVec[jj];
870 vector< double > coords = Field< vector< double > >::get( obj, fieldName );
871 for ( unsigned int kk = 0; kk < 7; ++kk) {
872 buffer[jj * 7 + kk] = coords[kk];
873 }
874 }
875 } else { // Check for things like Ca or chans in an elec compt
876 for (unsigned int jj = 0; jj < bit->objVec.size(); ++jj) {
877 ObjId pa = findParentElecCompt( bit->objVec[jj] );
878 vector< double > coords( 7, 0.0 );
879 if (pa != ObjId()) {
880 coords = Field< vector< double > >::get(pa, fieldName);
881 }
882 for ( unsigned int kk = 0; kk < 7; ++kk) {
883 buffer[jj * 7 + kk] = coords[kk];
884 }
885 }
886 }
887 hsize_t dims[2];
888 dims[0] = bit->data.size();
889 dims[1] = 7;
890 hid_t memspace = H5Screate_simple(2, dims, NULL);
891 hid_t dataspace = H5Screate_simple(2, dims, NULL);
892 hid_t dataset = H5Dcreate2(container, fieldName.c_str(), H5T_NATIVE_DOUBLE, dataspace, H5P_DEFAULT, H5P_DEFAULT, H5P_DEFAULT);
893 hid_t filespace = H5Dget_space(dataset);
894
895 if ( bit->className.find( "Pool" ) != string::npos ) {
896 writeScalarAttr<string>(dataset, "meshType", meshType);
897 }
898 // writeScalarAttr<string>(dataspace, "meshType", "NeuroMesh");
899
900 herr_t status = H5Dwrite(dataset, H5T_NATIVE_DOUBLE, memspace, filespace, H5P_DEFAULT, buffer);
901 if ( status < 0 ) {
902 cout << "Error: Failed to write coords as static entry\n";
903 }
904 }
905}

908{
909 // These can be large, exceed 64K limit of attributes. So write as
910 // datasets, not attributes.
911 for ( const string& fName : modelFileNames_ ) {
912 // string fPath = MODELFILEPATH + string("/") + fName;
913 string fPath = MODELFILEPATH;
914 std::ifstream f( fName );
915 auto ss = ostringstream{};
916 if ( f.is_open() ) {
917 ss << f.rdbuf();
918 string fstr = ss.str();
919 char* filebuf = (char*)calloc( ss.str().length()+1, sizeof(char) );
920 char** sources = (char**) calloc( 1, sizeof(char* ) );
921 sources[0] = filebuf;
922 strcpy( filebuf, fstr.c_str() );
923 hid_t fGroup = require_group(filehandle_, fPath);
924 hid_t ds = createStringDataset(fGroup, fName, (hsize_t)1, (hsize_t)1 );
925 hid_t memtype = H5Tcopy(H5T_C_S1);

927 assert(status >= 0);

929 status = H5Dwrite(ds, memtype, H5S_ALL, H5S_ALL, H5P_DEFAULT, sources );
930 assert(status >= 0);
931 free( filebuf );
932 free( sources );
933 } else {
934 cout << "Warning: NSDFWriter2::writeModelFiles Could not open file '" << fName << "'/n";
935 }

939void NSDFWriter2::writeModelTree()
940{
941 if (modelRoot_ == "")
942 return;
943 vector< string > tokens;
944 ObjId mRoot(modelRoot_);
945 string rootPath = MODELTREEPATH + string("/") + mRoot.element()->getName();
946 hid_t rootGroup = require_group(filehandle_, rootPath);
947 hid_t tmp;
948 htri_t exists;
949 herr_t status;
950 deque<Id> nodeQueue;
951 deque<hid_t> h5nodeQueue;
952 nodeQueue.push_back(mRoot);
953 h5nodeQueue.push_back(rootGroup);
954 // TODO: need to clarify what happens with array elements. We can
955 // have one node per vec and set a count field for the number of
956 // elements
957 while (nodeQueue.size() > 0){
958 ObjId node = nodeQueue.front();
959 nodeQueue.pop_front();
960 hid_t prev = h5nodeQueue.front();;
961 h5nodeQueue.pop_front();
962 vector < Id > children;
963 Neutral::children(node.eref(), children);
964 for ( unsigned int ii = 0; ii < children.size(); ++ii){
965 string name = children[ii].element()->getName();
966 // skip the system elements
967 if (children[ii].path() == "/Msgs"
968 || children[ii].path() == "/clock"
969 || children[ii].path() == "/classes"
970 || children[ii].path() == "/postmaster"){
971 continue;
972 }
973 exists = H5Lexists(prev, name.c_str(), H5P_DEFAULT);
974 if (exists > 0){
975 tmp = H5Gopen2(prev, name.c_str(), H5P_DEFAULT);
976 } else {
977 tmp = H5Gcreate2(prev, name.c_str(), H5P_DEFAULT, H5P_DEFAULT, H5P_DEFAULT);
978 }
979 writeScalarAttr< string >(tmp, "uid", children[ii].path());
980 nodeQueue.push_back(children[ii]);
981 h5nodeQueue.push_back(tmp);
982 }
983 status = H5Gclose(prev);
984 }
