415 }

417 // allocateObjMap( temp );
418 deAllocateModel();
419 allocateModel(temp);

421 // kinterface_->setNumPools( n );

428 // dinterface_->setNumPools( n );
429 dinterface_->setStoich(e.id());
430 }
431 zombifyModel(e, temp);
432 if(kinterface_) {
433 kinterface_->setDsolve(dsolve_);
434 kinterface_->updateRateTerms();
435 }
436}

551vector<unsigned int> Stoich::getPoolIdMap() const
552{
553 if(poolLookup_.size() == 0)
554 return vector<unsigned int>(1, 0);
555
556 unsigned int minId = 1000000;
557 unsigned int maxId = 0;
558 unsigned int maxIndex = 0;
559
560 for(auto i = poolLookup_.cbegin(); i != poolLookup_.end(); ++i) {
561 unsigned int j = i->first.value();
562 if(j < minId)
563 minId = j;
564 if(j > maxId)
565 maxId = j;
566 if(maxIndex < i->second)
567 maxIndex = i->second;
568 }
569 vector<unsigned int> ret(maxId - minId + 2, ~0U);
570 for(auto i = poolLookup_.cbegin(); i != poolLookup_.end(); ++i) {
571 unsigned int j = i->first.value() - minId;
572 ret[j] = i->second;
573 }
574 ret[ret.size() - 1] = minId;
575
576 return ret;
577}

670/**
671 * Checks if specified reac is off solver. As side-effect it compiles
672 * a vector of the pools that are off-solver, and the corresponding
673 * compartments for those pools
674 */
675static bool isOffSolverReac(const Element* e, Id myCompt,
676 // vector< Id >& offSolverPools,
677 vector<Id>& poolCompts, map<Id, Id>& poolComptMap)
678{
679 assert(myCompt != Id());
680 assert(myCompt.element()->cinfo()->isA("ChemCompt"));
681 bool ret = false;
682 vector<Id> neighbors;
683 e->getNeighbors(neighbors, e->cinfo()->findFinfo("subOut"));
684 vector<Id> n2;
685 e->getNeighbors(n2, e->cinfo()->findFinfo("prdOut"));
686 neighbors.insert(neighbors.end(), n2.begin(), n2.end());
687 for(vector<Id>::const_iterator j = neighbors.begin(); j != neighbors.end();
688 ++j) {
689 Id otherCompt = getCompt(*j);
690 if(myCompt != otherCompt) {
691 // offSolverPools.push_back( *j );
692 poolCompts.push_back(otherCompt);
693 poolComptMap[*j] = otherCompt; // Avoids duplication of pools
694 ret = true;
695 if(j->element()->cinfo()->isA("BufPool")) {
696 cout << "Warning: Avoid BufPool: " << j->path()
697 << "\n as reactant in cross-compartment reactions\n";
698 }
699 }
700 }
701 return ret;
702}

704/**
705 * Extracts and orders the compartments associated with a given reac.
706 */
707pair<Id, Id> extractCompts(const vector<Id>& compts)
708{
709 pair<Id, Id> ret;
710 for(vector<Id>::const_iterator i = compts.begin(); i != compts.end(); ++i) {
711 if(ret.first == Id()) {
712 ret.first = *i;
713 }
714 else if(ret.first != *i) {
715 if(ret.second == Id())
716 ret.second = *i;
717 else {
718 cout << "Error: Stoich::extractCompts: more than 2 "
719 "compartments\n";
720 assert(0);
721 }
722 }
723 }
724 if((ret.second != Id()) && ret.second < ret.first) {
725 Id temp = ret.first;
726 ret.first = ret.second;
727 ret.second = ret.first;
728 }
729
730 return ret;
731}

751 offSolverEnzCompts_.push_back(extractCompts(compts));
752 }
753 else if(e->cinfo()->isA("MMEnz")) {
754 offSolverMMenzVec_.push_back(*i);
755 offSolverMMenzCompts_.push_back(extractCompts(compts));
756 }
757 }
758 else {
759 temp.push_back(*i);
760 }
761 }
762 else {
763 temp.push_back(*i);
764 }
765 }
766
767 offSolverPoolMap_.clear();
768 for(map<Id, Id>::iterator i = poolComptMap.begin(); i != poolComptMap.end();
769 ++i) {
770 // fill in the map for activeOffSolverPools.
771 offSolverPoolMap_[i->second].push_back(i->first);

1418{

1469/**
1470 * This takes the specified forward and reverse half-reacs belonging
1471 * to the specified Reac, and builds them into the Stoich.
1472 */
1473unsigned int Stoich::innerInstallReaction(Id reacId, const vector<Id>& subs,
1474 const vector<Id>& prds)
1475{
1476 ZeroOrder* forward = makeHalfReaction(0, subs);
1477 ZeroOrder* reverse = makeHalfReaction(0, prds);
1478 unsigned int rateIndex = convertIdToReacIndex(reacId);
1479 unsigned int revRateIndex = rateIndex;
1480 if(useOneWay_) {
1481 rates_[rateIndex] = forward;
1482 revRateIndex = rateIndex + 1;
1483 rates_[revRateIndex] = reverse;
1484 }
1485 else {
1486 rates_[rateIndex] = new BidirectionalReaction(forward, reverse);
1487 }
1488
1489 vector<unsigned int> molIndex;
1490 vector<double> reacScaleSubstrates;
1491 vector<double> reacScaleProducts;
1492
1493 if(useOneWay_) {
1494 unsigned int numReactants = forward->getReactants(molIndex);
1495 for(unsigned int i = 0; i < numReactants; ++i) {
1496 int temp = N_.get(molIndex[i], rateIndex);
1497 N_.set(molIndex[i], rateIndex, temp - 1);
1498 temp = N_.get(molIndex[i], revRateIndex);
1499 N_.set(molIndex[i], revRateIndex, temp + 1);
1500 }
1501
1502 numReactants = reverse->getReactants(molIndex);
1503 for(unsigned int i = 0; i < numReactants; ++i) {
1504 int temp = N_.get(molIndex[i], rateIndex);
1505 N_.set(molIndex[i], rateIndex, temp + 1);
1506 temp = N_.get(molIndex[i], revRateIndex);
1507 N_.set(molIndex[i], revRateIndex, temp - 1);
1508 }
1509 }
1510 else {
1511 unsigned int numReactants = forward->getReactants(molIndex);
1512 for(unsigned int i = 0; i < numReactants; ++i) {
1513 int temp = N_.get(molIndex[i], rateIndex);
1514 N_.set(molIndex[i], rateIndex, temp - 1);
1515 }
1516
1517 numReactants = reverse->getReactants(molIndex);
1518 for(unsigned int i = 0; i < numReactants; ++i) {
1519 int temp = N_.get(molIndex[i], revRateIndex);
1520 N_.set(molIndex[i], rateIndex, temp + 1);
1521 }
1522 }
1523 return rateIndex;
1524}

1667void Stoich::installEnzyme(ZeroOrder* r1, ZeroOrder* r2, ZeroOrder* r3,
1668 Id enzId, Id enzMolId, const vector<Id>& prds)
1669{
1670 unsigned int rateIndex = convertIdToReacIndex(enzId);
1671
1672 if(useOneWay_) {
1673 rates_[rateIndex] = r1;
1674 rates_[rateIndex + 1] = r2;
1675 rates_[rateIndex + 2] = r3;
1676 }
1677 else {
1678 rates_[rateIndex] = new BidirectionalReaction(r1, r2);
1679 rates_[rateIndex + 1] = r3;
1680 }
1681
1682 vector<unsigned int> poolIndex;
1683 unsigned int numReactants = r2->getReactants(poolIndex);
1684 assert(numReactants == 1); // Should be cplx as the only product
1685 unsigned int cplxPool = poolIndex[0];
1686
1687 if(useOneWay_) {
1688 numReactants = r1->getReactants(poolIndex); // Substrates
1689 for(unsigned int i = 0; i < numReactants; ++i) {
1690 int temp = N_.get(poolIndex[i], rateIndex); // terms for r1
1691 N_.set(poolIndex[i], rateIndex, temp - 1);
1692 temp = N_.get(poolIndex[i], rateIndex + 1); // terms for r2
1693 N_.set(poolIndex[i], rateIndex + 1, temp + 1);
1694 }
1695
1696 int temp = N_.get(cplxPool, rateIndex); // term for r1
1697 N_.set(cplxPool, rateIndex, temp + 1);
1698 temp = N_.get(cplxPool, rateIndex + 1); // term for r2
1699 N_.set(cplxPool, rateIndex + 1, temp - 1);
1700 }
1701 else // Regular bidirectional reactions.
1702 {
1703 numReactants = r1->getReactants(poolIndex); // Substrates
1704 for(unsigned int i = 0; i < numReactants; ++i) {
1705 int temp = N_.get(poolIndex[i], rateIndex);
1706 N_.set(poolIndex[i], rateIndex, temp - 1);
1707 }
1708 int temp = N_.get(cplxPool, rateIndex);
1709 N_.set(cplxPool, rateIndex, temp + 1);
1710 }
1711
1712 // Now assign reaction 3. The complex is the only substrate here.
1713 // Reac 3 is already unidirectional, so all we need to do to handle
1714 // one-way reactions is to get the index right.
1715 unsigned int reac3index = (useOneWay_) ? rateIndex + 2 : rateIndex + 1;
1716 int temp = N_.get(cplxPool, reac3index);
1717 N_.set(cplxPool, reac3index, temp - 1);
1718
1719 // For the products, we go to the prd list directly.
1720 for(unsigned int i = 0; i < prds.size(); ++i) {
1721 unsigned int j = convertIdToPoolIndex(prds[i]);
1722 int temp = N_.get(j, reac3index);
1723 N_.set(j, reac3index, temp + 1);
1724 }
1725 // Enz is also a product here.
1726 unsigned int enzPool = convertIdToPoolIndex(enzMolId);
1727 temp = N_.get(enzPool, reac3index);
1728 N_.set(enzPool, reac3index, temp + 1);
1729}
