86bool FastElim::hinesReorder( const vector< unsigned int >& parentVoxel )
87{
88 // First we fill in the vector that specifies the old row number
89 // assigned to each row of the reordered matrix.
90 assert( parentVoxel.size() == nrows_ );
91 vector< unsigned int > numKids( nrows_, 0 );
92 vector< unsigned int > lookupOldRowFromNew;
93 vector< bool > rowPending( nrows_, true );
94 unsigned int numDone = 0;
95 for ( unsigned int i = 0; i < nrows_; ++i )
96 {
97 if ( parentVoxel[i] != -1 )
98 numKids[ parentVoxel[i] ]++;
99 }
100 while ( numDone < nrows_ )
101 {
102 for ( unsigned int i = 0; i < nrows_; ++i )
103 {
104 if ( rowPending[i] && numKids[i] == 0 )
105 {
106 lookupOldRowFromNew.push_back( i );
107 rowPending[i] = false;
108 numDone++;
109 unsigned int pa = parentVoxel[i];
110 // Unsure what the root parent is. Assume it is -1
111 while ( pa != -1 && numKids[pa] == 1 )
112 {
113 assert( rowPending[pa] );
114 rowPending[pa] = false;
115 numDone++;
116 lookupOldRowFromNew.push_back( pa );
117 pa = parentVoxel[pa];
118 }
119 if ( pa != -1 )
120 {
121 assert( numKids[pa] > 0 );
122 numKids[pa]--;
123 }
124 }
125 }
126 }

133 // Then we fill in the reordered matrix. Note we need to reorder

172void sortByColumn( vector< unsigned int >& col, vector< double >& entry )
173{
174 unsigned int num = col.size();
175 assert( num == entry.size() );
176 // Stupid bubble sort, as we only have up to 5 entries and need to
177 // sort both the col and reorder the entries by the same sequence.
178 for ( unsigned int i = 0; i < num; ++i )
179 {
180 for ( unsigned int j = 1; j < num; ++j )
181 {
182 if ( col[j] < col[j-1] )
183 {
184 unsigned int temp = col[j];
185 col[j] = col[j-1];
186 col[j-1] = temp;
187 double v = entry[j];
188 entry[j] = entry[j-1];
189 entry[j-1] = v;
190 }
191 }
192 }
193}

297/**
298 * Builds the vector of forward ops: ratio, i, j
299 * RHS[i] = RHS[i] - RHS[j] * ratio
300 * This vec tells the routine which rows below have to be eliminated.
301 * This includes the rows if any in the tridiagonal band and also
302 * rows, if any, on branches.
303 */
304void FastElim::buildForwardElim( vector< unsigned int >& diag,
305 vector< Triplet< double > >& fops )
306{
307 vector< vector< unsigned int > > rowsToElim( nrows_ );
308 diag.clear();
309 for ( unsigned int i = 0; i < nrows_; ++i )
310 {
311 unsigned int rs = rowStart_[i];
312 unsigned int re = rowStart_[i+1];
313 for ( unsigned int j = rs; j < re; ++j )
314 {
315 unsigned int k = colIndex_[j];
316 if ( k == i )
317 {
318 diag.push_back(j);
319 }
320 else if ( k > i )
321 {
322 rowsToElim[ i ].push_back( k );
323 }
324 }
325 }
326 for ( unsigned int i = 0; i < nrows_; ++i )
327 {
328 double d = N_[diag[i]];
329 unsigned int diagend = rowStart_[ i + 1 ];
330 assert( diag[i] < diagend );
331 vector< unsigned int >& elim = rowsToElim[i];
332 for ( unsigned int j = 0; j < elim.size(); ++j )
333 {
334 unsigned int erow = elim[j];
335 if ( erow == i ) continue;
336 unsigned int rs = rowStart_[ erow ];
337 unsigned int re = rowStart_[ erow+1 ];
338 // assert( colIndex_[rs] == i );
339 double ratio = get( erow, i ) / d;
340 // double ratio = N_[rs]/N_[diag[i]];
341 for ( unsigned int k = diag[i]+1; k < diagend; ++k )
342 {
343 unsigned int col = colIndex_[k];
344 // findElimEntry, subtract it out.
345 for ( unsigned int q = rs; q < re; ++q )
346 {
347 if ( colIndex_[q] == col )

353 fops.push_back( Triplet< double >( ratio, i, erow) );

374/**
375 * Operations to be done on the RHS for the back sub are generated and
376 * put into the bops (backward ops) vector.
377 * col > row here, row is the entry being operated on, and col is given by
378 * rowsToSub.
379 * offDiagVal is the value on the off-diagonal at row,col.
380 * diagVal is the value on the diagonal at [row][row].
381 * RHS[row] = ( RHS[row] - offDiagVal * RHS[col] ) / diagVal
382 */
383void FastElim::buildBackwardSub( vector< unsigned int >& diag,
384 vector< Triplet< double > >& bops, vector< double >& diagVal )
385{
386 // This vec tells the routine which rows below have to be back-subbed.
387 // This includes the rows if any in the tridiagonal band and also
388 // rows, if any, on branches.
389 vector< vector< unsigned int > > rowsToSub( nrows_ );
390
391 for ( unsigned int i = 0; i < nrows_; ++i )
392 {
393 unsigned int d = diag[i] + 1;
394 unsigned int re = rowStart_[i+1];
395 for ( unsigned int j = d; j < re; ++j )
396 {
397 unsigned int k = colIndex_[j];
398 // At this point the row to sub is at (i, k). We need to go down
399 // to the (k,k) diagonal to sub it out.
400 rowsToSub[ k ].push_back( i );
401 }
402 }
403 for ( unsigned int i = 0; i < rowsToSub.size(); ++i )
404 {
405 cout << i << " : ";
406 for ( unsigned int j = 0; j < rowsToSub[i].size(); ++j )
407 {
408 cout << rowsToSub[i][j] << " ";
409 }
410 cout << endl;
411 }
412
413 diagVal.resize( 0 );
414 // Fill in the diagonal terms. Here we do all entries.
415 for ( unsigned int i = 0; i != nrows_ ; ++i )
416 {
417 diagVal.push_back( 1.0 / N_[diag[i]] );
418 }
419
420 // Fill in the back-sub operations. Note we don't need to check zero.
421 for ( unsigned int i = nrows_-1; i != 0 ; --i )
422 {
423 for ( unsigned int j = rowsToSub[i].size() - 1; j != -1; --j )
424 {
425 unsigned int k = rowsToSub[i][j];
426 double val = get( k, i ); //k is the row to go, i is the diag.
427 bops.push_back( Triplet< double >( val * diagVal[i], i, k ) );
428 }
429 }
430
431 for ( unsigned int i = 0; i < bops.size(); ++i )
432 {
433 cout << i << ": " << bops[i].a_ << " " <<
434 bops[i].b_ << " " << // diagonal index
435 bops[i].c_ << " " << // off-diagonal index
436 1.0 / diagVal[bops[i].b_] << // diagonal value.
437 endl;
438 }
439}

441void advance( vector< double >& y,
442 const vector< Triplet< double > >& ops, // has both fops and bops.
443 const vector< double >& diagVal )
444{
445 for ( vector< Triplet< double > >::const_iterator
446 i = ops.begin(); i != ops.end(); ++i )
447 y[i->c_] -= y[i->b_] * i->a_;
448
449 assert( y.size() == diagVal.size() );
450 vector< double >::iterator iy = y.begin();
451 for ( vector< double >::const_iterator
452 i = diagVal.begin(); i != diagVal.end(); ++i )
453 *iy++ *= *i;
454}

456double checkAns(
457 const double* m, unsigned int numCompts,
458 const double* ans, const double* rhs )
459{
460 vector< double > check( numCompts, 0.0 );
461 for ( unsigned int i = 0; i < numCompts; ++i )
462 {
463 for ( unsigned int j = 0; j < numCompts; ++j )
464 check[i] += m[i*numCompts + j] * ans[j];
465 }
466 double ret = 0.0;
467 for ( unsigned int i = 0; i < numCompts; ++i )
468 ret += (check[i] - rhs[i]) * (check[i] - rhs[i] );
469 return ret;
470}
