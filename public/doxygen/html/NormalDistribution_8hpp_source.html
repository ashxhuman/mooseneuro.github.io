136/**
137 * @Synopsis Get a Normally Distribution value from a given RNG.
138 *
139 * @tparam T
140 */
141/* ----------------------------------------------------------------------------*/
142template<typename T = double>
143class normal_distribution
144{
145
146public:
147
148 normal_distribution( T mean = 0.0, T sigma = 1.0) :
149 mean_(mean), sigma_(sigma)
150 {
151 mean_ = mean;
152 sigma_ = sigma;
153 }
154
155 ~normal_distribution() { ; }
156
157 double operator()( moose::MOOSE_RNG_DEFAULT_ENGINE& r )
158 {
159 unsigned long U=0, sign=0, i=0, j=0;
160 T x=0, y=0;
161
162 while (1)
163 {
164 U = r();
165 i = U & 0x0000007F; /* 7 bit to choose the step */
166 sign = U & 0x00000080; /* 1 bit for the sign */
167 j = U>>8; /* 24 bit for the x-value */
168
169 x = j*details::wtab[i];
170 if (j < details::ktab[i]) break;
171
172 if (i<127)
173 {
174 double y0 = 0, y1 = 0;
175 y0 = details::ytab[i];
176 y1 = details::ytab[i+1];
177 y = y1+(y0-y1)*r();
178 }
179 else
180 {
181 x = PARAM_R - std::log(1.0 - r()) / PARAM_R;
182 y = exp( - PARAM_R*(x-0.5* PARAM_R)) * r();
183 }
184 if (y < std::exp(-0.5*x*x)) break;
185 }
186 return sign ? sigma_*x : -sigma_*x;
187 }
188
189public:
190
191 T mean_;
192 T sigma_;
193};
