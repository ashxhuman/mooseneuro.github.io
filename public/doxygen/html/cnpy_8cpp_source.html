54{
55 string s{"("};

105void findHeader(std::fstream& fs, string& header )
106{
107 // Read header, till we hit newline character.
108 char ch;
109 fs.seekg(0); // Go to the begining.
110 header.clear();
111 while(! fs.eof())
112 {
113 fs.get(ch);
114 if( '\n' == ch )
115 break;
116 header.push_back(ch);
117 }
118 assert( header.size() >= __pre__.size() );
119}

121/**
122 * @brief Change shape in numpy header.
123 *
124 * @param
125 * @param data_len
126 * @param
127 */
128void changeHeaderShape(std::fstream& fs, const size_t data_len, const size_t numcols)
129{
130 string header{""};
131
132 // Find header. Search for newline.
133 findHeader(fs, header);
134 const size_t headerSize = header.size();
135
136 size_t shapePos = header.find( "'shape':" );
137 size_t lbrac = header.find( '(', shapePos );
138 size_t rbrac = header.find( ')', lbrac );
139 assert( lbrac > shapePos );
140 assert( rbrac > lbrac );
141
142 string prefixHeader = header.substr( 0, lbrac + 1 );
143 string postfixHeader = header.substr( rbrac );
144
145 string shapeStr = header.substr( lbrac+1, rbrac-lbrac-1);
146
147 vector<string> tokens;
148 split( tokens, shapeStr, "," );
149
150 string newShape = "";
151 for (size_t i = 0; i < tokens.size(); i++)
152 newShape += std::to_string( atoi( tokens[i].c_str() ) + data_len/numcols ) + ",";
153
154 string newHeader = prefixHeader + newShape + postfixHeader;
155 if( newHeader.size() < header.size() )
156 {
157 cout << "Warn: Modified header can not be smaller than old header" << endl;
158 }
159
160 // Resize to the old header size. Newline is not included in the header.
161 newHeader.resize(header.size());
162 newHeader += '\n'; // Add newline before writing.
163 fs.seekp(0);
164 fs.write(newHeader.c_str(), newHeader.size());
165}

167size_t writeHeader(std::fstream& fs, const vector<string>& colnames, const vector<size_t>& shape)
168{
169 // Heder are always at the begining of file.
170 fs.seekp(0);
171
172 // Write the format string. 8 bytes.
173 fs.write(&__pre__[0], __pre__.size());
174
175 char endianChar = cnpy2::BigEndianTest();
176 const char formatChar = 'd';
177
178 // Next 4 bytes are header length. This is computed again when data is
179 // appended to the file. We can have maximum of 2^32 bytes of header which
180 // ~4GB.
181
182 string header = ""; // This is the header to numpy file
183 header += "{'descr':[";
184 for( auto it = colnames.cbegin(); it != colnames.end(); it++ )
185 header += "('" + *it + "','" + endianChar + formatChar + "'),";
186
187 // shape is changed everytime we append the data. We use fixed number of
188 // character in shape. Its a int, we will use 13 chars to represent shape.
189 header += "], 'fortran_order':False,'shape':";
190 header += shapeToString(shape);
191 header += ",}";
192
193 // Add some extra sapce for safety.
194 header += string(12, ' ');
195
196 // FROM THE DOC: It is terminated by a newline (\n) and padded with spaces
197 // (\x20) to make the total of len(magic string) + 2 + len(length) +
198 // HEADER_LEN be evenly divisible by 64 for alignment purposes.
199 // pad with spaces so that preamble+headerlen+header is modulo 16 bytes.
200 // preamble is 8 bytes, header len is 4 bytes, total 12.
201 // header needs to end with \n
202 unsigned int remainder = 16 - (12 + header.size()) % 16;
203 header.insert(header.end(), remainder-1, ' ');
204 header += '\n'; // Add newline.
205
206 // Now write the size of header. Its 4 byte long in version 2.
207 uint32_t s = header.size();
208 fs.write((char*)&s, 4);
209 fs << header;
210 return fs.tellp();
211}

230void writeNumpy(const string& outfile, const vector<double>& data, const vector<string>& colnames)
231{
232
233 // In our application, we need to write a vector as matrix. We do not
234 // support the stacking of matrices.
235 vector<size_t> shape;
236
237 if( colnames.size() == 0)
238 return;
239
240 shape.push_back(data.size() / colnames.size());
241
242 // Create a new file.
243 std::fstream fs;
244 fs.open(outfile, std::ofstream::in | std::ofstream::out | std::ofstream::binary | std::ofstream::trunc);
245
246 /* In mode "w", open the file and write a header as well. When file is open
247 * in mode "a", we assume that file is alreay a valid numpy file.
248 */
249 if(! fs.is_open())
250 {
251 moose::showWarn( "Could not open file " + outfile );
252 return;
253 }
254
255 auto p = writeHeader( fs, colnames, shape );
256 fs.seekp(p);
257
258 // Here the previous character is '\n'
259
260 fs.write(reinterpret_cast<const char*>(data.data()), sizeof(double)*data.size());
261 fs.close();
262}

264void appendNumpy(const string& outfile, const vector<double>& vec, const vector<string>& colnames)
265{
266
267 std::fstream fs;
268 fs.open(outfile, std::fstream::in | std::fstream::out | std::fstream::binary);
269
270 if( ! fs.is_open() )
271 {
272 moose::showWarn( "Could not open " + outfile + " to write " );
273 return;
274 }
275
276 // And change the shape in header.
277 changeHeaderShape(fs, vec.size(), colnames.size());
278
279 // Go to the end.
280 fs.seekp(0, std::ios_base::end);
281 fs.write(reinterpret_cast<const char*>(&vec[0]), sizeof(double)*vec.size());
282 fs.close();
283}

285void readNumpy(const string& infile, vector<double>& data)
286{
287 cout << "Reading from " << infile << endl;
288 std::ifstream fs;
289 fs.open(infile, std::ios::in | std::ios::binary);
290
291 if(! fs.is_open())
292 {
293 cerr << "Could not open " << infile << endl;
294 return;
295 }
296
297 char ch;
298 fs.get(ch);
299 size_t nBytes = 1;
300 while(ch != '\n')
301 {
302 fs.get(ch);
303 nBytes += 1;
304 }
305
306 char buff[sizeof(double)];
307 double x;
308
309 fs.seekg(nBytes, std::ios_base::beg);

312 fs.read(buff, sizeof(double));
