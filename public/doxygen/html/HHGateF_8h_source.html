19/**
20 * This class handles a single gate on an HHChannel. It is equivalent to the
21 * m and h terms on the Hodgkin-Huxley Na channel, or the n term on the
22 * K channel. It stores the
23 * voltage-dependence (sometimes concentration-dependence) of the
24 * gating variables for opening the channel. It does so in a tabular form
25 * which can be directly filled using experimental data points.
26 * It also provides a set of
27 * utility functions for defining the gate in functional forms, and
28 * accessing those original functional forms.
29 * The HHGateF is
30 * accessed as a FieldElement, which means that it is available as a
31 * pointer on the HHChannel. HHGateFs are typically shared. This means that
32 * when you make a copy or a vector of an HHChannel, there is only a single
33 * HHGateF created, and its pointer is used by all the copies.
34 * The lookup functions are thread-safe.
35 * Field assignment to the HHGateF should be possible only from the
36 * original HHChannel, but all the others do have read permission.
37 * Whereas HHGate uses interpolation tables, HHGateF uses direct
38 * formula evaluation, hence slower but possibly more accurate.
39 */
40
41class HHGateF : public HHGateBase {
42public:
43 /**
44 * Dummy constructor, to keep Dinfo happy. Should never be used
45 */
46 HHGateF();
47
48 /**
49 * This constructor is the one meant to be used. It takes the
50 * originalId of the parent HHChannel as a required argument,
51 * so that any subsequent 'write' functions can be checked to
52 * see if they are legal. Also tracks its own Id.
53 */
54 HHGateF(Id originalChanId, Id originalGateId);
55 /// HHGates remain shared between copies of a channel, so it
56 /// should never be copied. Yet we need to define this because
57 /// eprtk parser deletes its copy assignment, which deletes
58 /// compiler generated copy assignment operator for HHGateF, which
59 /// raises error with Dinfo, which tries to reference the copy
60 /// operator.
61 HHGateF& operator=(const HHGateF&);
62 //////////////////////////////////////////////////////////
63 // LookupValueFinfos
64 //////////////////////////////////////////////////////////
65 /**
66 * lookupA: Look up the A vector from a double. Typically does
67 * so by direct scaling and offset to an integer lookup, using
68 * a fine enough table granularity that there is little error.
69 * Alternatively uses linear interpolation.
70 * The range of the double is predefined based on knowledge of
71 * voltage or conc ranges, and the granularity is specified by
72 * the xmin, xmax, and invDx fields.
73 */
74 double lookupA(double v) const;
75
76 /**
77 * lookupB: Look up the B vector from a double, similar to lookupA.
78 */
79 double lookupB(double v) const;

84 * Single call to get both A and B values by lookup
85 */
86 void lookupBoth(double v, double* A, double* B) const;

103 static const Cinfo* initCinfo();
104
105protected:
106 /// Whether the gate is expressed in tau-inf form. If false, it is
107 /// alpha-beta form
108 bool tauInf_;
109 exprtk::symbol_table<double> symTab_;
110 exprtk::expression<double> alpha_;

117 mutable double betav_;

122 string alphaExpr_;
123 string betaExpr_;
