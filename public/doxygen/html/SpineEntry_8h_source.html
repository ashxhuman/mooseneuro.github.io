13/**
14 * Helper class for the SpineMesh. Defines a single spine.
15 */
16
17class SpineEntry
18{
19 public:
20 /**

23 SpineEntry( Id shaft, Id head, unsigned int parent );

38 Id headId() const; /// Returns Id of head electrical compartment.
39
40 /**
41 * Generate list of matching CubeMesh entries to the single Head

43 */
44 void matchCubeMeshEntriesToHead( const ChemCompt* compt,

46 double granularity, vector< VoxelJunction >& ret ) const;

48 /**

76
77 double rootArea() const;
78 double headArea() const;
79 double diffusionLength() const;
80
81 /**
82 * Resizes SpineEntry. Takes original shaft base and scales up
83 * from there. Retains original shaft dimensions, only the
84 * head is scaled.
85 */
86 void setVolume( double volume );
87
88 /**
89 * moves Spine entry so that shaft base is now positioned at xyz.
90 */
91 void positionShaftBase( double x, double y, double z );
92
93 /**
94 * psdCoords is used to build the PsdMesh. The function returns 8
95 * coords to define the psd:
96 * centre xyz
97 * direction xyz
98 * dia,
99 * diffusion distance to middle of spine Head.
100 */
101 vector< double > psdCoords() const;
102
103 /**
104 * x1y1z1 x2y2z2 x3y3z3 r0 r1. Here x1 is farthest, x2 is
105 * junction between neck and shaft, and x3 is on the dend.
106 * r0 is radius of spine head, r1 of shaft.
107 */
108 vector< double > spineCoords() const;
109

114 /**

116 */
117 unsigned int parent_;
118
119 /// Id of electrical compartment in which this diffusive compt lives
120 Id shaftId_;
121 Id headId_;
122};
