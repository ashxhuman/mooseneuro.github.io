17/**
18 * This class manages id lookups for elements. Ids provide a uniform
19 * handle for every object, independent of which node they are located on.
20 */
21class Id {
22public:
23 //////////////////////////////////////////////////////////////
24 // Id creation
25 //////////////////////////////////////////////////////////////
26 /**
27 * Returns the root Id
28 */
29 Id();
30
31 /**
32 * Creates an id with the specified Element number
33 */
34 Id(unsigned int id);
35
36 /**
37 * Returns an id found by traversing the specified path
38 */
39 Id(const std::string& path);
40
41 /**
42 * Downconverts an OjbId to an Id
43 */
44 Id(const ObjId& oi);
45
46 /**
47 * Destroys an Id. Doesn't do anything much.
48 */
49 ~Id()
50 {
51 }
52
53 /**
54 * Reserves an id for assigning to an Element. Each time it is
55 * called a new id is reserved, even if previous ones have not been
56 * used yet.
57 */
58 static Id nextId();
59
60 /**
61 * Returns the number of Ids in use.
62 */
63 static unsigned int numIds();
64
65 /**
66 * The specified element is placed into current id.
67 */
68 void bindIdToElement(Element* e);
69
70 /**
71 * Cleanly deletes the associated Element, and zeroes out
72 * contents of elements vector at the location pointed to by
73 * this->id_.
74 */
75 void destroy() const;
76
77 //////////////////////////////////////////////////////////////
78 // Id info
79 //////////////////////////////////////////////////////////////
80 /**
81 * Returns the full pathname of the object on the id.
82 */
83 std::string path() const;
84
85 /**
86 * Returns the Element pointed to by the id
87 * If it is off-node, returns an allocated wrapper element with
88 * postmaster and id information. Calling func has to free it.
89 * This wrapper element may also point to UNKNOWN NODE, in which
90 * case the master node IdManager has to figure out where it
91 * belongs.
92 * Returns 0 on failure.
93 * Deprecated.
94 Element* operator()() const;
95 */
96
97 /**
98 * Returns the Element pointed to by the Id.
99 * Perhaps cleaner to use than operator()() as it is an explicit
100 * function.
101 */
102 Element* element() const;
103
104 // unsigned int index() const;
105
106 /**
107 * Returns the Eref to the element plus index
108 */
109 Eref eref() const;
110
111 /**
112 * Returns an id whose value is string-converted from the
113 * specified string.
114 */
115 static Id str2Id(const std::string& s);
116
117 /**
118 * Returns a string holding the ascii value of the id_ .
119 */
120 static std::string id2str(Id id);
121
122 unsigned int value() const;
123
124 //////////////////////////////////////////////////////////////
125 // Comparisons between ids
126 //////////////////////////////////////////////////////////////
127 bool operator==(const Id& other) const
128 {
129 // return id_ == other.id_ && index_ == other.index_;
130 return id_ == other.id_;
131 }
132
133 bool operator!=(const Id& other) const
134 {
135 // return id_ != other.id_ || index_ != other.index_;
136 return id_ != other.id_;
137 }
138
139 bool operator<(const Id& other) const
140 {
141 // return ( id_ < other.id_ ) ||
142 // ( id_ == other.id_ && index_ < other.index_ );
143 return (id_ < other.id_);
144 }
145
146 // The follwoing two functions check if the Id is associated with
147 // an existing element. Needed for handling objects that have been
148 // destroyed.
149 static bool isValid(Id id)
150 {
151 return (id.id_ < elements().size()) && (elements()[id.id_] != 0);
152 }
153
154 static bool isValid(unsigned int id)
155 {
156 return (id < elements().size()) && (elements()[id] != 0);
157 }
158
159 //////////////////////////////////////////////////////////////
160 /**
161 * Used to clean up all Elements when the simulation is terminated
162 */
163 static void clearAllElements();
164
165 /**
166 * Used to clean out any specific Id.
167 */
168 void zeroOut() const;
169 //////////////////////////////////////////////////////////////
170
171 friend ostream& operator<<(ostream& s, const Id& i);
172 friend istream& operator>>(istream& s, Id& i);
173
174private:
175 // static void setManager( Manager* m );
176 unsigned int id_; // Unique identifier for Element*
177 // unsigned int index_; // Index of array entry within element.
178 static vector<Element*>& elements();
179};

190 }
191};
192}
