27/**
28 * wildcardFieldComparison returns true if the value of the
29 * specified field matches the value in the comparsion string mid.
30 * Format is FIELD(name)=val
31 * If the format or the value does not match, return 0
32 */
33static bool wildcardFieldComparison( ObjId oid, const string& mid )
34{
35 // where = could be the usual comparison operators and val
36 // could be a number. No strings yet
37
38 auto pos = mid.find(')');
39 if ( pos == string::npos )
40 return 0;
41 string fieldName = mid.substr( 0, pos );
42 auto pos2 = mid.find_last_of( "=<>" );
43 if ( pos2 == string::npos )
44 return 0;
45 string op = mid.substr( pos + 1, pos2 - pos );
46
47 string testValue = mid.substr( pos2 + 1 );
48
49 if ( testValue.length() == 0 )
50 return 0;
51
52 /*
53 const Finfo* f = id()->findFinfo( fieldName );
54 if ( !f )
55 return 0;
56
57 string actualValue;
58 bool ret = f->strGet( id.eref(), actualValue );
59 */
60 string actualValue;
61
62 bool ret = SetGet::strGet( oid, fieldName, actualValue );
63 if ( ret == 0 )
64 return 0;
65 if ( op == "==" || op == "=" )
66 return ( testValue == actualValue );
67 if ( op == "!=" )
68 return ( testValue != actualValue );
69
70 double v1 = atof( actualValue.c_str() );
71 double v2 = atof( testValue.c_str() );
72 if ( op == ">" )
73 return ( v1 > v2 );
74 if ( op == ">=" )
75 return ( v1 >= v2 );
76 if ( op == "<" )
77 return ( v1 < v2 );
78 if ( op == "<=" )
79 return ( v1 <= v2 );
80
81 return 0;
82}

107/**
108 * This is the basic wildcardFind function, working on a single
109 * tree. It adds entries into the vector 'ret' with Ids found according
110 * to the path string. It preserves the order of the returned Ids
111 * as the order of elements traversed in the search. It does NOT
112 * eliminate duplicates. This is a depth-first search.
113 * Note that it does the dumb but backward compatible thing with
114 * Ids of arrays: it lists every entry.
115 *
116 * It returns the number of Ids found here.
117 */
118int simpleWildcardFind( const string& path, vector< ObjId >& ret)
119{
120 if ( path.length() == 0 )
121 return 0;
122 unsigned int n = ret.size();
123 vector< string > wildcards;
124 Shell::chopString( path, wildcards, ',' );
125 vector< string >::iterator i;
126 for ( i = wildcards.begin(); i != wildcards.end(); ++i )
127 innerFind( *i, ret );
128
129 return ret.size() - n;
130}

132static void myUnique(vector<ObjId>& ret)
133{
134 sort(ret.begin(), ret.end());
135 unsigned int i, j;
136 j = 0;
137 for (i = 1; i < ret.size(); i++)
138 {
139 if (ret[j] != ret[i])
140 {
141 ret[++j] = ret[i];
142 }
143 }
144 j++;
145 if (j < ret.size())
146 ret.resize(j);
147}

165/**
166 * singleLevelWildcard parses a single level of the path and returns all
167 * ids that match it. If there is a suitable doublehash, it will recurse
168 * into child elements.
169 * Returns # of ids found.
170 */
171int singleLevelWildcard( ObjId start, const string& path, vector< ObjId >& ret )
172{
173 if ( path.length() == 0 )
174 return 0;
175 unsigned int nret = ret.size();
176
177 string beforeBrace;
178 string insideBrace;
179 // This has to handle ghastly cases like foo[][FIELD(x)=12.3]
180 unsigned int index = findBraceContent( path, beforeBrace, insideBrace );
181 if ( beforeBrace == "##" )
182 // recursive.
183 return allChildren( start, index, insideBrace, ret );
184
185 vector< Id > kids;
186 Neutral::children( start.eref(), kids );
187 vector< Id >::iterator i;
188 for ( i = kids.begin(); i != kids.end(); i++ )
189 {
190 if ( matchName( ObjId( *i, ALLDATA ),
191 index, beforeBrace, insideBrace ) )
192 {
193 if ( index == ALLDATA )
194 {
195 for ( unsigned int j = 0; j < i->element()->numData(); ++j )
196 ret.push_back( ObjId( *i, j ) );
197 }
198 else if ( i->element()->hasFields() && index < i->element()->numField( start.dataIndex ) )
199 {
200 ret.push_back( ObjId( *i, start.dataIndex, index ) );
201 }
202 else if ( !i->element()->hasFields() && index < i->element()->numData() )
203 {
204 ret.push_back( ObjId( *i, index ) );
205 }
206 }
207 }
208
209 return ret.size() - nret;
210}

212/**
213 * Parses the name and separates out the stuff before the brace,
214 * the stuff inside it, and if present, the index which is also in a
215 * brace. Returns the index, and if not found, zero.
216 * Assume order is
217 * foo[index][insideBrace]
218 * or foo[index]
219 * or foo[insideBrace]
220 * or foo
221 *
222 * Note that for the index, an empty [] means ALLDATA, but the
223 * absence of the braces altogether means zero.
224 *
225 * Note also that if the name ends in the wildcard '#', then we assume
226 * that all indices are OK, unless overridden by a subsequent specific
227 * index via braces.
228 * So foo# gives all indices
229 * but foo#[3] only gives index == 3.
230 */
231unsigned int findBraceContent( const string& path, string& beforeBrace,
232 string& insideBrace )
233{
234 int index = 0;
235 beforeBrace = "";
236 insideBrace = "";
237
238 if ( path.length() == 0 )
239 return 0;
240 vector< string > names;
241 Shell::chopString( path, names, '[' );
242 if ( names.size() == 0 )
243 return 0;
244 if ( names.size() >= 1 )
245 beforeBrace = names[0];
246 unsigned int len = beforeBrace.length();
247 if ( len > 0 && beforeBrace[len -1] == '#' )
248 index = ALLDATA;
249 if ( names.size() >= 2 )
250 {
251 const string& n = names[1];
252 if ( n == "]" ) // A [] construct means use all indices.
253 {
254 index = ALLDATA;
255 }
256 else if ( isdigit( n[0] ) )
257 {
258 index = atoi( n.c_str() );
259 }
260 else // some complicated text construct for the brace
261 {
262 insideBrace = n.substr( 0, n.length() - 1 );
263 return index;
264 }
265 if ( names.size() == 3 ) // name[number][another_string]
266 {
267 string n1 = names[2].substr( 0, names[2].length() - 1 );
268 insideBrace = n1;
269 }
270 }
271 return index;
272}

306/**
307 * matchInsideBrace checks for element property matches
308 * Still has some legacy hacks for reading GENESIS code.
309 */
310bool matchInsideBrace( ObjId id, const string& inside )
311{
312 /* Map from Genesis class names to Moose class names */
313 // const map< string, string >& classNameMap = sliClassNameConvert();
314 if ( inside == "" )
315 return true; // empty means that there is no condition to apply.
316
317 if ( inside.substr(0, 4 ) == "TYPE" ||
318 inside.substr(0, 5 ) == "CLASS" ||
319 inside.substr(0, 3 ) == "ISA" )
320 {
321 auto pos = inside.rfind( "=" );
322 if ( pos == string::npos )
323 return false;
324 bool isEquality = ( inside[ pos - 1 ] != '!' );
325 string typeName = inside.substr( pos + 1 );
326 if ( typeName == "membrane" )
327 typeName = "Compartment";
328
329 if ( inside.substr( 0, 5 ) == "CLASS" && typeName == "channel" )
330 typeName = "HHChannel";
331
332 bool isEqual;
333 if ( inside.substr( 0, 3 ) == "ISA" )
334 {
335 isEqual = id.element()->cinfo()->isA( typeName );
336 }
337 else
338 {
339 isEqual = ( typeName == id.element()->cinfo()->name() );
340 }
341 /*
342 map< string, string >::const_iterator iter = classNameMap.find( typeName );
343 if ( iter != classNameMap.end() )
344 isEqual = ( iter->second == id()->className() );
345 else
346 isEqual = ( typeName == id()->className() );
347 */
348
349 return ( isEqual == isEquality );
350 }
351 else if ( inside.substr( 0, 6 ) == "FIELD(" )
352 {
353 if ( id.dataIndex == ALLDATA )
354 {
355 return wildcardFieldComparison( id.id, inside.substr( 6 ) );
356 }
357 else
358 {
359 return wildcardFieldComparison( id, inside.substr( 6 ) );
360 }
361 }
362
363 return false;
364}

366/**
367 * Returns true if the name matches the wildcard string. Doesn't care about
368 * following characters in 'name'. Single character wildcards are
369 * indicated with '?'
370 */
371bool alignedSingleWildcardMatch( const string& name, const string& wild )
372{
373 unsigned int len = wild.length();
374 if ( name.length() < len )
375 return false;
376 for ( unsigned int i = 0; i < len; i++ )
377 {
378 if ( wild[i] != '?' && name[i] != wild[i] )
379 return false;
380 }
381 return true;
382}

384/**
385 * Returns start index for match between string and wildcard using '?' to
386 * indicate single character matches.
387 * Scans forward along 'name' until it finds a match, or gives up.
388 * The entire wildcard must be matched, otherwise returns ~0U
389 * findWithSingleCharWildcard( "abc", 0, "a?c" ): return 0;
390 * findWithSingleCharWildcard( "xyzabc", 1, "a?c" ): return 3;
391 * findWithSingleCharWildcard( "xyzabb", 1, "a?c" ): return ~0U;
392 */
393unsigned int findWithSingleCharWildcard(
394 const string& name, unsigned int start, const string& wild )
395{
396 unsigned int len = wild.length();
397 if ( len + start > name.length() )
398 return ~0;
399 unsigned int end = 1 + name.length() - len;
400 for ( unsigned int i = start; i < end; ++i )
401 {
402 if ( alignedSingleWildcardMatch( name.substr(i), wild ) )
403 return i;
404 }
405 return ~0;
406}

408/**
409 * matchBeforeBrace checks to see if the wildcard string 'name' matches
410 * up with the name of the id.
411 * Rules:
412 * # may be used at multiple places in the wildcard.
413 * It substitutes for any number of characters.
414 *
415 * ? may be used any number of times in the wildcard, and
416 * must substitute exactly for characters.
417 *
418 * If bracesInName, then the Id name itself includes braces.
419 */
420bool matchBeforeBrace( ObjId id, const string& wild )
421{
422 if ( wild == "#" || wild == "##" )
423 return true;
424
425 string ename = id.element()->getName();
426 if ( wild == ename )
427 return true;
428
429 // Check if the wildcard string has any # or ? symbols.
430 if ( wild.find_first_of( "#?" ) == string::npos )
431 return false;
432
433 // Break the 'wild' into the sections that must match, at the #s.
434 // Then go through each of these sections doing a match to ename.
435 // If not found, then return false.
436 vector< string > chops;
437 Shell::chopString( wild, chops, '#' );
438 unsigned int prev = 0;
439 unsigned int start = 0;
440
441 for ( vector< string >::iterator
442 i = chops.begin(); i != chops.end(); ++i )
443 {
444 start = findWithSingleCharWildcard( ename, prev, *i );
445 if ( start == ~0U )
446 return false;
447 if ( prev == 0 && start > 0 && wild[0] != '#' )
448 return false;
449 prev = start + i->length();
450 }
451 return true;
452
453 /*
454
455 auto pre = name.find( "#" );
456 auto post = name.rfind( "#" );
457
458 // # may only be used once in the wildcard, but substitutes for any
459 // number of characters.
460 if ( pre != string::npos && post == pre ) {
461 if ( ename.length() < ( name.length() - post - 1 ) )
462 return false;
463 unsigned int epos = ename.length() - ( name.length() - post - 1 );
464 return ( name.substr( 0, pre ) == ename.substr( 0, pre ) &&
465 name.substr( post + 1 ) == ename.substr( epos ) );
466 }
467
468 // ? may be used any number of times in the wildcard, and
469 // must substitute exactly for characters.
470 if ( name.length() != ename.length() )
471 return 0;
472 for ( unsigned int i = 0; i < name.length(); i++ )
473 if ( name[i] != '?' && name[i] != ename[i] )
474 return false;
475 return true;
476 */
477}

479/**
480 * Recursive function to compare all descendants and cram matches into ret.
481 * Returns number of matches.
482 */
483int allChildren( ObjId start,
484 unsigned int index, const string& insideBrace, vector< ObjId >& ret )
485{
486 unsigned int nret = ret.size();
487 vector< Id > kids;
488 Neutral::children( start.eref(), kids );
489 vector< Id >::iterator i;
490 for ( i = kids.begin(); i != kids.end(); i++ )
491 {
492 if ( i->element()->hasFields() )
493 {
494 if ( matchInsideBrace( *i, insideBrace ) )
495 {
496 if ( index == ALLDATA )
497 {
498 ObjId oid( *i, start.dataIndex );
499 ret.push_back( oid );
500 }
501 else if (index < i->element()->numField( start.dataIndex ) )
502 {
503 ObjId oid( *i, start.dataIndex, index );
504 ret.push_back( oid );
505 }
506 }
507 }
508 else
509 {
510 for ( unsigned int j = 0; j < i->element()->numData(); ++j )
511 {
512 ObjId oid( *i, j );
513 allChildren( oid, index, insideBrace, ret );
514 if ( (index == ALLDATA || index == j) && matchInsideBrace( oid, insideBrace ) )
515 ret.push_back( oid );
516 }
517 }
518 }
519 return ret.size() - nret;
520}

522/**
523 * This is the main recursive function of the wildcarding scheme.
524 * It builds a wildcard list based on path. Puts found Ids into ret,
525 * and returns # found.
526 * The start ObjId is one that already matches.
527 * depth is the position on the path.
528 * This should work for multi-node wildcard searches since it only
529 * refers to messaging and basic Element information that is present on
530 * all nodes.
531 */
532int wildcardRelativeFind( ObjId start, const vector< string >& path,
533 unsigned int depth, vector< ObjId >& ret )
534{
535 int nret = 0;
536 vector< ObjId > currentLevelIds;
537 if ( depth == path.size() )
538 {
539 if ( ret.size() == 0 || ret.back() != start )
540 {
541 ret.push_back( start );
542 }
543 return 1;
544 }
545
546 if ( singleLevelWildcard( start, path[depth], currentLevelIds ) > 0 )
547 {
548 vector< ObjId >::iterator i;
549 for ( i = currentLevelIds.begin(); i != currentLevelIds.end(); ++i )
550 nret += wildcardRelativeFind( *i, path, depth + 1, ret );
551 }
552 return nret;
553}

555void wildcardTestFunc( ObjId* elist, unsigned int ne, const string& path )
556{
557 vector< ObjId > ret;
558 simpleWildcardFind( path, ret );
559 if ( ne != ret.size() )
560 {
561 cout << "!\nAssert '" << path << "' : expected " <<
562 ne << ", found " << ret.size() << "\n";
563 assert( 0 );
564 }
565 sort( ret.begin(), ret.end() );
566 for ( unsigned int i = 0; i < ne ; i++ )
567 {
568 if ( elist[ i ] != ret[ i ] )
569 {
570 cout << "!\nAssert " << path << ": item " << i <<
571 ": " << elist[i].element()->getName() << " != " <<
572 ret[i].element()->getName() << "\n";
573 assert( 0 );
574 }
575 }
576 cout << ".";
577}
