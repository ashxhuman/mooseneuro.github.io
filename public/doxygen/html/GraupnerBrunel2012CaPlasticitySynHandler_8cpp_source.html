344weightFactors GraupnerBrunel2012CaPlasticitySynHandler::updateCaWeightFactors( double currTime )
345{
346 double CaOld = Ca_;
347 double deltaT = currTime-lastCaUpdateTime_;
348 // Update Ca, but CaPre and CaPost are added in vProcess()
349 Ca_ *= exp(-deltaT/tauCa_);
350 lastCaUpdateTime_ = currTime;
351 weightFactors wUp; // by default all are set to 0.0
352
353 // calculate/approximate time spent above potentiation and depression thresholds
354 // see pg 13 of Higgins et al | October 2014 | Volume 10 | Issue 10 | e1003834 | PLOS Comp Biol
355 // starting from bottom condition, going upwards in the algorithm given in above paper
356 if (CaOld <= thetaD_)
357 {
358 }
359 else if (CaOld <= thetaP_)
360 {
361 //cout << "tD<Caold<tP" << "\n";
362 if (Ca_ <= thetaD_)
363 {
364 wUp.tD = tauCa_*log(CaOld/thetaD_);
365 //cout << "Ca<tD" << "\n";
366 }
367 else
368 {
369 wUp.tD = deltaT;
370 //cout << "Ca>tD" << "\n";
371 }
372 }
373 else
374 {
375 //cout << "Caold>tP" << "\n";
376 if (Ca_ <= thetaD_)
377 {
378 wUp.tP = tauCa_*log(CaOld/thetaP_);
379 wUp.tD = tauCa_*log(thetaP_/thetaD_);
380 //cout << "Ca<tD" << "\n";
381 //cout << "Caold = " << CaOld << "thetaP = " << thetaP_ << "\n";
382 }
383 else if (Ca_ <= thetaP_)
384 {
385 wUp.tP = tauCa_*log(CaOld/thetaP_);
386 wUp.tD = deltaT - wUp.tP;
387 //cout << "Ca<tP" << "\n";
388 }
389 else
390 {
391 wUp.tP = deltaT;
392 //cout << "Ca>tP" << "\n";
393 }
394 }
395 wUp.t0 = deltaT - wUp.tP - wUp.tD;
396
397 // Depending on tP and tD, I return A,B,C factors for weight update
398 // (see page 13 of Higgins et al 2014).
399 // A,B,C,D,E are used to compute the weight change for one or multiple synapses
400 // A is weight independent, B,D are weight dependent and C,E are accumulated noise
401 if (wUp.tP > 0)
402 {
403 double gPgD = gammaP_+gammaD_;
404 wUp.A = gammaP_/gPgD*(1.0-exp(-wUp.tP*gPgD/tauSyn_));
405 wUp.B = exp(-wUp.tP*gPgD/tauSyn_);
406 if (noisy_)
407 {
408 wUp.C = noiseSD_ * dist_(rng_) *
409 sqrt( ( 1.0-exp(-2*gPgD*wUp.tP/tauSyn_) ) / gPgD );
410 // cout << " A = " << wUp.A << " B = " << wUp.B << " C = " << wUp.C << "\n";
411 }
412 else
413 {
414 wUp.C = 0.0;
415 }
416 }
417
418 if (wUp.tD > 0)
419 {
420 wUp.D = exp(-wUp.tD*gammaD_/tauSyn_);
421 if (noisy_)
422 {
423 wUp.E = noiseSD_ * dist_( rng_ ) *
424 sqrt( ( 1.0-exp(-2*gammaD_*wUp.tD/tauSyn_) ) / 2.0/gammaD_ );
425 //cout << "D = " << wUp.D << " E = " << wUp.E << "\n";
426 }
427 else
428 {
429 wUp.E = 0.0;
430 }
431 }
432
433 //cout << currTime << " tD = " << wUp.tD << " tP = " << wUp.tP << "\n";
434 // tP, tD, A, B, C, D, E of wUp
435 // are set to 0.0 by default in struct constructor
436 return wUp; // return by value, i.e. copies the struct, so wUp going out of scope doesn't matter
437 // malloc and returning pointer is more expensive for small structs
438 // see: http://stackoverflow.com/questions/9590827/is-it-safe-to-return-a-struct-in-c-c
439}

441void GraupnerBrunel2012CaPlasticitySynHandler::updateWeight( Synapse* synPtr, weightFactors *wFacPtr )
442{
443 double newWeight = synPtr->getWeight();
444 //cout << " oldweight = " << newWeight << "\n";
445 if (wFacPtr->tP > 0.0)
446 {
447 newWeight = wFacPtr->A + wFacPtr->B*newWeight + wFacPtr->C;
448 //cout << " midweight = " << newWeight << "\n";
449 }
450 if (wFacPtr->tD > 0.0)
451 {
452 newWeight = wFacPtr->D*newWeight + wFacPtr->E; // update the weight again
453 //cout << " newweight = " << newWeight << "\n";
454 }
455
456 // potential is usually ignorable when t0 is small,
457 // i.e. Ca is mostly above thetaD or thetaP
458 //cout << "before bistable newWeight = " << newWeight << "\n";
459 if (bistable_)
460 {
461 double chi0 = pow((newWeight-0.5),2.0) / (newWeight*(newWeight-1));
462 double weightDeviation = 0.5*sqrt( 1.0 + 1.0/(chi0*exp(wFacPtr->t0/2.0/tauSyn_)-1.0) );
463 if (newWeight<0.5)
464 {
465 newWeight = 0.5 - weightDeviation;
466 }
467 else
468 {
469 newWeight = 0.5 + weightDeviation;
470 }
471 }
472 //cout << "after bistable newWeight = " << newWeight << "\n";
473
474 // clip weight within [weightMin,weightMax]
475 newWeight = std::max(weightMin_, std::min(newWeight, weightMax_));
476 //cout << " clipweight = " << newWeight << "\n";
477 //cout << " A = " << wFacPtr->A << " B = " << wFacPtr->B << " C = " << wFacPtr->C
478 // << " D = " << wFacPtr->D << " E = " << wFacPtr->E << " newW = "<< newWeight << "\n";
479 synPtr->setWeight( newWeight );
480}

482void GraupnerBrunel2012CaPlasticitySynHandler::vProcess( const Eref& e, ProcPtr p )
483{
484 double activation = 0.0;
485 double currTime = p->currTime;
486 bool CaFactorsUpdated = false; // Ca-decay and weight Factors updation is done only once
487 // if any pre-spike, pre-spike+delayD or post-spike occurs
488 // Ca is bumped by CaPre and CaPost on
489 // pre-spike+delayD and post-spike respectively
490 // At the end, if any event above occurred,
491 // change all pre-synaptic weights based on Ca
492 weightFactors wFacs;
493
494 // process pre-synaptic spike events for activation, Ca and weight update
495 while( !events_.empty() && events_.top().time <= currTime )
496 {
497 PreSynEvent currEvent = events_.top();
498
499 unsigned int synIndex = currEvent.synIndex;
500 // Warning, coder! 'STDPSynapse currSyn = synapses_[synIndex];' is wrong,
501 // it creates a new, shallow-copied object.
502 // We want only to point to the same object.
503 Synapse* currSynPtr = &synapses_[synIndex];
504
505 // activate the synapse for every pre-spike
506 // If the synapse has a delay, the weight could be updated during the delay!
507 // currEvent.weight is the weight before the delay!
508 // Might be better to use currSynPtr->getWeight()
509 // or even the latest updated weight below?!
510 // Using currSynPtr->getWeight().
511 // Still, weight update is event driven and not continuous
512 // so using currSynPtr->getWeight() takes care of
513 // weight updates due to events/spikes during the delay
514 // but doesn't take care of continuous weight evolution during the delay.
515 // The weight update is done
516 // after sending the current weight to activation
517 // Maybe I should use the updated weight to send to activation?!
518
519 // See: http://www.genesis-sim.org/GENESIS/Hyperdoc/Manual-26.html#synchan
520 // Send out weight / dt for every spike
521 // Since it is an impulse active only for one dt,
522 // need to send it divided by dt.
523 // Can connect activation to SynChan (double exp)
524 // or to LIF as an impulse to voltage.
525 //activation += currEvent.weight * weightScale_ / p->dt;
526 activation += currSynPtr->getWeight() * weightScale_ / p->dt;
527
528 // update only once for this time-step if an event occurs
529 if (!CaFactorsUpdated)
530 {
531 // update Ca and weightFactors
532 wFacs = updateCaWeightFactors( currTime );
533 CaFactorsUpdated = true;
534 }
535
536 events_.pop();
537 }
538 if ( activation != 0.0 )
539 SynHandlerBase::activationOut()->send( e, activation );
540
541 // process delayed pre-synaptic spike events for Ca and weight update
542 // delayD after pre-spike accounts for NMDA rise time
543 while( !delayDPreEvents_.empty() && delayDPreEvents_.top().time <= currTime )
544 {
545 // Update Ca, and add CaPre
546 // update only once for this time-step if an event occurs
547 if (!CaFactorsUpdated)
548 {
549 // update Ca and weightFactors
550 wFacs = updateCaWeightFactors( currTime );
551 CaFactorsUpdated = true;
552 }
553 Ca_ += CaPre_;
554
555 delayDPreEvents_.pop();
556 }
557
558 // process post-synaptic spike events for Ca and weight update
559 while( !postEvents_.empty() && postEvents_.top().time <= currTime )
560 {
561 // update Ca, then add CaPost
562 // update only once for this time-step if an event occurs
563 if (!CaFactorsUpdated)
564 {
565 // update Ca and weightFactors
566 wFacs = updateCaWeightFactors( currTime );
567 CaFactorsUpdated = true;
568 }
569 Ca_ += CaPost_;
570
571 postEvents_.pop();
572 }
573
574 // If any event has happened, update all pre-synaptic weights
575 // If you want individual Ca for each pre-synapse
576 // create individual SynHandlers for each
577 if (CaFactorsUpdated)
578 {
579 // Change weight of all synapses
580 for (unsigned int i=0; i<synapses_.size(); i++)
581 {
582 // Warning, coder! 'Synapse currSyn = synapses_[i];' is wrong,
583 // it creates a new, shallow-copied object.
584 // We want only to point to the same object.
585 Synapse* currSynPtrHere = &synapses_[i];
586 updateWeight(currSynPtrHere,&wFacs);
587 }
588 }
