303static double integ( double myN, double rf, double rb, double dt )
304{
305 const double EPSILON = 1e-12;
306 if ( myN > EPSILON && rf > EPSILON )
307 {
308 double C = exp( -rf * dt / myN );
309 myN *= C + ( rb / rf ) * ( 1.0 - C );
310 }
311 else
312 {
313 myN += ( rb - rf ) * dt;
314 }
315 if ( myN < 0.0 )
316 return 0.0;
317 return myN;
318}

320void Dsolve::calcJnDiff( const DiffJunction& jn, Dsolve* other, double dt)
321{
322 const double EPSILON = 1e-16;
323 assert( jn.otherPools.size() == jn.myPools.size() );
324 for ( unsigned int i = 0; i < jn.myPools.size(); ++i )
325 {
326 DiffPoolVec& myDv = pools_[ jn.myPools[i] ];
327 if ( myDv.getDiffConst() < EPSILON )
328 continue;
329 DiffPoolVec& otherDv = other->pools_[ jn.otherPools[i] ];
330 if ( otherDv.getDiffConst() < EPSILON )
331 continue;
332 // This geom mean is used in case we have the odd situation of
333 // different diffusion constants.
334 double effectiveDiffConst =
335 sqrt( myDv.getDiffConst() * otherDv.getDiffConst() );
336
337 for (auto j = jn.vj.cbegin(); j != jn.vj.end(); ++j )
338 {
339 double myN = myDv.getN( j->first );
340 double otherN = otherDv.getN( j->second );
341 // Here we do an exp Euler calculation
342 // rf is rate from self to other.
343 // double k = myDv.getDiffConst() * j->diffScale;
344 double k = effectiveDiffConst * j->diffScale;
345 double lastN = myN;
346 myN = integ( myN,
347 k * myN / j->firstVol,
348 k * otherN / j->secondVol,
349 dt
350 );
351 otherN += lastN - myN; // Simple mass conservation
352 if ( otherN < 0.0 ) // Avoid negatives
353 {
354 myN += otherN;
355 otherN = 0.0;
356 }
357 myDv.setN( j->first, myN );
358 otherDv.setN( j->second, otherN );
359 }
360 }

363void Dsolve::calcJnXfer( const DiffJunction& jn,
364 const vector< unsigned int >& srcXfer,
365 const vector< unsigned int >& destXfer,
366 Dsolve* srcDsolve, Dsolve* destDsolve )
367{
368 assert( destXfer.size() == srcXfer.size() );
369 for ( unsigned int i = 0; i < srcXfer.size(); ++i )
370 {
371 DiffPoolVec& srcDv = srcDsolve->pools_[ srcXfer[i] ];
372 DiffPoolVec& destDv = destDsolve->pools_[ destXfer[i] ];
373 for ( vector< VoxelJunction >::const_iterator
374 j = jn.vj.begin(); j != jn.vj.end(); ++j )
375 {
376 double prevSrc = srcDv.getPrev( j->first );
377 double prevDest = destDv.getPrev( j->second );
378 double srcN = srcDv.getN( j->first );
379 double destN = destDv.getN( j->second );
380 // Consider delta as sum of local dN, and reference as prevDest
381 // newN = (srcN - prevSrc + destN - prevDest) + prevDest
382 double newN = srcN + destN - prevSrc;
383 srcDv.setN( j->first, newN );
384 destDv.setN( j->second, newN );
385 }
386 }

389void Dsolve::calcJnChan( const DiffJunction& jn, Dsolve* other, double dt )
390{
391 // Each jn has some channels
392 // Each channel has a chanPool, an intPool and an extPool.
393 // chanPool and intPool must be on self, extPool is on other. In
394 // cases where the intPool is on other, it attempts to swap the
395 // int and ext pools, but this too could fail
396 // because the chanPool could be a third compartment, such as the memb
397 //
398 // Don't have a solution for this case as yet.
399 // Other alternative is to have a message to update the N of the chan,
400 // so it isn't in the domain of the solver at all except for here.
401 // In which case we will want to point to the Moose object for it.
402 //
403
404 for ( unsigned int i = 0; i < jn.myChannels.size(); ++i )
405 {
406 if ( channels_[ jn.myChannels[i] ].isLocal )
407 continue;
408 ConcChanInfo& myChan = channels_[ jn.myChannels[i] ];
409 DiffPoolVec& myDv = pools_[ myChan.myPool ];
410 DiffPoolVec& otherDv = other->pools_[ myChan.otherPool ];
411 DiffPoolVec& chanDv = pools_[ myChan.chanPool ];
412 for ( vector< VoxelJunction >::const_iterator
413 j = jn.vj.begin(); j != jn.vj.end(); ++j )
414 {
415
416 double myN = myDv.getN( j->first );
417 double lastN = myN;
418 double otherN = otherDv.getN( j->second );
419 double chanN = chanDv.getN( j->first );
420 // Stick in a conversion factor for the myN and otherN into
421 // concentrations. Note that SI is millimolar.
422 double perm = myChan.permeability * chanN / NA;
423 myN = integ( myN, perm * myN/j->firstVol,
424 perm * otherN/j->secondVol, dt );
425 otherN += lastN - myN; // Mass consv
426 if ( otherN < 0.0 ) // Avoid negatives
427 {
428 myN += otherN;
429 otherN = 0.0;
430 }
431 myDv.setN( j->first, myN );
432 otherDv.setN( j->second, otherN );
433 }
434 }
435}

437// Same as above, but now go through channels on other Dsolve.
438void Dsolve::calcOtherJnChan( const DiffJunction& jn, Dsolve* other, double dt )
439{
440 for ( unsigned int i = 0; i < jn.otherChannels.size(); ++i )
441 {
442 if ( other->channels_[ jn.otherChannels[i] ].isLocal )
443 continue;
444 ConcChanInfo& otherChan = other->channels_[ jn.otherChannels[i] ];
445 // This is the DiffPoolVec for the pools on the other Dsolve,
446 // the one with the channel.
447 // DiffPoolVec& otherDv = other->pools_[ jn.otherPools[otherChan.myPool] ];
448 DiffPoolVec& otherDv = other->pools_[ otherChan.myPool ];
449 // Local diffPoolVec.
450 // DiffPoolVec& myDv = pools_[ jn.myPools[otherChan.otherPool] ];
451 DiffPoolVec& myDv = pools_[ otherChan.otherPool ];
452 DiffPoolVec& chanDv = other->pools_[ otherChan.chanPool ];
453 for ( vector< VoxelJunction >::const_iterator
454 j = jn.vj.begin(); j != jn.vj.end(); ++j )
455 {
456
457 double myN = myDv.getN( j->first );
458 double lastN = myN;
459 double otherN = otherDv.getN( j->second );
460 double chanN = chanDv.getN( j->second );
461 // Stick in a conversion factor for the myN and otherN into
462 // concentrations. Note that SI is millimolar.
463 double perm = otherChan.permeability * chanN / NA;
464 myN = integ( myN, perm * myN/j->firstVol,
465 perm * otherN/j->secondVol, dt );
466 otherN += lastN - myN; // Mass consv
467 if ( otherN < 0.0 ) // Avoid negatives
468 {
469 myN += otherN;
470 otherN = 0.0;
471 }
472 myDv.setN( j->first, myN );
473 otherDv.setN( j->second, otherN );
474 }
475 }

478void Dsolve::calcLocalChan( double dt )
479{
480 // There may be some chans which connect within the compartment.
481 // This is odd biologically, but happens for modelling convenience.
482 // The channels have a flag indicating if this is so. It is a simple
483 // calculation, but lacks numerical accuracy as it is outside the
484 // solver framework.
485
486 ChemCompt* cc = reinterpret_cast< ChemCompt* >( compartment_.eref().data() );
487 const vector< double >& vols = cc->vGetVoxelVolume();
488 for (auto ch = channels_.begin(); ch != channels_.end(); ++ch )
489 {
490 if ( ch->isLocal )
491 {
492 DiffPoolVec& myDv = pools_[ ch->myPool ];
493 DiffPoolVec& otherDv = pools_[ ch->otherPool ];
494 DiffPoolVec& chanDv = pools_[ ch->chanPool ];
495 assert( myDv.getNumVoxels() == numVoxels_ );
496 for (unsigned int i = 0; i < numVoxels_; ++i )
497 {
498 double myN = myDv.getN( i );
499 double lastN = myN;
500 double otherN = otherDv.getN( i );
501 double chanN = chanDv.getN( i );
502 double vol = vols[i];
503 // Stick in a conversion factor for the myN and otherN into
504 // concentrations. Note that SI is millimolar.
505 double perm = ch->permeability * chanN / NA;
506 myN = integ( myN, perm * myN/vol, perm * otherN/vol, dt );
507 otherN += lastN - myN; // Mass consv
508 if ( otherN < 0.0 ) // Avoid negatives
509 {
510 myN += otherN;
511 otherN = 0.0;
512 }
513 myDv.setN( i, myN );
514 otherDv.setN( i, otherN );
515 }
516 }
517 }
518}

521/**
522 * Computes flux through a junction between diffusion solvers.
523 * Most used at junctions on spines and PSDs, but can also be used
524 * when a given diff solver is decomposed. At present the lookups
525 * on the other diffusion solver assume that the data is on the local
526 * node. Once this works well I can figure out how to do across nodes.
527 * Note that we split the diffusion and channel calculations before and
528 * after then calcJnXfer calculations. This improves accuracy by 5x.
529 */
530void Dsolve::calcJunction( const DiffJunction& jn, double dt )
531{
532 Id oid( jn.otherDsolve );
533 assert ( oid != Id() );
534 assert ( oid.element()->cinfo()->isA( "Dsolve" ) );
535
536 Dsolve* other = reinterpret_cast< Dsolve* >( oid.eref().data() );
537 calcJnDiff( jn, other, dt/2.0 );
538
539 calcJnChan( jn, other, dt/2.0 );
540 calcOtherJnChan( jn, other, dt/2.0 );
541
542 calcJnXfer( jn, jn.myXferSrc, jn.otherXferDest, this, other );
543 calcJnXfer( jn, jn.otherXferSrc, jn.myXferDest, other, this );
544
545 calcJnDiff( jn, other, dt/2.0 );
546
547 calcJnChan( jn, other, dt/2.0 );
548 calcOtherJnChan( jn, other, dt/2.0 );
549}

629void Dsolve::fillConcChans( const vector< ObjId >& chans )
630{
631 static const Cinfo* ccc = Cinfo::find( "ConcChan" );
632 static const Finfo* inPoolFinfo = ccc->findFinfo( "inPool" );
633 static const Finfo* outPoolFinfo = ccc->findFinfo( "outPool" );
634 static const Finfo* chanPoolFinfo = ccc->findFinfo( "setNumChan" );
635 FuncId fin = static_cast< const DestFinfo* >( inPoolFinfo )->getFid();
636 FuncId fout = static_cast< const DestFinfo* >(outPoolFinfo )->getFid();
637 FuncId fchan =
638 static_cast< const DestFinfo* >(chanPoolFinfo )->getFid();
639
640 // Find the in pools and the chan pools on the current compt.
641 // Save the Id of the outPool as an integer.
642 // Use these and the permeability to create the ConcChanInfo.
643 for ( auto i = chans.begin(); i != chans.end(); ++i )
644 {
645 vector< Id > ret;
646 if (i->element()->getNeighbors( ret, inPoolFinfo ) == 0 ) return;
647 ObjId inPool( ret[0] );
648 ret.clear();
649 if (i->element()->getNeighbors( ret, outPoolFinfo ) == 0 ) return;
650 ObjId outPool( ret[0] );
651 ret.clear();
652 if (i->element()->getNeighbors( ret, chanPoolFinfo ) == 0 ) return;
653 ObjId chanPool( ret[0] );
654 ret.clear();
655 unsigned int outPoolValue = outPool.id.value();
656 bool swapped = false;
657 bool isLocal = false;
658 if ( !( inPool.bad() || chanPool.bad() ) )
659 {
660 unsigned int inPoolIndex = convertIdToPoolIndex( inPool.id );
661 unsigned int chanPoolIndex = convertIdToPoolIndex(chanPool.id);
662 if ( inPoolIndex == ~0U ) // Swap in and out as chan is symm
663 {
664 inPoolIndex = convertIdToPoolIndex( outPool.id );
665 outPoolValue = inPool.id.value();
666 swapped = true;
667 }
668 else
669 {
670 unsigned int outPoolIndex = convertIdToPoolIndex( outPool.id );
671 // edge case where both in and out are on same compartment.
672 // Not much sense in bio, but a common convenience hack.
673 if ( outPoolIndex != ~0U )
674 {
675 outPoolValue = outPoolIndex;
676 isLocal = true;
677 }
678
679 }

802/**
803 * build: This function is called either by setStoich or setPath.
804 * By this point the diffusion consts etc will be assigned to the
805 * poolVecs.
806 * This requires
807 * - Stoich should be assigned, OR
808 * - A 'path' should be assigned which has been traversed to find pools.
809 * - compartment should be assigned so we know how many voxels.
810 * - If Stoich, its 'path' should be set so we know numPools. It needs
811 * to know the numVoxels from the compartment. At the time of
812 * path setting the zombification is done, which takes the Id of
813 * the solver.
814 * - After this build can be done. Just reinit doesn't make sense since
815 * the build does a lot of things which should not be repeated for
816 * each reinit.
817 */
818
819void Dsolve::build( double dt, const MeshCompt *m )
820{
821 if ( doubleEq( dt, dt_ ) )
822 return;
823 if ( compartment_ == Id() )
824 {
825 cout << "Dsolve::build: Warning: No compartment defined. \n"
826 "Did you forget to assign 'stoich.dsolve = this' ?\n";
827 return;
828 }
829 dt_ = dt;
830 unsigned int numVoxels = m->getNumEntries();
831
832 for ( unsigned int i = 0; i < numLocalPools_; ++i )
833 {
834 bool debugFlag = false;
835 vector< unsigned int > diagIndex;
836 vector< double > diagVal;
837 vector< Triplet< double > > fops;
838 FastMatrixElim elim( numVoxels, numVoxels );
839 if ( elim.buildForDiffusion(
840 m->getParentVoxel(), m->getVoxelVolume(),
841 m->getVoxelArea(), m->getVoxelLength(),
842 pools_[i].getDiffConst(), pools_[i].getMotorConst(), dt ) )
843 {
844 vector< unsigned int > parentVoxel = m->getParentVoxel();
845 assert( elim.checkSymmetricShape() );
846 vector< unsigned int > lookupOldRowsFromNew;
847 elim.hinesReorder( parentVoxel, lookupOldRowsFromNew );
848 assert( elim.checkSymmetricShape() );
849 pools_[i].setNumVoxels( numVoxels_ );
850 elim.buildForwardElim( diagIndex, fops );
851 elim.buildBackwardSub( diagIndex, fops, diagVal );
852 elim.opsReorder( lookupOldRowsFromNew, fops, diagVal );
853 if (debugFlag )
854 elim.print();
855 }
856 pools_[i].setOps( fops, diagVal );
857 }
858}

954 * static void mapXfersBetweenDsolves(...)
955 * Build a list of all the molecules that should transfer instantaneously
956 * to another compartment, for cross-compartment reactions.
957 * Here we look for src names of the form <name>_xfer_<destComptName>
958 * For example, if we had an enzyme in compartment 'src', whose product

960 * the name of the enzyme product in compartment src would be
961 * bar_xfer_dest

977 string poolName = pool.element()->getName();
