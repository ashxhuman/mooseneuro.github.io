215void Spine::setShaftLength( const Eref& e, double len )
216{
217 if ( len < minimumSize_ )
218 len = minimumSize_;
219 else if ( len > maximumSize_ )
220 len = maximumSize_;
221 vector< Id > sl = parent_->spineIds( e.fieldIndex() );
222 if ( sl.size() > 1 &&
223 sl[0].element()->cinfo()->isA( "CompartmentBase" ) )
224 {
225 double origDia = Field< double >::get( sl[0], "diameter" );
226 double dx = Field< double >::get( sl[0], "x" );
227 double dy = Field< double >::get( sl[0], "y" );
228 double dz = Field< double >::get( sl[0], "z" );
229 SetGet2< double, double >::set(
230 sl[0], "setGeomAndElec", len, origDia );
231
232 dx = Field< double >::get( sl[0], "x" ) - dx;
233 dy = Field< double >::get( sl[0], "y" ) - dy;
234 dz = Field< double >::get( sl[0], "z" ) - dz;
235
236 SetGet3< double, double, double >::set( sl[1], "displace",
237 dx, dy, dz );
238 // Here we've set the electrical and geometrical stuff. Now to
239 // do the diffusion. Chem doesn't come into the picture for the
240 // spine shaft.
241 // Assume the scaleDiffusion function propagates changes into the
242 // VoxelJunctions used by the Dsolve.
243 parent_->scaleShaftDiffusion( e.fieldIndex(), len, origDia );
244 }
245}

284void Spine::setHeadLength( const Eref& e, double len )
285{
286 if ( len < minimumSize_ )
287 len = minimumSize_;
288 else if ( len > maximumSize_ )
289 len = maximumSize_;
290
291 vector< Id > sl = parent_->spineIds( e.fieldIndex() );
292 if ( sl.size() > 1 &&
293 sl[1].element()->cinfo()->isA( "CompartmentBase") )
294 {
295 double origDia = Field< double >::get( sl[1], "diameter" );
296 double origLen = Field< double >::get( sl[1], "length" );
297 SetGet2< double, double >::set(
298 sl[1], "setGeomAndElec", len, origDia );
299 // Here we've set the electrical and geometrical stuff. Now to
300 // do the diffusion.
301 // Assume the scaleDiffusion function propagates changes into the
302 // VoxelJunctions used by the Dsolve.
303 parent_->scaleHeadDiffusion( e.fieldIndex(), len, origDia );
304 // Now scale the chem stuff. The PSD mesh is assumed to scale only
305 // with top surface area of head, so it is not affected here.
306 parent_->scaleBufAndRates( e.fieldIndex(), len/origLen, 1.0 );
307 }

361// Handle like a cylinder of equal length and dia. vol = PI*dia*dia*len/4
362void Spine::setHeadVolume( const Eref& e, double volume )
363{
364 if ( volume < 0 )
365 volume = 0.0;
366 double dia = pow( volume * 4.0 / PI, 1.0/3.0 );
367 if ( dia < minimumSize_ )
368 volume = pow( minimumSize_, 3.0 ) * PI / 4.0;
369 else if ( dia > maximumSize_ )
370 volume = pow( maximumSize_, 3.0 ) * PI / 4.0;
371
372 vector< Id > sl = parent_->spineIds( e.fieldIndex() );
373 if ( sl.size() > 1 &&
374 sl[0].element()->cinfo()->isA( "CompartmentBase") )
375 {
376 double origLen = Field< double >::get( sl[1], "length" );
377 double origDia = Field< double >::get( sl[1], "diameter" );
378 double oldVolume = origLen * origDia * origDia * PI / 4.0;
379 double ratio = pow( volume / oldVolume, 1.0/3.0 );
380
381 SetGet2< double, double >::set(
382 sl[1], "setGeomAndElec", origLen * ratio, origDia * ratio );
383 parent_->scaleHeadDiffusion( e.fieldIndex(), origLen * ratio, origDia * ratio );
384 parent_->scaleBufAndRates( e.fieldIndex(), ratio, ratio );
385 }

393void Spine::setTotalLength( const Eref& e, double len )
394{
395 double shaftLen = getShaftLength( e );
396 double headLen = getHeadLength( e );
397 double totLen = shaftLen + headLen;
398
399 shaftLen *= len / totLen;
400 headLen *= len / totLen;
401
402 /// Don't set this if either of them is out of range.
403 if ( shaftLen < minimumSize_ || shaftLen > maximumSize_ ||
404 headLen < minimumSize_ || headLen > maximumSize_ )
405 return;
406
407 setShaftLength( e, shaftLen );
408 setHeadLength( e, headLen );
409}
