23/**
24 * Returns the dot product of the specified row with the
25 * vector v. v corresponds to the vector of reaction rates.
26 * v must have nColumns entries.
27 */
28double KinSparseMatrix::computeRowRate(
29 unsigned int row, const vector< double >& v
30) const
31{
32 assert( nColumns() == 0 || row < nRows() );
33 assert( v.size() == nColumns() );
34 const int* entry = 0;
35 const unsigned int* colIndex = 0;
36 unsigned int numEntries = getRow( row, &entry, &colIndex );
37 const int* end = entry + numEntries;
38
39 double ret = 0.0;
40 for ( const int* i = entry; i != end; ++i )
41 {
42 ret += *i * v[ *colIndex++ ];
43 assert(! std::isnan(ret));
44 }
45 //assert ( !std::isnan(ret) );
46 return ret;
47}

50/**
51 * Has to operate on transposed matrix
52 * row argument refers to reac# in this transformed situation.
53 * Fills up 'deps' with reac#s that depend on the row argument.
54 * Does NOT ensure that list is unique.
55 */
56void KinSparseMatrix::getGillespieDependence(
57 unsigned int row, vector< unsigned int >& deps
58) const
59{
60 deps.resize( 0 );
61 // vector< unsigned int > deps;
62 for ( unsigned int i = 0; i < nrows_; ++i )
63 {
64 // i is index for reac # here. Note that matrix is transposed.
65 unsigned int j = rowStart_[ row ];
66 unsigned int jend = rowStart_[ row + 1 ];
67 unsigned int k = rowStart_[ i ];
68 unsigned int kend = rowStart_[ i + 1 ];
69
70 while ( j < jend && k < kend )
71 {
72 if ( colIndex_[ j ] == colIndex_[ k ] )
73 {
74 /* Pre 28 Nov 2015. Why below zero? Shouldn't it be any?
75 if ( N_[ k ] < 0 ) {
76 deps.push_back( i );
77 }
78 */
79 assert( round( N_[k] ) != 0 );
80 deps.push_back( i );
81 ++j;
82 ++k;
83 }
84 else if ( colIndex_[ j ] < colIndex_[ k ] )
85 {
86 ++j;
87 }
88 else if ( colIndex_[ j ] > colIndex_[ k ] )
89 {
90 ++k;
91 }
92 else
93 {
94 assert( 0 );
95 }
96 }
97 }
98}

100/**
101 * This too operates on the transposed matrix, because we need to get all
102 * the molecules for a given reac: a column in the original N matrix.
103 * Direction [-1,+1] specifies whether the reaction is forward or backward.

105void KinSparseMatrix::fireReac( unsigned int reacIndex, vector< double >& S, double direction )
106const
107{
108 assert( ncolumns_ == S.size() && reacIndex < nrows_ );
109 unsigned int rowBeginIndex = rowStart_[ reacIndex ];
110 // vector< int >::const_iterator rowEnd = N_.begin() + rowStart_[ reacIndex + 1];
111 vector< int >::const_iterator rowBegin =
112 N_.begin() + rowBeginIndex;
113 vector< int >::const_iterator rowEnd =
114 N_.begin() + rowTruncated_[ reacIndex ];
115 vector< unsigned int >::const_iterator molIndex =
116 colIndex_.begin() + rowBeginIndex;
117
118 for ( vector< int >::const_iterator i = rowBegin; i != rowEnd; ++i )
119 {
120 double& x = S[ *molIndex++ ];
121 x += *i * direction;
122 x *= (x > 0 );
123 // assert( S[ *molIndex ] + *i * direction >= 0.0 );
124 // S[ *molIndex++ ] += *i * direction;

126}

128/**
129 * This function generates a new internal list of rowEnds, such that
130 * they are all less than the maxColumnIndex.
131 * It is used because in fireReac we don't want to update all the
132 * molecules, only those that are variable.
133 */
134void KinSparseMatrix::truncateRow( unsigned int maxColumnIndex )
135{
136 rowTruncated_.resize( nrows_, 0 );
137 if ( colIndex_.size() == 0 )
138 return;
139 for ( unsigned int i = 0; i < nrows_; ++i )
140 {
141 unsigned int endCol = rowStart_[ i ];
142 for ( unsigned int j = rowStart_[ i ];
143 j < rowStart_[ i + 1 ]; ++j )
144 {
145 if ( colIndex_[ j ] < maxColumnIndex )
146 {
147 endCol = j + 1;
148 }
149 else
150 {
151 break;
152 }
153 }
154 rowTruncated_[ i ] = endCol;
155 }
156}
