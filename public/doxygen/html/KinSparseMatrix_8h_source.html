13class KinSparseMatrix: public SparseMatrix< int >
14{
15public:
16
17 /**
18 * Returns all non-zero column indices, for the specified row.
19 * This gives reac #s in orig matrix, and molecule #s in the
20 * transposed matrix
21 * Not needed. The getRow function does all this, more efficiently.
22 int getRowIndices(
23 unsigned int row, vector< unsigned int >& indices );
24 */
25
26 /**
27 * Returns the dot product of the specified row with the
28 * vector v. v corresponds to the vector of reaction rates.
29 * v must have nColumns entries.
30 */
31 double computeRowRate(
32 unsigned int row, const vector< double >& v
33 ) const;
34
35
36 /**
37 * Does a special self-product of the specified row. Output
38 * is the set of nonzero indices in the product
39 * abs( Rij ) * neg( Rjk ) for the specified index i, where
40 * neg( val ) is true only if val < 0.
41 */
42 void getGillespieDependence(
43 unsigned int row, vector< unsigned int >& cols
44 ) const;
45
46 /**
47 * Fires a stochastic reaction: It undergoes a single transition
48 * This operation updates the mol concs due to the reacn.
49 * Direction is +1 or -1, specifies direction of reaction
50 */
51 void fireReac( unsigned int reacIndex, vector< double >& S,
52 double direction ) const;
53
54 /**
55 * This function generates a new internal list of rowEnds, such
56 * that they are all less than the maxColumnIndex.
57 * It is used because in fireReac we don't want to update all the
58 * molecules, only those that are variable.
59 */
60 void truncateRow( unsigned int maxColumnIndex );
61
62private:
63 /**
64 * End colIndex for rows (molecules in the transposed matrix)
65 * so that only variable molecules are below the colIndex.
66 */
67 vector< unsigned int > rowTruncated_;
