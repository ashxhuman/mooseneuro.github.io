292/// Virtual function to return coords of mesh Entry.
293/// For Endo mesh, coords are middle of parent followed by radius.
294vector< double > EndoMesh::getCoordinates( unsigned int fid ) const
295{
296 vector< double > temp = parent_->getCoordinates( fid );
297 vector< double > ret;
298 double vol = getMeshEntryVolume( fid );
299 if ( temp.size() > 6 ) {
300 ret.resize( 4 );
301 ret[0] = 0.5 * (temp[0] + temp[3] );
302 ret[1] = 0.5 * (temp[1] + temp[4] );
303 ret[2] = 0.5 * (temp[2] + temp[5] );
304 ret[3] = 2.0 * pow( vol*3.0/(4.0 * PI), 1.0/3.0 ); // Assume sphere
305 }
306 return ret;
307}

438void EndoMesh::matchMeshEntries( const ChemCompt* other,
439 vector< VoxelJunction >& ret ) const
440{
441 /**
442 * Consider concentric EndoMeshes in an outer cyl/sphere of radius r_0.
443 * Radii, from out to in, are r_0, r_1, r_2.... r_innermost.
444 * Consider that we are on EndoMesh n.
445 * The diffusion length in each case is 1/2(r_n-1 - r_n+1).
446 * For the innermost we use r_n+1 = 0
447 * The diffusion XA is the area of the EndoMesh.
448 */
449 vector< double > vol = other->vGetVoxelVolume();
450 const MeshCompt* mc = static_cast< const MeshCompt* >( other );
451 vector< double > len = mc->getVoxelLength();
452 assert( vol.size() == len.size() );
453 ret.resize( vol.size() );
454 vector< double > myVol = vGetVoxelVolume();
455 assert( myVol.size() == vol.size() );
456 vector< double > myArea = getVoxelArea();
457 assert( myArea.size() == vol.size() );
458
459 for ( unsigned int i = 0; i < vol.size(); ++i ) {
460 double rSurround = sqrt( vol[i] / (len[i] * PI ) );
461 ret[i].first = ret[i].second = i;
462 ret[i].firstVol = myVol[i];
463 ret[i].secondVol = vol[i];
464
465 // For now we don't consider internal EndoMeshes.
466 ret[i].diffScale = 2.0 * myArea[i] / rSurround;
467 }
468}
