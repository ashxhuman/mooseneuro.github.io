199/**
200 * This assumes that diffLength is the quantity to preserve, over numEntries.
201 * So when the compartment changes volume, numEntries changes. diffLength will
202 * be fine-tuned to be a clean multiple.
203 */
204void CylMesh::updateCoords( const Eref& e, const vector< double >& concs )
205{
206 double temp = sqrt(
207 ( x1_ - x0_ ) * ( x1_ - x0_ ) +
208 ( y1_ - y0_ ) * ( y1_ - y0_ ) +
209 ( z1_ - z0_ ) * ( z1_ - z0_ )
210 );
211
212 if ( doubleEq( temp, 0.0 ) ) {
213 cout << "Error: CylMesh::updateCoords:\n"
214 "total length of compartment = 0 with these parameters\n";
215 return;
216 }
217 totLen_ = temp;
218
219 temp = totLen_ / diffLength_;
220 if ( temp < 1.0 ) {
221 diffLength_ = totLen_;
222 numEntries_ = 1;
223 } else {
224 numEntries_ = static_cast< unsigned int >( round ( temp ) );
225 diffLength_ = totLen_ / numEntries_;
226 }
227 rSlope_ = ( r1_ - r0_ ) / numEntries_;
228 lenSlope_ = diffLength_ * rSlope_ * 2 / ( r0_ + r1_ );
229
230 // dx2_[0] = diffLength_;
231 // dx2_[1] = diffLength_;
232 buildStencil();
233 setChildConcs( e, concs, 0 );
234}

465/**
466 * diffLength = length constant for diffusive spread
467 * len = length of each mesh entry
468 * totLen = total length of cylinder
469 * diffLength = k * r^2
470 * Each entry has the same number of diffLengths, L = len/diffLength.
471 * Thinner entries have shorter diffLength.
472 * This gives a moderately nasty quadratic.
473 * However, as len(i) is prop to diffLength(i),
474 * and diffLength(i) is prop to r(i)^2
475 * and the cyl-mesh is assumed a gently sloping cone
476 * we get len(i) is prop to (r0 + slope.x)^2
477 * and ignoring the 2nd-order term we have
478 * len(i) is approx proportional to x position.
479 *
480 * dr/dx = (r1-r0)/len
481 * ri = r0 + i * dr/dx
482 * r(i+1)-ri = (r1-r0)/numEntries
483 * len = k * r^2
484 * we get k from integ_r0,r1( len.dr ) = totLen
485 * So k.r^3/3 | r0, r1 = totLen
486 * => k/3 * ( r1^3 - r0^3) = totLen
487 * => k = 3 * totLen / (r1^3 - r0^3);
488 * This is bad if r1 == r0, and is generally unpleasant.
489 *
490 * Simple definition of rSlope:
491 * rSlope is measured per meshEntry, not per length:
492 * rSlope = ( r1 - r0 ) / numEntries;
493 * Let's just compute len0 from r0 and diffLength.
494 * len0/diffLength = 2 * r0 / (r0 + r1)
495 * so len0 = diffLength * 2 * r0 / (r0 + r1)
496 * and dlen/dx = lenSlope = diffLength * rSlope * 2/(r0 + r1)
497 *
498 * Drop the following calculations:
499 * // dlen/dx = dr/dx * dlen/dr = ( (r1-r0)/len ) * 2k.r
500 * // To linearize, let 2r = r0 + r1.
501 * // so dlen/dx = ( (r1-r0)/len ) * k * ( r0 + r1 )
502 * // len(i) = len0 + i * dlen/dx
503 * // len0 = totLen/numEntries - ( numEntries/2 ) * dlen/dx
504 */
505
506/// Virtual function to return volume of mesh Entry.
507double CylMesh::getMeshEntryVolume( unsigned int fid ) const
508{
509 double len0 = diffLength_ * 2 * r0_ / ( r0_ + r1_ );
510
511 double ri = r0_ + (fid + 0.5) * rSlope_;
512 double leni = len0 + ( fid + 0.5 ) * lenSlope_;
513
514 return leni * ri * ri * PI;
515}

517/// Virtual function to return coords of mesh Entry.
518/// For Cylindrical mesh, coords are x1y1z1 x2y2z2 r0 r1 phi0 phi1
519vector< double > CylMesh::getCoordinates( unsigned int fid ) const
520{
521 vector< double > ret(10, 0.0);
522 double len0 = diffLength_ * 2 * r0_ / ( r0_ + r1_ );
523 // double len0 = diffLength_ * 2 * ( r0_ + rSlope_ / 0.5) / ( r0_ + r1_ );
524 double lenStart = len0 + lenSlope_ * 0.5;
525
526 double axialStart = fid * lenStart + ( ( fid * (fid - 1 ) )/2 ) * lenSlope_;
527 // fid * totLen_/numEntries_ + (fid - frac ) * lenSlope_;
528 double axialEnd =
529 (fid + 1) * lenStart + ( ( fid * (fid + 1 ) )/2 ) * lenSlope_;
530 // (fid + 1) * totLen_/numEntries_ + (fid - frac + 1.0) * lenSlope_;
531
532 ret[0] = x0_ + (x1_ - x0_ ) * axialStart/totLen_;
533 ret[1] = y0_ + (y1_ - y0_ ) * axialStart/totLen_;
534 ret[2] = z0_ + (z1_ - z0_ ) * axialStart/totLen_;
535
536 ret[3] = x0_ + (x1_ - x0_ ) * axialEnd/totLen_;
537 ret[4] = y0_ + (y1_ - y0_ ) * axialEnd/totLen_;
538 ret[5] = z0_ + (z1_ - z0_ ) * axialEnd/totLen_;
539
540 ret[6] = r0_ + fid * rSlope_;
541 ret[7] = r0_ + (fid + 1.0) * rSlope_;
542
543 ret[8] = 0;
544 ret[9] = 0;
545
546 return ret;
547}

549/// Virtual function to return diffusion X-section area for each neighbor
550vector< double > CylMesh::getDiffusionArea( unsigned int fid ) const
551{
552 if ( numEntries_ <= 1 )
553 return vector< double >( 0 );
554
555 double rlow = r0_ + fid * rSlope_;
556 double rhigh = r0_ + (fid + 1.0) * rSlope_;
557
558 if ( fid == 0 ) {
559 if ( isToroid_ ) {
560 vector < double > ret( 2 );
561 ret[0] = rlow * rlow * PI;
562 ret[1] = rhigh * rhigh * PI;
563 return ret;
564 } else {
565 return vector < double >( 1, rhigh * rhigh * PI );
566 }
567 }
568
569 if ( fid == (numEntries_ - 1 ) ) {
570 if ( isToroid_ ) {
571 vector < double > ret( 2 );
572 ret[0] = rlow * rlow * PI;
573 ret[1] = r0_ * r0_ * PI; // Wrapping around
574 return ret;
575 } else {
576 return vector < double >( 1, rlow * rlow * PI );
577 }
578 }
579 vector< double > ret( 2 );
580 ret[0] = rlow * rlow * PI;
581 ret[1] = rhigh * rhigh * PI;
582 return ret;
583}

671void CylMesh::innerBuildDefaultMesh( const Eref& e,
672 double volume, unsigned int numEntries )
673{
674 /// Single voxel cylinder with diameter = length.
675 /// vol = volume = pi.r^2.len.
676 /// So len = 2r, volume = pi*r^2*2r = 2pi*r^3 so r = (volume/2pi)^(1/3)
677 double r = pow( ( volume / ( PI * 2 ) ), 1.0 / 3 );
678 vector< double > coords( 9, 0 );
679 coords[3] = 2 * r;
680 coords[6] = r;
681 coords[7] = r;
682 coords[8] = 2 * r / numEntries;
683 setCoords( e, coords );
684}

706const vector< double >& CylMesh::vGetVoxelMidpoint() const
707{
708 static vector< double > midpoint( numEntries_ * 3, 0.0 );
709 midpoint.resize( numEntries_ * 3 );
710 double dx = ( x1_ - x0_ ) / numEntries_;
711 double dy = ( y1_ - y0_ ) / numEntries_;
712 double dz = ( z1_ - z0_ ) / numEntries_;
713 vector< double >::iterator j = midpoint.begin();
714 for ( unsigned int i = 0; i < numEntries_; ++i )
715 *j++ = x0_ + dx * i;
716 for ( unsigned int i = 0; i < numEntries_; ++i )
717 *j++ = y0_ + dy * i;
718 for ( unsigned int i = 0; i < numEntries_; ++i )
719 *j++ = z0_ + dz * i;
720
721 return midpoint;
722}
723
724const vector< double >& CylMesh::getVoxelArea() const
725{
726 static vector< double > area;
727 area.resize( numEntries_ );
728 for ( unsigned int i = 0; i < numEntries_; ++i ) {
729 double frac = ( 0.5 + static_cast< double >( i ) ) /
730 static_cast< double >( numEntries_ );
731 double r = r0_ * ( 1.0 - frac ) + r1_ * frac;
732 area[i] = r * r * PI;
733 }
734 return area;
735}

812//////////////////////////////////////////////////////////////////
813// Utility function to set up Stencil for diffusion
814//////////////////////////////////////////////////////////////////
815void CylMesh::buildStencil()
816{
817 setStencilSize( numEntries_, numEntries_ );
818 for ( unsigned int i = 0; i < numEntries_; ++i ) {
819 double rLow = r0_ + i * rSlope_;
820 double rHigh = r0_ + (i + 1.0) * rSlope_;
821 double aLow = rLow * rLow * PI;
822 double aHigh = rHigh * rHigh * PI;
823 vector< double > entry;
824 vector< unsigned int > colIndex;
825 if ( i == 0 ) {
826 colIndex.push_back( 1 );
827 entry.push_back( aHigh / diffLength_ );
828 if ( isToroid_ ) {
829 colIndex.push_back( numEntries_ - 1 );
830 entry.push_back( aLow / diffLength_ );
831 }
832 } else if ( i == numEntries_ - 1 ) {
833 if ( isToroid_ ) {
834 colIndex.push_back( 0 );
835 if ( r0_ < r1_ )
836 entry.push_back( r0_ * r0_ * PI / diffLength_ );
837 else
838 entry.push_back( r1_ * r1_ * PI / diffLength_ );
839 }
840 colIndex.push_back( numEntries_ - 2 );
841 entry.push_back( aLow / diffLength_ );
842 } else { // Mostly it is in the middle.
843 colIndex.push_back( i - 1 );
844 entry.push_back( aLow / diffLength_ );
845 colIndex.push_back( i + 1 );
846 entry.push_back( aHigh / diffLength_ );
847 }
848 addRow( i, entry, colIndex );
849 }
850 innerResetStencil();
851}

888// Look for end-to-end diffusion, not sideways for now.
889// Very straightforward but tedious because of the different permutations.
890// Could readily add cylinder side to other end.
891void CylMesh::matchCylMeshEntries( const CylMesh* other,
892 vector< VoxelJunction >& ret ) const
893{
894 const double EPSILON = 1e-3;
895 ret.clear();
896 // Should really estimate the distance from the centre of the smaller
897 // cylinder cap disk to the plane of the larger disk, provided it is
898 // within the radius of the disk. The subsequent calculations are the
899 // same though.
900 double dr1 = // startOfSelf-to-startOfOther
901 distance(x0_ - other->x0_, y0_ - other->y0_, z0_ - other->z0_ );
902 double dr2 = // endOfSelf-to-endOfOther
903 distance(x1_ - other->x1_, y1_ - other->y1_, z1_ - other->z1_ );
904 double dr3 = // endOfSelf-to-startOfOther
905 distance(x1_ - other->x0_, y1_ - other->y0_, z1_ - other->z0_ );
906 double dr4 = // startOfSelf-to-endOfOther
907 distance(x0_ - other->x1_, y0_ - other->y1_, z0_ - other->z1_ );
908
909 if ( dr1 <= dr2 && dr1 <= dr3 && dr1 <= dr4 ) {
910 if ( ( dr1/totLen_ < EPSILON && dr1/other->totLen_ < EPSILON ) ) {
911 double xda;
912 if ( r0_ < other->r0_ )
913 xda = 2 * r0_ * r0_ * PI / ( diffLength_ + other->diffLength_ );
914 else
915 xda = 2 * other->r0_ * other->r0_ * PI /
916 ( diffLength_ + other->diffLength_ );
917 ret.push_back( VoxelJunction( 0, 0, xda ) );
918 ret.back().first = 0;
919 ret.back().second = 0;
920 ret.back().firstVol = getMeshEntryVolume( 0 );
921 ret.back().secondVol = other->getMeshEntryVolume( 0 );
922 }
923 } else if ( dr2 <= dr3 && dr2 <= dr4 ) {
924 if ( ( dr2/totLen_ < EPSILON && dr2/other->totLen_ < EPSILON ) ) {
925 double xda;
926 if ( r1_ < other->r1_ )
927 xda = 2 * r1_ * r1_ * PI / ( diffLength_ + other->diffLength_ );
928 else
929 xda = 2 * other->r1_ * other->r1_ * PI /
930 ( diffLength_ + other->diffLength_ );
931 ret.push_back( VoxelJunction( numEntries_ - 1,
932 other->numEntries_ - 1, xda ) );
933 ret.back().first = numEntries_;
934 ret.back().second = other->numEntries_ - 1;
935 ret.back().firstVol = getMeshEntryVolume( numEntries_ - 1 );
936 ret.back().secondVol = other->getMeshEntryVolume( other->numEntries_ - 1 );
937 }
938 } else if ( dr3 <= dr4 ) {
939 if ( ( dr3/totLen_ < EPSILON && dr3/other->totLen_ < EPSILON ) ) {
940 double xda;
941 if ( r1_ < other->r0_ )
942 xda = 2 * r1_ * r1_ * PI / ( diffLength_ + other->diffLength_ );
943 else
944 xda = 2 * other->r0_ * other->r0_ * PI /
945 ( diffLength_ + other->diffLength_ );
946 ret.push_back( VoxelJunction( numEntries_ - 1, 0, xda ) );
947 ret.back().first = numEntries_ - 1;
948 ret.back().second = 0;
949 ret.back().firstVol = getMeshEntryVolume( numEntries_ - 1 );
950 ret.back().secondVol = other->getMeshEntryVolume( 0 );
951 }
952 } else {
953 if ( ( dr4/totLen_ < EPSILON && dr4/other->totLen_ < EPSILON ) ) {
954 double xda;
955 if ( r0_ < other->r1_ )
956 xda = 2 * r0_ * r0_ * PI / ( diffLength_ + other->diffLength_ );
957 else
958 xda = 2 * other->r1_ * other->r1_ * PI /
959 ( diffLength_ + other->diffLength_ );
960 ret.push_back( VoxelJunction( 0, other->numEntries_ - 1, xda ));
961 ret.back().first = 0;
962 ret.back().second = other->numEntries_ - 1;
963 ret.back().firstVol = getMeshEntryVolume( 0 );
964 ret.back().secondVol = other->getMeshEntryVolume( other->numEntries_ - 1 );
965 }
966 }

985void fillPointsOnCircle(
986 const Vec& u, const Vec& v, const Vec& q,
987 double h, double r, vector< double >& area,
988 const CubeMesh* other
989 )
990{
991 // fine-tune the h spacing so it is integral around circle.
992 // This will cause small errors in area estimate but they will
993 // be anisotropic. The alternative will have large errors toward
994 // 360 degrees, but not elsewhere.
995 unsigned int numAngle = static_cast<unsigned int>(floor( 2.0 * PI * r / h + 0.5 ));
996 assert( numAngle > 0 );
997 double dtheta = 2.0 * PI / numAngle;
998 double dArea = h * dtheta * r;
999 // March along points on surface of circle centred at q.
1000 for ( unsigned int j = 0; j < numAngle; ++j ) {
1001 double theta = j * dtheta;
1002 double c = cos( theta );
1003 double s = sin( theta );
1004 double p0 = q.a0() + r * ( u.a0() * c + v.a0() * s );
1005 double p1 = q.a1() + r * ( u.a1() * c + v.a1() * s );
1006 double p2 = q.a2() + r * ( u.a2() * c + v.a2() * s );
1007 unsigned int index = other->spaceToIndex( p0, p1, p2 );
1008 if ( index != CubeMesh::EMPTY )
1009 area[index] += dArea;
1010 }
1011}

1013void CylMesh::matchCubeMeshEntries( const CubeMesh* other,
1014 vector< VoxelJunction >& ret ) const
1015{
1016 const double EPSILON = 1e-18;
1017 Vec a( x1_ - x0_, y1_ - y0_, z1_ - z0_ );
1018 Vec u;
1019 Vec v;
1020 a.orthogonalAxes( u, v );
1021
1022 double h = selectGridVolume( other->getDx() );
1023
1024 unsigned int num = static_cast<unsigned int>(floor( 0.1 + diffLength_ / h ));
1025 // March along axis of cylinder.
1026 // q is the location of the point along axis.
1027 for ( unsigned int i = 0; i < numEntries_; ++i ) {
1028 vector< double >area( other->getNumEntries(), 0.0 );
1029 for ( unsigned int j = 0; j < num; ++j ) {
1030 unsigned int m = i * num + j;
1031 double frac = ( m * h + h/2.0 ) / totLen_;
1032 double q0 = x0_ + a.a0() * frac;
1033 double q1 = y0_ + a.a1() * frac;
1034 double q2 = z0_ + a.a2() * frac;
1035 // get radius of cylinder at this point.
1036 double r = r0_ + ( m * h + h / 2.0 ) * rSlope_;
1037 fillPointsOnCircle( u, v, Vec( q0, q1, q2 ),
1038 h, r, area, other );
1039 }
1040 // Go through all cubeMesh entries and compute diffusion
1041 // cross-section. Assume this is through a membrane, so the
1042 // only factor relevant is area. Not the distance.
1043 for ( unsigned int k = 0; k < area.size(); ++k ) {
1044 if ( area[k] > EPSILON ) {
1045 ret.push_back( VoxelJunction( i, k, area[k] ) );
1046 }
1047 }
1048 }

1074// this is the function that does the actual calculation.
1075double CylMesh::nearest( double x, double y, double z,
1076 double& linePos, double& r ) const
1077{
1078 // Consider r0 = x0,y0,z0 and r1 = x1, y1, z1, and r = x,y,z.
1079 // Fraction along cylinder = k
1080 //
1081 // Then, point p along line from r0 to r1 is
1082 // p = k( r0-r1) + r1.
1083 //
1084 // Solving,
1085 // k = (r0 - r1).(r - r1) / (|r0-r1|^2)
1086 //
1087
1088 double dist = distance( x1_ - x0_, y1_ - y0_, z1_ - z0_ );
1089 double k = dotprd(
1090 x1_ - x0_, y1_ - y0_, z1_ - z0_,
1091 x - x0_, y - y0_, z - z0_ ) / ( dist * dist );
1092
1093 // x2, y2, z2 are the coords of the nearest point.
1094 double x2 = k * (x1_ - x0_) + x0_;
1095 double y2 = k * (y1_ - y0_) + y0_;
1096 double z2 = k * (z1_ - z0_) + z0_;
1097
1098 double ret = distance( x - x2, y - y2, z - z2 );
1099 linePos = k;
1100 r = r0_ + k * numEntries_ * rSlope_;
1101 return ret;
1102}
