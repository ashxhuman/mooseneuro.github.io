77 * The initProc function is for a second phase of 'process'
78 * operations. It sends the axial and raxial messages
79 * to other compartments. It has to be executed out of phase
80 * with the main process so that all compartments are
81 * equivalent and there is no calling order dependence in
82 * the results.
83 */
84 void vInitProc( const Eref& e, ProcPtr p );

95 */
96 void vHandleChannel( const Eref& e, double Gk, double Ek);
97
98 /**
99 * handleRaxial handles incoming raxial message data.
100 */
101 void vHandleRaxial( double Ra, double Vm);
102
103 /**
104 * handleAxial handles incoming axial message data.
105 */
106 void vHandleAxial( double Vm);
107
108 /**
109 * Injects a constantly updated current into the compartment.
110 * Unlike the 'inject' field, this injected current is
111 * applicable only for a single timestep. So this is meant to
112 * be used as the destination of a message rather than as a
113 * one-time assignment.
114 */
115 void vInjectMsg( const Eref& e , double current);
116
117
118 /**
119 * Injects a constantly updated current into the
120 * compartment, with a probability prob. Note that it isn't
121 * the current amplitude that is random, it is the presence
122 * or absence of the current that is probabilistic.
123 */
124
125 void vRandInject( const Eref& e , double prob, double current);
126
127 /// Assigns the solver to the zombie
128 void vSetSolver( const Eref& e, Id hsolve );
129
130 /**

133 static const Cinfo* initCinfo();
