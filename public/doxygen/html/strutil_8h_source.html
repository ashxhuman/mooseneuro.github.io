105/* ----------------------------------------------------------------------------*/
106template <typename T = double>
107std::string vectorToCSV(const std::vector<T>& vec)
108{
109 std::stringstream ss;
110 for(size_t i = 0; i < vec.size(); i++)
111 ss << vec[i] << ',';
112 auto res = ss.str();
113 if(',' == res.back())
114 res.pop_back();
115 return res;
116}
