25bool isWhitespaceOnly(const std::string& s)
26{
27 if (s.empty()) {
28 return true;
29 }
30 for (char c : s) {
31 if (!std::isspace(static_cast<unsigned char>(c))) {
32 return false; // Found a non-whitespace character
33 }
34 }
35 return true; // All characters were whitespace
36}

38ReadSwc::ReadSwc(const string& fname)
39{
40 ifstream fin(fname.c_str());
41 if (!fin) {
42 cerr << "ReadSwc:: could not open file " << fname << endl;
43 return;
44 }
45
46 string temp;
47 int badSegs = 0;
48 while (getline(fin, temp)) {
49 if (isWhitespaceOnly(temp))
50 continue;
51 auto pos = temp.find_first_not_of("\t ");
52 if (pos == string::npos)
53 continue;
54 if (temp[pos] == '#')
55 continue;
56
57 SwcSegment t(temp);
58 if (t.OK())
59 segs_.push_back(SwcSegment(temp));
60 else
61 badSegs++;
62 }
63 bool valid = validate();
64 if (valid) {
65 assignKids();
66 cleanMultipointSoma();
67 cleanZeroLength();
68 parseBranches();
69 }
70 cout << "ReadSwc: " << fname << " : NumSegs = " << segs_.size()
71 << ", bad = " << badSegs << ", Validated = " << valid
72 << ", numBranches = " << branches_.size() << endl;
73 diagnostics();
74}

76bool ReadSwc::validate() const
77{
78 int numStart = 0;
79 int numOrphans = 0;
80 int badIndex = 0;
81 int badRadius = 0;
82 for (unsigned int i = 0; i < segs_.size(); ++i) {
83 const SwcSegment& s = segs_[i];
84 if (s.myIndex() != i + 1)
85 badIndex++;
86 if (s.parent() == ~0U) {
87 numStart++;
88 }
89 else {
90 if (s.parent() > i) {
91 numOrphans++;
92 }
93 }
94 if (s.radius() < MinRadius) {
95 badRadius++;
96 }
97 }
98 bool valid = (numStart == 1 && numOrphans == 0 && badRadius == 0);
99 if (!valid) {
100 cout << "ReadSwc::validate() failed: \nNumSegs = " << segs_.size()
101 << ", numStart = " << numStart << ", orphans = " << numOrphans
102 << ", badIndex = " << badIndex << ", badRadius = " << badRadius
103 << ", numBranches = " << branches_.size() << endl;
104 }
105 return valid;
106}

122void ReadSwc::cleanZeroLength()
123{
124 static double EPSILON = 1e-2; // Assume units in microns.
125 for (unsigned int i = 1; i < segs_.size(); ++i) {
126 SwcSegment& s = segs_[i];
127 SwcSegment& pa = segs_[s.parent() - 1];
128 if (s.distance(pa) < EPSILON) {
129 // Remove the zero length child from pa.kids_
130 vector<int> temp;
131 for (unsigned int j = 0; j < pa.kids().size(); ++j) {
132 if (static_cast<unsigned int>(pa.kids()[j]) != s.myIndex())
133 temp.push_back(pa.kids()[j]);
134 }
135 // Go through all kids of s and reparent them.
136 for (unsigned int j = 0; j < s.kids().size(); ++j) {
137 SwcSegment& kid = segs_[s.kids()[j] - 1];
138 kid.setParent(pa.myIndex());
139 temp.push_back(kid.myIndex());
140 }
141 pa.replaceKids(temp);
142 s.setBad();
143 cout << "ReadSwc:: Cleaned zero length " << s.myIndex() << endl;
144 }
145 }

148void ReadSwc::cleanMultipointSoma()
149{
150 // Find the root soma segment (parent == ~0U)
151 int rootIdx = -1;
152 for (unsigned int i = 0; i < segs_.size(); ++i) {
153 if (segs_[i].parent() == ~0U) {
154 rootIdx = static_cast<int>(i);
155 break;
156 }
157 }
158 if (rootIdx < 0 || segs_[rootIdx].type() != SwcSegment::SOMA)
159 return;
160
161 SwcSegment& soma = segs_[rootIdx];
162
163 // Warn if the soma does not match the NeuroMorpho.org 3-point convention:
164 // exactly 2 direct soma children, all sharing x,z,radius with root,
165 // and y-offsets of exactly ±radius.
166
167 static const double EPS = 1e-3; // microns
168 vector<int> directSomaKids;
169 for (int k : soma.kids()) {
170 if (segs_[k - 1].type() == SwcSegment::SOMA)
171 directSomaKids.push_back(k);
172 }
173 if (directSomaKids.size() == 2) {
174 const SwcSegment& s2 = segs_[directSomaKids[0] - 1];
175 const SwcSegment& s3 = segs_[directSomaKids[1] - 1];
176 double xs = soma.vec().a0(), ys = soma.vec().a1(), zs = soma.vec().a2(),
177 rs = soma.radius();
178 bool xzMatch =
179 fabs(s2.vec().a0() - xs) < EPS && fabs(s2.vec().a2() - zs) < EPS &&
180 fabs(s3.vec().a0() - xs) < EPS && fabs(s3.vec().a2() - zs) < EPS;
181 bool radiiMatch =
182 fabs(s2.radius() - rs) < EPS && fabs(s3.radius() - rs) < EPS;
183 bool yMatch = (fabs(s2.vec().a1() - (ys - rs)) < EPS &&
184 fabs(s3.vec().a1() - (ys + rs)) < EPS) ||
185 (fabs(s3.vec().a1() - (ys - rs)) < EPS &&
186 fabs(s2.vec().a1() - (ys + rs)) < EPS);
187 if (!xzMatch || !radiiMatch || !yMatch)
188 cout << "ReadSwc::cleanMultipointSoma: Warning: 3-point soma "
189 "does not match "
190 << "NeuroMorpho.org convention"
191 << (!xzMatch ? " [x,z coords differ]" : "")
192 << (!radiiMatch ? " [radii differ]" : "")
193 << (!yMatch ? " [y-offsets != ±r]" : "") << endl;
194 }
195 else if (!directSomaKids.empty()) {
196 // 1 child: may be 2-point (Arbor) soma or a chained soma.
197 // >2 children: contour soma or non-standard file.
198 cout << "ReadSwc::cleanMultipointSoma: Note: soma has "
199 << directSomaKids.size() << " direct soma-type child(ren) "
200 << "(NeuroMorpho.org standard expects 2)." << endl;
201 }
202
203 // 2-point soma (Arbor style): single direct SOMA child with no chained
204 // SOMA grandchildren beneath it. Both points define the cylinder geometry
205 // — do not collapse.
206 if (directSomaKids.size() == 1) {
207 int onlyKidIdx = directSomaKids[0];
208 bool hasSomaGrandkids = false;
209 for (int gk : segs_[onlyKidIdx - 1].kids()) {
210 if (segs_[gk - 1].type() == SwcSegment::SOMA) {
211 hasSomaGrandkids = true;
212 break;
213 }
214 }
215 if (!hasSomaGrandkids) {
216 cout << "ReadSwc::cleanMultipointSoma: 2-point soma detected; "
217 "preserving geometry."
218 << endl;
219 return;
220 }
221 // else: chained soma (child has soma grandkids) — fall through to collapse.
222 }
223
224 // Use a queue to handle chained soma segments (e.g. 1->2->3->dendrite)
225 // not just direct children of root.
226 vector<int> toProcess = soma.kids();
227 vector<int> newKids;
228 while (!toProcess.empty()) {
229 vector<int> nextToProcess;
230 for (int k : toProcess) {
231 SwcSegment& kid = segs_[k - 1];
232 if (kid.type() == SwcSegment::SOMA) {
233 for (int grandkid : kid.kids()) {
234 segs_[grandkid - 1].setParent(soma.myIndex());
235 nextToProcess.push_back(grandkid);
236 }
237 kid.setBad();
238 cout << "ReadSwc:: Merged multi-point soma segment " << k
239 << " into root soma" << endl;
240 }
241 else {
242 newKids.push_back(k);
243 }
244 }
245 toProcess = nextToProcess;
246 }
247 soma.replaceKids(newKids);
248}

250void ReadSwc::traverseBranch(const SwcSegment& s, double& len, double& L,
251 vector<int>& cable) const
252{
253 const SwcSegment* prev = &s;
254 cable.resize(1, s.myIndex()); // Always include the starting seg.
255 // Note that the cable is filled up with entries in reverse order.
256
257 if (s.parent() == ~0U) {
258 len = s.radius();
259 L = sqrt(len);
260 return;
261 }
262
263 do {
264 // Beware the indexing!
265 const SwcSegment& pa = segs_[prev->parent() - 1];
266 len += pa.distance(*prev);
267 L += pa.L();
268 cable.push_back(pa.myIndex());
269 prev = &pa;
270 } while ((prev->parent() != ~0U) && (prev->kids().size() == 1));
271 cable.pop_back(); // Get rid of the last entry, it is on the parent.
272}

274void ReadSwc::parseBranches()
275{
276 // Fill vector of all branches.
277 for (unsigned int i = 0; i < segs_.size(); ++i) {
278 const SwcSegment& s = segs_[i];
279 // Branch endpoints: root soma (always), forks (≥2 kids), or leaves (0
280 // kids). Root must be explicit: after soma chain cleanup it may have
281 // exactly 1 kid.
282 if (s.OK() && (s.parent() == ~0U || s.kids().size() != 1)) {
283 vector<int> cable;
284 // int branchIndex = branches_.
285 // branches_.push_back( i + 1 );
286 double len = 0;
287 double L = 0;
288 traverseBranch(s, len, L, cable);
289 // branchGeomLength_.push_back( len );
290 // branchElectroLength_.push_back( L );
291 SwcBranch br(branches_.size(), s, len, L, cable);
292 branches_.push_back(br);
293 }
294 }
295 // Assign the parent of each branch. This is known because the
296 // parent of the first segment in the branch is the last segment
297 // in the parent branch. I construct a reverse lookup table to find
298 // the branch # from its last segment number.
299 vector<int> reverseSeg(segs_.size() + 1, 0);
300 for (unsigned int i = 0; i < branches_.size(); ++i)
301 reverseSeg[branches_[i].segs_.back()] = i;
302 for (unsigned int i = 0; i < branches_.size(); ++i) {
303 int parentSeg = segs_[branches_[i].segs_[0] - 1].parent();
304 if (parentSeg == ~0U) // root - no parent
305 continue;
306
307 assert(parentSeg != 0); // Note that segment indices start from 1
308 branches_[i].setParent(reverseSeg[parentSeg]);
309 }
310}

326static Id makeCompt(Id parent, const string& __name, const SwcSegment& seg,
327 const SwcSegment& pa, double RM, double RA, double CM)
328{
329 Shell* shell = reinterpret_cast<Shell*>(Id().eref().data());
330 double len = seg.radius() * 2.0;
331 Id compt;
332 double x0, y0, z0;
333 if (seg.parent() != ~0U) {
334 len = seg.distance(pa);
335 x0 = pa.vec().a0();
336 y0 = pa.vec().a1();
337 z0 = pa.vec().a2();
338 }
339 else {
340 x0 = seg.vec().a0() - len;
341 y0 = seg.vec().a1();
342 z0 = seg.vec().a2();
343 }
344 assert(len > 0.0);
345 compt = shell->doCreate("Compartment", parent, __name, 1);
346 Eref er = compt.eref();
347 moose::CompartmentBase* cptr =
348 reinterpret_cast<moose::CompartmentBase*>(compt.eref().data());
349 double xa = seg.radius() * seg.radius() * PI * 1e-12;
350 len *= 1e-6;
351 double dia = seg.radius() * 2.0e-6;
352 cptr->setRm(er, RM / (len * dia * PI));
353 cptr->setRa(er, RA * len / xa);
354 cptr->setCm(er, CM * (len * dia * PI));
355 cptr->setDiameter(dia);
356 cptr->setLength(len);
357 cptr->setX0(x0 * 1e-6);
358 cptr->setY0(y0 * 1e-6);
359 cptr->setZ0(z0 * 1e-6);
360 cptr->setX(seg.vec().a0() * 1e-6);
361 cptr->setY(seg.vec().a1() * 1e-6);
362 cptr->setZ(seg.vec().a2() * 1e-6);
363 return compt;
364}

366bool ReadSwc::testIfOnlyBasalsArePresent() const
367{
368 /// Some SWCs label all non-soma segments as basals.
369 unsigned int numDend = 0;
370 unsigned int numBasal = 0;
371 for (unsigned int i = 0; i < branches_.size(); ++i) {
372 const SwcBranch& br = branches_[i];
373 for (unsigned int j = 0; j < br.segs_.size(); ++j) {
374 const SwcSegment& seg = segs_[br.segs_[j] - 1];
375 numBasal += (seg.type() == SwcSegment::BASAL);
376 numDend += (seg.type() == SwcSegment::DEND);
377 }
378 }
379 return ((numBasal > 0) && (numDend == 0));
380}

382bool ReadSwc::build(Id parent, double lambda, double RM, double RA, double CM)
383{
384 Shell* shell = reinterpret_cast<Shell*>(Id().eref().data());
385 vector<Id> compts(segs_.size());
386 unsigned int numSomas = 0;
387 unsigned int numRootAxons = 0;
388 unsigned int numRootBasals = 0;
389 unsigned int numRootDends = 0;
390 vector<unsigned int> numBranchesOnMyParent(branches_.size(), 0);
391 vector<string> parentName(segs_.size());
392 string segName;
393 string basalName = "basal";
394 if (testIfOnlyBasalsArePresent())
395 basalName = "dend";
396 for (unsigned int i = 0; i < branches_.size(); ++i) {
397 SwcBranch& br = branches_[i];
398 unsigned int myBranchIdx = numBranchesOnMyParent[br.parent()];
399 numBranchesOnMyParent[br.parent()]++;
400 for (unsigned int j = 0; j < br.segs_.size(); ++j) {
401 stringstream ss;
402 Id compt;
403 SwcSegment& seg = segs_[br.segs_[j] - 1];
404 unsigned int paIndex = seg.parent();
405 if (paIndex == ~0U) // root soma
406 {
407 // Check for 2-point soma: a live SOMA-type direct child whose
408 // coordinates define the distal end of the soma cylinder.
409 const SwcSegment* distalSoma = nullptr;
410 for (int k : seg.kids()) {
411 if (segs_[k - 1].OK() &&
412 segs_[k - 1].type() == SwcSegment::SOMA) {
413 distalSoma = &segs_[k - 1];
414 break;
415 }
416 }
417 if (distalSoma) {
418 // 2-point soma: seg is proximal (root), distalSoma is distal.
419 // makeCompt uses the first arg as the distal segment to get
420 // len = distance(distal, proximal) since distal->parent != ~0U.
421 compt = makeCompt(parent, "soma", *distalSoma, seg, RM, RA, CM);
422 // Pre-fill the distal soma's slot so it's found when its
423 // branch or cable entry is processed later.
424 compts[distalSoma->myIndex() - 1] = compt;
425 }
426 else {
427 // 1-point soma: artificial cylinder of length = 2*r.
428 compt = makeCompt(parent, "soma", seg, seg, RM, RA, CM);
429 }
430 numSomas++;
431 }
432 else if (seg.type() == SwcSegment::SOMA) {
433 // Distal end of a 2-point soma: compartment was already created
434 // when the root soma was processed and stored in compts[].
435 compt = compts[seg.myIndex() - 1];
436 if (compt == Id()) {
437 // Defensive fallback (should not occur with valid SWC).
438 SwcSegment& pa = segs_[paIndex - 1];
439 compt = makeCompt(parent, "soma", seg, pa, RM, RA, CM);
440 numSomas++;
441 }
442 // Do NOT increment numSomas again — already counted at root soma.
443 }
444 else {
445 SwcSegment& pa = segs_[paIndex - 1];
446 if (pa.type() != seg.type()) {
447 if (seg.type() == SwcSegment::AXON) {
448 ss << "axon" << numRootAxons;
449 segName = ss.str();
450 numRootAxons++;
451 }
452 else if (seg.type() == SwcSegment::BASAL) {
453 ss << basalName << numRootBasals;
454 segName = ss.str();
455 numRootBasals++;
456 }
457 else { // Everything else is a dend.
458 ss << "dend" << numRootDends;
459 segName = ss.str();
460 numRootDends++;
461 }
462 }
463 else if (j == 0) { // extend name with branch idx
464 const string& paName =
465 compts[paIndex - 1].element()->getName();
466 string paBranchName = paName.substr(0, paName.rfind('_'));
467 ss << paBranchName << "." << myBranchIdx << "_0";
468 segName = ss.str();
469 }
470 else {
471 const string& paName =
472 compts[paIndex - 1].element()->getName();
473 string paBranchName = paName.substr(0, paName.rfind('_'));
474 ss << paBranchName << "_" << j;
475 segName = ss.str();
476 }
477 compt = makeCompt(parent, segName, seg, pa, RM, RA, CM);
478 assert(compt != Id());
479 assert(compts[paIndex - 1] != Id());
480 shell->doAddMsg("Single", compts[paIndex - 1], "axial", compt,
481 "raxial");
482 }
483 assert(compt != Id());
484 compts[seg.myIndex() - 1] = compt;
485 }
486 }
487 return true;
488}
