144void DiffPoolVec::advance( double dt )
145{
146 if ( ops_.size() == 0 ) return;
147
148 for (auto i = ops_.cbegin(); i != ops_.end(); ++i )
149 n_[i->c_] -= n_[i->b_] * i->a_;
150
151 assert( n_.size() == diagVal_.size() );
152
153 auto iy = n_.begin();
154 for ( auto i = diagVal_.cbegin(); i != diagVal_.end(); ++i )
155 *iy++ *= *i;
156}

158void DiffPoolVec::reinit( const vector< double >& vols ) // Not called by the clock, but by parent.
159{
160 const double NA_ = 6.0221415e23;
161 assert( n_.size() == concInit_.size() );
162 // vector< double > vols( concInit_.size(), 1.0 );
163 vector< double > nInit( concInit_.size(), 0.0 );
164 for ( size_t i = 0; i < concInit_.size(); ++i )
165 nInit[i] = concInit_[i] * NA_ * vols[i];
166
167 prev_ = n_ = nInit;
168}
