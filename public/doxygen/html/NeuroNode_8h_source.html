87 //////////////////////////////////////////////////////////////
88 // A set of calls for safely generating a tree by traversing
89 // compartment messages. Designed to be somewhat resilient, even
90 // when given a badly set up neuron.
91 //////////////////////////////////////////////////////////////
92 /**
93 * Finds all the compartments connected to current node, put them
94 * all into the 'children' vector even if they may be 'parent' by
95 * the messaging. This is because this function has to be robust
96 * enough to sort this out

98 void findConnectedCompartments(

102 /**
103 * Go through nodes vector and eliminate entries that have zero
104 * children, that is, are not connected to any others.
105 * Need to clean up 'children_' list after this is called.
106 */
107 static unsigned int removeDisconnectedNodes(
108 vector< NeuroNode >& nodes );
109
110 /**
111 * Find the start node, typically the soma, of a model. In terms of
112 * the solution, this should be the node at the root of the tree.
113 * Returns index in nodes vector.
114 * Technically the matrix solution could begin from any terminal
115 * branch, but it helps to keep the soma identical to the root of
116 * the tree.
117 *
118 * Uses two heuristics to locate the start node: Looks for the
119 * node with the largest diameter, and also looks for node(s)
120 * with 'soma' in their name. If these disagree then it goes
121 * with the 'soma' node. If there are
122 * many of the soma nodes, it goes with the fattest.
123 */
124 static unsigned int findStartNode(
125 const vector< NeuroNode >& nodes );
126
127 /**
128 * Traverses the nodes list starting from the 'start' node, and
129 * sets up correct parent-child information. This involves removing
130 * the identified 'parent' node from the 'children_' vector and
131 * assigning it to the parent_ field.
132 * Then it redoes the entire nodes vector (with due care for
133 * indexing of children and parents)
134 * so that it is in the correct order for a depth-first traversal.
135 * This means that you can take any entry in the list, and the
136 * immediately following entries will be all the descendants,
137 * if any.
138 */
139 static void traverse(
140 vector< NeuroNode >& nodes, unsigned int start );
141
142 /**
143 * Helper recursive function for traversing nodes to build tree.
144 */
145 void innerTraverse(
146 vector< NeuroNode >& tree,
147 const vector< NeuroNode >& nodes,
148 vector< unsigned int >& seen
149 ) const;
150
151 /**
152 * This function takes a list of elements that include connected
153 * compartments, and constructs a tree of nodes out of them. The
154 * generated nodes vector starts with the soma, and is a depth-first
155 * sequence of nodes. This is meant to be insensitive to vagaries
156 * in how the user has set up the compartment messaging, provided
157 * that there is at least one recognized message between connected
158 * compartments.
159 */
160 static void buildTree( vector< NeuroNode >& nodes,
161 vector< ObjId > elist );
162 static void buildSpinyTree(
163 vector< ObjId >& elist, vector< NeuroNode >& nodes,
164 vector< Id >& shaftId, vector< Id >& headId,
165 vector< unsigned int >& spineParent );
166 void setParentAndChildren( unsigned int index, int dendParent,
167 vector< NeuroNode >& nodes,
168 const unordered_map< Id, unsigned int >& dendMap );
169
170 /**
171 * Trims off all spines from tree. Does so by identifying a set of
172 * reasonable names: shaft, head, spine, and variants in capitals.
173 * Having done this it builds two matching vectors of vector of
174 * shafts and heads, which is a hack that assumes that there are
175 * no sub-branches in spines.
176 * The returned nodes vector has non spine/shaft compartments only.
177 * The returned shaftId vector has all the shaft compartments.
178 * The returned headId vector has all the shaft compartments.
179 * The returned parent vector has the indices of the parent
180 * node for each shaft.
181 * There should be exactly the same number of entries in the
182 * shaftId, headId and parent vectors.
183 */
184 static void filterSpines( vector< NeuroNode >& nodes,
185 vector< Id >& shaftId, vector< Id >& headId,
186 vector< unsigned int >& parent );
187 private:
188 /**
189 * Index of parent NeuroNode, typically a diffusive compartment.
190 * In the special case where the junction to the parent electrical
191 * compartment is NOT at the end of the parent compartment, this
192 * refers instead to a dummy NeuroNode which has the coordinates.
193 *
194 * One of the nodes, typically the soma, will have no parent.
195 * This is indicated by the value ~0U.
196 */
197 unsigned int parent_;
198
199 /**
200 * Index of children of this NeuroNode.
201 */
202 vector< unsigned int >children_;
203
204 /**
205 * Index of starting MeshEntry handled by this NeuroNode. Assumes
206 * a block of contiguous fids are handled by each NeuroNode.
207 */
208 unsigned int startFid_;
209
210
211 /// Id of electrical compartment in which this diffusive compt lives
212 Id elecCompt_;
213
214 /**
215 * Special case for soma, perhaps for spine heads.
216 * When true, xyz are centre, and dia is dia.
217 */
218 bool isSphere_;
219
220 // For spines we will also need a double to indicate position along
221 // parent dendrite.
222 //
223};
