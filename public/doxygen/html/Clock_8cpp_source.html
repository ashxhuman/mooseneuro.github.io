11/**
12 * The Clock manages simulation scheduling, in a close
13 * collaboration with the Tick.
14 * This version does this using an array of child Ticks, which
15 * it manages directly.
16 *
17 * Simulation scheduling requires that certain functions of
18 * groups of objects be called in a strict sequence according to
19 * a range of timesteps, dt. For example, the numerical integration
20 * function of a compartment may be called every 10 microseconds.
21 * Furthermore, there may be ion channels in the simulation which also
22 * have to be called every 10 microseconds, but always after all
23 * the compartments have been called. Finally,
24 * the graphical update function to plot the compartment voltage
25 * may be called every 1 millisecond of simulation time.
26 * The whole sequence has to be repeated till the runtime of the
27 * simulation is complete.
28 *
29 * This scheduler uses integral multiples of the base timestep dt_.
30 *
31 * The system works like this:
32 * 1. Assign times to each Tick. This is divided by dt_ and the rounded
33 * value is used for the integral multiple. Zero means the tick is not
34 * scheduled.
35 * 2. The process call goes through all active ticks in order every
36 * timestep. Each Tick increments its counter and decides if it is
37 * time to fire.
38 * 4. The Reinit call goes through all active ticks in order, just once.
39 * 5. We connect up the Ticks to their target objects.
40 * 6. We begin the simulation by calling 'start' or 'step' on the Clock.
41 * 'step' executes exactly one timestep (of the minimum dt_),
42 * visiting all ticks as above..
43 * 'start' executes an integral number of such timesteps.
44 * 7. To interrupt the simulation at some intermediate time, call 'stop'.
45 * This lets the system complete its current step.
46 * 8. To restart the simulation from where it left off, use the same
47 * 'start' or 'step' function on the Clock. As all the ticks
48 * retain their state, the simulation can resume smoothly.
49 */

79// Static func for building the process vectors.
80static vector< SrcFinfo1< ProcPtr >* > buildProcessVec( const string& name )
81{
82 vector< SrcFinfo1< ProcPtr >* > vec;
83 vec.resize( Clock::numTicks );
84 for ( unsigned int i = 0; i < Clock::numTicks; ++i )
85 {
86 stringstream ss;
87 ss << name << i;
88 stringstream ss2;
89 ss2 << name << " for Tick " << i;

91 }
92 return vec;

582/**
583 * A little nasty because we want to ensure that the main clock dt is
584 * set intelligently from the assignment here.
585 */
586void Clock::setTickDt( unsigned int i, double v )
587{
588 unsigned int numUsed = 0;
589 if ( v < minimumDt )
590 {
591 cout << "Warning: Clock::setTickDt: " << v <<
592 " is smaller than minimum allowed timestep " <<
593 minimumDt << endl;
594 cout << "dt not set\n";
595 return;
596 }
597 for ( unsigned int j = 0; j < numTicks; ++j )
598 numUsed += ( ticks_[j] != 0 );
599
600 if ( numUsed == 0 )
601 {
602 dt_ = v;
603 }
604 else if ( dt_ > v )
605 {
606 for ( unsigned int j = 0; j < numTicks; ++j )
607 if ( ticks_[j] != 0 )
608 ticks_[j] = static_cast<unsigned int>(round( ( ticks_[j] * dt_ ) / v ));
609 dt_ = v;
610 }
611
612 if ( checkTickNum( "setTickDt", i ) )
613 ticks_[i] = static_cast<unsigned int>(round( v / dt_ ));
614}

617{
618 if ( i < Clock::numTicks )
619 return ticks_[i] * dt_;
620 return 0.0;

671{
672 activeTicks_.resize(0);
673 activeTicksMap_.resize(0);
674 stride_ = ~0U;
675 for ( unsigned int i = 0; i < ticks_.size(); ++i )
676 {
677 if ( ticks_[i] > 0 &&
678 e.element()->hasMsgs( processVec()[i]->getBindIndex() ) )
679 {
680 activeTicks_.push_back( ticks_[i] );
681 activeTicksMap_.push_back( i );
682 if ( ticks_[i] > 0 && stride_ > ticks_[i] )
683 stride_ = ticks_[i];
684 }
685 }
686 // Should really do the HCF of N numbers here to get the stride.
687}

689/**
690 * Start has to happen gracefully: If the simulation was stopped for any
691 * reason, it has to pick up where it left off.
692 * The "runtime" argument is the additional time to run the simulation.
693 */
694void Clock::handleStart( const Eref& e, double runtime, bool notify )
695{
696 notify_ = notify;
697 if ( stride_ == 0 || stride_ == ~0U )
698 stride_ = 1;
699 unsigned long n = static_cast<unsigned long>(round( runtime / ( stride_ * dt_ ) ));
700 handleStep( e, n );
701}
702
703void Clock::handleStep( const Eref& e, unsigned long numSteps )
704{
705 numSteps *= stride_;
706 if ( isRunning_ || doingReinit_ )
707 {
708 cout << "Clock::handleStart: Warning: simulation already in progress.\n Command ignored\n";
709 return;
710 }
711
712 time_t rawtime;
713 struct tm * timeinfo;
714 char now[80];
715
716 buildTicks( e );
717 if (nSteps_ == 0 )
718 info_.setFirstStep();
719 else
720 info_.setContinue();
721 assert( currentStep_ == nSteps_ );
722 assert( activeTicks_.size() == activeTicksMap_.size() );
723 nSteps_ += numSteps;
724 runTime_ = nSteps_ * dt_;
725 for ( isRunning_ = (activeTicks_.size() > 0 );
726 isRunning_ && currentStep_ < nSteps_; currentStep_ += stride_ )
727 {
728 // Curr time is end of current step.
729 unsigned long endStep = currentStep_ + stride_;
730 currentTime_ = info_.currTime = dt_ * endStep;
731
732#if 0
733
734//#if PARALLELIZE_CLOCK_USING_CPP11_ASYNC
735
736 // NOTE: It does not produce very promising results. The challenge here
737 // is doing load-balancing.
738 // TODO: To start with, we can put one solver on one thread and everything
739 // else onto 1 thread. Each Hsove, Ksolve, and Gsolve can take its own
740 // thread and rest are on different threads.
741
742 unsigned int nTasks = activeTicks_.size( );
743 unsigned int numThreads_ = 3;
744 unsigned int blockSize = 1 + (nTasks / numThreads_);
745
746 for( unsigned int i = 0; i < numThreads_; ++i )
747 {
748 std::async( std::launch::async, [this,blockSize,i,nTasks,endStep,e]
749 {
750 unsigned int mapI = i * blockSize;
751 // Get the block we want to run in paralle.
752 for( unsigned int ii = i * blockSize; ii < min((i+1) * blockSize, nTasks); ii++ )
753 {
