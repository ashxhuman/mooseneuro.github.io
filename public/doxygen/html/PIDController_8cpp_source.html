291void PIDController::process(const Eref& e, ProcPtr proc )
292{
293 double dt = proc->dt;
294 e_previous_ = error_;
295 error_ = command_ - sensed_;
296 e_integral_ += 0.5 * (error_ + e_previous_) * dt;
297 e_derivative_ = (error_ - e_previous_) / dt;
298 output_ = gain_ * (error_ + e_integral_ / tau_i_ + e_derivative_ * tau_d_);
299 if (output_ > saturation_){
300 output_ = saturation_;
301 e_integral_ -= 0.5 * (error_ + e_previous_) * dt;
302 } else if (output_ < -saturation_){
303 output_ = -saturation_;
304 e_integral_ -= 0.5 * (error_ + e_previous_) * dt;
305 }
306 outputOut()->send(e, output_);
307}

310void PIDController::reinit(const Eref& e, ProcPtr proc )
311{
312 if ( tau_i_ <= 0.0 ){
313 tau_i_ = proc->dt;
314 }
315 if ( tau_d_ < 0.0 ){
316 tau_d_ = proc->dt / 4;
317 }
318 sensed_ = 0.0;
319 output_ = 0;
320 error_ = 0;
321 e_previous_ = error_;
322 e_integral_ = 0;
323 e_derivative_ = 0;
324 outputOut()->send(e, output_);
325}
