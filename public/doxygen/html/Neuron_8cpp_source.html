610bool parseDistrib( vector< vector < string > >& lines,
611 const vector< string >& distrib )
612{
613 lines.clear();
614 vector< string > temp;
615 for ( unsigned int i = 0; i < distrib.size(); ++i )
616 {
617 if ( distrib[i] == "" )
618 {
619 if ( temp.size() < 4 )
620 {
621 cout << "Warning: Neuron::parseDistrib: <4 args: " <<
622 temp.size() << endl;
623 return false;
624 }
625 if ( temp.size() % 2 == 1 )
626 {
627 cout << "Warning: Neuron::parseDistrib: : odd # of args:"
628 << temp.size() << endl;
629 return false;
630 }
631 lines.push_back( temp );
632 temp.clear();
633 }

703static void assignParam( Id obj, const string& field,
704 double val, double len, double dia )
705{
706 if ( obj.element()->cinfo()->isA( "ChanBase" ) )
707 {
708 if ( field == "Gbar" )
709 {
710 if ( val > 0 )
711 Field< double >::set( obj, "Gbar", val * len * dia * PI );
712 }
713 else if ( field == "Ek" )
714 {
715 Field< double >::set( obj, "Ek", val );
716 }
717 }
718 else if ( obj.element()->cinfo()->isA( "CaConcBase" ) )
719 {
720 Field< double >::set( obj, "length", len );
721 Field< double >::set( obj, "diameter", dia );
722 // cout << "len, dia = " << len << ", " << dia << endl;
723 if ( field == "CaBasal" || field == "tau" || field == "thick" ||
724 field == "floor" || field == "ceiling" )
725 {
726 Field< double >::set( obj, field, val );
727 }
728 else if ( field == "B" )
729 {
730 // The system obeys dC/dt = B*I_Ca - C/tau
731 // So B = arg/( vol * F ). Here the system provides vol and F,
732 // so the arg is just a scale factor over this, typically
733 // to be thought of in terms of buffering. Small B is more
734 // buffering. This field is deprecated but used in legacy
735 // GENESIS scripts.
736 Field< double >::set( obj, "B", val /
737 ( FaradayConst * len * dia * dia * PI / 4.0 ) );
738 }
739 }

1363 for ( unsigned int i = 0; i < self.kids().size(); ++i )
1364 {
1365 SwcSegment& kid = segs[ self.kids()[i] ];

1370 double Ra = Field< double >::get( kidId, "Ra" );
1371 // Note that sqrt( Rm/Ra ) = lambda/length = 1/L.
1372 double electrotonicLength = sqrt( Ra / Rm );
1373 double e = eSoma + electrotonicLength;
1374 traverseCumulativeDistance( kid, segs, lookupId,
1375 segmentLength, electrotonicLength, p, e );
1376 }

1599/**
1600 * Utility function to return a coordinate system where
1601 * z is the direction of a dendritic segment,
1602 * x is the direction of spines outward from soma and perpendicular to z
1603 * and y is the perpendicular to x and z.
1604 */
1605static double coordSystem( Id soma, Id dend, Vec& x, Vec& y, Vec& z )
1606{
1607 static const double EPSILON = 1e-20;
1608 double x0 = Field< double >::get( dend, "x0" );
1609 double y0 = Field< double >::get( dend, "y0" );
1610 double z0 = Field< double >::get( dend, "z0" );
1611 double x1 = Field< double >::get( dend, "x" );
1612 double y1 = Field< double >::get( dend, "y" );
1613 double z1 = Field< double >::get( dend, "z" );
1614
1615 Vec temp( x1-x0, y1-y0, z1-z0 );
1616 double len = temp.length();
1617 z = Vec( temp.a0()/len, temp.a1()/len, temp.a2()/len );
1618
1619 double sx0 = Field< double >::get( soma, "x0" );
1620 double sy0 = Field< double >::get( soma, "y0" );
1621 double sz0 = Field< double >::get( soma, "z0" );
1622 Vec temp2( x0 - sx0, y0-sy0, z0-sz0 );
1623 Vec y2 = temp.crossProduct( z );
1624 double ylen = y2.length();
1625 double ytemp = 1.0;
1626 while ( ylen < EPSILON )
1627 {
1628 Vec t( ytemp , sqrt( 2.0 ), 0.0 );
1629 y2 = t.crossProduct( z );
1630 ylen = y2.length();
1631 ytemp += 1.0;
1632 }
1633 y = Vec( y2.a0()/ylen, y2.a1()/ylen, y2.a2()/ylen );
1634 x = z.crossProduct( y );
1635 assert( doubleEq( x.length(), 1.0 ) );
1636 return len;
1637}

1639/**
1640 * Utility function to change coords of spine so as to reorient it.
1641 */
1642static void reorientSpine( vector< Id >& spineCompts,
1643 vector< Vec >& coords,
1644 Vec& parentPos, double pos, double angle,
1645 Vec& x, Vec& y, Vec& z )
1646{
1647 double c = cos( angle );
1648 double s = sin( angle );
1649 double omc = 1.0 - c;
1650
1651 Vec rot0( z.a0()*z.a0()*omc + c,
1652 z.a1()*z.a0()*omc - z.a2()*s ,
1653 z.a2()*z.a0()*omc + z.a1()*s );
1654
1655 Vec rot1( z.a0()*z.a1()*omc + z.a2()*s,
1656 z.a1()*z.a1()*omc + c,
1657 z.a2()*z.a1()*omc - z.a0()*s );
1658
1659 Vec rot2( z.a0()*z.a2()*omc - z.a1()*s,
1660 z.a1()*z.a2()*omc + z.a0()*s,
1661 z.a2()*z.a2()*omc + c );
1662
1663 Vec translation = z * pos + parentPos;
1664 // Vec translation = parentPos;
1665 vector< Vec > ret( coords.size() );
1666 for ( unsigned int i = 0; i < coords.size(); ++i )
1667 {
1668 ret[i] = Vec( rot0.dotProduct( coords[i] ) + translation.a0(),
1669 rot1.dotProduct( coords[i] ) + translation.a1(),
1670 rot2.dotProduct( coords[i] ) + translation.a2() );
1671
1672 }
1673 assert( spineCompts.size() * 2 == ret.size() );
1674
1675 for ( unsigned int i = 0; i < spineCompts.size(); ++i )
1676 {
1677 unsigned int j = 2 * i;
1678 Field< double >::set( spineCompts[i], "x0", ret[j].a0() );
1679 Field< double >::set( spineCompts[i], "y0", ret[j].a1() );
1680 Field< double >::set( spineCompts[i], "z0", ret[j].a2() );
1681 // cout << "(" << ret[j].a0() << ", " << ret[j].a1() << ", " << ret[j].a2() << ")";
1682 j = j + 1;
1683 Field< double >::set( spineCompts[i], "x", ret[j].a0() );
1684 Field< double >::set( spineCompts[i], "y", ret[j].a1() );
1685 Field< double >::set( spineCompts[i], "z", ret[j].a2() );
1686 // cout << "(" << ret[j].a0() << ", " << ret[j].a1() << ", " << ret[j].a2() << ")\n";
1687 }

1690/**
1691 * Utility function to add a single spine to the given parent.
1692 * Returns vector of added spine contents.
1693 * parent is parent compartment for this spine.
1694 * spineProto is just that.
1695 * pos is position (in metres ) along parent compartment
1696 * angle is angle (in radians) to rotate spine wrt x in plane xy.
1697 * Size is size scaling factor, 1 leaves as is.
1698 * x, y, z are unit vectors. Z is along the parent compt.
1699 * We first shift the spine over so that it is offset by the parent compt
1700 * diameter.
1701 * We then need to reorient the spine which lies along (i,0,0) to
1702 * lie along x. X is a unit vector so this is done simply by
1703 * multiplying each coord of the spine by x.
1704 * Finally we rotate the spine around the z axis by the specified angle
1705 * k is index of this spine.
1706 */
1707
1708static vector< Id > addSpine( Id parentCompt, Id spineProto,
1709 double pos, double angle,
1710 Vec& x, Vec& y, Vec& z,
1711 double size,
1712 unsigned int k )
1713{
1714 Shell* shell = reinterpret_cast< Shell* >( Id().eref().data() );
1715 Id parentObject = Neutral::parent( parentCompt );
1716 stringstream sstemp;
1717 sstemp << k;
1718 string kstr = sstemp.str();
1719 Id spine = shell->doCopy( spineProto, parentObject, "_spine" + kstr,
1720 1, false, false );
1721 vector< Id > kids;
1722 Neutral::children( spine.eref(), kids );
1723 double x0 = Field< double >::get( parentCompt, "x0" );
1724 double y0 = Field< double >::get( parentCompt, "y0" );
1725 double z0 = Field< double >::get( parentCompt, "z0" );
1726 double parentRadius = Field< double >::get( parentCompt, "diameter" )/2;
1727 Vec ppos( x0, y0, z0 );
1728 // First, build the coordinates vector for the spine. Assume that its
1729 // major axis is along the unit vector [1,0,0].
1730 vector< Vec > coords;
1731 vector< Id > ret; // Filtered to make sure that it returns only compts
1732 for ( vector< Id >::iterator i = kids.begin(); i != kids.end(); ++i )
1733 {
1734 if ( i->element()->cinfo()->isA( "CompartmentBase" ) )
1735 {
1736 i->element()->setName( i->element()->getName() + kstr );
1737 x0 = Field< double >::get( *i, "x0" ) * size;
1738 y0 = Field< double >::get( *i, "y0" ) * size;
1739 z0 = Field< double >::get( *i, "z0" ) * size;
1740 coords.push_back( Vec( x0 + parentRadius, y0, z0 ) );
1741 double x = Field< double >::get( *i, "x" ) * size;
1742 double y = Field< double >::get( *i, "y" ) * size;
1743 double z = Field< double >::get( *i, "z" ) * size;
1744 double dia = Field< double >::get( *i, "diameter" ) * size;
1745 // Field< double >::set( *i, "diameter", dia );
1746
1747 double len = sqrt(
1748 (x-x0)*(x-x0) +
1749 (y-y0)*(y-y0) +
1750 (z-z0)*(z-z0) );
1751 SetGet2< double, double >::set( *i, "setGeomAndElec", len, dia );
1752
1753
1754 coords.push_back( Vec( x + parentRadius, y, z ) );
1755 // scaleSpineCompt( *i, size );
1756 shell->doMove( *i, parentObject );
1757 ret.push_back( *i );
1758 }
1759 }
1760 // Then, take the projection of this along the x vector passed in.
1761 for( vector< Vec >::iterator i = coords.begin(); i != coords.end(); ++i)
1762 *i = x * i->a0();
1763 shell->doDelete( spine ); // get rid of the holder for the spine copy.
1764 shell->doAddMsg( "Single", parentCompt, "axial", kids[0], "raxial" );
1765 reorientSpine( kids, coords, ppos, pos, angle, x, y, z );
1766
1767 return ret;
1768}
