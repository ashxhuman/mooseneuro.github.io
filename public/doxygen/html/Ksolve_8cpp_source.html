267void Ksolve::setMethod( string method )
268{
269 std::transform(method.begin(), method.end(), method.begin(), ::tolower);
270 // If user is trying to set ksolve method after ksolve has been initialized,
271 // show a warning.
272 if( isBuilt_ )
273 {
274 moose::showWarn(
275 "You are trying to set Ksolve::method after moose::Stoich has been "
276 " initialized. This will be ignored. Please do before ksolve is assigned to "
277 " moose::Stoich."
278 );
279 return;
280 }
281
282#if USE_GSL
283 if ( method == "rk5" || method == "gsl" )
284 {
285 method_ = "rk5";
286 }
287 else if ( method == "rk4" || method == "rk2" ||
288 method == "rk8" || method == "rkck" || method == "lsoda" )
289 {
290 method_ = method;
291 }
292 else
293 {
294 cout << "Warning: Ksolve::setMethod: '" << method <<
295 "' is not known, using default rk5\n";
296 method_ = "rk5";
297 }
298#elif USE_BOOST_ODE
299 // TODO: Check for boost related methods.
300 method_ = method;
301#endif
302}

381void Ksolve::setStoich( Id stoich )
382{
383 assert( stoich.element()->cinfo()->isA( "Stoich" ) );
384 stoich_ = stoich;
385 stoichPtr_ = reinterpret_cast< Stoich* >( stoich.eref().data() );
386
387 if ( !isBuilt_ )
388 {
389 OdeSystem ode;
390 ode.epsAbs = epsAbs_;
391 ode.epsRel = epsRel_;
392 // ode.initStepSize = getEstimatedDt();
393 ode.initStepSize = 0.01; // This will be overridden at reinit.
394 ode.method = method_;
395
396#ifdef USE_GSL
397 ode.gslSys.dimension = stoichPtr_->getNumAllPools();
398 if ( ode.gslSys.dimension == 0 )
399 {
400 stoichPtr_ = 0;
401 return; // No pools, so don't bother.
402 }
403 innerSetMethod( ode, method_ );
404 ode.gslSys.function = &VoxelPools::gslFunc;
405 ode.gslSys.jacobian = 0;
406 innerSetMethod( ode, method_ );
407 unsigned int numVoxels = pools_.size();
408 for ( unsigned int i = 0 ; i < numVoxels; ++i )
409 {
410 ode.gslSys.params = &pools_[i];
411 pools_[i].setStoich( stoichPtr_, &ode );
412 // pools_[i].setIntDt( ode.initStepSize ); // We're setting it up anyway
413 }
414#elif USE_BOOST_ODE
415 ode.dimension = stoichPtr_->getNumAllPools();
416 if ( ode.dimension == 0 )
417 return; // No pools, so don't bother.
418 unsigned int numVoxels = pools_.size();
419 for ( unsigned int i = 0 ; i < numVoxels; ++i )
420 {
421 pools_[i].setStoich( stoichPtr_, &ode );
422 }
423#endif
424 isBuilt_ = true;
425 }

539double Ksolve::getEstimatedDt() const
540{
541 static const double EPSILON = 1e-15;
542 vector< double > s( stoichPtr_->getNumAllPools(), 1.0 );
543 vector< double > v( stoichPtr_->getNumRates(), 0.0 );
544 double maxVel = 0.0;
545 if ( pools_.size() > 0.0 )
546 {
547 pools_[0].updateReacVelocities( &s[0], v );
548 for ( vector< double >::iterator
549 i = v.begin(); i != v.end(); ++i )
550 if ( maxVel < *i )
551 maxVel = *i;
552 }
553 if ( maxVel < EPSILON )
554 return 0.1; // Based on typical sig pathway reac rates.
555 // Heuristic: the largest velocity times dt should be 10% of mol conc.
556 return 0.1 / maxVel;
557}

559//////////////////////////////////////////////////////////////
560// Process operations.
561//////////////////////////////////////////////////////////////
562void Ksolve::process( const Eref& e, ProcPtr p )
563{
564 if ( isBuilt_ == false )
565 return;
566
567 //t0_ = high_resolution_clock::now();
568
569 // First, handle incoming diffusion values, update S with those.
570 if ( dsolvePtr_ )
571 {
572 vector< double > dvalues( 4 );
573 dvalues[0] = 0;
574 dvalues[1] = getNumLocalVoxels();
575 dvalues[2] = 0;
576 dvalues[3] = stoichPtr_->getNumVarPools();
577
578 dsolvePtr_->getBlock( dvalues );
579 // Second, set the prev_ value in DiffPoolVec
580 dsolvePtr_->setPrev();
581 setBlock( dvalues );
582 }
583
584 if( 1 == numThreads_ || 1 == pools_.size() )
585 {
586 if( numThreads_ > 1 )
587 {
588 cerr << "Warn: Not enough voxels for multithreading. "
589 << "Reverting to serial mode. " << endl;
590 numThreads_ = 1;
591 }
592
593 for ( unsigned int i = 0; i < pools_.size(); i++ )
594 pools_[i].advance( p );
595 }
596 else
597 {
598 std::vector<std::future<size_t>> vecFutures;
599
600 // lambdas is faster than std::bind
601 for (auto interval : intervals_)
602 {
603 vecFutures.push_back(
604 std::async( std::launch::async
605 , &Ksolve::advance_chunk
606 , this
607 , interval.first
608 , interval.second, p
609 )
610 );
611 }
612 size_t tot = 0;
613 for (auto &v : vecFutures )
614 tot += v.get();
615 assert(tot == pools_.size());
616 }
617
618 // Assemble and send the integrated values off for the Dsolve.
619 if ( dsolvePtr_ )
620 {
621 vector< double > kvalues( 4 );
622 kvalues[0] = 0;
623 kvalues[1] = getNumLocalVoxels();
624 kvalues[2] = 0;
625 kvalues[3] = stoichPtr_->getNumVarPools();
626 getBlock( kvalues );
627 dsolvePtr_->setBlock( kvalues );
628
629 // Now use the values in the Dsolve to update junction fluxes
630 // for diffusion, channels, and xreacs
631 dsolvePtr_->updateJunctions( p->dt );
632 }
633
634 //t1_ = high_resolution_clock::now();
635 //moose::addSolverProf( "Ksolve", duration_cast<duration<double>> (t1_ - t0_ ).count(), 1 );
636}

643size_t Ksolve::advance_chunk( const size_t begin, const size_t end, ProcPtr p )
644{
645 size_t tot = 0;
646 for (size_t i = begin; i < std::min(end, pools_.size()); i++)
647 {
648 pools_[i].advance( p );
649 tot += 1;
650 }
651 return tot;
652}

655void Ksolve::reinit( const Eref& e, ProcPtr p )
656{
657 if ( !stoichPtr_ )
658 return;
659
660 if ( isBuilt_ )
661 {
662 for ( unsigned int i = 0 ; i < pools_.size(); ++i ) {
663 pools_[i].setNumVoxels( pools_.size() );
664 pools_[i].reinit( p->dt );
665 }
666 }
667 else
668 {
669 cout << "Warning:Ksolve::reinit: Reaction system not initialized\n";
670 return;
671 }
672
673 if(numThreads_ > pools_.size())
674 numThreads_ = pools_.size();
675
676 if(numThreads_ > 1)
677 cout << "Info: Multi-threaded Ksolve (" << numThreads_ << " threads)."
678 << endl;
679
680 // Recompute the partition of interval.
681 intervals_.clear();
682 moose::splitIntervalInNParts(pools_.size(), numThreads_, intervals_);
683}

698/**
699 * updateRateTerms obtains the latest parameters for the rates_ vector,
700 * and has each of the pools update its parameters including rescaling
701 * for volumes.
702 */
703void Ksolve::updateRateTerms( unsigned int index )
704{
705 if ( index == ~0U )
706 {
707 for ( unsigned int i = 0 ; i < pools_.size(); ++i )
708 {
709 pools_[i].updateAllRateTerms( stoichPtr_->getRateTerms(),
710 stoichPtr_->getNumCoreRates() );
711 }
712 }
713 else if ( index < stoichPtr_->getNumRates() )
714 {
715 for ( unsigned int i = 0 ; i < pools_.size(); ++i )
716 pools_[i].updateRateTerms( stoichPtr_->getRateTerms(),
717 stoichPtr_->getNumCoreRates(), index );
718 }
719}

726unsigned int Ksolve::getPoolIndex( const Eref& e ) const
727{
728 if (!stoichPtr_) {

737unsigned int Ksolve::getVoxelIndex( const Eref& e ) const
738{
739 unsigned int ret = e.dataIndex();
740 if ( ret < startVoxel_ || ret >= startVoxel_ + pools_.size() )
741 return OFFNODE;
742 return ret - startVoxel_;

878void Ksolve::getBlock( vector< double >& values ) const
879{
880 unsigned int startVoxel = static_cast<unsigned int>(values[0]);
881 unsigned int numVoxels = static_cast<unsigned int>(values[1]);
882 unsigned int startPool = static_cast<unsigned int>(values[2]);
883 unsigned int numPools = static_cast<unsigned int>(values[3]);
884
885 assert( startVoxel >= startVoxel_ );
886 assert( numVoxels <= pools_.size() );
887 assert( pools_.size() > 0 );
888 assert( numPools + startPool <= pools_[0].size() );
889 values.resize( 4 + numVoxels * numPools );
890
891 for ( unsigned int i = 0; i < numVoxels; ++i )
892 {
893 const double* v = pools_[ startVoxel + i ].S();
894 for ( unsigned int j = 0; j < numPools; ++j )
895 {
896 values[ 4 + j * numVoxels + i] = v[ j + startPool ];
897 }
898 }
899}
