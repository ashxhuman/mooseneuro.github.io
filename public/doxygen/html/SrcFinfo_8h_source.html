99 void send( const Eref& er, T arg ) const
100 {
101 const vector< MsgDigest >& md = er.msgDigest( getBindIndex() );
102 for ( vector< MsgDigest >::const_iterator
103 i = md.begin(); i != md.end(); ++i ) {
104 const OpFunc1Base< T >* f =
105 dynamic_cast< const OpFunc1Base< T >* >( i->func );
106 assert( f );
107 for ( vector< Eref >::const_iterator
108 j = i->targets.begin(); j != i->targets.end(); ++j ) {
109 if ( j->dataIndex() == ALLDATA ) {
110 Element* e = j->element();
111 unsigned int start = e->localDataStart();
112 unsigned int end = start + e->numLocalData();
113 for ( unsigned int k = start; k < end; ++k )
114 f->op( Eref( e, k ), arg );
115 } else {
116 f->op( *j, arg );
117 // Need to send stuff offnode too here. The
118 // target in this case is just the src Element.
119 // Its ObjId gets stuffed into the send buf.
120 // On the other node it will execute
121 // its own send command with the passed in args.
122 }
123 }
124 }
125 }

157 /**
158 * Each target gets successive entries from the arg vector.
159 * Rolls over if the # of targets exceeds vector size.
160 * Fails totally if the targets are off-node.
161 */
162 void sendVec( const Eref& er, const vector< T >& arg ) const
163 {
164 if ( arg.size() == 0 )
165 return;
166 const vector< MsgDigest >& md = er.msgDigest( getBindIndex() );
167 unsigned int argPos = 0;
168 for ( vector< MsgDigest >::const_iterator
169 i = md.begin(); i != md.end(); ++i ) {
170 const OpFunc1Base< T >* f =
171 dynamic_cast< const OpFunc1Base< T >* >( i->func );
172 assert( f );
173 for ( vector< Eref >::const_iterator
174 j = i->targets.begin(); j != i->targets.end(); ++j ) {
175 if ( j->dataIndex() == ALLDATA ) {
176 Element* e = j->element();
177 unsigned int start = e->localDataStart();
178 unsigned int end = start + e->numLocalData();
179 for ( unsigned int k = start; k < end; ++k ) {
180 f->op( Eref( e, k ), arg[ argPos++ ] );
181 if ( argPos >= arg.size() )
182 argPos = 0;
183 }
184 } else {
185 f->op( *j, arg[ argPos++ ] );
186 if ( argPos >= arg.size() )
187 argPos = 0;
188 // Need to send stuff offnode too here. The
189 // target in this case is just the src Element.
190 // Its ObjId gets stuffed into the send buf.
191 // On the other node it will execute
192 // its own send command with the passed in args.
193 }
194 }
195 }
196 }

270 void sendBuffer( const Eref& e, double* buf ) const
271 {
272 // Note that buf2val alters the buf ptr as a side-effect, in
273 // order to advance to the next argument. However, the order
274 // in which argument evaluation is called within a multi-
275 // argument function is undefined. So we have to explicitly
276 // ensure the ordering of argument calls.

278 send( e, arg1, Conv< T2 >::buf2val( &buf ) );
