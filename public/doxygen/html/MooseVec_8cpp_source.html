89ObjId MooseVec::getItem(const int index) const
90{
91 // Handle negative indexing.
92 size_t i = (index < 0) ? size() + index : static_cast<size_t>(index);
93 if(i >= size()) {
94 throw nb::index_error(("Index " + to_string(i) + " out of range").c_str());
95 }
96 if(oid_.element()->hasFields()) {
97 return getFieldItem(i);
98 }
99 return getDataItem(i);
100}

102vector<ObjId> MooseVec::getItemRange(const nb::slice& slice) const
103{
104 auto [start, stop, step, length] = slice.compute(size());
105 vector<ObjId> res;
106 res.reserve(length);
107 for(size_t ii = start; ii < stop; ii += step) {
108 res.push_back(getItem(static_cast<int>(ii)));
109 }
110 return res;
111}

123nb::object MooseVec::getAttribute(const string& name)
124{
125 // Special id level attributes
126 if(name == "numData") {
127 return nb::cast(Field<unsigned int>::get(oid_, "numData"));
128 }
129 if(name == "numField") {
130 return nb::cast(Field<unsigned int>::get(oid_, "numField"));
131 }
132
133 // If type is double, int, bool etc, then return the numpy array. else
134 // return the list of python object.
135 auto cinfo = oid_.element()->cinfo();
136 auto finfo = cinfo->findFinfo(name);
137 if (!finfo) {
138 throw nb::attribute_error((name + " not found on " + path()).c_str());
139 }
140
141 auto rttType = finfo->rttiType();
142 if(rttType == "double")
143 return nb::cast(getAttributeNumpy<double>(name));
144 if(rttType == "unsigned int")
145 return nb::cast(getAttributeNumpy<unsigned int>(name));
146 if(rttType == "int")
147 return nb::cast(getAttributeNumpy<int>(name));
148 if(rttType == "bool")
149 return nb::cast(getAttributeNumpy<bool>(name));
150
151 string finfoType = cinfo->getFinfoType(finfo);
152 if(finfoType == "LookupValueFinfo") {
153 return nb::cast(VecLookupField(oid_, finfo));
154 }
155 if(finfoType == "FieldElementFinfo") {
156 return nb::cast(VecElementField(oid_, finfo));
157 }
158 cerr << "DEBUG: None of the simply handled types: " << finfoType << endl;
159 // For complex types, return list objects
160 nb::list result;
161 for(size_t ii = 0; ii < size(); ii++){
162 result.append(getFieldGeneric(getItem(ii), name));
163 }
164 return result;
165}

178bool MooseVec::setAttribute(const string& name, const nb::object& val)
179{
180 auto cinfo = oid_.element()->cinfo();
181 auto finfo = cinfo->findFinfo(name);
182 if (!finfo) {
183 throw nb::attribute_error((name + " not found on " + path()).c_str());
184 }
185
186 auto rttType = finfo->rttiType();
187
188 bool isIterable = nb::isinstance<nb::iterable>(val) && !nb::isinstance<nb::str>(val);
189
190 if(isIterable) {
191 if(rttType == "double")
192 return setAttrOneToOne<double>(name, nb::cast<vector<double>>(val));
193 if(rttType == "unsigned int")
194 return setAttrOneToOne<unsigned int>(
195 name, nb::cast<vector<unsigned int>>(val));
196 if(rttType == "int")
197 return setAttrOneToOne<int>(
198 name, nb::cast<vector<int>>(val));
199 if(rttType == "bool")
200 return setAttrOneToOne<bool>(name, nb::cast<vector<bool>>(val));
201 if(rttType == "string")
202 return setAttrOneToOne<string>(name, nb::cast<vector<string>>(val));
203 }
204 else {
205 if(rttType == "double")
206 return setAttrOneToAll<double>(name, nb::cast<double>(val));
207 if(rttType == "unsigned int")
208 return setAttrOneToAll<unsigned int>(name,
209 nb::cast<unsigned int>(val));
210 if(rttType == "int")
211 return setAttrOneToAll< int>(name,
212 nb::cast<int>(val));
213 if(rttType == "bool")
214 return setAttrOneToAll<bool>(name, nb::cast<bool>(val));
215 if (rttType == "string")
216 return setAttrOneToAll<string>(name, nb::cast<string>(val));
217 }
218
219 throw nb::type_error(("Unsupported type: " + rttType).c_str());
220}
