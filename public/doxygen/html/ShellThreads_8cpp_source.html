25/**
26 * Launches Parser. Blocking when the parser blocks.
27 */
28void Shell::launchParser()
29{
30 Id shellId;
31 Shell* s = reinterpret_cast< Shell* >( shellId.eref().data() );
32 bool quit = 0;
33
34 cout << "moose : " << flush;
35 while ( !quit ) {
36 string temp;
37 cin >> temp;
38 if ( temp == "quit" || temp == "q" ) {
39 s->doQuit();
40 quit = 1;
41 }
42 }
43 cout << "\nQuitting Moose\n" << flush;
44}
