47 "Name",
48 "HHGateF2D",
49 "Author",
50 "Subhasis Ray, 2025, CHINTA. This is based on HHGate2D"
51 " implementation by Niraj Dudani, 2009, NCBS.",
52 "Description",
53 "HHGateF2D: Gate for Hodkgin-Huxley type channels, equivalent to the"
54 " m and h terms on the Na squid channel and the n term on K.\n"
55 " This is specialized for dependency on two variables, voltage"
56 " and concentration (usually [Ca2+]).\n"
57 " Unlike HHGate2D, HHGateF2D evaluates the formulas for the gate"

63 " The formulas must be specified as in the form f(v, c),"
64 " where v and c are the variable names. For example:\n"
65 " \"1500/(1 + (c / 1.5e-4 * exp(-77 * v)))\"\n"
66 " While the names correspond to voltage and concentration, they can be"
67 " any two parameters mapped to the corresponding fields in"
68 " HHChannelF2D.\n"
69 " Additionally, like HHGateF, HHGateF2D provides the following "
70 " predefined variable names to facilitate intermediate "
71 " calculations:\n"
72 " `alpha` for forward rate,\n"
73 " `beta` for backward rate,\n"
74 " `tau` for time constant, and\n"
75 " `inf` for steady state open fraction\n"

137double HHGateF2D::lookupA(vector<double> v) const
138{
139 if(v.size() < 2) {
140 cerr << "Error: HHGateF2D::lookupA: 2 real numbers needed to lookup "
141 "2D table.\n";
142 return 0.0;
143 }
144
145 if(v.size() > 2) {
146 cerr << "Error: HHGateF2D::getAValue: Only 2 real numbers needed to "
147 "lookup 2D table. "
148 "Using only first 2.\n";
149 }
150 v_ = v[0];
151 conc_ = v[1];
152 return tauInf_ ? beta_.value() / alpha_.value() : alpha_.value();
153}
154
155double HHGateF2D::lookupB(vector<double> v) const
156{
157 if(v.size() < 2) {
158 cerr << "Error: HHGateF2D::lookupB: 2 real numbers needed to lookup "
159 "2D table.\n";
160 return 0.0;
161 }
162
163 if(v.size() > 2) {
164 cerr << "Error: HHGateF2D::getAValue: Only 2 real numbers needed to "
165 "lookup 2D table. "
166 "Using only first 2.\n";
167 }
168
169 return tauInf_ ? 1.0 / alpha_.value() : alpha_.value() + beta_.value();
170}
