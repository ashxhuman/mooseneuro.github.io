90/* Write to a csv file. */
91void StreamerBase::writeToCSVFile( const string& filepath, const OpenMode openmode
92 , const vector<double>& data, const vector<string>& columns )
93{
94 string m = (openmode == WRITE_STR)?"w":"a";
95 FILE* fp = fopen( filepath.c_str(), m.c_str());
96
97 if( NULL == fp )
98 {
99 LOG( moose::warning, "Failed to open " << filepath );
100 return;
101 }
102
103 // If writing in "w" mode, write the header first.
104 if(openmode == WRITE_STR)
105 {
106 string headerText = "";
107 for( vector<string>::const_iterator it = columns.begin();
108 it != columns.end(); it++ )
109 headerText += ( *it + delimiter_ );
110 headerText += eol;
111 fprintf(fp, "%s", headerText.c_str());
112 }
113
114 string text = "";
115 for( unsigned int i = 0; i < data.size(); i+=columns.size() )
116 {
117 // Start of a new row.
118 for( unsigned int ii = 0; ii < columns.size(); ii++ )
119 text += moose::toString( data[i+ii] ) + delimiter_;
120
121 // At the end of each row, we remove the delimiter_ and append newline_.
122 *(text.end()-1) = eol;
123 }
124 fprintf(fp, "%s", text.c_str() );
125 fclose(fp);
126}
