13/**
14 * The CubeMesh represents a chemically identified compartment shaped
15 * like a cuboid. This is not really an effective geometry for most
16 * neurons because it would have to be rather finely subdivided to fit
17 * a typical dendrite or soma volume, but it is general.
18 */

315 /**
316 * For spherical mesh, coords are xyz r0 r1 theta0 theta1 phi0 phi1
317 * For Cylindrical mesh, coords are x1y1z1 x2y2z2 r0 r1 phi0 phi1
318 * For cuboid mesh, coords are x1y1z1 x2y2z2
319 * For tetrahedral mesh, coords are x1y1z1 x2y2z2 x3y3z3 x4y4z4
320 * Later worry about planar meshes. Those would be surfaces
321 */
322
323 /**
324 * Mesh to Space lookup. Indexed by linear mesh index, from 0 to
325 * number of actual mesh entries (occupied cuboids). Returns
326 * spatial index, from 0 to nx * ny * nz - 1.
327 * Needed whenever the cuboid mesh is not filling the entire volume
328 * of the cube, that is, in most cases.
329 */
330 vector< unsigned int > m2s_;
331
332 /**
333 * Space to Mesh lookup. Indexed by spatial index, from
334 * 0 to nx * ny * nz - 1. Specifically, point x y z is indexed as
335 * ( z * ny + y ) * nx + x. Returns mesh index to look up molecules
336 * etc in the specific volume. In case the spatial location is
337 * outside the included volume of the mesh, returns ~0.
338 */
339 vector< unsigned int > s2m_;
340
341 /**
342 * Vector of spatial meshIndices comprising surface of volume in
343 * CubeMesh.
344 */
345 vector< unsigned int > surface_;
