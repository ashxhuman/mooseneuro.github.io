73// Function to convert it into its fully separated path.
74string Id::path() const
75{
76 string ret = Neutral::path(eref());
77
78 // FIXME: Monday 09 March 2020 12:30:27 PM IST, Dilawar Singh
79 // This beaks the path comparison. Getting x.path from x returned in a
80 // list by moose.wildcardFind() and getting path from here doesn't math
81 // when this is enabled.
82 // If we want to remove [0] then it should be done globally and not just
83 // here.
84
85 // Trim off trailing []
86 assert(ret.length() > 0);
87 while (ret.back() == ']') {
88 size_t pos = ret.find_last_of('[');
89 if (pos != string::npos && pos > 0) {
90 ret = ret.substr(0, pos);
91 }
92 }
93
94 return ret;
95}

150void Id::bindIdToElement(Element* e)
151{

154 elements().reserve(elements().size() + 1000);
