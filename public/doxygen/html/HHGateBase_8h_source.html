9#ifndef _HHGateBase_h
10#define _HHGateBase_h
11
12/**
13 * This class is the base for gates in Hodgkin-Huxley type channels
14 * (HHGates). HHGates are typically shared. This means that when you
15 * make a copy or a vector of an HHChannel, there is only a single
16 * HHGate created, and its pointer is used by all the copies. The
17 * lookup functions are thread-safe. Field assignment to the HHGate
18 * should be possible only from the original HHChannel, but all the
19 * others do have read permission.
20 * This base class only factors out the tracking of the ownership.
21 */
22
23class HHGateBase {
24public:
25 /**
26 * Dummy constructor, to keep Dinfo happy. Should never be used
27 */
28 HHGateBase();
29
30 /**
31 * This constructor is the one meant to be used. It takes the
32 * originalId of the parent HHChannel as a required argument,
33 * so that any subsequent 'write' functions can be checked to
34 * see if they are legal. Also tracks its own Id.
35 */
36 HHGateBase(Id originalChanId, Id originalGateId);
37
38 /////////////////////////////////////////////////////////////////
39 // Utility funcs
40 /////////////////////////////////////////////////////////////////
41 /**
42 * Checks if the provided Id is the one that the HHGate was created
43 * on. If true, fine, otherwise complains about trying to set the
44 * field.
45 */
46 bool checkOriginal(Id id, const string& field) const;
47
48 /**
49 * isOriginalChannel returns true if the provided Id is the Id of
50 * the channel on which the HHGate was created.
51 */
52 bool isOriginalChannel(Id id) const;
53
54 /**
55 * isOriginalChannel returns true if the provided Id is the Id of
56 * the Gate created at the same time as the original channel.
57 */
58 bool isOriginalGate(Id id) const;
59
60 /**
61 * Returns the Id of the original Channel.
62 */
63 Id originalChannelId() const;
64
65 /**
66 * Returns the Id of the original Gate.
67 */
68 Id originalGateId() const;
69
70 static const Cinfo* initCinfo();
71
72protected:
73 /**
74 * Id of original channel, the one which has actually allocated it,
75 * All other Elements have to treat the values as readonly.
76 */
77 Id originalChanId_;
78
79 /**
80 * Id of original Gate, the one which was created with the original
81 * channel.
82 * All other Elements have to treat the values as readonly.
83 */
84 Id originalGateId_;
85};
