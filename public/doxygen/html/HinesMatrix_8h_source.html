25struct JunctionStruct
26{
27 JunctionStruct( unsigned int i, unsigned int r ) :
28 index( i ),
29 rank( r )
30 {
31 ;
32 }
33
34 bool operator< ( const JunctionStruct& other ) const
35 {
36 return ( index < other.index );
37 }
38
39 unsigned int index; ///< Hines index of the compartment.
40 unsigned int rank; ///< Number of elements "remaining" in this
41 ///< compartment's group: i.e., number of children
42 ///< with a larger Hines index, +1 for the parent.
43};
