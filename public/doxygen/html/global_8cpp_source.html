47int checkPath(const string& path)
48{
49 if (path.size() < 1)
50 return EMPTY_PATH;

81// MOOSE suffixes [0] to all elements to path. Remove [0] with null
82// character whenever possible. For n > 0, [n] should not be touched. Its
83// the user job to take the pain and write the correct path.
84string createMOOSEPath(const string& path)
85{
86 string s = path; /* Local copy */
87 // Remove [0] from paths. They will be annoying for normal users.
88 std::string::size_type n = 0;
89 string zeroIndex("[0]");
90 while ((n = s.find(zeroIndex, n)) != std::string::npos)
91 s.erase(n, zeroIndex.size());
92 return s;
93}

95/**
96 * @brief Create directories recursively needed to open the given file p.
97 *
98 * @param path When successfully created, returns created path, else
99 * convert path to a filename by replacing '/' by '_'.
100 */
101bool createParentDirs(const string& path)
102{
103 // Remove the filename from the given path so we only have the
104 // directory.
105 string p = path;
106 bool failed = false;
107 size_t pos = p.find_last_of('/');
108 if (pos != std::string::npos)
109 p = p.substr(0, pos);
110 else /* no parent directory to create */
111 return true;
112
113 if (p.size() == 0)
114 return true;
115
116 string command("mkdir -p ");
117 command += p;
118 int ret = system(command.c_str());
119 struct stat info;
120 if (stat(p.c_str(), &info) != 0)
121 {
122 LOG(moose::warning, "cannot access " << p);
123 return false;
124 }
125 else if (info.st_mode & S_IFDIR)
126 {
127 LOG(moose::info, "Created directory " << p);
128 return true;
129 }
130 else
131 {
132 LOG(moose::warning, p << " is no directory");
133 return false;
134 }
135 return true;
136}

174/* --------------------------------------------------------------------------*/
175/**
176 * @Synopsis Generate a suitable column name based on given path. Replace
177 * [\d+] with \d+; and remove [\0+]. Replace / with given delim (default '.')
178 *
179 * @Param path (string).
180 * @Param delim (char)
181 * @Param maxLevel (int). Max number of parents to include. Default 1.
182 *
183 * @Returns Reformatted path suitable for column name.
184 *
185 * Example:
186 * -------
187 * /a[12]/d[0]/e[1]/d[0] -> e1.d
188 * /a[12]/d[0] -> e12.d
189 * /a[00]/b[0] -> a.b
190 */
191/* ----------------------------------------------------------------------------*/
192string moosePathToColumnName(const string& path, char delim, size_t maxParents)
193{
194 string s(path);
195 static std::regex e0("\\[(0+)\\]"); // Remove [0+]
196 static std::regex e1("\\[(\\d+)\\]"); // Replace [xyz] by xyz
197 s = std::regex_replace(s, e0, "");
198 s = std::regex_replace(s, e1, "$1");
199
200 string s2;
201 string colname = "";
202 size_t nBreak = 0;
203
204 // Keep as many parents as required by maxParents. Using reverse magic.
205 for (auto rit = s.rbegin(); rit != s.rend(); rit++)
206 {
207 if (*rit == '/')
208 {
209 colname = s2 + delim + colname;
210 s2 = "";
211 nBreak += 1;
212 if (nBreak == (1 + maxParents))
213 break;
214 }
215 else
216 s2 = *rit + s2;
217 }
218 colname.pop_back();
219 return colname;
220}
