29SwcSegment::SwcSegment( const string& line )
30 :
31 geometricalDistanceFromSoma_( 0.0 ),
32 electrotonicDistanceFromSoma_( 0.0 )
33{
34 vector< string > args;
35 stringstream ss( line );
36 string temp;
37 while (ss >> temp ) {
38 args.push_back( temp );
39 }
40 if ( args.size() == 7 ) {
41 myIndex_ = atoi( args[0].c_str() );
42 type_ = atoi( args[1].c_str() );
43 double x = atof( args[2].c_str() );
44 double y = atof( args[3].c_str() );
45 double z = atof( args[4].c_str() );
46 v_ = Vec( x, y, z );
47 radius_ = atof( args[5].c_str() );
48 int pa = atoi( args[6].c_str() );
49 if ( pa > 0 )
50 parent_ = pa;
51 else
52 parent_ = ~0U;
53 } else {
54 type_ = UNDEF;
55 }
56}

77void SwcSegment::figureOutType()

90SwcBranch::SwcBranch( int i, const SwcSegment& start, double len, double L,
91 const vector< int >& cable )
92 : SwcSegment( start ),
93 r0( start.radius() ),
94 r1( start.radius() ),
95 geomLength( len ),
96 electroLength( L )
97{
98 myIndex_ = i;
99 parent_ = 0;
100 kids_.resize( 0 );
101 segs_.resize( cable.size() );
102 // Put the contents of cable into segs, in reverse order.
103 vector< int >::const_reverse_iterator j = cable.rbegin();
104 vector< int >::iterator k = segs_.begin();
105 for ( k = segs_.begin(); k != segs_.end(); ++k )
106 *k = *j++;
107}
